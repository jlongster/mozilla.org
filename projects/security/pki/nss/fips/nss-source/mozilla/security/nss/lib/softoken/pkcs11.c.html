   38 /*
   39  * This file implements PKCS 11 on top of our existing security modules
   40  *
   41  * For more information about PKCS 11 See PKCS 11 Token Inteface Standard.
   42  *   This implementation has two slots:
   43  *        slot 1 is our generic crypto support. It does not require login.
   44  *   It supports Public Key ops, and all they bulk ciphers and hashes. 
   45  *   It can also support Private Key ops for imported Private keys. It does 
   46  *   not have any token storage.
   47  *        slot 2 is our private key support. It requires a login before use. It
   48  *   can store Private Keys and Certs as token objects. Currently only private
   49  *   keys and their associated Certificates are saved on the token.
   50  *
   51  *   In this implementation, session objects are only visible to the session
   52  *   that created or generated them.
   53  */

  456 static char *
  457 sftk_setStringName(const char *inString, char *buffer, int buffer_length)
  458 {
  459     int full_length, string_length;
  460 
  461     full_length = buffer_length -1;
  462     string_length = PORT_Strlen(inString);
  463     /* 
  464      *  shorten the string, respecting utf8 encoding
  465      *  to do so, we work backward from the end 
  466      *  bytes looking from the end are either:
  467      *    - ascii [0x00,0x7f]
  468      *    - the [2-n]th byte of a multibyte sequence 
  469      *        [0x3F,0xBF], i.e, most significant 2 bits are '10'
  470      *    - the first byte of a multibyte sequence [0xC0,0xFD],
  471      *        i.e, most significant 2 bits are '11'
  472      *
  473      *    When the string is too long, we lop off any trailing '10' bytes,
  474      *  if any. When these are all eliminated we lop off
  475      *  one additional byte. Thus if we lopped any '10'
  476      *  we'll be lopping a '11' byte (the first byte of the multibyte sequence),
  477      *  otherwise we're lopping off an ascii character.
  478      *
  479      *    To test for '10' bytes, we first AND it with 
  480      *  11000000 (0xc0) so that we get 10000000 (0x80) if and only if
  481      *  the byte starts with 10. We test for equality.
  482      */
  483     while ( string_length > full_length ) {
  484         /* need to shorten */
  485         while ( string_length > 0 && 
  486               ((inString[string_length-1]&(char)0xc0) == (char)0x80)) {
  487             /* lop off '10' byte */
  488             string_length--;
  489         }
  490         /* 
  491          * test string_length in case bad data is received
  492          * and string consisted of all '10' bytes,
  493          * avoiding any infinite loop
  494          */
  495         if ( string_length ) {
  496             /* remove either '11' byte or an asci byte */
  497             string_length--;
  498         }
  499     }
  500     PORT_Memset(buffer,' ',full_length);
  501     buffer[full_length] = 0;
  502     PORT_Memcpy(buffer,inString,string_length);
  503     return buffer;
  504 }

  594 /*
  595  * check the consistancy and initialize a Certificate Object 
  596  */
  597 static CK_RV
  598 sftk_handleCertObject(SFTKSession *session,SFTKObject *object)
  599 {
  600     CK_CERTIFICATE_TYPE type;
  601     SFTKAttribute *attribute;
  602     CK_RV crv;
  603 
  604     /* certificates must have a type */
  605     if ( !sftk_hasAttribute(object,CKA_CERTIFICATE_TYPE) ) {
  606         return CKR_TEMPLATE_INCOMPLETE;
  607     }
  608 
  609     /* we can't store any certs private */
  610     if (sftk_isTrue(object,CKA_PRIVATE)) {
  611         return CKR_ATTRIBUTE_VALUE_INVALID;
  612     }
  613         
  614     /* We only support X.509 Certs for now */
  615     attribute = sftk_FindAttribute(object,CKA_CERTIFICATE_TYPE);
  616     if (attribute == NULL) return CKR_TEMPLATE_INCOMPLETE;
  617     type = *(CK_CERTIFICATE_TYPE *)attribute->attrib.pValue;
  618     sftk_FreeAttribute(attribute);
  619 
  620     if (type != CKC_X_509) {
  621         return CKR_ATTRIBUTE_VALUE_INVALID;
  622     }
  623 
  624     /* X.509 Certificate */
  625 
  626     /* make sure we have a cert */
  627     if ( !sftk_hasAttribute(object,CKA_VALUE) ) {
  628         return CKR_TEMPLATE_INCOMPLETE;
  629     }
  630 
  631     /* in PKCS #11, Subject is a required field */
  632     if ( !sftk_hasAttribute(object,CKA_SUBJECT) ) {
  633         return CKR_TEMPLATE_INCOMPLETE;
  634     }
  635 
  636     /* in PKCS #11, Issuer is a required field */
  637     if ( !sftk_hasAttribute(object,CKA_ISSUER) ) {
  638         return CKR_TEMPLATE_INCOMPLETE;
  639     }
  640 
  641     /* in PKCS #11, Serial is a required field */
  642     if ( !sftk_hasAttribute(object,CKA_SERIAL_NUMBER) ) {
  643         return CKR_TEMPLATE_INCOMPLETE;
  644     }
  645 
  646     /* add it to the object */
  647     object->objectInfo = NULL;
  648     object->infoFree = (SFTKFree) NULL;
  649     
  650     /* now just verify the required date fields */
  651     crv = sftk_defaultAttribute(object, CKA_ID, NULL, 0);
  652     if (crv != CKR_OK) { return crv; }
  653 
  654     if (sftk_isTrue(object,CKA_TOKEN)) {
  655         SFTKSlot *slot = session->slot;
  656         SECItem derCert;
  657         NSSLOWCERTCertificate *cert;
  658          NSSLOWCERTCertTrust *trust = NULL;
  659          NSSLOWCERTCertTrust userTrust = 
  660                 { CERTDB_USER, CERTDB_USER, CERTDB_USER };
  661          NSSLOWCERTCertTrust defTrust = 
  662                 { CERTDB_TRUSTED_UNKNOWN, 
  663                         CERTDB_TRUSTED_UNKNOWN, CERTDB_TRUSTED_UNKNOWN };
  664         char *label = NULL;
  665         char *email = NULL;
  666         SECStatus rv;
  667         PRBool inDB = PR_TRUE;
  668         NSSLOWCERTCertDBHandle *certHandle = sftk_getCertDB(slot);
  669         NSSLOWKEYDBHandle *keyHandle = NULL;
  670 
  671         if (certHandle == NULL) {
  672             return CKR_TOKEN_WRITE_PROTECTED;
  673         }
  674 
  675         /* get the der cert */ 
  676         attribute = sftk_FindAttribute(object,CKA_VALUE);
  677         PORT_Assert[1](attribute);
  678 
  679         derCert.type = 0;
  680         derCert.data = (unsigned char *)attribute->attrib.pValue;
  681         derCert.len = attribute->attrib.ulValueLen ;
  682 
  683         label = sftk_getString(object,CKA_LABEL);
  684 
  685         cert =  nsslowcert_FindCertByDERCert(certHandle, &derCert);
  686         if (cert == NULL) {
  687             cert = nsslowcert_DecodeDERCertificate(&derCert, label);
  688             inDB = PR_FALSE;
  689         }
  690         if (cert == NULL) {
  691             if (label) PORT_Free(label);
  692                 sftk_FreeAttribute(attribute);
  693             sftk_freeCertDB(certHandle);
  694             return CKR_ATTRIBUTE_VALUE_INVALID;
  695         }
  696 
  697         keyHandle = sftk_getKeyDB(slot);
  698         if (keyHandle) {
  699             if (nsslowkey_KeyForCertExists(keyHandle,cert)) {
  700                 trust = &userTrust;
  701             }
  702             sftk_freeKeyDB(keyHandle);
  703         }
  704 
  705         if (!inDB) {
  706             if (!trust) trust = &defTrust;
  707             rv = nsslowcert_AddPermCert(certHandle, cert, label, trust);
  708         } else {
  709             rv = trust ? nsslowcert_ChangeCertTrust(certHandle,cert,trust) :
  710                                 SECSuccess;
  711         }
  712 
  713         if (label) PORT_Free(label);
  714         sftk_FreeAttribute(attribute);
  715 
  716         if (rv != SECSuccess) {
  717             sftk_freeCertDB(certHandle);
  718             nsslowcert_DestroyCertificate(cert);
  719             return CKR_DEVICE_ERROR;
  720         }
  721 
  722         /*
  723          * Add a NULL S/MIME profile if necessary.
  724          */
  725         email = sftk_getString(object,CKA_NETSCAPE_EMAIL);
  726         if (email) {
  727             certDBEntrySMime *entry;
  728 
  729             entry = nsslowcert_ReadDBSMimeEntry(certHandle,email);
  730             if (!entry) {
  731                     nsslowcert_SaveSMimeProfile(certHandle, email, 
  732                                                 &cert->derSubject, NULL, NULL);
  733             } else {
  734                  nsslowcert_DestroyDBEntry((certDBEntry *)entry);
  735             }
  736             PORT_Free(email);
  737         }
  738         sftk_freeCertDB(certHandle);
  739         object->handle=sftk_mkHandle(slot,&cert->certKey,SFTK_TOKEN_TYPE_CERT);
  740         nsslowcert_DestroyCertificate(cert);
  741     }
  742 
  743     return CKR_OK;
  744 }

  746 unsigned int
  747 sftk_MapTrust(CK_TRUST trust, PRBool clientAuth)
  748 {
  749     unsigned int trustCA = clientAuth ? CERTDB_TRUSTED_CLIENT_CA :
  750                                                         CERTDB_TRUSTED_CA;
  751     switch (trust) {
  752     case CKT_NETSCAPE_TRUSTED:
  753         return CERTDB_VALID_PEER|CERTDB_TRUSTED;
  754     case CKT_NETSCAPE_TRUSTED_DELEGATOR:
  755         return CERTDB_VALID_CA|trustCA;
  756     case CKT_NETSCAPE_UNTRUSTED:
  757         return CERTDB_NOT_TRUSTED;
  758     case CKT_NETSCAPE_MUST_VERIFY:
  759         return 0;
  760     case CKT_NETSCAPE_VALID: /* implies must verify */
  761         return CERTDB_VALID_PEER;
  762     case CKT_NETSCAPE_VALID_DELEGATOR: /* implies must verify */
  763         return CERTDB_VALID_CA;
  764     default:
  765         break;
  766     }
  767     return CERTDB_TRUSTED_UNKNOWN;
  768 }

  856             }
  857             sftk_FreeAttribute(trust);
  858         }
  859         trust = sftk_FindAttribute(object,CKA_TRUST_CODE_SIGNING);
  860         if (trust) {
  861             if (trust->attrib.ulValueLen == sizeof(CK_TRUST)) {
  862                 PORT_Memcpy(&signTrust,trust->attrib.pValue,
  863                                                         sizeof(signTrust));
  864             }
  865             sftk_FreeAttribute(trust);
  866         }
  867         stepUp = CK_FALSE;
  868         trust = sftk_FindAttribute(object,CKA_TRUST_STEP_UP_APPROVED);
  869         if (trust) {
  870             if (trust->attrib.ulValueLen == sizeof(CK_BBOOL)) {
  871                 stepUp = *(CK_BBOOL*)trust->attrib.pValue;
  872             }
  873             sftk_FreeAttribute(trust);
  874         }
  875 
  876         /* preserve certain old fields */
  877         if (cert->trust) {
  878             dbTrust.sslFlags =
  879                           cert->trust->sslFlags & CERTDB_PRESERVE_TRUST_BITS;
  880             dbTrust.emailFlags=
  881                         cert->trust->emailFlags & CERTDB_PRESERVE_TRUST_BITS;
  882             dbTrust.objectSigningFlags = 
  883                 cert->trust->objectSigningFlags & CERTDB_PRESERVE_TRUST_BITS;
  884         }
  885 
  886         dbTrust.sslFlags |= sftk_MapTrust(sslTrust,PR_FALSE);
  887         dbTrust.sslFlags |= sftk_MapTrust(clientTrust,PR_TRUE);
  888         dbTrust.emailFlags |= sftk_MapTrust(emailTrust,PR_FALSE);
  889         dbTrust.objectSigningFlags |= sftk_MapTrust(signTrust,PR_FALSE);
  890         if (stepUp) {
  891             dbTrust.sslFlags |= CERTDB_GOVT_APPROVED_CA;
  892         }
  893 
  894         rv = nsslowcert_ChangeCertTrust(certHandle,cert,&dbTrust);
  895         object->handle=sftk_mkHandle(slot,&cert->certKey,SFTK_TOKEN_TYPE_TRUST);
  896         nsslowcert_DestroyCertificate(cert);
  897         sftk_freeCertDB(certHandle);
  898         if (rv != SECSuccess) {
  899            return CKR_DEVICE_ERROR;
  900         }
  901     }
  902 
  903     return CKR_OK;
  904 }

 1072 static CK_RV
 1073 sftk_handlePublicKeyObject(SFTKSession *session, SFTKObject *object,
 1074                                                          CK_KEY_TYPE key_type)
 1075 {
 1076     CK_BBOOL encrypt = CK_TRUE;
 1077     CK_BBOOL recover = CK_TRUE;
 1078     CK_BBOOL wrap = CK_TRUE;
 1079     CK_BBOOL derive = CK_FALSE;
 1080     CK_BBOOL verify = CK_TRUE;
 1081     CK_ATTRIBUTE_TYPE pubKeyAttr = CKA_VALUE;
 1082     CK_RV crv;
 1083 
 1084     switch (key_type) {
 1085     case CKK_RSA:
 1086         crv = sftk_ConstrainAttribute(object, CKA_MODULUS,
 1087                                                  RSA_MIN_MODULUS_BITS, 0, 0);
 1088         if (crv != CKR_OK) {
 1089             return crv;
 1090         }
 1091         crv = sftk_ConstrainAttribute(object, CKA_PUBLIC_EXPONENT, 2, 0, 0);
 1092         if (crv != CKR_OK) {
 1093             return crv;
 1094         }
 1095         pubKeyAttr = CKA_MODULUS;
 1096         break;
 1097     case CKK_DSA:
 1098         crv = sftk_ConstrainAttribute(object, CKA_SUBPRIME, 
 1099                                                 DSA_Q_BITS, DSA_Q_BITS, 0);
 1100         if (crv != CKR_OK) {
 1101             return crv;
 1102         }
 1103         crv = sftk_ConstrainAttribute(object, CKA_PRIME, 
 1104                                         DSA_MIN_P_BITS, DSA_MAX_P_BITS, 64);
 1105         if (crv != CKR_OK) {
 1106             return crv;
 1107         }
 1108         crv = sftk_ConstrainAttribute(object, CKA_BASE, 1, DSA_MAX_P_BITS, 0);
 1109         if (crv != CKR_OK) {
 1110             return crv;
 1111         }
 1112         crv = sftk_ConstrainAttribute(object, CKA_VALUE, 1, DSA_MAX_P_BITS, 0);
 1113         if (crv != CKR_OK) {
 1114             return crv;
 1115         }
 1116         encrypt = CK_FALSE;
 1117         recover = CK_FALSE;
 1118         wrap = CK_FALSE;
 1119         break;
 1120     case CKK_DH:
 1121         crv = sftk_ConstrainAttribute(object, CKA_PRIME, 
 1122                                         DH_MIN_P_BITS, DH_MAX_P_BITS, 0);
 1123         if (crv != CKR_OK) {
 1124             return crv;
 1125         }
 1126         crv = sftk_ConstrainAttribute(object, CKA_BASE, 1, DH_MAX_P_BITS, 0);
 1127         if (crv != CKR_OK) {
 1128             return crv;
 1129         }
 1130         crv = sftk_ConstrainAttribute(object, CKA_VALUE, 1, DH_MAX_P_BITS, 0);
 1131         if (crv != CKR_OK) {
 1132             return crv;
 1133         }
 1134         verify = CK_FALSE;
 1135         derive = CK_TRUE;
 1136         encrypt = CK_FALSE;
 1137         recover = CK_FALSE;
 1138         wrap = CK_FALSE;
 1139         break;
 1140 #ifdef NSS_ENABLE_ECC
 1141     case CKK_EC:
 1142         if ( !sftk_hasAttribute(object, CKA_EC_PARAMS)) {
 1143             return CKR_TEMPLATE_INCOMPLETE;
 1144         }
 1145         if ( !sftk_hasAttribute(object, CKA_EC_POINT)) {
 1146             return CKR_TEMPLATE_INCOMPLETE;
 1147         }
 1148         pubKeyAttr = CKA_EC_POINT;
 1149         derive = CK_TRUE;    /* for ECDH */
 1150         verify = CK_TRUE;    /* for ECDSA */
 1151         encrypt = CK_FALSE;
 1152         recover = CK_FALSE;
 1153         wrap = CK_FALSE;
 1154         break;
 1155 #endif /* NSS_ENABLE_ECC */
 1156     default:
 1157         return CKR_ATTRIBUTE_VALUE_INVALID;
 1158     }
 1159 
 1160     /* make sure the required fields exist */
 1161     crv = sftk_defaultAttribute(object,CKA_SUBJECT,NULL,0);
 1162     if (crv != CKR_OK)  return crv; 
 1163     crv = sftk_defaultAttribute(object,CKA_ENCRYPT,&encrypt,sizeof(CK_BBOOL));
 1164     if (crv != CKR_OK)  return crv; 
 1165     crv = sftk_defaultAttribute(object,CKA_VERIFY,&verify,sizeof(CK_BBOOL));
 1166     if (crv != CKR_OK)  return crv; 
 1167     crv = sftk_defaultAttribute(object,CKA_VERIFY_RECOVER,
 1168                                                 &recover,sizeof(CK_BBOOL));
 1169     if (crv != CKR_OK)  return crv; 
 1170     crv = sftk_defaultAttribute(object,CKA_WRAP,&wrap,sizeof(CK_BBOOL));
 1171     if (crv != CKR_OK)  return crv; 
 1172     crv = sftk_defaultAttribute(object,CKA_DERIVE,&derive,sizeof(CK_BBOOL));
 1173     if (crv != CKR_OK)  return crv; 
 1174 
 1175     object->objectInfo = sftk_GetPubKey(object,key_type, &crv);
 1176     if (object->objectInfo == NULL) {
 1177         return crv;
 1178     }
 1179     object->infoFree = (SFTKFree) nsslowkey_DestroyPublicKey;
 1180 
 1181     if (sftk_isTrue(object,CKA_TOKEN)) {
 1182         SFTKSlot *slot = session->slot;
 1183         NSSLOWKEYPrivateKey *priv;
 1184         SECItem pubKey;
 1185         NSSLOWKEYDBHandle *keyHandle = NULL;
 1186 
 1187         crv = sftk_Attribute2SSecItem(NULL,&pubKey,object,pubKeyAttr);
 1188         if (crv != CKR_OK) return crv;
 1189 
 1190         PORT_Assert[1](pubKey.data);
 1191         keyHandle = sftk_getKeyDB(slot);
 1192         if (keyHandle == NULL) {
 1193             PORT_Free(pubKey.data);
 1194             return CKR_TOKEN_WRITE_PROTECTED;
 1195         }
 1196         if (keyHandle->version != 3) {
 1197             unsigned char buf[SHA1_LENGTH];
 1198             SHA1_HashBuf(buf,pubKey.data,pubKey.len);
 1199             PORT_Memcpy(pubKey.data,buf,sizeof(buf));
 1200             pubKey.len = sizeof(buf);
 1201         }
 1202         /* make sure the associated private key already exists */
 1203         /* only works if we are logged in */
 1204         priv = nsslowkey_FindKeyByPublicKey(keyHandle, &pubKey, slot->password);
 1205         sftk_freeKeyDB(keyHandle);
 1206         if (priv == NULL) {
 1207             PORT_Free(pubKey.data);
 1208             return crv;
 1209         }
 1210         nsslowkey_DestroyPrivateKey(priv);
 1211 
 1212         object->handle = sftk_mkHandle(slot, &pubKey, SFTK_TOKEN_TYPE_PUB);
 1213         PORT_Free(pubKey.data);
 1214     }
 1215 
 1216     return CKR_OK;
 1217 }

 1399          * applied to rsa private keys anyway (other private keys did not
 1400          * get their attributes NULL'ed out */
 1401     }
 1402     return CKR_OK;
 1403 }

 1409 /* Validate secret key data, and set defaults */
 1410 static CK_RV
 1411 validateSecretKey(SFTKSession *session, SFTKObject *object, 
 1412                                         CK_KEY_TYPE key_type, PRBool isFIPS)
 1413 {
 1414     CK_RV crv;
 1415     CK_BBOOL cktrue = CK_TRUE;
 1416     CK_BBOOL ckfalse = CK_FALSE;
 1417     SFTKAttribute *attribute = NULL;
 1418     unsigned long requiredLen;
 1419 
 1420     crv = sftk_defaultAttribute(object,CKA_SENSITIVE,
 1421                                 isFIPS?&cktrue:&ckfalse,sizeof(CK_BBOOL));
 1422     if (crv != CKR_OK)  return crv; 
 1423     crv = sftk_defaultAttribute(object,CKA_EXTRACTABLE,
 1424                                                 &cktrue,sizeof(CK_BBOOL));
 1425     if (crv != CKR_OK)  return crv; 
 1426     crv = sftk_defaultAttribute(object,CKA_ENCRYPT,&cktrue,sizeof(CK_BBOOL));
 1427     if (crv != CKR_OK)  return crv; 
 1428     crv = sftk_defaultAttribute(object,CKA_DECRYPT,&cktrue,sizeof(CK_BBOOL));
 1429     if (crv != CKR_OK)  return crv; 
 1430     crv = sftk_defaultAttribute(object,CKA_SIGN,&ckfalse,sizeof(CK_BBOOL));
 1431     if (crv != CKR_OK)  return crv; 
 1432     crv = sftk_defaultAttribute(object,CKA_VERIFY,&ckfalse,sizeof(CK_BBOOL));
 1433     if (crv != CKR_OK)  return crv; 
 1434     crv = sftk_defaultAttribute(object,CKA_WRAP,&cktrue,sizeof(CK_BBOOL));
 1435     if (crv != CKR_OK)  return crv; 
 1436     crv = sftk_defaultAttribute(object,CKA_UNWRAP,&cktrue,sizeof(CK_BBOOL));
 1437     if (crv != CKR_OK)  return crv; 
 1438 
 1439     if ( !sftk_hasAttribute(object, CKA_VALUE)) {
 1440         return CKR_TEMPLATE_INCOMPLETE;
 1441     }
 1442     /* the next two bits get modified only in the key gen and token cases */
 1443     crv = sftk_forceAttribute(object,CKA_ALWAYS_SENSITIVE,
 1444                                                 &ckfalse,sizeof(CK_BBOOL));
 1445     if (crv != CKR_OK)  return crv; 
 1446     crv = sftk_forceAttribute(object,CKA_NEVER_EXTRACTABLE,
 1447                                                 &ckfalse,sizeof(CK_BBOOL));
 1448     if (crv != CKR_OK)  return crv; 
 1449 
 1450     /* some types of keys have a value length */
 1451     crv = CKR_OK;
 1452     switch (key_type) {
 1453     /* force CKA_VALUE_LEN to be set */
 1454     case CKK_GENERIC_SECRET:
 1455     case CKK_RC2:
 1456     case CKK_RC4:
 1457 #if NSS_SOFTOKEN_DOES_RC5
 1458     case CKK_RC5:
 1459 #endif
 1460 #ifdef NSS_SOFTOKEN_DOES_CAST
 1461     case CKK_CAST:
 1462     case CKK_CAST3:
 1463     case CKK_CAST5:
 1464 #endif
 1465 #if NSS_SOFTOKEN_DOES_IDEA
 1466     case CKK_IDEA:
 1467 #endif
 1468         attribute = sftk_FindAttribute(object,CKA_VALUE);
 1469         /* shouldn't happen */
 1470         if (attribute == NULL) return CKR_TEMPLATE_INCOMPLETE;
 1471         crv = sftk_forceAttribute(object, CKA_VALUE_LEN, 
 1472                         &attribute->attrib.ulValueLen, sizeof(CK_ULONG));
 1473         sftk_FreeAttribute(attribute);
 1474         break;
 1475     /* force the value to have the correct parity */
 1476     case CKK_DES:
 1477     case CKK_DES2:
 1478     case CKK_DES3:
 1479     case CKK_CDMF:
 1480         attribute = sftk_FindAttribute(object,CKA_VALUE);
 1481         /* shouldn't happen */
 1482         if (attribute == NULL) 
 1483             return CKR_TEMPLATE_INCOMPLETE;
 1484         requiredLen = sftk_MapKeySize(key_type);
 1485         if (attribute->attrib.ulValueLen != requiredLen) {
 1486             sftk_FreeAttribute(attribute);
 1487             return CKR_KEY_SIZE_RANGE;
 1488         }
 1489         sftk_FormatDESKey((unsigned char*)attribute->attrib.pValue,
 1490                                                  attribute->attrib.ulValueLen);
 1491         sftk_FreeAttribute(attribute);
 1492         break;
 1493     default:
 1494         break;
 1495     }
 1496 
 1497     return crv;
 1498 }

 1500 #define SFTK_KEY_MAX_RETRIES 10 /* don't hang if we are having problems with the rng */
 1501 #define SFTK_KEY_ID_SIZE 18 /* don't use either SHA1 or MD5 sizes */
 1502 /*
 1503  * Secret keys must have a CKA_ID value to be stored in the database. This code
 1504  * will generate one if there wasn't one already. 
 1505  */
 1506 static CK_RV
 1507 sftk_GenerateSecretCKA_ID(NSSLOWKEYDBHandle *handle, SECItem *id, char *label)
 1508 {
 1509     unsigned int retries;
 1510     SECStatus rv = SECSuccess;
 1511     CK_RV crv = CKR_OK;
 1512 
 1513     id->data = NULL;
 1514     if (label) {
 1515         id->data = (unsigned char *)PORT_Strdup(label);
 1516         if (id->data == NULL) {
 1517             return CKR_HOST_MEMORY;
 1518         }
 1519         id->len = PORT_Strlen(label)+1;
 1520         if (!nsslowkey_KeyForIDExists(handle,id)) { 
 1521             return CKR_OK;
 1522         }
 1523         PORT_Free(id->data);
 1524         id->data = NULL;
 1525         id->len = 0;
 1526     }
 1527     id->data = (unsigned char *)PORT_Alloc(SFTK_KEY_ID_SIZE);
 1528     if (id->data == NULL) {
 1529         return CKR_HOST_MEMORY;
 1530     }
 1531     id->len = SFTK_KEY_ID_SIZE;
 1532 
 1533     retries = 0;
 1534     do {
 1535         rv = RNG_GenerateGlobalRandomBytes(id->data,id->len);
 1536     } while (rv == SECSuccess && nsslowkey_KeyForIDExists(handle,id) && 
 1537                                 (++retries <= SFTK_KEY_MAX_RETRIES));
 1538 
 1539     if ((rv != SECSuccess) || (retries > SFTK_KEY_MAX_RETRIES)) {
 1540         if (rv != SECSuccess) {
 1541             sftk_fatalError[1] = PR_TRUE;
 1542         }
 1543         crv = CKR_DEVICE_ERROR; /* random number generator is bad */
 1544         PORT_Free(id->data);
 1545         id->data = NULL;
 1546         id->len = 0;
 1547     }
 1548     return crv;
 1549 }

 1551 /*
 1552  * check the consistancy and initialize a Secret Key Object 
 1553  */
 1554 static CK_RV
 1555 sftk_handleSecretKeyObject(SFTKSession *session,SFTKObject *object,
 1556                                         CK_KEY_TYPE key_type, PRBool isFIPS)
 1557 {
 1558     CK_RV crv;
 1559     NSSLOWKEYPrivateKey *privKey   = NULL;
 1560     NSSLOWKEYDBHandle   *keyHandle = NULL;
 1561     SECItem pubKey;
 1562     char *label = NULL;
 1563 
 1564     pubKey.data = 0;
 1565 
 1566     /* First validate and set defaults */
 1567     crv = validateSecretKey(session, object, key_type, isFIPS);
 1568     if (crv != CKR_OK) goto loser;
 1569 
 1570     /* If the object is a TOKEN object, store in the database */
 1571     if (sftk_isTrue(object,CKA_TOKEN)) {
 1572         SFTKSlot *slot = session->slot;
 1573         SECStatus rv = SECSuccess;
 1574         keyHandle = sftk_getKeyDB(slot);
 1575 
 1576         if (keyHandle == NULL) {
 1577             return CKR_TOKEN_WRITE_PROTECTED;
 1578         }
 1579 
 1580         label = sftk_getString(object,CKA_LABEL);
 1581 
 1582         crv = sftk_Attribute2SecItem(NULL, &pubKey, object, CKA_ID);  
 1583                                                 /* Should this be ID? */
 1584         if (crv != CKR_OK) goto loser;
 1585 
 1586         /* if we don't have an ID, generate one */
 1587         if (pubKey.len == 0) {
 1588             if (pubKey.data) { 
 1589                 PORT_Free(pubKey.data);
 1590                 pubKey.data = NULL;
 1591             }
 1592             crv = sftk_GenerateSecretCKA_ID(keyHandle, &pubKey, label);
 1593             if (crv != CKR_OK) goto loser;
 1594 
 1595             crv = sftk_forceAttribute(object, CKA_ID, pubKey.data, pubKey.len);
 1596             if (crv != CKR_OK) goto loser;
 1597         }
 1598 
 1599         privKey = sftk_mkSecretKeyRep(object);
 1600         if (privKey == NULL) {
 1601             crv = CKR_HOST_MEMORY;
 1602             goto loser;
 1603         }
 1604 
 1605         rv = nsslowkey_StoreKeyByPublicKey(keyHandle,
 1606                         privKey, &pubKey, label, slot->password);
 1607         if (rv != SECSuccess) {
 1608             crv = CKR_DEVICE_ERROR;
 1609             goto loser;
 1610         }
 1611 
 1612         object->handle = sftk_mkHandle(slot,&pubKey,SFTK_TOKEN_TYPE_KEY);
 1613     }
 1614 
 1615 loser:
 1616     if (keyHandle) sftk_freeKeyDB(keyHandle);
 1617     if (label) PORT_Free(label);
 1618     if (privKey) nsslowkey_DestroyPrivateKey(privKey);
 1619     if (pubKey.data) PORT_Free(pubKey.data);
 1620 
 1621     return crv;
 1622 }

 1780 /* 
 1781  * Handle Object does all the object consistancy checks, automatic attribute
 1782  * generation, attribute defaulting, etc. If handleObject succeeds, the object
 1783  * will be assigned an object handle, and the object installed in the session
 1784  * or stored in the DB.
 1785  */
 1786 CK_RV
 1787 sftk_handleObject(SFTKObject *object, SFTKSession *session)
 1788 {
 1789     SFTKSlot *slot = session->slot;
 1790     CK_BBOOL ckfalse = CK_FALSE;
 1791     CK_BBOOL cktrue = CK_TRUE;
 1792     SFTKAttribute *attribute;
 1793     CK_RV crv;
 1794 
 1795     /* make sure all the base object types are defined. If not set the
 1796      * defaults */
 1797     crv = sftk_defaultAttribute(object,CKA_TOKEN,&ckfalse,sizeof(CK_BBOOL));
 1798     if (crv != CKR_OK) return crv;
 1799     crv = sftk_defaultAttribute(object,CKA_PRIVATE,&ckfalse,sizeof(CK_BBOOL));
 1800     if (crv != CKR_OK) return crv;
 1801     crv = sftk_defaultAttribute(object,CKA_LABEL,NULL,0);
 1802     if (crv != CKR_OK) return crv;
 1803     crv = sftk_defaultAttribute(object,CKA_MODIFIABLE,&cktrue,sizeof(CK_BBOOL));
 1804     if (crv != CKR_OK) return crv;
 1805 
 1806     /* don't create a private object if we aren't logged in */
 1807     if ((!slot->isLoggedIn) && (slot->needLogin) &&
 1808                                 (sftk_isTrue(object,CKA_PRIVATE))) {
 1809         return CKR_USER_NOT_LOGGED_IN;
 1810     }
 1811 
 1812 
 1813     if (((session->info.flags & CKF_RW_SESSION) == 0) &&
 1814                                 (sftk_isTrue(object,CKA_TOKEN))) {
 1815         return CKR_SESSION_READ_ONLY;
 1816     }
 1817         
 1818     /* PKCS #11 object ID's are unique for all objects on a
 1819      * token */
 1820     PZ_Lock(slot->objectLock);
 1821     object->handle = slot->tokenIDCount++;
 1822     PZ_Unlock(slot->objectLock);
 1823 
 1824     /* get the object class */
 1825     attribute = sftk_FindAttribute(object,CKA_CLASS);
 1826     if (attribute == NULL) {
 1827         return CKR_TEMPLATE_INCOMPLETE;
 1828     }
 1829     object->objclass = *(CK_OBJECT_CLASS *)attribute->attrib.pValue;
 1830     sftk_FreeAttribute(attribute);
 1831 
 1832     /* now handle the specific. Get a session handle for these functions
 1833      * to use */
 1834     switch (object->objclass) {
 1835     case CKO_DATA:
 1836         crv = sftk_handleDataObject(session,object);
 1837         break;
 1838     case CKO_CERTIFICATE:
 1839         crv = sftk_handleCertObject(session,object);
 1840         break;
 1841     case CKO_NETSCAPE_TRUST:
 1842         crv = sftk_handleTrustObject(session,object);
 1843         break;
 1844     case CKO_NETSCAPE_CRL:
 1845         crv = sftk_handleCrlObject(session,object);
 1846         break;
 1847     case CKO_NETSCAPE_SMIME:
 1848         crv = sftk_handleSMimeObject(session,object);
 1849         break;
 1850     case CKO_PRIVATE_KEY:
 1851     case CKO_PUBLIC_KEY:
 1852     case CKO_SECRET_KEY:
 1853         crv = sftk_handleKeyObject(session,object);
 1854         break;
 1855     case CKO_KG_PARAMETERS:
 1856         crv = sftk_handleKeyParameterObject(session,object);
 1857         break;
 1858     default:
 1859         crv = CKR_ATTRIBUTE_VALUE_INVALID;
 1860         break;
 1861     }
 1862 
 1863     /* can't fail from here on out unless the pk_handlXXX functions have
 1864      * failed the request */
 1865     if (crv != CKR_OK) {
 1866         return crv;
 1867     }
 1868 
 1869     /* now link the object into the slot and session structures */
 1870     if (sftk_isToken(object->handle)) {
 1871         sftk_convertSessionToToken(object);
 1872     } else {
 1873         object->slot = slot;
 1874         sftk_AddObject(session,object);
 1875     }
 1876 
 1877     return CKR_OK;
 1878 }

 1880 /*
 1881  * ******************** Public Key Utilities ***************************
 1882  */
 1883 /* Generate a low public key structure from an object */
 1884 NSSLOWKEYPublicKey *sftk_GetPubKey(SFTKObject *object,CK_KEY_TYPE key_type, 
 1885                                                                 CK_RV *crvp)
 1886 {
 1887     NSSLOWKEYPublicKey *pubKey;
 1888     PLArenaPool *arena;
 1889     CK_RV crv;
 1890 
 1891     if (object->objclass != CKO_PUBLIC_KEY) {
 1892         *crvp = CKR_KEY_TYPE_INCONSISTENT;
 1893         return NULL;
 1894     }
 1895 
 1896     if (sftk_isToken(object->handle)) {
 1897 /* ferret out the token object handle */
 1898     }
 1899 
 1900     /* If we already have a key, use it */
 1901     if (object->objectInfo) {
 1902         *crvp = CKR_OK;
 1903         return (NSSLOWKEYPublicKey *)object->objectInfo;
 1904     }
 1905 
 1906     /* allocate the structure */
 1907     arena = PORT_NewArena(DER_DEFAULT_CHUNKSIZE);
 1908     if (arena == NULL) {
 1909         *crvp = CKR_HOST_MEMORY;
 1910         return NULL;
 1911     }
 1912 
 1913     pubKey = (NSSLOWKEYPublicKey *)
 1914                         PORT_ArenaAlloc(arena,sizeof(NSSLOWKEYPublicKey));
 1915     if (pubKey == NULL) {
 1916             PORT_FreeArena(arena,PR_FALSE);
 1917         *crvp = CKR_HOST_MEMORY;
 1918         return NULL;
 1919     }
 1920 
 1921     /* fill in the structure */
 1922     pubKey->arena = arena;
 1923     switch (key_type) {
 1924     case CKK_RSA:
 1925         pubKey->keyType = NSSLOWKEYRSAKey;
 1926         crv = sftk_Attribute2SSecItem(arena,&pubKey->u.rsa.modulus,
 1927                                                         object,CKA_MODULUS);
 1928             if (crv != CKR_OK) break;
 1929             crv = sftk_Attribute2SSecItem(arena,&pubKey->u.rsa.publicExponent,
 1930                                                 object,CKA_PUBLIC_EXPONENT);
 1931         break;
 1932     case CKK_DSA:
 1933         pubKey->keyType = NSSLOWKEYDSAKey;
 1934         crv = sftk_Attribute2SSecItem(arena,&pubKey->u.dsa.params.prime,
 1935                                                         object,CKA_PRIME);
 1936             if (crv != CKR_OK) break;
 1937         crv = sftk_Attribute2SSecItem(arena,&pubKey->u.dsa.params.subPrime,
 1938                                                         object,CKA_SUBPRIME);
 1939             if (crv != CKR_OK) break;
 1940         crv = sftk_Attribute2SSecItem(arena,&pubKey->u.dsa.params.base,
 1941                                                         object,CKA_BASE);
 1942             if (crv != CKR_OK) break;
 1943             crv = sftk_Attribute2SSecItem(arena,&pubKey->u.dsa.publicValue,
 1944                                                         object,CKA_VALUE);
 1945         break;
 1946     case CKK_DH:
 1947         pubKey->keyType = NSSLOWKEYDHKey;
 1948         crv = sftk_Attribute2SSecItem(arena,&pubKey->u.dh.prime,
 1949                                                         object,CKA_PRIME);
 1950             if (crv != CKR_OK) break;
 1951         crv = sftk_Attribute2SSecItem(arena,&pubKey->u.dh.base,
 1952                                                         object,CKA_BASE);
 1953             if (crv != CKR_OK) break;
 1954             crv = sftk_Attribute2SSecItem(arena,&pubKey->u.dh.publicValue,
 1955                                                         object,CKA_VALUE);
 1956         break;
 1957 #ifdef NSS_ENABLE_ECC
 1958     case CKK_EC:
 1959         pubKey->keyType = NSSLOWKEYECKey;
 1960         crv = sftk_Attribute2SSecItem(arena,
 1961                                       &pubKey->u.ec.ecParams.DEREncoding,
 1962                                       object,CKA_EC_PARAMS);
 1963         if (crv != CKR_OK) break;
 1964 
 1965         /* Fill out the rest of the ecParams structure 
 1966          * based on the encoded params
 1967          */
 1968         if (EC_FillParams(arena, &pubKey->u.ec.ecParams.DEREncoding,
 1969                     &pubKey->u.ec.ecParams) != SECSuccess) {
 1970             crv = CKR_DOMAIN_PARAMS_INVALID;
 1971             break;
 1972         }
 1973             
 1974         crv = sftk_Attribute2SSecItem(arena,&pubKey->u.ec.publicValue,
 1975                                       object,CKA_EC_POINT);
 1976         break;
 1977 #endif /* NSS_ENABLE_ECC */

 4429 static SECStatus
 4430 sftk_key_collect(DBT *key, DBT *data, void *arg)
 4431 {
 4432     sftkKeyData *keyData;
 4433     NSSLOWKEYPrivateKey *privKey = NULL;
 4434     SECItem tmpDBKey;
 4435     SFTKSlot *slot;
 4436     
 4437     keyData = (sftkKeyData *)arg;
 4438     slot = keyData->slot;
 4439 
 4440     tmpDBKey.data = key->data;
 4441     tmpDBKey.len = key->size;
 4442     tmpDBKey.type = siBuffer;
 4443 
 4444     PORT_Assert[1](keyData->keyHandle);
 4445     if (!keyData->strict && keyData->id) {
 4446         SECItem result;
 4447         PRBool haveMatch= PR_FALSE;
 4448         unsigned char hashKey[SHA1_LENGTH];
 4449         result.data = hashKey;
 4450         result.len = sizeof(hashKey);
 4451 
 4452         if (keyData->id->len == 0) {
 4453             /* Make sure this isn't a NSC_KEY */
 4454             privKey = nsslowkey_FindKeyByPublicKey(keyData->keyHandle, 
 4455                                         &tmpDBKey, keyData->slot->password);
 4456             if (privKey) {
 4457                 haveMatch = isSecretKey(privKey) ?
 4458                                 (PRBool)(keyData->classFlags & NSC_KEY) != 0:
 4459                                 (PRBool)(keyData->classFlags & 
 4460                                               (NSC_PRIVATE|NSC_PUBLIC)) != 0;
 4461                 nsslowkey_DestroyPrivateKey(privKey);
 4462             }
 4463         } else {
 4464             SHA1_HashBuf( hashKey, key->data, key->size ); /* match id */
 4465             haveMatch = SECITEM_ItemsAreEqual(keyData->id,&result);
 4466             if (!haveMatch && ((unsigned char *)key->data)[0] == 0) {
 4467                 /* This is a fix for backwards compatibility.  The key
 4468                  * database indexes private keys by the public key, and
 4469                  * versions of NSS prior to 3.4 stored the public key as
 4470                  * a signed integer.  The public key is now treated as an
 4471                  * unsigned integer, with no leading zero.  In order to
 4472                  * correctly compute the hash of an old key, it is necessary
 4473                  * to fallback and detect the leading zero.
 4474                  */
 4475                 SHA1_HashBuf(hashKey, 
 4476                              (unsigned char *)key->data + 1, key->size - 1);
 4477                 haveMatch = SECITEM_ItemsAreEqual(keyData->id,&result);
 4478             }
 4479         }
 4480         if (haveMatch) {
 4481             if (keyData->classFlags & NSC_PRIVATE)  {
 4482                 sftk_addHandle(keyData->searchHandles,
 4483                         sftk_mkHandle(slot,&tmpDBKey,SFTK_TOKEN_TYPE_PRIV));
 4484             }
 4485             if (keyData->classFlags & NSC_PUBLIC) {
 4486                 sftk_addHandle(keyData->searchHandles,
 4487                         sftk_mkHandle(slot,&tmpDBKey,SFTK_TOKEN_TYPE_PUB));
 4488             }
 4489             if (keyData->classFlags & NSC_KEY) {
 4490                 sftk_addHandle(keyData->searchHandles,
 4491                         sftk_mkHandle(slot,&tmpDBKey,SFTK_TOKEN_TYPE_KEY));
 4492             }
 4493         }
 4494         return SECSuccess;
 4495     }
 4496 
 4497     privKey = nsslowkey_FindKeyByPublicKey(keyData->keyHandle, &tmpDBKey, 
 4498                                                  keyData->slot->password);
 4499     if ( privKey == NULL ) {
 4500         goto loser;
 4501     }
 4502 
 4503     if (isSecretKey(privKey)) {
 4504         if ((keyData->classFlags & NSC_KEY) && 
 4505                 sftk_tokenMatch(keyData->slot, &tmpDBKey, SFTK_TOKEN_TYPE_KEY,
 4506                         keyData->template, keyData->templ_count)) {
 4507             sftk_addHandle(keyData->searchHandles,
 4508                 sftk_mkHandle(keyData->slot, &tmpDBKey, SFTK_TOKEN_TYPE_KEY));
 4509         }
 4510     } else {
 4511         if ((keyData->classFlags & NSC_PRIVATE) && 
 4512                 sftk_tokenMatch(keyData->slot, &tmpDBKey, SFTK_TOKEN_TYPE_PRIV,
 4513                         keyData->template, keyData->templ_count)) {
 4514             sftk_addHandle(keyData->searchHandles,
 4515                 sftk_mkHandle(keyData->slot,&tmpDBKey,SFTK_TOKEN_TYPE_PRIV));
 4516         }
 4517         if ((keyData->classFlags & NSC_PUBLIC) && 
 4518                 sftk_tokenMatch(keyData->slot, &tmpDBKey, SFTK_TOKEN_TYPE_PUB,
 4519                         keyData->template, keyData->templ_count)) {
 4520             sftk_addHandle(keyData->searchHandles,
 4521                 sftk_mkHandle(keyData->slot, &tmpDBKey,SFTK_TOKEN_TYPE_PUB));
 4522         }
 4523     }
 4524 
 4525 loser:
 4526     if ( privKey ) {
 4527         nsslowkey_DestroyPrivateKey(privKey);
 4528     }
 4529     return(SECSuccess);
 4530 }

 4542     keyHandle = sftk_getKeyDB(slot);

 4544         return;

 4548         privKey = nsslowkey_FindKeyByPublicKey(keyHandle, key_id, slot->password);

 4568         if (keyHandle->version != 3) {

 4570         }

 4572         /* all soft token id's are SHA1_HASH_LEN's */

 4575         }
 4576         if (found) {
 4577            /* if we already found some keys, don't do the traversal */
 4578            goto loser;

 4589     keyData.strict = mustStrict ? mustStrict : NSC_STRICT;
 4590 
 4591     nsslowkey_TraverseKeys(keyHandle, sftk_key_collect, &keyData);
 4592 loser:
 4593     sftk_freeKeyDB(keyHandle);
 4594         
 4595 }

 4972             }
 4973             break;
 4974         case CKA_TOKEN:
 4975             if (pTemplate[i].ulValueLen != sizeof(CK_BBOOL)) {
 4976                 classFlags = 0;
 4977             }
 4978             if (*((CK_BBOOL *)pTemplate[i].pValue) == CK_TRUE) {
 4979                 *tokenOnly = PR_TRUE;
 4980             } else {
 4981                 classFlags = 0;
 4982             }
 4983             break;
 4984         case CKA_CERT_SHA1_HASH:
 4985             classFlags &= NSC_TRUST;
 4986             copy = &cert_sha1_hash; break;
 4987         case CKA_CERT_MD5_HASH:
 4988             classFlags &= NSC_TRUST;
 4989             copy = &cert_md5_hash; break;
 4990         case CKA_CERTIFICATE_TYPE:
 4991             if (pTemplate[i].ulValueLen != sizeof(CK_CERTIFICATE_TYPE)) {
 4992                 classFlags = 0;
 4993             }
 4994             classFlags &= NSC_CERT;
 4995             if (*((CK_CERTIFICATE_TYPE *)pTemplate[i].pValue) != CKC_X_509) {
 4996                 classFlags = 0;
 4997             }
 4998             break;
 4999         case CKA_ID:
 5000             copy = &key_id;
 5001             classFlags &= (NSC_CERT|NSC_PRIVATE|NSC_KEY|NSC_PUBLIC);
 5002             break;
 5003         case CKA_NETSCAPE_KRL:
 5004             if (pTemplate[i].ulValueLen != sizeof(CK_BBOOL)) {
 5005                 classFlags = 0;
 5006             }
 5007             classFlags &= NSC_CRL;
 5008             isKrl = (PRBool)(*((CK_BBOOL *)pTemplate[i].pValue) == CK_TRUE);
 5009             break;
 5010         case CKA_MODIFIABLE:
 5011              break;
 5012         case CKA_KEY_TYPE:
 5013         case CKA_DERIVE:
 5014             classFlags &= NSC_PUBLIC|NSC_PRIVATE|NSC_KEY;
 5015             break;
 5016         case CKA_VERIFY_RECOVER:
 5017             classFlags &= NSC_PUBLIC;
 5018             break;
 5019         case CKA_SIGN_RECOVER:
 5020             classFlags &= NSC_PRIVATE;
 5021             break;
 5022         case CKA_ENCRYPT:
 5023         case CKA_VERIFY:
 5024         case CKA_WRAP:
 5025             classFlags &= NSC_PUBLIC|NSC_KEY;
 5026             break;
 5027         case CKA_DECRYPT:
 5028         case CKA_SIGN:
 5029         case CKA_UNWRAP:
 5030         case CKA_ALWAYS_SENSITIVE:
 5031         case CKA_EXTRACTABLE:
 5032         case CKA_NEVER_EXTRACTABLE:
 5033             classFlags &= NSC_PRIVATE|NSC_KEY;
 5034             break;
 5035         /* can't be a certificate if it doesn't match one of the above 
 5036          * attributes */
 5037         default: 
 5038              classFlags  = 0;
 5039              break;
 5040         }
 5041          if (copy) {
 5042             copy->data = (unsigned char*)pTemplate[i].pValue;
 5043             copy->len = pTemplate[i].ulValueLen;
 5044         }
 5045         copy = NULL;
 5046     }
 5047 
 5048 
 5049     /* certs */
 5050     if (classFlags & (NSC_CERT|NSC_TRUST)) {
 5051         sftk_searchCertsAndTrust(slot,&derCert,&name,&derSubject,
 5052                                  &issuerSN, &email,classFlags,search, 
 5053                                 pTemplate, ulCount);
 5054     }
 5055 
 5056     /* keys */
 5057     if (classFlags & (NSC_PRIVATE|NSC_PUBLIC|NSC_KEY)) {
 5058         PRBool mustStrict = ((classFlags & NSC_KEY) != 0) && (name.len != 0);
 5059         sftk_searchKeys(slot, &key_id, isLoggedIn, classFlags, search,
 5060                          mustStrict, pTemplate, ulCount);
 5061     }
 5062 
 5063     /* crl's */
 5064     if (classFlags & NSC_CRL) {
 5065         sftk_searchCrls(slot, &derSubject, isKrl, classFlags, search,
 5066                         pTemplate, ulCount);
 5067     }
 5068     /* Add S/MIME entry stuff */
 5069     if (classFlags & NSC_SMIME) {
 5070         sftk_searchSMime(slot, &email, search, pTemplate, ulCount);
 5071     }
 5072     return CKR_OK;
 5073 }

 5076 /* NSC_FindObjectsInit initializes a search for token and session objects 
 5077  * that match a template. */
 5078 CK_RV NSC_FindObjectsInit(CK_SESSION_HANDLE hSession,
 5079                             CK_ATTRIBUTE_PTR pTemplate,CK_ULONG ulCount)
 5080 {
 5081     SFTKSearchResults *search = NULL, *freeSearch = NULL;
 5082     SFTKSession *session = NULL;
 5083     SFTKSlot *slot = sftk_SlotFromSessionHandle(hSession);
 5084     PRBool tokenOnly = PR_FALSE;
 5085     CK_RV crv = CKR_OK;
 5086     PRBool isLoggedIn;
 5087     
 5088     session = sftk_SessionFromHandle(hSession);
 5089     if (session == NULL) {
 5090         crv = CKR_SESSION_HANDLE_INVALID;
 5091         goto loser;
 5092     }
 5093    
 5094     search = (SFTKSearchResults *)PORT_Alloc(sizeof(SFTKSearchResults));
 5095     if (search == NULL) {
 5096         crv = CKR_HOST_MEMORY;
 5097         goto loser;
 5098     }
 5099     search->handles = (CK_OBJECT_HANDLE *)
 5100                 PORT_Alloc(sizeof(CK_OBJECT_HANDLE) * NSC_SEARCH_BLOCK_SIZE);
 5101     if (search->handles == NULL) {
 5102         crv = CKR_HOST_MEMORY;
 5103         goto loser;
 5104     }
 5105     search->index = 0;
 5106     search->size = 0;
 5107     search->array_size = NSC_SEARCH_BLOCK_SIZE;
 5108     isLoggedIn = (PRBool)((!slot->needLogin) || slot->isLoggedIn);
 5109 
 5110     crv = sftk_searchTokenList(slot, search, pTemplate, ulCount, &tokenOnly,
 5111                                                                 isLoggedIn);
 5112     if (crv != CKR_OK) {
 5113         goto loser;
 5114     }
 5115     
 5116     /* build list of found objects in the session */
 5117     if (!tokenOnly) {
 5118         crv = sftk_searchObjectList(search, slot->tokObjects, 
 5119                                 slot->tokObjHashSize, slot->objectLock, 
 5120                                         pTemplate, ulCount, isLoggedIn);
 5121     }
 5122     if (crv != CKR_OK) {
 5123         goto loser;
 5124     }
 5125 
 5126     if ((freeSearch = session->search) != NULL) {
 5127         session->search = NULL;
 5128         sftk_FreeSearch(freeSearch);
 5129     }
 5130     session->search = search;
 5131     sftk_FreeSession(session);
 5132     return CKR_OK;
 5133 
 5134 loser:
 5135     if (search) {
 5136         sftk_FreeSearch(search);
 5137     }
 5138     if (session) {
 5139         sftk_FreeSession(session);
 5140     }
 5141     return crv;
 5142 }

 5145 /* NSC_FindObjects continues a search for token and session objects 
 5146  * that match a template, obtaining additional object handles. */
 5147 CK_RV NSC_FindObjects(CK_SESSION_HANDLE hSession,
 5148     CK_OBJECT_HANDLE_PTR phObject,CK_ULONG ulMaxObjectCount,
 5149                                             CK_ULONG_PTR pulObjectCount)
 5150 {
 5151     SFTKSession *session;
 5152     SFTKSearchResults *search;
 5153     int        transfer;
 5154     int left;
 5155 
 5156     *pulObjectCount = 0;
 5157     session = sftk_SessionFromHandle(hSession);
 5158     if (session == NULL) return CKR_SESSION_HANDLE_INVALID;
 5159     if (session->search == NULL) {
 5160         sftk_FreeSession(session);
 5161         return CKR_OK;
 5162     }
 5163     search = session->search;
 5164     left = session->search->size - session->search->index;
 5165     transfer = ((int)ulMaxObjectCount > left) ? left : ulMaxObjectCount;
 5166     if (transfer > 0) {
 5167         PORT_Memcpy(phObject,&search->handles[search->index],
 5168                                         transfer*sizeof(CK_OBJECT_HANDLE_PTR));
 5169     } else {
 5170        *phObject = CK_INVALID_HANDLE;
 5171     }
 5172 
 5173     search->index += transfer;
 5174     if (search->index == search->size) {
 5175         session->search = NULL;
 5176         sftk_FreeSearch(search);
 5177     }
 5178     *pulObjectCount = transfer;
 5179     sftk_FreeSession(session);
 5180     return CKR_OK;
 5181 }

 5185 {
 5186     SFTKSession *session;
 5187     SFTKSearchResults *search;
 5188

 5195         sftk_FreeSearch(search);
 5196     }
 5197     return CKR_OK;
 5198 }
