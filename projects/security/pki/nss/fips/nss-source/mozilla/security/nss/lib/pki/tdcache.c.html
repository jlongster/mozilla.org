  156 NSS_IMPLEMENT PRStatus
  157 nssTrustDomain_InitializeCache (
  158   NSSTrustDomain *td,

  162     NSSArena *arena;
  163     nssTDCertificateCache *cache = td->cache;
  164 #ifdef DEBUG_CACHE

  166     PR_ASSERT(s_log);

  171         return PR_FAILURE;

  195     if (!cache->nickname[1]) {
  196         goto loser;
  197     }
  198     /* Create the email --> list of subject lists hash */
  199     cache->email = nssHash_CreateString(arena, cacheSize);
  200     if (!cache->email) {
  201         goto loser;
  202     }
  203     cache->arena = arena;
  204     td->cache = cache;
  205 #ifdef DEBUG_CACHE
  206     PR_LOG[2](s_log, PR_LOG[2]_DEBUG, ("Cache initialized."));
  207 #endif
  208     return PR_SUCCESS;
  209 loser:
  210     PZ_DestroyLock(cache->lock[1][2]);
  211     nssArena_Destroy(arena);
  212     td->cache = NULL;
  213 #ifdef DEBUG_CACHE
  214     PR_LOG[2](s_log, PR_LOG[2]_DEBUG, ("Cache initialization failed[1][2][3][4][5][6]."));
  215 #endif
