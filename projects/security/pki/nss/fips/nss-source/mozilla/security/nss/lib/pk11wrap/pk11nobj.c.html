  384 /*
  385  * Return a list of CRLs matching specified issuer and type
  386  * CRLs are not allocated in the list's arena, but rather in their own,
  387  * arena, so that they can be used individually in the CRL cache .
  388  * CRLs are always partially decoded for efficiency.
  389  */
  390 SECStatus pk11_RetrieveCrls(CERTCrlHeadNode *nodes, SECItem* issuer,
  391                             void *wincx)
  392 {
  393     pk11TraverseSlot creater;
  394     CK_ATTRIBUTE theTemplate[2];
  395     CK_ATTRIBUTE *attrs;
  396     CK_OBJECT_CLASS crlClass = CKO_NETSCAPE_CRL;
  397     crlOptions options;
  398 
  399     attrs = theTemplate;
  400     PK11_SETATTRS[1][2](attrs, CKA_CLASS, &crlClass, sizeof(crlClass)); attrs++;
  401 
  402     options.head[1] = nodes;
  403 
  404     /* - do a partial decoding - we don't need to decode the entries while
  405        fetching
  406        - don't copy the DER for optimal performance - CRL can be very large
  407        - have the CRL objects adopt the DER, so SEC_DestroyCrl will free it
  408        - keep bad CRL objects. The CRL cache is interested in them, for
  409          security purposes. Bad CRL objects are a sign of something amiss.
  410     */
  411 
  412     options.decodeOptions = CRL_DECODE_SKIP_ENTRIES | CRL_DECODE_DONT_COPY_DER |
  413                             CRL_DECODE_ADOPT_HEAP_DER | CRL_DECODE_KEEP_BAD_CRL;
  414     if (issuer)
  415     {
  416         PK11_SETATTRS[1][2](attrs, CKA_SUBJECT, issuer->data[2], issuer->len[1][2][3][6][9][10][13]); attrs++;
  417     }
  418 
  419     creater.callback = pk11_RetrieveCrlsCallback;
  420     creater.callbackArg = (void *) &options;
  421     creater.findTemplate = theTemplate;
  422     creater.templateCount = (attrs - theTemplate);
  423 
  424     return pk11_TraverseAllSlots(PK11_TraverseSlot, &creater, PR_FALSE, wincx);

  564 PK11_FindSMimeProfile(PK11SlotInfo **slot, char *emailAddr,

  566 {
  567     CK_OBJECT_CLASS smimeClass = CKO_NETSCAPE_SMIME;

  574         { CKA_SUBJECT, NULL, 0 },

  576     };

  578     int tsize = sizeof(theTemplate)/sizeof(theTemplate[0]);
  579     CK_OBJECT_HANDLE smimeh = CK_INVALID_HANDLE;
  580     CK_ATTRIBUTE *attrs = theTemplate;
  581     CK_RV crv;
  582     SECItem *emailProfile = NULL;
  583 
  584     if (!emailAddr || !emailAddr[0]) {
  585         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  586         return NULL;
  587     }
  588 
  589     PK11_SETATTRS[1][2](attrs, CKA_SUBJECT, name->data[2], name->len[1][2][3][6][9][10][13]); attrs++;
  590     PK11_SETATTRS[1][2](attrs, CKA_CLASS, &smimeClass, sizeof(smimeClass)); attrs++;
  591     PK11_SETATTRS[1][2](attrs, CKA_NETSCAPE_EMAIL, emailAddr, strlen(emailAddr)); 
  592                                                                     attrs++;
  593 
  594     if (*slot) {
  595             smimeh = pk11_FindObjectByTemplate(*slot,theTemplate,tsize);
  596     } else {
  597         PK11SlotList *list = PK11_GetAllTokens(CKM_INVALID_MECHANISM,
  598                                                         PR_FALSE,PR_TRUE,NULL);
  599         PK11SlotListElement *le;
  600 
  601         /* loop through all the slots */
  602         for (le = list->head[1]; le; le = le->next) {
  603             smimeh = pk11_FindObjectByTemplate(le->slot,theTemplate,tsize);
  604             if (smimeh != CK_INVALID_HANDLE) {
  605                 *slot = slot.c.dep.html#PK11_ReferenceSlot">PK11_ReferenceSlot(le->slot);
  606                 break;
  607             }
  608         }
  609         PK11_FreeSlotList(list);
  610     }
  611     
  612     if (smimeh == CK_INVALID_HANDLE) {
  613         PORT_SetError(SEC_ERROR_NO_KRL);
  614         return NULL;
  615     }
  616 
  617     if (profileTime) {
  618             PK11_SETATTRS[1][2](smimeData, CKA_NETSCAPE_SMIME_TIMESTAMP, NULL, 0);
  619     } 
  620     
  621     crv = PK11_GetAttributes(NULL,*slot,smimeh,smimeData,2);
  622     if (crv != CKR_OK) {
  623         PORT_SetError(PK11_MapError (crv));
  624         goto loser;
  625     }
  626 
  627     if (!profileTime) {
  628         SECItem profileSubject;
  629 
  630         profileSubject.data[2] = (unsigned char*) smimeData[0].pValue;
  631         profileSubject.len[1][2][3][6][9][10][13] = smimeData[0].ulValueLen;
  632         if (!SECITEM_ItemsAreEqual(&profileSubject,name)) {
  633             goto loser;
  634         }
  635     }
  636 
  637     emailProfile = (SECItem *)PORT_ZAlloc(sizeof(SECItem));    
  638     if (emailProfile == NULL) {
  639         goto loser;
  640     }
  641 
  642     emailProfile->data[2] = (unsigned char*) smimeData[1].pValue;
  643     emailProfile->len[1][2][3][6][9][10][13] = smimeData[1].ulValueLen;
  644 
  645     if (profileTime) {
  646         *profileTime = (SECItem *)PORT_ZAlloc(sizeof(SECItem));    
  647         if (*profileTime) {
  648             (*profileTime)->data[2] = (unsigned char*) smimeData[0].pValue;
  649             (*profileTime)->len[1][2][3][6][9][10][13] = smimeData[0].ulValueLen;
  650         }
  651     }
  652 
  653 loser:
  654     if (emailProfile == NULL) {
  655         if (smimeData[1].pValue) {
  656             PORT_Free(smimeData[1].pValue);
  657         }
  658     }
  659     if (profileTime == NULL || *profileTime == NULL) {
  660         if (smimeData[0].pValue) {
  661             PORT_Free(smimeData[0].pValue);
  662         }
  663     }
  664     return emailProfile;
  665 }

  739 /* import the CRL into the token */
  740 
  741 CERTSignedCrl* PK11_ImportCRL(PK11SlotInfo * slot, SECItem *derCRL, char *url,
  742     int type, void *wincx, types.h.dep.html#PRInt32">PRInt32 importOptions, types.h.dep.html#PRArenaPool">PRArenaPool* arena,
  743     PRInt32 decodeoptions)
  744 {
  745     CERTSignedCrl *newCrl, *crl;
  746     SECStatus rv[1];
  747     CERTCertificate *caCert = NULL;
  748 
  749     newCrl = crl = NULL;
  750 
  751     do {
  752         newCrl = CERT_DecodeDERCrlWithFlags(arena, derCRL, type,
  753                                             decodeoptions);
  754         if (newCrl == NULL) {
  755             if (type == SEC_CRL_TYPE) {
  756                 /* only promote error when the error code is too generic */
  757                 if (PORT_GetError () == SEC_ERROR_BAD_DER)
  758                     PORT_SetError(SEC_ERROR_CRL_INVALID);
  759                 } else {
  760                 PORT_SetError(SEC_ERROR_KRL_INVALID);
  761             }
  762             break;                
  763         }
  764 
  765         if (0 == (importOptions & CRL_IMPORT_BYPASS_CHECKS)){
  766             CERTCertDBHandle* handle = CERT_GetDefaultCertDB();
  767             PR_ASSERT(handle != NULL);
  768             caCert = CERT_FindCertByName (handle,
  769                                           &newCrl->crl.derName);
  770             if (caCert == NULL) {
  771                 PORT_SetError(SEC_ERROR_UNKNOWN_ISSUER);            
  772                 break;
  773             }
  774 
  775             /* If caCert is a v3 certificate, make sure that it can be used for
  776                crl signing purpose */
  777             rv[1] = CERT_CheckCertUsage (caCert, KU_CRL_SIGN);
  778             if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  779                 break;
  780             }
  781 
  782             rv[1] = CERT_VerifySignedData(&newCrl->signatureWrap, caCert,
  783                                        PR_Now[1][2][3][4][5](), wincx);
  784             if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  785                 if (type == SEC_CRL_TYPE) {
  786                     PORT_SetError(SEC_ERROR_CRL_BAD_SIGNATURE);
  787                 } else {
  788                     PORT_SetError(SEC_ERROR_KRL_BAD_SIGNATURE);
  789                 }
  790                 break;
  791             }
  792         }
  793 
  794         crl = crl_storeCRL(slot, url, newCrl, derCRL, type);
  795 
  796     } while (0);
  797 
  798     if (crl == NULL) {
  799         SEC_DestroyCrl (newCrl);
  800     }
  801     if (caCert) {
  802         CERT_DestroyCertificate(caCert);
  803     }
  804     return (crl);
