   37 /*
   38  * Public prototypes for base64 encoding/decoding.
   39  *
   40  * $Id$
   41  */
   42 #ifndef _NSSB64_H_
   43 #define _NSSB64_H_
   44 
   45 #include "seccomon.h"
   46 #include "nssb64t.h"
   47 
   48 SEC_BEGIN_PROTOS
   49 
   50 /*
   51  * Functions to start a base64 decoding/encoding context.
   52  */
   53 
   54 extern NSSBase64Decoder *
   55 NSSBase64Decoder_Create (PRInt32 (*output_fn) (void *, const unsigned char *,
   56                                                PRInt32),
   57                          void *output_arg);
   58 
   59 extern NSSBase64Encoder *
   60 NSSBase64Encoder_Create (PRInt32 (*output_fn) (void *, const char *, PRInt32),
   61                          void *output_arg);
   62 
   63 /*
   64  * Push data through the decoder/encoder, causing the output_fn (provided
   65  * to Create) to be called with the decoded/encoded data.
   66  */
   67 
   68 extern SECStatus
   69 NSSBase64Decoder_Update (NSSBase64Decoder *data, const char *buffer,
   70                          PRUint32 size);
   71 
   72 extern SECStatus
   73 NSSBase64Encoder_Update (NSSBase64Encoder *data, const unsigned char *buffer,
   74                          PRUint32 size);
   75 
   76 /*
   77  * When you're done processing, call this to close the context.
   78  * If "abort_p" is false, then calling this may cause the output_fn
   79  * to be called one last time (as the last buffered data is flushed out).
   80  */
   81 
   82 extern SECStatus
   83 NSSBase64Decoder_Destroy (NSSBase64Decoder *data, PRBool abort_p);
   84 
   85 extern SECStatus
   86 NSSBase64Encoder_Destroy (NSSBase64Encoder *data, PRBool abort_p);
   87 
   88 /*
   89  * Perform base64 decoding from an ascii string "inStr" to an Item.
   90  * The length of the input must be provided as "inLen".  The Item
   91  * may be provided (as "outItemOpt"); you can also pass in a NULL
   92  * and the Item will be allocated for you.
   93  *
   94  * In any case, the data within the Item will be allocated for you.
   95  * All allocation will happen out of the passed-in "arenaOpt", if non-NULL.
   96  * If "arenaOpt" is NULL, standard allocation (heap) will be used and
   97  * you will want to free the result via SECITEM_FreeItem.
   98  *
   99  * Return value is NULL on error, the Item (allocated or provided) otherwise.
  100  */
  101 extern SECItem *
  102 NSSBase64_DecodeBuffer (PRArenaPool *arenaOpt, SECItem *outItemOpt,
  103                         const char *inStr, unsigned int inLen);
  104 
  105 /*
  106  * Perform base64 encoding of binary data "inItem" to an ascii string.
  107  * The output buffer may be provided (as "outStrOpt"); you can also pass
  108  * in a NULL and the buffer will be allocated for you.  The result will
  109  * be null-terminated, and if the buffer is provided, "maxOutLen" must
  110  * specify the maximum length of the buffer and will be checked to
  111  * supply sufficient space space for the encoded result.  (If "outStrOpt"
  112  * is NULL, "maxOutLen" is ignored.)
  113  *
  114  * If "outStrOpt" is NULL, allocation will happen out of the passed-in
  115  * "arenaOpt", if *it* is non-NULL, otherwise standard allocation (heap)
  116  * will be used.
  117  *
  118  * Return value is NULL on error, the output buffer (allocated or provided)
  119  * otherwise.
  120  */
  121 extern char *
  122 NSSBase64_EncodeItem (PRArenaPool *arenaOpt, char *outStrOpt,
  123                       unsigned int maxOutLen, SECItem *inItem);
  124 
  125 SEC_END_PROTOS[1]
  126 
  127 #endif /* _NSSB64_H_ */
