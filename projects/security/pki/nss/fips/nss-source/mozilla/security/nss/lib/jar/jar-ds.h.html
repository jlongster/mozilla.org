   40 /* Typedefs */
   41 typedef struct ZZLinkStr ZZLink;
   42 typedef struct ZZListStr ZZList;
   43 
   44 /*
   45 ** Circular linked list. Each link contains a pointer to the object that
   46 ** is actually in the list.
   47 */ 
   48 struct ZZLinkStr 
   49 {
   50     ZZLink *next;
   51     ZZLink *prev;
   52     JAR_Item *thing;
   53 };
   54 
   55 struct ZZListStr 
   56 {
   57     ZZLink link;
   58 };
   59 
   60 #define ZZ_InitList(lst)             \
   61 {                                     \
   62     (lst)->link.next = &(lst)->link; \
   63     (lst)->link.prev = &(lst)->link; \
   64     (lst)->link.thing = 0;             \
   65 }
   66 
   67 #define ZZ_ListEmpty(lst) \
   68     ((lst)->link.next == &(lst)->link)
   69 
   70 #define ZZ_ListHead(lst) \
   71     ((lst)->link.next)
   72 
   73 #define ZZ_ListTail(lst) \
   74     ((lst)->link.prev)
   75 
   76 #define ZZ_ListIterDone(lst,lnk) \
   77     ((lnk) == &(lst)->link)
   78 
   79 #define ZZ_AppendLink(lst,lnk)            \
   80 {                                    \
   81     (lnk)->next = &(lst)->link;            \
   82     (lnk)->prev = (lst)->link.prev; \
   83     (lst)->link.prev->next = (lnk); \
   84     (lst)->link.prev = (lnk);            \
   85 }
   86 
   87 #define ZZ_InsertLink(lst,lnk)            \
   88 {                                    \
   89     (lnk)->next = (lst)->link.next; \
   90     (lnk)->prev = &(lst)->link;            \
   91     (lst)->link.next->prev = (lnk); \
   92     (lst)->link.next = (lnk);            \
   93 }
   94 
   95 #define ZZ_RemoveLink(lnk)             \
   96 {                                     \
   97     (lnk)->next->prev = (lnk)->prev; \
   98     (lnk)->prev->next = (lnk)->next; \
   99     (lnk)->next = 0;                     \
  100     (lnk)->prev = 0;                     \
  101 }
  102 
  103 extern ZZLink *ZZ_NewLink (JAR_Item *thing);
  104 extern void ZZ_DestroyLink (ZZLink *link);
  105 extern ZZList *ZZ_NewList (void);
  106 extern void ZZ_DestroyList (ZZList *list);
  107 
  108 
  109 #endif /* __JAR_DS_h_ */
