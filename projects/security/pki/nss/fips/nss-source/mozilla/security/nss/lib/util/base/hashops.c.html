   51 static void * PR_CALLBACK
   52 nss_arena_hash_alloc_table
   53 (
   54   void *pool,
   55   PRSize size[4][8][9]
   56 )
   57 {
   58   NSSArena *arena = (NSSArena *)NULL;
   59 
   60 #ifdef NSSDEBUG
   61   if( (void *)NULL != arena ) {
   62     if( PR_SUCCESS != nssArena_verifyPointer(arena) ) {
   63       return (void *)NULL;
   64     }
   65   }
   66 #endif /* NSSDEBUG */
   67 
   68   return nss_ZAlloc(arena, size[4][8][9]);
   69 }
   70 
   71 static void PR_CALLBACK
   72 nss_arena_hash_free_table
   73 (
   74   void *pool, 
   75   void *item
   76 )
   77 {
   78   (void)nss_ZFreeIf(item);
   79 }
   80 
   81 static PLHashEntry * PR_CALLBACK
   82 nss_arena_hash_alloc_entry
   83 (
   84   void *pool,
   85   const void *key[1]
   86 )
   87 {
   88   NSSArena *arena = NULL;
   89 
   90 #ifdef NSSDEBUG
   91   if( (void *)NULL != arena ) {
   92     if( PR_SUCCESS != nssArena_verifyPointer(arena) ) {
   93       return (void *)NULL;
   94     }
   95   }
   96 #endif /* NSSDEBUG */
   97 
   98   return nss_ZNEW(arena, PLHashEntry);
   99 }
  100 
  101 static void PR_CALLBACK
  102 nss_arena_hash_free_entry
  103 (
  104   void *pool,
  105   PLHashEntry *he,
  106   PRUintn flag
  107 )
  108 {
  109   if( HT_FREE_ENTRY == flag ) {
  110     (void)nss_ZFreeIf(he);
  111   }
  112 }
