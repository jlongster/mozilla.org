  103 /*
  104  * LOG2_BUCKETS_PER_SESSION_LOCK must be a prime number.
  105  * With SESSION_HASH_SIZE=1024, LOG2 can be 9, 5, 1, or 0.
  106  * With SESSION_HASH_SIZE=4096, LOG2 can be 11, 9, 5, 1, or 0.
  107  *
  108  * HASH_SIZE   LOG2_BUCKETS_PER   BUCKETS_PER_LOCK  NUMBER_OF_BUCKETS
  109  * 1024        9                  512               2
  110  * 1024        5                  32                32
  111  * 1024        1                  2                 512
  112  * 1024        0                  1                 1024
  113  * 4096        11                 2048              2
  114  * 4096        9                  512               8
  115  * 4096        5                  32                128
  116  * 4096        1                  2                 2048
  117  * 4096        0                  1                 4096
  118  */
  119 #define LOG2_BUCKETS_PER_SESSION_LOCK 1

  317 /*
  318  * slots (have sessions and objects)
  319  *
  320  * The array of sessionLock's protect the session hash table (head[])
  321  * as well as the reference count of session objects in that bucket
  322  * (head[]->refCount),  objectLock protects all elements of the token
  323  * object hash table (tokObjects[], tokenIDCount, and tokenHashTable),
  324  * slotLock protects the remaining protected elements:
  325  * password, isLoggedIn, ssoLoggedIn, and sessionCount,
  326  * and pwCheckLock serializes the key database password checks in
  327  * NSC_SetPIN and NSC_Login.
  328  *
  329  * Each of the fields below has the following lifetime as commented
  330  * next to the fields:
  331  *   invariant  - This value is set when the slot is first created and
  332  * never changed until it is destroyed.
  333  *   per load   - This value is set when the slot is first created, or 
  334  * when the slot is used to open another directory. Between open and close
  335  * this field does not change.
  336  *   variable - This value changes through the normal process of slot operation.
  337  *      - reset. The value of this variable is cleared during an open/close 
  338  *   cycles.
  339  *      - preserved. The value of this variable is preserved over open/close
  340  *   cycles.
  341  */
  342 struct SFTKSlotStr {
  343     CK_SLOT_ID                slotID;                        /* invariant */
  344     PZLock                *slotLock;                /* invariant */
  345     PZLock                **sessionLock;                /* invariant */
  346     unsigned int        numSessionLocks;        /* invariant */
  347     unsigned long        sessionLockMask;        /* invariant */
  348     PZLock                *objectLock;                /* invariant */
  349     PRLock                *pwCheckLock;                /* invariant */
  350     SECItem                *password;                /* variable - reset */
  351     PRBool                present;                /* variable -set */
  352     PRBool                hasTokens;                /* per load */
  353     PRBool                isLoggedIn;                /* variable - reset */
  354     PRBool                ssoLoggedIn;                /* variable - reset */
  355     PRBool                needLogin;                /* per load */
  356     PRBool                DB_loaded;                /* per load */
  357     PRBool                readOnly;                /* per load */
  358     PRBool                optimizeSpace;                /* invariant */
  359     NSSLOWCERTCertDBHandle *certDB;                /* per load */
  360     NSSLOWKEYDBHandle        *keyDB;                        /* per load */
  361     int                        minimumPinLen;                /* per load */
  362     PRInt32                sessionIDCount;                /* atomically incremented */
  363                                                 /* (preserved) */
  364     int                        sessionIDConflict;         /* not protected by a lock */
  365                                                     /* (preserved) */
  366     int                        sessionCount;           /* variable - reset */
  367     PRInt32             rwSessionCount;            /* set by atomic operations */
  368                                                   /* (reset) */
  369     int                        tokenIDCount;              /* variable - perserved */
  370     int                        index;                        /* invariant */
  371     PLHashTable                *tokenHashTable;        /* invariant */
  372     SFTKObject                **tokObjects;                /* variable - reset */
  373     unsigned int        tokObjHashSize;                /* invariant */
  374     SFTKSession                **head;                        /* variable -reset */
  375     unsigned int        sessHashSize;                /* invariant */
  376     char                tokDescription[33];        /* per load */
  377     char                slotDescription[64];        /* invariant */
  378 };

  444 /* queueing helper macros */
  445 #define sftk_hash(value,size) \
  446         ((PRUint32)((value) * SHMULTIPLIER) & (size-1))
  447 #define sftkqueue_add(element,id,head,hash_size) \
  448         { int tmp[1] = sftk_hash(id,hash_size); \
  449         (element)->next = (head)[tmp[1]]; \
  450         (element)->prev = NULL; \
  451         if ((head)[tmp[1]]) (head)[tmp[1]]->prev = (element); \
  452         (head)[tmp[1]] = (element); }
  453 #define sftkqueue_find(element,id,head,hash_size) \
  454         for( (element) = (head)[sftk_hash(id,hash_size)]; (element) != NULL; \
  455                                          (element) = (element)->next) { \
  456             if ((element)->handle == (id)) { break; } }
  457 #define sftkqueue_is_queued(element,id,head,hash_size) \
  458         ( ((element)->next) || ((element)->prev) || \
  459          ((head)[sftk_hash(id,hash_size)] == (element)) )
  460 #define sftkqueue_delete(element,id,head,hash_size) \
  461         if ((element)->next) (element)->next->prev = (element)->prev; \
  462         if ((element)->prev) (element)->prev->next = (element)->next; \
  463            else (head)[sftk_hash(id,hash_size)] = ((element)->next); \
  464         (element)->next = NULL; \
  465         (element)->prev = NULL; \
  466 
  467 #define sftkqueue_init_element(element) \
  468     (element)->prev = NULL;
  469 
  470 #define sftkqueue_add2(element, id, index, head) \
  471     {                                            \
  472         (element)->next = (head)[index];         \
  473         if ((head)[index])                       \
  474             (head)[index]->prev = (element);     \
  475         (head)[index] = (element);               \
  476     }
  477 
  478 #define sftkqueue_find2(element, id, index, head) \
  479     for ( (element) = (head)[index];              \
  480           (element) != NULL;                      \
  481           (element) = (element)->next) {          \
  482         if ((element)->handle == (id)) { break; } \
  483     }
  484 
  485 #define sftkqueue_delete2(element, id, index, head) \
  486         if ((element)->next) (element)->next->prev = (element)->prev; \
  487         if ((element)->prev) (element)->prev->next = (element)->next; \
  488            else (head)[index] = ((element)->next);

  490 #define sftkqueue_clear_deleted_element(element) \
  491         (element)->next = NULL; \
  492         (element)->prev = NULL; \
