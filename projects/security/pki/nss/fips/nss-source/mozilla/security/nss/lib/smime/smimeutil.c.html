  174 /*
  175  * NSS_SMIME_EnableCipher - this function locally records the user's preference
  176  */
  177 SECStatus 
  178 NSS_SMIMEUtil_EnableCipher(unsigned long which, PRBool on)
  179 {
  180     unsigned long mask;
  181     int mapi;
  182 
  183     mask = which & CIPHER_FAMILYID_MASK;
  184 
  185     PORT_Assert[1] (mask == CIPHER_FAMILYID_SMIME);
  186     if (mask != CIPHER_FAMILYID_SMIME)
  187         /* XXX set an error! */
  188             return SECFailure;
  189 
  190     mapi = smime_mapi_by_cipher[1](which);
  191     if (mapi < 0)
  192         /* XXX set an error */
  193         return SECFailure;
  194 
  195     /* do we try to turn on a forbidden cipher? */
  196     if (!smime_cipher_map[1][mapi].allowed && on) {
  197         PORT_SetError (SEC_ERROR_BAD_EXPORT_ALGORITHM);
  198         return SECFailure;
  199     }
  200 
  201     if (smime_cipher_map[1][mapi].enabled != on)
  202         smime_cipher_map[1][mapi].enabled = on;
  203 
  204     return SECSuccess;
  205 }

  473                 cipher_votes[strong_mapi] += pref;
  474                 pref--;
  475             } 
  476 
  477             /* always cast (possibly less) votes for the weak algorithm */
  478             cipher_abilities[weak_mapi]++;
  479             cipher_votes[weak_mapi] += pref;
  480         }
  481         if (profile != NULL)
  482             SECITEM_FreeItem(profile, PR_TRUE);
  483     }
  484 
  485     /* find cipher that is agreeable by all recipients and that has the most votes */
  486     max = 0;
  487     for (mapi = 0; mapi < smime_cipher_map_count; mapi++) {
  488         /* if not all of the recipients can do this, forget it */
  489         if (cipher_abilities[mapi] != rcount)
  490             continue;
  491         /* if cipher is not enabled or not allowed by policy, forget it */
  492         if (!smime_cipher_map[1][mapi].enabled || !smime_cipher_map[1][mapi].allowed)
  493             continue;
  494         /* if we're not doing fortezza, but the cipher is fortezza, forget it */
  495         if (!scert_is_fortezza  && (smime_cipher_map[1][mapi].cipher == SMIME_FORTEZZA))
  496             continue;
  497         /* now see if this one has more votes than the last best one */
  498         if (cipher_votes[mapi] >= max) {
  499             /* if equal number of votes, prefer the ones further down in the list */
  500             /* with the expectation that these are higher rated ciphers */
  501             chosen_cipher = smime_cipher_map[1][mapi].cipher;
  502             max = cipher_votes[mapi];
  503         }
  504     }
  505     /* if no common cipher was found, chosen_cipher stays at the default */
  506 
  507 done:
  508     if (poolp != NULL)
  509         PORT_FreeArena (poolp, PR_FALSE);
  510 
  511     return chosen_cipher;
  512 }
