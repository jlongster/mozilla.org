   43 #ifndef _H_MPLOGIC_
   44 #define _H_MPLOGIC_
   45 
   46 #include "mpi.h"
   47 
   48 /*
   49   The logical operations treat an mp_int as if it were a bit vector,
   50   without regard to its sign (an mp_int is represented in a signed
   51   magnitude format).  Values are treated as if they had an infinite
   52   string of zeros left of the most-significant bit.
   53  */
   54 
   55 /* Parity results                    */
   56 
   57 #define MP_EVEN       MP_YES
   58 #define MP_ODD        MP_NO
   59 
   60 /* Bitwise functions                 */
   61 
   62 mp_err mpl_not(mp_int *a, mp_int *b);            /* one's complement  */
   63 mp_err mpl_and(mp_int *a, mp_int *b, mp_int *c); /* bitwise AND       */
   64 mp_err mpl_or(mp_int *a, mp_int *b, mp_int *c);  /* bitwise OR        */
   65 mp_err mpl_xor(mp_int *a, mp_int *b, mp_int *c); /* bitwise XOR       */
   66 
   67 /* Shift functions                   */
   68 
   69 mp_err mpl_rsh(const mp_int *a, mp_int *b, mp_digit d);   /* right shift    */
   70 mp_err mpl_lsh(const mp_int *a, mp_int *b, mp_digit d);   /* left shift     */
   71 
   72 /* Bit count and parity              */
   73 
   74 mp_err mpl_num_set(mp_int *a, int *num);         /* count set bits    */
   75 mp_err mpl_num_clear(mp_int *a, int *num);       /* count clear bits  */
   76 mp_err mpl_parity(mp_int *a);                    /* determine parity  */
   77 
   78 /* Get & Set the value of a bit */
   79 
   80 mp_err mpl_set_bit(mp_int *a, mp_size bitNum, mp_size value);
   81 mp_err mpl_get_bit(const mp_int *a, mp_size bitNum);
   82 mp_err mpl_get_bits(const mp_int *a, mp_size lsbNum, mp_size numBits);
   83 mp_err mpl_significant_bits(const mp_int *a);
   84 
   85 #endif /* end _H_MPLOGIC_ */
