   95 /* Both classic ASN.1 and QuickDER have a feature that removes leading zeroes
   96    out of SEC_ASN1_INTEGER if the caller sets siUnsignedInteger in the type
   97    field of the target SECItem prior to calling the decoder. Otherwise, the
   98    type field is ignored and untouched. For SECItem that are dynamically
   99    allocated (from POINTER, SET OF, SEQUENCE OF) the decoder sets the type
  100    field to siBuffer. */
  101 
  102 extern SECStatus SEC_ASN1DecodeItem(PRArenaPool *pool, void *dest,
  103                                     const SEC_ASN1Template *t,
  104                                     const SECItem *src);

  178 /*
  179  * Find the appropriate subtemplate for the given template.
  180  * This may involve calling a "chooser" function, or it may just
  181  * be right there.  In either case, it is expected to *have* a
  182  * subtemplate; this is asserted in debug builds (in non-debug
  183  * builds, NULL will be returned).
  184  *
  185  * "thing" is a pointer to the structure being encoded/decoded
  186  * "encoding", when true, means that we are in the process of encoding
  187  *        (as opposed to in the process of decoding)
  188  */
  189 extern const SEC_ASN1Template *
  190 SEC_ASN1GetSubtemplate (const SEC_ASN1Template *inTemplate, void *thing,
  191                         PRBool encoding);
