   51 /* Converts a point P(px, py) from affine coordinates to Jacobian
   52  * projective coordinates R(rx, ry, rz). Assumes input is already
   53  * field-encoded using field_enc, and returns output that is still
   54  * field-encoded. */
   55 mp_err
   56 ec_GFp_pt_aff2jac(const mp_int *px, const mp_int *py, mp_int *rx,
   57                                   mp_int *ry, mp_int *rz, const ECGroup *group)
   58 {
   59         mp_err res = MP_OKAY;
   60 
   61         if (ec_GFp_pt_is_inf_aff(px, py) == MP_YES) {
   62                 MP_CHECKOK(ec_GFp_pt_set_inf_jac(rx, ry, rz));
   63         } else {
   64                 MP_CHECKOK(mp_copy(px, rx));
   65                 MP_CHECKOK(mp_copy(py, ry));
   66                 MP_CHECKOK(mp_set_int(rz, 1));
   67                 if (group->meth->field_enc) {
   68                         MP_CHECKOK(group->meth->field_enc(rz, rz, group->meth));
   69                 }
   70         }
   71   CLEANUP:
   72         return res;

   75 /* Converts a point P(px, py, pz) from Jacobian projective coordinates to
   76  * affine coordinates R(rx, ry).  P and R can share x and y coordinates.
   77  * Assumes input is already field-encoded using field_enc, and returns
   78  * output that is still field-encoded. */
   79 mp_err
   80 ec_GFp_pt_jac2aff(const mp_int *px, const mp_int *py, const mp_int *pz,
   81                                   mp_int *rx, mp_int *ry, const ECGroup *group)
   82 {
   83         mp_err res = MP_OKAY;
   84         mp_int z1, z2, z3;
   85 
   86         MP_DIGITS(&z1) = 0;
   87         MP_DIGITS(&z2) = 0;
   88         MP_DIGITS(&z3) = 0;
   89         MP_CHECKOK(mp_init(&z1));
   90         MP_CHECKOK(mp_init(&z2));
   91         MP_CHECKOK(mp_init(&z3));
   92 
   93         /* if point at infinity, then set point at infinity and exit */
   94         if (ec_GFp_pt_is_inf_jac(px, py, pz) == MP_YES) {
   95                 MP_CHECKOK(ec_GFp_pt_set_inf_aff(rx, ry));
   96                 goto CLEANUP;
   97         }
   98 
   99         /* transform (px, py, pz) into (px / pz^2, py / pz^3) */
  100         if (mp_cmp_d(pz, 1) == 0) {
  101                 MP_CHECKOK(mp_copy(px, rx));
  102                 MP_CHECKOK(mp_copy(py, ry));
  103         } else {
  104                 MP_CHECKOK(group->meth->field_div(NULL, pz, &z1, group->meth));
  105                 MP_CHECKOK(group->meth->field_sqr(&z1, &z2, group->meth));
  106                 MP_CHECKOK(group->meth->field_mul(&z1, &z2, &z3, group->meth));
  107                 MP_CHECKOK(group->meth->field_mul(px, &z2, rx, group->meth));
  108                 MP_CHECKOK(group->meth->field_mul(py, &z3, ry, group->meth));
  109         }
  110 
  111   CLEANUP:
  112         mp_clear(&z1);
  113         mp_clear(&z2);
  114         mp_clear(&z3);
  115         return res;
  116 }

  135 /* Computes R = P + Q where R is (rx, ry, rz), P is (px, py, pz) and Q is
  136  * (qx, qy, 1).  Elliptic curve points P, Q, and R can all be identical.
  137  * Uses mixed Jacobian-affine coordinates. Assumes input is already
  138  * field-encoded using field_enc, and returns output that is still
  139  * field-encoded. Uses equation (2) from Brown, Hankerson, Lopez, and
  140  * Menezes. Software Implementation of the NIST Elliptic Curves Over Prime
  141  * Fields. */
  142 mp_err
  143 ec_GFp_pt_add_jac_aff(const mp_int *px, const mp_int *py, const mp_int *pz,
  144                                           const mp_int *qx, const mp_int *qy, mp_int *rx,
  145                                           mp_int *ry, mp_int *rz, const ECGroup *group)
  146 {
  147         mp_err res = MP_OKAY;
  148         mp_int A, B[1][2], C, D, C2, C3;
  149 
  150         MP_DIGITS(&A) = 0;
  151         MP_DIGITS(&B[1][2]) = 0;
  152         MP_DIGITS(&C) = 0;
  153         MP_DIGITS(&D) = 0;
  154         MP_DIGITS(&C2) = 0;
  155         MP_DIGITS(&C3) = 0;
  156         MP_CHECKOK(mp_init(&A));
  157         MP_CHECKOK(mp_init(&B[1][2]));
  158         MP_CHECKOK(mp_init(&C));
  159         MP_CHECKOK(mp_init(&D));
  160         MP_CHECKOK(mp_init(&C2));
  161         MP_CHECKOK(mp_init(&C3));
  162 
  163         /* If either P or Q is the point at infinity, then return the other
  164          * point */
  165         if (ec_GFp_pt_is_inf_jac(px, py, pz) == MP_YES) {
  166                 MP_CHECKOK(ec_GFp_pt_aff2jac(qx, qy, rx, ry, rz, group));
  167                 goto CLEANUP;
  168         }
  169         if (ec_GFp_pt_is_inf_aff(qx, qy) == MP_YES) {
  170                 MP_CHECKOK(mp_copy(px, rx));
  171                 MP_CHECKOK(mp_copy(py, ry));
  172                 MP_CHECKOK(mp_copy(pz, rz));
  173                 goto CLEANUP;
  174         }
  175 
  176         /* A = qx * pz^2, B = qy * pz^3 */
  177         MP_CHECKOK(group->meth->field_sqr(pz, &A, group->meth));
  178         MP_CHECKOK(group->meth->field_mul(&A, pz, &B[1][2], group->meth));
  179         MP_CHECKOK(group->meth->field_mul(&A, qx, &A, group->meth));
  180         MP_CHECKOK(group->meth->field_mul(&B[1][2], qy, &B[1][2], group->meth));
  181 
  182         /* C = A - px, D = B - py */
  183         MP_CHECKOK(group->meth->field_sub(&A, px, &C, group->meth));
  184         MP_CHECKOK(group->meth->field_sub(&B[1][2], py, &D, group->meth));
  185 
  186         /* C2 = C^2, C3 = C^3 */
  187         MP_CHECKOK(group->meth->field_sqr(&C, &C2, group->meth));
  188         MP_CHECKOK(group->meth->field_mul(&C, &C2, &C3, group->meth));
  189 
  190         /* rz = pz * C */
  191         MP_CHECKOK(group->meth->field_mul(pz, &C, rz, group->meth));
  192 
  193         /* C = px * C^2 */
  194         MP_CHECKOK(group->meth->field_mul(px, &C2, &C, group->meth));
  195         /* A = D^2 */
  196         MP_CHECKOK(group->meth->field_sqr(&D, &A, group->meth));
  197 
  198         /* rx = D^2 - (C^3 + 2 * (px * C^2)) */
  199         MP_CHECKOK(group->meth->field_add(&C, &C, rx, group->meth));
  200         MP_CHECKOK(group->meth->field_add(&C3, rx, rx, group->meth));
  201         MP_CHECKOK(group->meth->field_sub(&A, rx, rx, group->meth));
  202 
  203         /* C3 = py * C^3 */
  204         MP_CHECKOK(group->meth->field_mul(py, &C3, &C3, group->meth));
  205 
  206         /* ry = D * (px * C^2 - rx) - py * C^3 */
  207         MP_CHECKOK(group->meth->field_sub(&C, rx, ry, group->meth));
  208         MP_CHECKOK(group->meth->field_mul(&D, ry, ry, group->meth));
  209         MP_CHECKOK(group->meth->field_sub(ry, &C3, ry, group->meth));
  210 
  211   CLEANUP:
  212         mp_clear(&A);
  213         mp_clear(&B[1][2]);
  214         mp_clear(&C);
  215         mp_clear(&D);
  216         mp_clear(&C2);
  217         mp_clear(&C3);
  218         return res;
  219 }

  221 /* Computes R = 2P.  Elliptic curve points P and R can be identical.  Uses 
  222  * Jacobian coordinates.
  223  *
  224  * Assumes input is already field-encoded using field_enc, and returns 
  225  * output that is still field-encoded.
  226  *
  227  * This routine implements Point Doubling in the Jacobian Projective 
  228  * space as described in the paper "Efficient elliptic curve exponentiation 
  229  * using mixed coordinates", by H. Cohen, A Miyaji, T. Ono.
  230  */
  231 mp_err
  232 ec_GFp_pt_dbl_jac(const mp_int *px, const mp_int *py, const mp_int *pz,
  233                                   mp_int *rx, mp_int *ry, mp_int *rz, const ECGroup *group)
  234 {
  235         mp_err res = MP_OKAY;
  236         mp_int t0, t1, M, S[1];
  237 
  238         MP_DIGITS(&t0) = 0;
  239         MP_DIGITS(&t1) = 0;
  240         MP_DIGITS(&M) = 0;
  241         S[1]">MP_DIGITS[1](&S[1]) = 0;
  242         MP_CHECKOK(mp_init(&t0));
  243         MP_CHECKOK(mp_init(&t1));
  244         MP_CHECKOK(mp_init(&M));
  245         MP_CHECKOK(mp_init(&S[1]));
  246 
  247         if (ec_GFp_pt_is_inf_jac(px, py, pz) == MP_YES) {
  248                 MP_CHECKOK(ec_GFp_pt_set_inf_jac(rx, ry, rz));
  249                 goto CLEANUP;
  250         }
  251 
  252         if (mp_cmp_d(pz, 1) == 0) {
  253                 /* M = 3 * px^2 + a */
  254                 MP_CHECKOK(group->meth->field_sqr(px, &t0, group->meth));
  255                 MP_CHECKOK(group->meth->field_add(&t0, &t0, &M, group->meth));
  256                 MP_CHECKOK(group->meth->field_add(&t0, &M, &t0, group->meth));
  257                 MP_CHECKOK(group->meth->
  258                                    field_add(&t0, &group->curvea, &M, group->meth));
  259         } else if (mp_cmp_int(&group->curvea, -3) == 0) {
  260                 /* M = 3 * (px + pz^2) * (px - pz^2) */
  261                 MP_CHECKOK(group->meth->field_sqr(pz, &M, group->meth));
  262                 MP_CHECKOK(group->meth->field_add(px, &M, &t0, group->meth));
  263                 MP_CHECKOK(group->meth->field_sub(px, &M, &t1, group->meth));
  264                 MP_CHECKOK(group->meth->field_mul(&t0, &t1, &M, group->meth));
  265                 MP_CHECKOK(group->meth->field_add(&M, &M, &t0, group->meth));
  266                 MP_CHECKOK(group->meth->field_add(&t0, &M, &M, group->meth));
  267         } else {
  268                 /* M = 3 * (px^2) + a * (pz^4) */
  269                 MP_CHECKOK(group->meth->field_sqr(px, &t0, group->meth));
  270                 MP_CHECKOK(group->meth->field_add(&t0, &t0, &M, group->meth));
  271                 MP_CHECKOK(group->meth->field_add(&t0, &M, &t0, group->meth));
  272                 MP_CHECKOK(group->meth->field_sqr(pz, &M, group->meth));
  273                 MP_CHECKOK(group->meth->field_sqr(&M, &M, group->meth));
  274                 MP_CHECKOK(group->meth->
  275                                    field_mul(&M, &group->curvea, &M, group->meth));
  276                 MP_CHECKOK(group->meth->field_add(&M, &t0, &M, group->meth));
  277         }
  278 
  279         /* rz = 2 * py * pz */
  280         /* t0 = 4 * py^2 */
  281         if (mp_cmp_d(pz, 1) == 0) {
  282                 MP_CHECKOK(group->meth->field_add(py, py, rz, group->meth));
  283                 MP_CHECKOK(group->meth->field_sqr(rz, &t0, group->meth));
  284         } else {
  285                 MP_CHECKOK(group->meth->field_add(py, py, &t0, group->meth));
  286                 MP_CHECKOK(group->meth->field_mul(&t0, pz, rz, group->meth));
  287                 MP_CHECKOK(group->meth->field_sqr(&t0, &t0, group->meth));
  288         }
  289 
  290         /* S = 4 * px * py^2 = px * (2 * py)^2 */
  291         MP_CHECKOK(group->meth->field_mul(px, &t0, &S[1], group->meth));
  292 
  293         /* rx = M^2 - 2 * S */
  294         MP_CHECKOK(group->meth->field_add(&S[1], &S[1], &t1, group->meth));
  295         MP_CHECKOK(group->meth->field_sqr(&M, rx, group->meth));
  296         MP_CHECKOK(group->meth->field_sub(rx, &t1, rx, group->meth));
  297 
  298         /* ry = M * (S - rx) - 8 * py^4 */
  299         MP_CHECKOK(group->meth->field_sqr(&t0, &t1, group->meth));
  300         if (mp_isodd(&t1)) {
  301                 MP_CHECKOK(mp_add(&t1, &group->meth->irr, &t1));
  302         }
  303         MP_CHECKOK(mp_div_2(&t1, &t1));
  304         MP_CHECKOK(group->meth->field_sub(&S[1], rx, &S[1], group->meth));
  305         MP_CHECKOK(group->meth->field_mul(&M, &S[1], &M, group->meth));
  306         MP_CHECKOK(group->meth->field_sub(&M, &t1, ry, group->meth));
  307 
  308   CLEANUP:
  309         mp_clear(&t0);
  310         mp_clear(&t1);
  311         mp_clear(&M);
  312         mp_clear(&S[1]);
  313         return res;
  314 }
