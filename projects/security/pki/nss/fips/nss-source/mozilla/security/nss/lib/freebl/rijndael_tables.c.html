   94 /* GF_MULTIPLY
   95  *
   96  * multiply two bytes represented in GF(2**8), mod (x**4 + 1)
   97  */
   98 PRUint8 gf_multiply(PRUint8 a, PRUint8 b)
   99 {
  100     PRUint8 res = 0;
  101     while (b > 0) {
  102         res = (b & 0x01) ? res ^ a : res;
  103         a = (a & 0x80) ? ((a << 1) ^ 0x1b) : (a << 1);
  104         b >>= 1;
  105     }
  106     return res;
  107 }
