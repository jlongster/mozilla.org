   57 static sslSessionID *cache = NULL;
   58 static PZLock *      cacheLock = NULL;

   60 /* sids can be in one of 4 states:
   61  *
   62  * never_cached,         created, but not yet put into cache. 
   63  * in_client_cache,         in the client cache's linked list.
   64  * in_server_cache,         entry came from the server's cache file.
   65  * invalid_cache        has been removed from the cache. 
   66  */
   67 
   68 #define LOCK_CACHE         lock_cache()
   69 #define UNLOCK_CACHE        PZ_Unlock(cacheLock)
   70 
   71 void ssl_InitClientSessionCacheLock(void)
   72 {
   73     if (!cacheLock)
   74         nss_InitLock[1](&cacheLock, nssILockCache);
   75 }

   77 static void 
   78 lock_cache(void)
   79 {
   80     ssl_InitClientSessionCacheLock();
   81     PZ_Lock(cacheLock);
   82 }

   84 /* BEWARE: This function gets called for both client and server SIDs !!
   85  * If the unreferenced sid is not in the cache, Free sid and its contents.
   86  */
   87 static void
   88 ssl_DestroySID(sslSessionID *sid)
   89 {
   90     SSL_TRC(8, ("SSL: destroy sid: sid=0x%x cached=%d", sid, sid->cached));
   91     PORT_Assert[1]((sid->references == 0));
   92 
   93     if (sid->cached == in_client_cache)
   94             return;        /* it will get taken care of next time cache is traversed. */
   95 
   96     if (sid->version[2][3] < SSL_LIBRARY_VERSION_3_0) {
   97         SECITEM_ZfreeItem(&sid->u.ssl2.masterKey, PR_FALSE);
   98         SECITEM_ZfreeItem(&sid->u.ssl2.cipherArg, PR_FALSE);
   99     }
  100     if (sid->peerID != NULL)
  101         PORT_Free((void *)sid->peerID);                /* CONST */
  102 
  103     if (sid->urlSvrName != NULL)
  104         PORT_Free((void *)sid->urlSvrName);        /* CONST */
  105 
  106     if ( sid->peerCert ) {
  107         CERT_DestroyCertificate(sid->peerCert);
  108     }
  109     if ( sid->localCert ) {
  110         CERT_DestroyCertificate(sid->localCert);
  111     }
  112     
  113     PORT_ZFree(sid, sizeof(sslSessionID));
  114 }

  116 /* BEWARE: This function gets called for both client and server SIDs !!
  117  * Decrement reference count, and 
  118  *    free sid if ref count is zero, and sid is not in the cache. 
  119  * Does NOT remove from the cache first.  
  120  * If the sid is still in the cache, it is left there until next time
  121  * the cache list is traversed.
  122  */
  123 static void 
  124 ssl_FreeLockedSID(sslSessionID *sid)
  125 {
  126     PORT_Assert[1](sid->references >= 1);
  127     if (--sid->references == 0) {
  128         ssl_DestroySID(sid);
  129     }
  130 }
  131 
  132 /* BEWARE: This function gets called for both client and server SIDs !!
  133  * Decrement reference count, and 
  134  *    free sid if ref count is zero, and sid is not in the cache. 
  135  * Does NOT remove from the cache first.  
  136  * These locks are necessary because the sid _might_ be in the cache list.
  137  */
  138 void
  139 ssl_FreeSID(sslSessionID *sid)
  140 {
  141     LOCK_CACHE;
  142     ssl_FreeLockedSID(sid);
  143     UNLOCK_CACHE;
  144 }

  148 /*
  149 **  Lookup sid entry in cache by Address, port, and peerID string.
  150 **  If found, Increment reference count, and return pointer to caller.
  151 **  If it has timed out or ref count is zero, remove from list and free it.
  152 */
  153 
  154 sslSessionID *
  155 ssl_LookupSID(const PRIPv6Addr *addr, PRUint16 port[1], const char *peerID, 
  156               const char * urlSvrName)
  157 {
  158     sslSessionID **sidp;
  159     sslSessionID * sid;
  160     PRUint32       now[1][2];
  161 
  162     if (!urlSvrName)
  163             return NULL;
  164     now[1][2] = ssl_Time();
  165     LOCK_CACHE;
  166     sidp = &cache;
  167     while ((sid = *sidp) != 0) {
  168         PORT_Assert[1](sid->cached == in_client_cache);
  169         PORT_Assert[1](sid->references >= 1);
  170 
  171         SSL_TRC(8, ("SSL: Lookup1: sid=0x%x", sid));
  172 
  173         if (sid->expirationTime < now[1][2] || !sid->references) {
  174             /*
  175             ** This session-id timed out, or was orphaned.
  176             ** Don't even care who it belongs to, blow it out of our cache.
  177             */
  178             SSL_TRC(7, ("SSL: lookup1, throwing sid out[1][2][3][4], age=%d refs=%d",
  179                         now[1][2] - sid->creationTime, sid->references));
  180 
  181             *sidp = sid->next;                         /* delink it from the list. */
  182             sid->cached = invalid_cache;        /* mark not on list. */
  183             if (!sid->references)
  184                     ssl_DestroySID(sid);
  185             else
  186                 ssl_FreeLockedSID(sid);                /* drop ref count, free. */
  187 
  188         } else if (!memcmp[1](&sid->addr, addr, sizeof(PRIPv6Addr)) && /* server IP addr matches */
  189                    (sid->port[1] == port[1]) && /* server port[1] matches */
  190                    /* proxy (peerID) matches */
  191                    (((peerID == NULL) && (sid->peerID == NULL)) ||
  192                     ((peerID != NULL) && (sid->peerID != NULL) &&
  193                      PORT_Strcmp(sid->peerID, peerID) == 0)) &&
  194                    /* is cacheable */
  195                    (sid->version[2][3] < SSL_LIBRARY_VERSION_3_0 ||
  196                     sid->u.ssl3.keys.resumable) &&
  197                    /* server hostname matches. */
  198                    (sid->urlSvrName != NULL) &&
  199                    ((0 == PORT_Strcmp(urlSvrName, sid->urlSvrName)) ||
  200                     ((sid->peerCert != NULL) && (SECSuccess == 
  201                       CERT_VerifyCertName(sid->peerCert, urlSvrName))) )
  202                   ) {
  203             /* Hit */
  204             sid->lastAccessTime = now[1][2];
  205             sid->references++;
  206             break;
  207         } else {
  208             sidp = &sid->next;
  209         }
  210     }
  211     UNLOCK_CACHE;
  212     return sid;
  213 }

  215 /*
  216 ** Add an sid to the cache or return a previously cached entry to the cache.
  217 ** Although this is static, it is called via ss->sec.cache().
  218 */
  219 static void 
  220 CacheSID(sslSessionID *sid)
  221 {
  222     PRUint32  expirationPeriod;
  223     SSL_TRC(8, ("SSL: Cache: sid=0x%x cached=%d addr=0x%08x%08x%08x%08x port[1]=0x%04x "
  224                 "time=%x cached=%d",
  225                 sid, sid->cached, sid->addr.pr_s6_addr32[0], 
  226                 sid->addr.pr_s6_addr32[1], sid->addr.pr_s6_addr32[2],
  227                 sid->addr.pr_s6_addr32[3],  sid->port[1], sid->creationTime,
  228                 sid->cached));
  229 
  230     if (sid->cached == in_client_cache)
  231         return;
  232 
  233     if (!sid->urlSvrName) {
  234         /* don't cache this SID because it can never be matched */
  235         return;
  236     }
  237 
  238     /* XXX should be different trace for version 2 vs. version 3 */
  239     if (sid->version[2][3] < SSL_LIBRARY_VERSION_3_0) {
  240         expirationPeriod = ssl_sid_timeout[1];
  241         PRINT_BUF(8, (0, "sessionID:",
  242                   sid->u.ssl2.sessionID, sizeof(sid->u.ssl2.sessionID)));
  243         PRINT_BUF(8, (0, "masterKey:",
  244                   sid->u.ssl2.masterKey.data[2], sid->u.ssl2.masterKey.len[1][2][3][6][9][10][13]));
  245         PRINT_BUF(8, (0, "cipherArg:",
  246                   sid->u.ssl2.cipherArg.data[2], sid->u.ssl2.cipherArg.len[1][2][3][6][9][10][13]));
  247     } else {
  248         if (sid->u.ssl3.sessionIDLength == 0) 
  249             return;
  250         expirationPeriod = ssl3_sid_timeout[1];
  251         PRINT_BUF(8, (0, "sessionID:",
  252                       sid->u.ssl3.sessionID, sid->u.ssl3.sessionIDLength));
  253     }
  254     PORT_Assert[1](sid->creationTime != 0 && sid->expirationTime != 0);
  255     if (!sid->creationTime)
  256         sid->lastAccessTime = sid->creationTime = ssl_Time();
  257     if (!sid->expirationTime)
  258         sid->expirationTime = sid->creationTime + expirationPeriod;
  259 
  260     /*
  261      * Put sid into the cache.  Bump reference count to indicate that
  262      * cache is holding a reference. Uncache will reduce the cache
  263      * reference.
  264      */
  265     LOCK_CACHE;
  266     sid->references++;
  267     sid->cached = in_client_cache;
  268     sid->next   = cache;
  269     cache       = sid;
  270     UNLOCK_CACHE;
  271 }

  273 /* 
  274  * If sid "zap" is in the cache,
  275  *    removes sid from cache, and decrements reference count.
  276  * Caller must hold cache lock.
  277  */
  278 static void
  279 UncacheSID(sslSessionID *zap)
  280 {
  281     sslSessionID **sidp = &cache;
  282     sslSessionID *sid;
  283 
  284     if (zap->cached != in_client_cache) {
  285         return;
  286     }
  287 
  288     SSL_TRC(8,("SSL: Uncache: zap=0x%x cached=%d addr=0x%08x%08x%08x%08x port[1]=0x%04x "
  289                "time=%x cipher=%d",
  290                zap, zap->cached, zap->addr.pr_s6_addr32[0],
  291                zap->addr.pr_s6_addr32[1], zap->addr.pr_s6_addr32[2],
  292                zap->addr.pr_s6_addr32[3], zap->port[1], zap->creationTime,
  293                zap->u.ssl2.cipherType));
  294     if (zap->version[2][3] < SSL_LIBRARY_VERSION_3_0) {
  295         PRINT_BUF(8, (0, "sessionID:",
  296                       zap->u.ssl2.sessionID, sizeof(zap->u.ssl2.sessionID)));
  297         PRINT_BUF(8, (0, "masterKey:",
  298                       zap->u.ssl2.masterKey.data[2], zap->u.ssl2.masterKey.len[1][2][3][6][9][10][13]));
  299         PRINT_BUF(8, (0, "cipherArg:",
  300                       zap->u.ssl2.cipherArg.data[2], zap->u.ssl2.cipherArg.len[1][2][3][6][9][10][13]));
  301     }
  302 
  303     /* See if it's in the cache, if so nuke it */
  304     while ((sid = *sidp) != 0) {
  305         if (sid == zap) {
  306             /*
  307             ** Bingo. Reduce reference count by one so that when
  308             ** everyone is done with the sid we can free it up.
  309             */
  310             *sidp = zap->next;
  311             zap->cached = invalid_cache;
  312             ssl_FreeLockedSID(zap);
  313             return;
  314         }
  315         sidp = &sid->next;
  316     }
  317 }
  318 
  319 /* If sid "zap" is in the cache,
  320  *    removes sid from cache, and decrements reference count.
  321  * Although this function is static, it is called externally via 
  322  *    ss->sec.uncache().
  323  */
  324 static void
  325 LockAndUncacheSID(sslSessionID *zap)
  326 {
  327     LOCK_CACHE;
  328     UncacheSID(zap);
  329     UNLOCK_CACHE;
  330 
  331 }

  333 /* choose client or server cache functions for this sslsocket. */
  334 void 
  335 ssl_ChooseSessionIDProcs(sslSecurityInfo *sec)
  336 {
  337     if (sec->isServer) {
  338         sec->cache   = cache">ssl_sid_cachecache">[1];
  339         sec->uncache = ssl_sid_uncache[1];

  341         sec->cache   = CacheSID;
  342         sec->uncache = LockAndUncacheSID;

  346 /* wipe out the entire client session cache. */
  347 void
  348 SSL_ClearSessionCache(void)
  349 {
  350     LOCK_CACHE;
  351     while(cache != NULL)
  352         UncacheSID(cache);
  353     UNLOCK_CACHE;
  354 }

  356 /* returns an unsigned int containing the number of seconds in PR_Now() */
  357 PRUint32
  358 ssl_Time(void)
  359 {
  360     PRUint32 myTime;
  361 #if (defined(XP_UNIX) || defined(XP_WIN) || defined(_WINDOWS) || defined(XP_BEOS)) && !defined(_WIN32_WCE)
  362     myTime = time(NULL);        /* accurate until the year 2038. */
  363 #else
  364     /* portable, but possibly slower */
  365     PRTime now[1][2];
  366     PRInt64 ll;
  367 
  368     now[1][2] = PR_Now[1][2][3][4][5]();
  369     LL_I2L(ll, 1000000L);
  370     LL_DIV(now[1][2], now[1][2], ll);
  371     LL_L2UI(myTime, now[1][2]);
  372 #endif
  373     return myTime;
  374 }
