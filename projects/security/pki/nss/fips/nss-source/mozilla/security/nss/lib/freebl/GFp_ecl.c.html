   77 /* Computes R = P + Q based on IEEE P1363 A.10.1.
   78  * Elliptic curve points P, Q, and R can all be identical.
   79  * Uses affine coordinates.
   80  */
   81 mp_err
   82 GFp_ec_pt_add_aff(const mp_int *p, const mp_int *a, const mp_int *px,
   83     const mp_int *py, const mp_int *qx, const mp_int *qy, 
   84     mp_int *rx, mp_int *ry)
   85 {
   86     mp_err err = MP_OKAY;
   87     mp_int lambda, temp, xtemp, ytemp;
   88 
   89     CHECK_MPI_OK( mp_init(&lambda) );
   90     CHECK_MPI_OK( mp_init(&temp) );
   91     CHECK_MPI_OK( mp_init(&xtemp) );
   92     CHECK_MPI_OK( mp_init(&ytemp) );
   93     /* if P = inf, then R = Q */
   94     if (GFp_ec_pt_is_inf_aff(px, py) == 0) {
   95         CHECK_MPI_OK( mp_copy(qx, rx) );
   96         CHECK_MPI_OK( mp_copy(qy, ry) );
   97         err = MP_OKAY;
   98         goto cleanup;
   99     }
  100     /* if Q = inf, then R = P */
  101     if (GFp_ec_pt_is_inf_aff(qx, qy) == 0) {
  102         CHECK_MPI_OK( mp_copy(px, rx) );
  103         CHECK_MPI_OK( mp_copy(py, ry) );
  104         err = MP_OKAY;
  105         goto cleanup;
  106     }
  107     /* if px != qx, then lambda = (py-qy) / (px-qx) */
  108     if (mp_cmp(px, qx) != 0) {
  109         CHECK_MPI_OK( mp_submod(py, qy, p, &ytemp) );
  110         CHECK_MPI_OK( mp_submod(px, qx, p, &xtemp) );
  111         CHECK_MPI_OK( mp_invmod(&xtemp, p, &xtemp) );
  112         CHECK_MPI_OK( mp_mulmod(&ytemp, &xtemp, p, &lambda) );
  113     } else {
  114         /* if py != qy or qy = 0, then R = inf */
  115         if (((mp_cmp(py, qy) != 0)) || (mp_cmp_z">mp_cmp_z(qy) == 0)) {
  116             mp_zero(rx);
  117             mp_zero(ry);
  118             err = MP_OKAY;
  119             goto cleanup;
  120         }
  121         /* lambda = (3qx^2+a) / (2qy) */
  122         CHECK_MPI_OK( mp_sqrmod[1](qx, p, &xtemp) );
  123         mp_set(&temp, 0x3);
  124         CHECK_MPI_OK( mp_mulmod(&xtemp, &temp, p, &xtemp) );
  125         CHECK_MPI_OK( mp_addmod(&xtemp, a, p, &xtemp) );
  126         mp_set(&temp, 0x2);
  127         CHECK_MPI_OK( mp_mulmod(qy, &temp, p, &ytemp) );
  128         CHECK_MPI_OK( mp_invmod(&ytemp, p, &ytemp) );
  129         CHECK_MPI_OK( mp_mulmod(&xtemp, &ytemp, p, &lambda) );
  130     }
  131     /* rx = lambda^2 - px - qx */
  132     CHECK_MPI_OK( mp_sqrmod[1](&lambda, p, &xtemp) );
  133     CHECK_MPI_OK( mp_submod(&xtemp, px, p, &xtemp) );
  134     CHECK_MPI_OK( mp_submod(&xtemp, qx, p, &xtemp) );
  135     /* ry = (x1-x2) * lambda - y1 */
  136     CHECK_MPI_OK( mp_submod(qx, &xtemp, p, &ytemp) );
  137     CHECK_MPI_OK( mp_mulmod(&ytemp, &lambda, p, &ytemp) );
  138     CHECK_MPI_OK( mp_submod(&ytemp, qy, p, &ytemp) );
  139     CHECK_MPI_OK( mp_copy(&xtemp, rx) );
  140     CHECK_MPI_OK( mp_copy(&ytemp, ry) );
  141 
  142 cleanup:
  143     mp_clear(&lambda);
  144     mp_clear(&temp);
  145     mp_clear(&xtemp);
  146     mp_clear(&ytemp);
  147     return err;
  148 }

  289 /* Converts a point P(px, py, pz) from Jacobian projective coordinates to
  290  * affine coordinates R(rx, ry).  P and R can share x and y coordinates.
  291  */
  292 mp_err
  293 GFp_ec_pt_jac2aff(const mp_int *px, const mp_int *py, const mp_int *pz,
  294     const mp_int *p, mp_int *rx, mp_int *ry)
  295 {
  296     mp_err err = MP_OKAY;
  297     mp_int z1, z2, z3;
  298     MP_DIGITS(&z1) = 0;
  299     MP_DIGITS(&z2) = 0;
  300     MP_DIGITS(&z3) = 0;
  301     CHECK_MPI_OK( mp_init(&z1) );
  302     CHECK_MPI_OK( mp_init(&z2) );
  303     CHECK_MPI_OK( mp_init(&z3) );
  304 
  305     /* if point at infinity, then set point at infinity and exit */
  306     if (GFp_ec_pt_is_inf_jac(px, py, pz) == MP_YES) {
  307         CHECK_MPI_OK( GFp_ec_pt_set_inf_aff(rx, ry) );
  308         goto cleanup;
  309     }
  310 
  311     /* transform (px, py, pz) into (px / pz^2, py / pz^3) */
  312     if (mp_cmp_d(pz, 1) == 0) {
  313         CHECK_MPI_OK( mp_copy(px, rx) );
  314         CHECK_MPI_OK( mp_copy(py, ry) );
  315     } else {
  316         CHECK_MPI_OK( mp_invmod(pz, p, &z1) );
  317         CHECK_MPI_OK( mp_sqrmod[1](&z1, p, &z2) );
  318         CHECK_MPI_OK( mp_mulmod(&z1, &z2, p, &z3) );
  319         CHECK_MPI_OK( mp_mulmod(px, &z2, p, rx) );
  320         CHECK_MPI_OK( mp_mulmod(py, &z3, p, ry) );
  321     }
  322         
  323 cleanup:
  324     mp_clear(&z1);
  325     mp_clear(&z2);
  326     mp_clear(&z3);
  327         return err;
  328 }

  349 /* Computes R = P + Q where R is (rx, ry, rz), P is (px, py, pz) and 
  350  * Q is (qx, qy, qz).  Elliptic curve points P, Q, and R can all be 
  351  * identical. Uses Jacobian coordinates.
  352  *
  353  * This routine implements Point Addition in the Jacobian Projective 
  354  * space as described in the paper "Efficient elliptic curve exponentiation 
  355  * using mixed coordinates", by H. Cohen, A Miyaji, T. Ono.
  356  */
  357 mp_err
  358 GFp_ec_pt_add_jac(const mp_int *p, const mp_int *a, const mp_int *px,
  359     const mp_int *py, const mp_int *pz, const mp_int *qx, 
  360     const mp_int *qy, const mp_int *qz, mp_int *rx, mp_int *ry, mp_int *rz)
  361 {
  362     mp_err err = MP_OKAY;
  363     mp_int n0, u1, u2, s1, s2, H[1], G[1];
  364     MP_DIGITS(&n0) = 0;
  365     MP_DIGITS(&u1) = 0;
  366     MP_DIGITS(&u2) = 0;
  367     MP_DIGITS(&s1) = 0;
  368     MP_DIGITS(&s2) = 0;
  369     MP_DIGITS(&H[1]) = 0;
  370     MP_DIG[1]ITS(&G[1]) = 0;
  371     CHECK_MPI_OK( mp_init(&n0) );
  372     CHECK_MPI_OK( mp_init(&u1) );
  373     CHECK_MPI_OK( mp_init(&u2) );
  374     CHECK_MPI_OK( mp_init(&s1) );
  375     CHECK_MPI_OK( mp_init(&s2) );
  376     CH[1]ECK_MPI_OK( mp_init(&H[1]) );
  377     CHECK_MPI_OK( mp_init(&G[1]) );
  378 
  379     /* Use point double if pointers are equal. */
  380     if ((px == qx) && (py == qy) && (pz == qz)) {
  381         err = GFp_ec_pt_dbl_jac(p, a, px, py, pz, rx, ry, rz);
  382         goto cleanup;
  383     }
  384         
  385     /* If either P or Q is the point at infinity, then return 
  386      * the other point 
  387      */
  388     if (GFp_ec_pt_is_inf_jac(px, py, pz) == MP_YES) {
  389         CHECK_MPI_OK( mp_copy(qx, rx) );
  390         CHECK_MPI_OK( mp_copy(qy, ry) );
  391         CHECK_MPI_OK( mp_copy(qz, rz) );
  392         goto cleanup;
  393     }
  394     if (GFp_ec_pt_is_inf_jac(qx, qy, qz) == MP_YES) {
  395         CHECK_MPI_OK( mp_copy(px, rx) );
  396         CHECK_MPI_OK( mp_copy(py, ry) );
  397         CHECK_MPI_OK( mp_copy(pz, rz) );
  398         goto cleanup;
  399     }
  400         
  401     /* Compute u1 = px * qz^2, s1 = py * qz^3 */
  402     if (mp_cmp_d(qz, 1) == 0) {
  403         CHECK_MPI_OK( mp_copy(px, &u1) );
  404         CHECK_MPI_OK( mp_copy(py, &s1) );
  405     } else {
  406         CHECK_MPI_OK( mp_sqrmod[1](qz, p, &n0) );
  407         CHECK_MPI_OK( mp_mulmod(px, &n0, p, &u1) );
  408         CHECK_MPI_OK( mp_mulmod(&n0, qz, p, &n0) );
  409         CHECK_MPI_OK( mp_mulmod(py, &n0, p, &s1) );
  410     }
  411 
  412     /* Compute u2 = qx * pz^2, s2 = qy * pz^3 */
  413     if (mp_cmp_d(pz, 1) == 0) {
  414         CHECK_MPI_OK( mp_copy(qx, &u2) );
  415         CHECK_MPI_OK( mp_copy(qy, &s2) );
  416     } else {
  417         CHECK_MPI_OK( mp_sqrmod[1](pz, p, &n0) );
  418         CHECK_MPI_OK( mp_mulmod(qx, &n0, p, &u2) );
  419         CHECK_MPI_OK( mp_mulmod(&n0, pz, p, &n0) );
  420         CHECK_MPI_OK( mp_mulmod(qy, &n0, p, &s2) );
  421     }
  422 
  423     /* Compute H = u2 - u1 ; G = s2 - s1 */
  424     CH[1]ECK_MPI_OK( mp_submod(&u2, &u1, p, &H[1]) );
  425     CHECK_MPI_OK( mp_submod(&s2, &s1, p, &G[1]) );
  426 
  427     if (mp_cmp_z(&H[1]) == 0) {
  428         if (mp_cmp_z(&G[1]) == 0) {
  429             /* P = Q; double */
  430             err = GFp_ec_pt_dbl_jac(p, a, px, py, pz, 
  431                                     rx, ry, rz);
  432             goto cleanup;
  433         } else {
  434             /* P = -Q; return point at infinity */
  435             CHECK_MPI_OK( GFp_ec_pt_set_inf_jac(rx, ry, rz) );
  436             goto cleanup;
  437         }
  438     }
  439 
  440     /* rz = pz * qz * H */
  441     if (mp_cmp_d(pz, 1) == 0) {
  442         if (mp_cmp_d(qz, 1) == 0) {
  443             /* if pz == qz == 1, then rz = H */
  444             CH[1]ECK_MPI_OK( mp_copy(&H[1], rz) );
  445         } else {
  446             CH[1]ECK_MPI_OK( mp_mulmod(qz, &H[1], p, rz) );
  447         }
  448     } else {
  449         if (mp_cmp_d(qz, 1) == 0) {
  450             CH[1]ECK_MPI_OK( mp_mulmod(pz, &H[1], p, rz) );
  451         } else {
  452             CHECK_MPI_OK( mp_mulmod(pz, qz, p, &n0) );
  453             CH[1]ECK_MPI_OK( mp_mulmod(&n0, &H[1], p, rz) );
  454         }
  455     }
  456         
  457     /* rx = G^2 - H^3 - 2 * u1 * H^2 */
  458     CHECK_MPI_OK( mp_sqrmod[1](&G[1], p, rx) );
  459     CH[1]ECK_MPI_OK( mp_sqrmod[1](&H[1], p, &n0) );
  460     CHECK_MPI_OK( mp_mulmod(&n0, &u1, p, &u1) );
  461     CHECK_MPI_OK( mp_addmod(&u1, &u1, p, &u2) );
  462     CH[1]ECK_MPI_OK( mp_mulmod(&H[1], &n0, p, &H[1]) );
  463     CH[1]ECK_MPI_OK( mp_submod(rx, &H[1], p, rx) );
  464     CHECK_MPI_OK( mp_submod(rx, &u2, p, rx) );
  465
