   43 #include <stdio.h>
   44 #include <stdlib.h>
   45 #include <string.h>
   46 #include <ctype.h>
   47 #include <limits.h>
   48 
   49 #include <time.h>
   50 
   51 #include "mpi.h"
   52 #include "mpprime.h"
   53 
   54 int main(int argc, char *argv[])
   55 {
   56   int        ix, num, prec = 8;
   57   double     d1, d2;
   58   clock_t    start, finish;
   59   time_t     seed;
   60   mp_int     a, c, d;
   61 
   62   seed = time(NULL);
   63 
   64   if(argc < 2) {
   65     fprintf(stderr, "Usage: %s <num-tests> [<precision>]\n", argv[0]);
   66     return 1;
   67   }
   68 
   69   if((num = atoi(argv[1])) < 0)
   70     num = -num;
   71 
   72   if(!num) {
   73     fprintf(stderr, "%s: must perform at least 1 test\n", argv[0]);
   74     return 1;
   75   }
   76 
   77   if(argc > 2) {
   78     if((prec = atoi(argv[2])) <= 0)
   79       prec = 8;
   80     else
   81       prec = (prec + (DIGIT_BIT - 1)) / DIGIT_BIT;
   82   }
   83   
   84   printf("Test 3a: Multiplication vs squaring timing test\n"
   85          "Precision:  %d digits (%u bits)\n"
   86          "# of tests: %d\n\n", prec, prec * DIGIT_BIT, num);
   87 
   88   mp_init_size(&a, prec);
   89 
   90   mp_init(&c); mp_init(&d);
   91 
   92   printf("Verifying accuracy ... \n");
   93   srand((unsigned int)seed);
   94   for(ix = 0; ix < num; ix++) {
   95     mpp_random_size(&a, prec);
   96     mp_mul(&a, &a, &c);
   97     mp_sqr[1](&a, &d);
   98 
   99     if(mp_cmp(&c, &d) != 0) {
  100       printf("Error!  Results not accurate:\n");
  101       printf("a = "); mp_print(&a, stdout); fputc('\n', stdout);
  102       printf("c = "); mp_print(&c, stdout); fputc('\n', stdout);
  103       printf("d = "); mp_print(&d, stdout); fputc('\n', stdout);
  104       mp_sub(&c, &d, &d);
  105       printf("dif "); mp_print(&d, stdout); fputc('\n', stdout);
  106       mp_clear(&c); mp_clear(&d);
  107       mp_clear(&a);
  108       return 1;
  109     }
  110   }
  111   printf("Accuracy is confirmed for the %d test samples\n", num);
  112   mp_clear(&d);
  113 
  114   printf("Testing squaring ... \n");
  115   srand((unsigned int)seed);
  116   start = clock();
  117   for(ix = 0; ix < num; ix++) {
  118     mpp_random_size(&a, prec);
  119     mp_sqr[1](&a, &c);
  120   }
  121   finish = clock();
  122 
  123   d2 = (double)(finish - start) / CLOCKS_PER_SEC;
  124 
  125   printf("Testing multiplication ... \n");
  126   srand((unsigned int)seed);
  127   start = clock();
  128   for(ix = 0; ix < num; ix++) {
  129     mpp_random(&a);
  130     mp_mul(&a, &a, &c);
  131   }
  132   finish = clock();
  133 
  134   d1 = (double)(finish - start) / CLOCKS_PER_SEC;
  135 
  136   printf("Multiplication time: %.3f sec (%.3f each)\n", d1, d1 / num);
  137   printf("Squaring time:       %.3f sec (%.3f each)\n", d2, d2 / num);
  138   printf("Improvement:         %.2f%%\n", (1.0 - (d2 / d1)) * 100.0);
  139 
  140   mp_clear(&c);
  141   mp_clear(&a);
  142 
  143   return 0;
  144 }
