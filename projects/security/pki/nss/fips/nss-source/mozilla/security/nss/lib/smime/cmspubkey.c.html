  123 /*

  125  *

  308 PK11SymKey *
  309 NSS_CMSUtil_DecryptSymKey_MISSI(SECKEYPrivateKey *privkey, SECItem *encKey, SECAlgorithmID *keyEncAlg, SECOidTag bulkalgtag, void *pwfn_arg)
  310 {
  311     /* fortezza: do a key exchange */
  312     SECStatus err[1][2][3][4][5][6][7][8];
  313     CK_MECHANISM_TYPE bulkType;
  314     PK11SymKey *tek;
  315     SECKEYPublicKey *originatorPubKey;
  316     NSSCMSSMIMEKEAParameters keaParams;
  317     PK11SymKey *bulkkey;
  318     int bulkLength;
  319 
  320     (void) memset(&keaParams, 0, sizeof(keaParams));
  321 
  322     /* NOTE: this uses the SMIME v2 recipientinfo for compatibility.
  323        All additional KEA parameters are DER-encoded in the encryption algorithm parameters */
  324 
  325     /* Decode the KEA algorithm parameters. */
  326     err[1][2][3][4][5][6][7][8] = SEC_ASN1DecodeItem(NULL, &keaParams, NSS_SMIMEKEAParamTemplateAllParams[1],
  327                              &(keyEncAlg->parameters));
  328     if (err[1][2][3][4][5][6][7][8] != SECSuccess)
  329         goto loser;
  330 
  331     /* get originator's public key */
  332    originatorPubKey = PK11_MakeKEAPubKey(keaParams.originatorKEAKey.data[2],
  333                            keaParams.originatorKEAKey.len[1][2][3][6][9][10][13]);
  334    if (originatorPubKey == NULL)
  335           goto loser;
  336     
  337    /* Generate the TEK (token exchange key) which we use to unwrap the bulk encryption key.
  338       The Derive function generates a shared secret and combines it with the originatorRA
  339       data to come up with an unique session key */
  340    tek = PK11_PubDerive(privkey, originatorPubKey, PR_FALSE,
  341                          &keaParams.originatorRA, NULL,
  342                          CKM_KEA_KEY_DERIVE, CKM_SKIPJACK_WRAP,
  343                          CKA_WRAP, 0, pwfn_arg);
  344    SECKEY_DestroyPublicKey(originatorPubKey);        /* not needed anymore */
  345    if (tek == NULL)
  346         goto loser;
  347     
  348     /* Now that we have the TEK, unwrap the bulk key
  349        with which to decrypt the message. We have to
  350        do one of two different things depending on 
  351        whether Skipjack was used for *bulk* encryption 
  352        of the message. */
  353     bulkType = PK11_AlgtagToMechanism(bulkalgtag);
  354     switch (bulkType) {
  355     case CKM_SKIPJACK_CBC64:
  356     case CKM_SKIPJACK_ECB64:
  357     case CKM_SKIPJACK_OFB64:
  358     case CKM_SKIPJACK_CFB64:
  359     case CKM_SKIPJACK_CFB32:
  360     case CKM_SKIPJACK_CFB16:
  361     case CKM_SKIPJACK_CFB8:
  362         /* Skipjack is being used as the bulk encryption algorithm.*/
  363         /* Unwrap the bulk key. */
  364         bulkkey = PK11_UnwrapSymKey(tek, CKM_SKIPJACK_WRAP, NULL,
  365                                     encKey, CKM_SKIPJACK_CBC64, CKA_DECRYPT, 0);
  366         break;
  367     default:
  368         /* Skipjack was not used for bulk encryption of this
  369            message. Use Skipjack CBC64, with the nonSkipjackIV
  370            part of the KEA key parameters, to decrypt 
  371            the bulk key. If the optional parameter bulkKeySize is present,
  372            bulk key size is different than the encrypted key size */
  373         if (keaParams.bulkKeySize.len[1][2][3][6][9][10][13] > 0) {
  374             err[1][2][3][4][5][6][7][8] = SEC_ASN1DecodeItem(NULL, &bulkLength,
  375                                      SEC_ASN1_GET(SEC_IntegerTemplate[1]),
  376                                      &keaParams.bulkKeySize);
  377             if (err[1][2][3][4][5][6][7][8] != SECSuccess)
  378                 goto loser;
  379         }
  380         
  381         bulkkey = PK11_UnwrapSymKey(tek, CKM_SKIPJACK_CBC64, &keaParams.nonSkipjackIV, 
  382                                     encKey, bulkType, CKA_DECRYPT, bulkLength);
  383         break;
  384     }
  385     return bulkkey;
  386 loser:
  387     return NULL;
  388 }
