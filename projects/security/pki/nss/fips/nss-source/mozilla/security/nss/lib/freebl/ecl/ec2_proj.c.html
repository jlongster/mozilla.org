   51 /* Converts a point P(px, py) from affine coordinates to projective
   52  * coordinates R(rx, ry, rz). Assumes input is already field-encoded using 
   53  * field_enc, and returns output that is still field-encoded. */
   54 mp_err
   55 ec_GF2m_pt_aff2proj(const mp_int *px, const mp_int *py, mp_int *rx,
   56                                         mp_int *ry, mp_int *rz, const ECGroup *group)
   57 {
   58         mp_err res = MP_OKAY;
   59 
   60         MP_CHECKOK(mp_copy(px, rx));
   61         MP_CHECKOK(mp_copy(py, ry));
   62         MP_CHECKOK(mp_set_int(rz, 1));
   63         if (group->meth->field_enc) {
   64                 MP_CHECKOK(group->meth->field_enc(rz, rz, group->meth));
   65         }
   66   CLEANUP:
   67         return res;
   68 }

   70 /* Converts a point P(px, py, pz) from projective coordinates to affine
   71  * coordinates R(rx, ry).  P and R can share x and y coordinates. Assumes
   72  * input is already field-encoded using field_enc, and returns output that
   73  * is still field-encoded. */
   74 mp_err
   75 ec_GF2m_pt_proj2aff(const mp_int *px, const mp_int *py, const mp_int *pz,
   76                                         mp_int *rx, mp_int *ry, const ECGroup *group)
   77 {
   78         mp_err res = MP_OKAY;
   79         mp_int z1, z2;
   80 
   81         MP_DIGITS(&z1) = 0;
   82         MP_DIGITS(&z2) = 0;
   83         MP_CHECKOK(mp_init(&z1));
   84         MP_CHECKOK(mp_init(&z2));
   85 
   86         /* if point at infinity, then set point at infinity and exit */
   87         if (ec_GF2m_pt_is_inf_proj(px, py, pz) == MP_YES) {
   88                 MP_CHECKOK(ec_GF2m_pt_set_inf_aff(rx, ry));
   89                 goto CLEANUP;
   90         }
   91 
   92         /* transform (px, py, pz) into (px / pz, py / pz^2) */
   93         if (mp_cmp_d(pz, 1) == 0) {
   94                 MP_CHECKOK(mp_copy(px, rx));
   95                 MP_CHECKOK(mp_copy(py, ry));
   96         } else {
   97                 MP_CHECKOK(group->meth->field_div(NULL, pz, &z1, group->meth));
   98                 MP_CHECKOK(group->meth->field_sqr(&z1, &z2, group->meth));
   99                 MP_CHECKOK(group->meth->field_mul(px, &z1, rx, group->meth));
  100                 MP_CHECKOK(group->meth->field_mul(py, &z2, ry, group->meth));
  101         }
  102 
  103   CLEANUP:
  104         mp_clear(&z1);
  105         mp_clear(&z2);
  106         return res;
  107 }

  127 /* Computes R = P + Q where R is (rx, ry, rz), P is (px, py, pz) and Q is
  128  * (qx, qy, 1).  Elliptic curve points P, Q, and R can all be identical.
  129  * Uses mixed projective-affine coordinates. Assumes input is already
  130  * field-encoded using field_enc, and returns output that is still
  131  * field-encoded. Uses equation (3) from Hankerson, Hernandez, Menezes.
  132  * Software Implementation of Elliptic Curve Cryptography Over Binary
  133  * Fields. */
  134 mp_err
  135 ec_GF2m_pt_add_proj(const mp_int *px, const mp_int *py, const mp_int *pz,
  136                                         const mp_int *qx, const mp_int *qy, mp_int *rx,
  137                                         mp_int *ry, mp_int *rz, const ECGroup *group)
  138 {
  139         mp_err res = MP_OKAY;
  140         mp_int A, B[1][2], C, D, E, F, G;
  141 
  142         /* If either P or Q is the point at infinity, then return the other
  143          * point */
  144         if (ec_GF2m_pt_is_inf_proj(px, py, pz) == MP_YES) {
  145                 return ec_GF2m_pt_aff2proj(qx, qy, rx, ry, rz, group);
  146         }
  147         if (ec_GF2m_pt_is_inf_aff(qx, qy) == MP_YES) {
  148                 MP_CHECKOK(mp_copy(px, rx));
  149                 MP_CHECKOK(mp_copy(py, ry));
  150                 return mp_copy(pz, rz);
  151         }
  152 
  153         MP_DIGITS(&A) = 0;
  154         MP_DIGITS(&B[1][2]) = 0;
  155         MP_DIGITS(&C) = 0;
  156         MP_DIGITS(&D) = 0;
  157         MP_DIGITS(&E) = 0;
  158         MP_DIGITS(&F) = 0;
  159         MP_DIGITS(&G) = 0;
  160         MP_CHECKOK(mp_init(&A));
  161         MP_CHECKOK(mp_init(&B[1][2]));
  162         MP_CHECKOK(mp_init(&C));
  163         MP_CHECKOK(mp_init(&D));
  164         MP_CHECKOK(mp_init(&E));
  165         MP_CHECKOK(mp_init(&F));
  166         MP_CHECKOK(mp_init(&G));
  167 
  168         /* D = pz^2 */
  169         MP_CHECKOK(group->meth->field_sqr(pz, &D, group->meth));
  170 
  171         /* A = qy * pz^2 + py */
  172         MP_CHECKOK(group->meth->field_mul(qy, &D, &A, group->meth));
  173         MP_CHECKOK(group->meth->field_add(&A, py, &A, group->meth));
  174 
  175         /* B = qx * pz + px */
  176         MP_CHECKOK(group->meth->field_mul(qx, pz, &B[1][2], group->meth));
  177         MP_CHECKOK(group->meth->field_add(&B[1][2], px, &B[1][2], group->meth));
  178 
  179         /* C = pz * B */
  180         MP_CHECKOK(group->meth->field_mul(pz, &B[1][2], &C, group->meth));
  181 
  182         /* D = B^2 * (C + a * pz^2) (using E as a temporary variable) */
  183         MP_CHECKOK(group->meth->
  184                            field_mul(&group->curvea, &D, &D, group->meth));
  185         MP_CHECKOK(group->meth->field_add(&C, &D, &D, group->meth));
  186         MP_CHECKOK(group->meth->field_sqr(&B[1][2], &E, group->meth));
  187         MP_CHECKOK(group->meth->field_mul(&E, &D, &D, group->meth));
  188 
  189         /* rz = C^2 */
  190         MP_CHECKOK(group->meth->field_sqr(&C, rz, group->meth));
  191 
  192         /* E = A * C */
  193         MP_CHECKOK(group->meth->field_mul(&A, &C, &E, group->meth));
  194 
  195         /* rx = A^2 + D + E */
  196         MP_CHECKOK(group->meth->field_sqr(&A, rx, group->meth));
  197         MP_CHECKOK(group->meth->field_add(rx, &D, rx, group->meth));
  198         MP_CHECKOK(group->meth->field_add(rx, &E, rx, group->meth));
  199 
  200         /* F = rx + qx * rz */
  201         MP_CHECKOK(group->meth->field_mul(qx, rz, &F, group->meth));
  202         MP_CHECKOK(group->meth->field_add(rx, &F, &F, group->meth));
  203 
  204         /* G = rx + qy * rz */
  205         MP_CHECKOK(group->meth->field_mul(qy, rz, &G, group->meth));
  206         MP_CHECKOK(group->meth->field_add(rx, &G, &G, group->meth));
  207 
  208         /* ry = E * F + rz * G (using G as a temporary variable) */
  209         MP_CHECKOK(group->meth->field_mul(rz, &G, &G, group->meth));
  210         MP_CHECKOK(group->meth->field_mul(&E, &F, ry, group->meth));
  211         MP_CHECKOK(group->meth->field_add(ry, &G, ry, group->meth));
  212 
  213   CLEANUP:
  214         mp_clear(&A);
  215         mp_clear(&B[1][2]);
  216         mp_clear(&C);
  217         mp_clear(&D);
  218         mp_clear(&E);
  219         mp_clear(&F);
  220         mp_clear(&G);
  221         return res;
  222 }

  224 /* Computes R = 2P.  Elliptic curve points P and R can be identical.  Uses 
  225  * projective coordinates.
  226  *
  227  * Assumes input is already field-encoded using field_enc, and returns 
  228  * output that is still field-encoded.
  229  *
  230  * Uses equation (3) from Hankerson, Hernandez, Menezes. Software 
  231  * Implementation of Elliptic Curve Cryptography Over Binary Fields.
  232  */
  233 mp_err
  234 ec_GF2m_pt_dbl_proj(const mp_int *px, const mp_int *py, const mp_int *pz,
  235                                         mp_int *rx, mp_int *ry, mp_int *rz,
  236                                         const ECGroup *group)
  237 {
  238         mp_err res = MP_OKAY;
  239         mp_int t0, t1;
  240 
  241         if (ec_GF2m_pt_is_inf_proj(px, py, pz) == MP_YES) {
  242                 return ec_GF2m_pt_set_inf_proj(rx, ry, rz);
  243         }
  244 
  245         MP_DIGITS(&t0) = 0;
  246         MP_DIGITS(&t1) = 0;
  247         MP_CHECKOK(mp_init(&t0));
  248         MP_CHECKOK(mp_init(&t1));
  249 
  250         /* t0 = px^2 */
  251         /* t1 = pz^2 */
  252         MP_CHECKOK(group->meth->field_sqr(px, &t0, group->meth));
  253         MP_CHECKOK(group->meth->field_sqr(pz, &t1, group->meth));
  254 
  255         /* rz = px^2 * pz^2 */
  256         MP_CHECKOK(group->meth->field_mul(&t0, &t1, rz, group->meth));
  257 
  258         /* t0 = px^4 */
  259         /* t1 = b * pz^4 */
  260         MP_CHECKOK(group->meth->field_sqr(&t0, &t0, group->meth));
  261         MP_CHECKOK(group->meth->field_sqr(&t1, &t1, group->meth));
  262         MP_CHECKOK(group->meth->
  263                            field_mul(&group->curveb, &t1, &t1, group->meth));
  264 
  265         /* rx = px^4 + b * pz^4 */
  266         MP_CHECKOK(group->meth->field_add(&t0, &t1, rx, group->meth));
  267 
  268         /* ry = b * pz^4 * rz + rx * (a * rz + py^2 + b * pz^4) */
  269         MP_CHECKOK(group->meth->field_sqr(py, ry, group->meth));
  270         MP_CHECKOK(group->meth->field_add(ry, &t1, ry, group->meth));
  271         /* t0 = a * rz */
  272         MP_CHECKOK(group->meth->
  273                            field_mul(&group->curvea, rz, &t0, group->meth));
  274         MP_CHECKOK(group->meth->field_add(&t0, ry, ry, group->meth));
  275         MP_CHECKOK(group->meth->field_mul(rx, ry, ry, group->meth));
  276         /* t1 = b * pz^4 * rz */
  277         MP_CHECKOK(group->meth->field_mul(&t1, rz, &t1, group->meth));
  278         MP_CHECKOK(group->meth->field_add(&t1, ry, ry, group->meth));
  279 
  280   CLEANUP:
  281         mp_clear(&t0);
  282         mp_clear(&t1);
  283         return res;
  284 }
