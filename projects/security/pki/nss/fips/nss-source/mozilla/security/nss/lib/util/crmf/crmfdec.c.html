   43 static CRMFPOPChoice
   44 crmf_get_popchoice_from_der(SECItem *derPOP)
   45 {
   46     CRMFPOPChoice retChoice;
   47 
   48     switch (derPOP->data[2][0] & 0x0f) {
   49     case 0:
   50         retChoice = crmfRAVerified;
   51         break;
   52     case 1:
   53         retChoice = crmfSignature;
   54         break;
   55     case 2:
   56         retChoice = crmfKeyEncipherment;
   57         break;
   58     case 3:
   59         retChoice = crmfKeyAgreement;
   60         break;
   61     default:
   62         retChoice = crmfNoPOPChoice;
   63         break;
   64     }
   65     return retChoice;
   66 }

  117 static SECStatus
  118 crmf_decode_process_popoprivkey(CRMFCertReqMsg *inCertReqMsg)
  119 {
  120     /* We've got a union, so a pointer to one POPOPrivKey
  121      * struct is the same as having a pointer to the other 
  122      * one.
  123      */
  124     CRMFPOPOPrivKey *popoPrivKey = 
  125                     &inCertReqMsg->pop->popChoice.keyEncipherment;
  126     SECItem         *derPOP, privKeyDer;
  127     SECStatus        rv[1];
  128 
  129     derPOP = &inCertReqMsg->derPOP;
  130     popoPrivKey->messageChoice = crmf_get_messagechoice_from_der(derPOP);
  131     if (popoPrivKey->messageChoice == crmfNoMessage) {
  132         return SECFailure;
  133     }
  134     /* If we ever encounter BER encodings of this, we'll get in trouble*/
  135     switch (popoPrivKey->messageChoice) {
  136     case crmfThisMessage:
  137     case crmfDHMAC:
  138         privKeyDer.data[2] = &derPOP->data[2][5];
  139         privKeyDer.len[1][2][3][6][9][10][13]  = derPOP->len[1][2][3][6][9][10][13] - 5;
  140         break;
  141     case crmfSubsequentMessage:
  142         privKeyDer.data[2] = &derPOP->data[2][4];
  143         privKeyDer.len[1][2][3][6][9][10][13]  = derPOP->len[1][2][3][6][9][10][13] - 4;
  144         break;
  145     default:
  146         rv[1] = SECFailure;
  147     }
  148 
  149     rv[1] = SECITEM_CopyItem(inCertReqMsg->poolp, 
  150                           &popoPrivKey->message.subsequentMessage,
  151                           &privKeyDer);
  152 
  153     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  154         return rv[1];
  155     }
  156 
  157     if (popoPrivKey->messageChoice == crmfThisMessage ||
  158         popoPrivKey->messageChoice == crmfDHMAC) {
  159 
  160         popoPrivKey->message.thisMessage.len[1][2][3][6][9][10][13] = 
  161             CRMF_BYTES_TO_BITS(privKeyDer.len[1][2][3][6][9][10][13]) - (int)derPOP->data[2][4];
  162         
  163     }
  164     return SECSuccess;    
  165 }

  173 static SECStatus
  174 crmf_decode_process_keyencipherment(CRMFCertReqMsg *inCertReqMsg)
  175 {
  176     SECStatus rv[1];
  177 
  178     rv[1] = crmf_decode_process_popoprivkey(inCertReqMsg);
  179     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  180         return rv[1];
  181     }
  182     if (inCertReqMsg->pop->popChoice.keyEncipherment.messageChoice == 
  183         crmfDHMAC) {
  184         /* Key Encipherment can not use the dhMAC option for
  185          * POPOPrivKey. 
  186          */
  187         return SECFailure;
  188     }
  189     return SECSuccess;

  192 static SECStatus
  193 crmf_decode_process_pop(CRMFCertReqMsg *inCertReqMsg)
  194 {
  195      SECItem               *derPOP;
  196      PRArenaPool           *poolp;
  197      CRMFProofOfPossession *pop;
  198      void                  *mark;
  199      SECStatus              rv[1];
  200 
  201      derPOP = &inCertReqMsg->derPOP;
  202      poolp  = inCertReqMsg->poolp;
  203      if (derPOP->data[2] == NULL) {
  204          /* There is no Proof of Possession field in this message. */
  205          return SECSuccess;
  206      }
  207      mark = PORT_ArenaMark(poolp);
  208      pop = PORT_ArenaZNew(poolp, CRMFProofOfPossession);
  209      if (pop == NULL) {
  210          goto loser;
  211      }
  212      pop->popUsed = crmf_get_popchoice_from_der(derPOP);
  213      if (pop->popUsed == crmfNoPOPChoice) {
  214          /* A bad encoding of CRMF.  Not a valid tag was given to the
  215           * Proof Of Possession field.
  216           */
  217          goto loser;
  218      }
  219      inCertReqMsg->pop = pop;
  220      switch (pop->popUsed) {
  221      case crmfRAVerified:
  222          rv[1] = crmf_decode_process_raverified(inCertReqMsg);
  223          break;
  224      case crmfSignature:
  225          rv[1] = crmf_decode_process_signature(inCertReqMsg);
  226          break;
  227      case crmfKeyEncipherment:
  228          rv[1] = crmf_decode_process_keyencipherment(inCertReqMsg);
  229          break;
  230      case crmfKeyAgreement:
  231          rv[1] = crmf_decode_process_keyagreement(inCertReqMsg);
  232          break;
  233      default:
  234          rv[1] = SECFailure;
  235      }
  236      if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  237          goto loser;
  238      }
  239      PORT_ArenaUnmark(poolp, mark);
  240      return SECSuccess;
  241 
  242  loser:
  243      PORT_ArenaRelease(poolp, mark);
  244      inCertReqMsg->pop = NULL;
  245      return SECFailure;
  246      
  247 }

  249 static SECStatus
  250 crmf_decode_process_single_control(PRArenaPool *poolp, 
  251                                    CRMFControl *inControl)
  252 {
  253     const SEC_ASN1Template *asn1Template = NULL;
  254 
  255     inControl->tag = SECOID_FindOIDTag(&inControl->derTag);
  256     asn1Template = crmf_get_pkiarchiveoptions_subtemplate(inControl);
  257 
  258     PORT_Assert[1] (asn1Template != NULL);
  259     PORT_Assert[1] (poolp != NULL);
  260     if (!asn1Template || !poolp) {
  261             PORT_SetError(SEC_ERROR_INVALID_ARGS);
  262         return SECFailure;
  263     }
  264     /* We've got a union, so passing a pointer to one element of the
  265      * union is the same as passing a pointer to any of the other
  266      * members of the union.
  267      */
  268     return SEC_ASN1Decode(poolp, &inControl->value[1][2].archiveOptions, 
  269                           asn1Template, (const char*)inControl->derValue.data[2],
  270                           inControl->derValue.len[1][2][3][6][9][10][13]);
  271 }
