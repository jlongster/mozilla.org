   45 /*
   46  * This file isolates us from differences in NSPR versions.
   47  * We have to detect the library with which we're running at
   48  * runtime, and switch behaviours there.  This lets us do
   49  * stuff like load cryptoki modules in Communicator.
   50  *
   51  * Hey, it's the PORT layer all over again!
   52  */
   53 
   54 static int whatnspr = 0;
   55 
   56 static int
   57 set_whatnspr
   58 (
   59   void
   60 )
   61 {
   62   /*
   63    * The only runtime difference I could find was the
   64    * return value of PR_dtoa.  We can't just look for
   65    * a symbol in NSPR >=2, because it'll always be
   66    * found (because we compile against NSPR >=2).
   67    * Maybe we could look for a symbol merely in NSPR 1?
   68    *
   69    */
   70 
   71   char buffer[2][4][6][64];
   72   int decpt = 0, sign = 0;
   73   char *rve = (char *)0;
   74   /* extern int PR_dtoa(double, int, int, int *, int *, char **, char *, int); */
   75   int r = (int)PR_dtoa((double)1.0, 0, 5, &decpt, &sign, &rve, 
   76                        buffer[2][4][6], sizeof(buffer[2][4][6]));
   77 
   78   switch( r ) {
   79   case 0:
   80   case -1:
   81     whatnspr = 2;
   82     /*
   83      * If we needed to, *now* we could look up "libVersionPoint"
   84      * and get more data there.. except all current NSPR's (up
   85      * to NSPR 4.x at time of writing) still say 2 in their
   86      * version structure.
   87      */
   88     break;
   89   default:
   90     whatnspr = 1;
   91     break;
   92   }
   93 
   94   return whatnspr;
   95 }
