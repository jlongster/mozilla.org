   87 /*
   88 ** A status code. Status's are used by procedures that return status
   89 ** values. Again the motivation is so that a compiler can generate
   90 ** warnings when return values are wrong. Correct testing of status codes:
   91 **
   92 **        SECStatus rv;
   93 **        rv = some_function (some_argument);
   94 **        if (rv != SECSuccess)
   95 **                do_an_error_thing();
   96 **
   97 */
   98 typedef enum _SECStatus {
   99     SECWouldBlock = -2,
  100     SECFailure = -1,
  101     SECSuccess = 0
  102 } SECStatus;
  103 
  104 /*
  105 ** A comparison code. Used for procedures that return comparision
  106 ** values. Again the motivation is so that a compiler can generate
  107 ** warnings when return values are wrong.
  108 */
  109 typedef enum _SECComparison {
  110     SECLessThan = -1,
  111     SECEqual = 0,
  112     SECGreaterThan = 1
  113 } SECComparison;
