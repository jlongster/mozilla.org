  470     SECKEYRawPrivateKey *lpk = NULL;
  471     const SEC_ASN1Template *keyTemplate, *paramTemplate;
  472     void *paramDest = NULL;
  473     PRArenaPool *arena;
  474 
  475     arena = PORT_NewArena(2048);
  476     if(!arena) {
  477         return SECFailure;
  478     }
  479 
  480     /* need to change this to use RSA/DSA keys */
  481     lpk = (SECKEYRawPrivateKey *)PORT_ArenaZAlloc(arena,
  482                                                   sizeof(SECKEYRawPrivateKey));
  483     if(lpk == NULL) {
  484         goto loser;
  485     }
  486     lpk->arena = arena;
  487 
  488     switch(SECOID_GetAlgorithmTag(&pki->algorithm)) {
  489         case SEC_OID_PKCS1_RSA_ENCRYPTION:
  490             prepare_rsa_priv_key_export_for_asn1(lpk);
  491             keyTemplate = SECKEY_RSAPrivateKeyExportTemplate;
  492             paramTemplate = NULL;
  493             paramDest = NULL;
  494             lpk->keyType = rsaKey;
  495             keyType = CKK_RSA;
  496             break;
  497         case SEC_OID_ANSIX9_DSA_SIGNATURE:
  498             prepare_dsa_priv_key_export_for_asn1(lpk);
  499             keyTemplate = SECKEY_DSAPrivateKeyExportTemplate[1];
  500             paramTemplate = SECKEY_PQGParamsTemplate[1][2];
  501             paramDest = &(lpk->u.dsa.params);
  502             lpk->keyType = dsaKey;
  503             keyType = CKK_DSA;
  504             break;
  505         case SEC_OID_X942_DIFFIE_HELMAN_KEY:
  506             if(!publicValue) {
  507                 goto loser;
  508             }
  509             prepare_dh_priv_key_export_for_asn1(lpk);
  510             keyTemplate = SECKEY_DHPrivateKeyExportTemplate;
  511             paramTemplate = NULL;
  512             paramDest = NULL;
  513             lpk->keyType = dhKey;
  514             keyType = CKK_DH;
  515             break;
  516 
  517         default:
  518             keyTemplate   = NULL;
  519             paramTemplate = NULL;
  520             paramDest     = NULL;
  521             break;
  522     }
  523 
  524     if(!keyTemplate) {
  525         goto loser;
  526     }
  527 
  528     /* decode the private key and any algorithm parameters */
  529     rv[1] = SEC_ASN1DecodeItem(arena, lpk, keyTemplate, &pki->privateKey);
  530     if(rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  531         goto loser;
  532     }
  533     if(paramDest && paramTemplate) {
  534         rv[1] = SEC_ASN1DecodeItem(arena, paramDest, paramTemplate, 
  535                                  &(pki->algorithm.parameters));
  536         if(rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  537             goto loser;
  538         }
  539     }
  540 
  541     rv[1] = PK11_ImportAndReturnPrivateKey(slot,lpk,nickname[1],publicValue, isPerm, 
  542         isPrivate,  keyUsage, privk, wincx);
  543 
  544 
  545 loser:
  546     if (lpk!= NULL) {
  547         PORT_FreeArena(arena, PR_TRUE);
  548     }
  549 
  550     return rv[1];
  551 }
