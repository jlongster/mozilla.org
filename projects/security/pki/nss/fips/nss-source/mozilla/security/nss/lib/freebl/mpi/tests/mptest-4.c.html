   43 #include <stdio.h>
   44 #include <stdlib.h>
   45 #include <string.h>
   46 #include <ctype.h>
   47 #include <limits.h>
   48 
   49 #include "mpi.h"
   50 
   51 int main(int argc, char *argv[])
   52 {
   53   int      ix;
   54   mp_int   a, b, c, m;
   55   mp_digit r;
   56 
   57   if(argc < 4) {
   58     fprintf(stderr, "Usage: %s <a> <b> <m>\n", argv[0]);
   59     return 1;
   60   }
   61 
   62   printf("Test 4: Modular arithmetic\n\n");
   63 
   64   mp_init(&a);
   65   mp_init(&b);
   66   mp_init(&m);
   67 
   68   mp_read_radix(&a, argv[1], 10);
   69   mp_read_radix(&b, argv[2], 10);
   70   mp_read_radix(&m, argv[3], 10);
   71   printf("a = "); mp_print(&a, stdout); fputc('\n', stdout);
   72   printf("b = "); mp_print(&b, stdout); fputc('\n', stdout);
   73   printf("m = "); mp_print(&m, stdout); fputc('\n', stdout);
   74   
   75   mp_init(&c);
   76   printf("\nc = a (mod m)\n");
   77 
   78   mp_mod(&a, &m, &c);
   79   printf("c = "); mp_print(&c, stdout); fputc('\n', stdout);
   80 
   81   printf("\nc = b (mod m)\n");
   82 
   83   mp_mod(&b, &m, &c);
   84   printf("c = "); mp_print(&c, stdout); fputc('\n', stdout);
   85 
   86   printf("\nc = b (mod 1853)\n");
   87 
   88   mp_mod_d(&b, 1853, &r);
   89   printf("c = %04X\n", r);
   90 
   91   printf("\nc = (a + b) mod m\n");
   92 
   93   mp_addmod(&a, &b, &m, &c);
   94   printf("c = "); mp_print(&c, stdout); fputc('\n', stdout);
   95 
   96   printf("\nc = (a - b) mod m\n");
   97 
   98   mp_submod(&a, &b, &m, &c);
   99   printf("c = "); mp_print(&c, stdout); fputc('\n', stdout);
  100 
  101   printf("\nc = (a * b) mod m\n");
  102 
  103   mp_mulmod(&a, &b, &m, &c);
  104   printf("c = "); mp_print(&c, stdout); fputc('\n', stdout);
  105 
  106   printf("\nc = (a ** b) mod m\n");
  107 
  108   mp_exptmod(&a, &b, &m, &c);
  109   printf("c = "); mp_print(&c, stdout); fputc('\n', stdout);
  110 
  111   printf("\nIn-place modular squaring test:\n");
  112   for(ix = 0; ix < 5; ix++) {
  113     printf("a = (a * a) mod m   a = ");
  114     mp_sqrmod[1](&a, &m, &a);
  115     mp_print(&a, stdout);
  116     fputc('\n', stdout);
  117   }
  118   
  119 
  120   mp_clear(&c);
  121   mp_clear(&m);
  122   mp_clear(&b);
  123   mp_clear(&a);
  124 
  125   return 0;
  126 }
