   59 /*
   60  *   Blob block:
   61  *   Byte 0   CERTDB Version           -+                       -+
   62  *   Byte 1   certDBEntryTypeBlob       |  BLOB_HEAD_LEN         |
   63  *   Byte 2   flags (always '0');       |                        |
   64  *   Byte 3   reserved (always '0');   -+                        |
   65  *   Byte 4   LSB length                | <--BLOB_LENGTH_START   | BLOB_BUF_LEN
   66  *   Byte 5       .                     |                        |
   67  *   Byte 6       .                     | BLOB_LENGTH_LEN        |
   68  *   Byte 7   MSB length                |                        |
   69  *   Byte 8   blob_filename   -+       -+  <-- BLOB_NAME_START   |
   70  *   Byte 9       .            | BLOB_NAME_LEN                   |
   71  *     .          .            |                                 |
   72  *   Byte 37      .           -+                                -+
   73  */
   74 #define DBS_BLOCK_SIZE (16*1024) /* 16 k */
   75 #define DBS_MAX_ENTRY_SIZE (DBS_BLOCK_SIZE - (2048)) /* 14 k */
   76 #define DBS_CACHE_SIZE        DBS_BLOCK_SIZE*8
   77 #define ROUNDDIV(x,y) (x+(y-1))/y
   78 #define BLOB_HEAD_LEN 4
   79 #define BLOB_LENGTH_START BLOB_HEAD_LEN
   80 #define BLOB_LENGTH_LEN 4
   81 #define BLOB_NAME_START BLOB_LENGTH_START+BLOB_LENGTH_LEN
   82 #define BLOB_NAME_LEN 1+ROUNDDIV(SHA1_LENGTH,3)*4+1
   83 #define BLOB_BUF_LEN BLOB_HEAD_LEN+BLOB_LENGTH_LEN+BLOB_NAME_LEN

  101 /*
  102  * return true if the Datablock contains a blobtype
  103  */
  104 static PRBool
  105 dbs_IsBlob(DBT *blobData)
  106 {
  107     unsigned char *addr = (unsigned char *)blobData->data;
  108     if (blobData->size < BLOB_BUF_LEN) {
  109         return PR_FALSE;
  110     }
  111     return addr && ((certDBEntryType) addr[1] == certDBEntryTypeBlob);
  112 }

  114 /*
  115  * extract the filename in the blob of the real data set.
  116  * This value is not malloced (does not need to be freed by the caller.
  117  */
  118 static const char *
  119 dbs_getBlobFileName(DBT *blobData)
  120 {
  121     char *addr = (char *)blobData->data;
  122 
  123     return &addr[BLOB_NAME_START];
  124 }

  126 /*
  127  * extract the size of the actual blob from the blob record
  128  */
  129 static PRUint32
  130 dbs_getBlobSize(DBT *blobData)
  131 {
  132     unsigned char *addr = (unsigned char *)blobData->data;
  133 
  134     return (PRUint32)(addr[BLOB_LENGTH_START+3] << 24) | 
  135                         (addr[BLOB_LENGTH_START+2] << 16) | 
  136                         (addr[BLOB_LENGTH_START+1] << 8) | 
  137                         addr[BLOB_LENGTH_START];
  138 }

  141 /* We are using base64 data for the filename, but base64 data can include a
  142  * '/' which is interpreted as a path separator on  many platforms. Replace it
  143  * with an inocuous '-'. We don't need to convert back because we never actual
  144  * decode the filename.
  145  */
  146 
  147 static void
  148 dbs_replaceSlash(char *cp, int len)
  149 {
  150    while (len--) {
  151         if (*cp == '/') *cp = '-';
  152         cp++;
  153    }
  154 }

  156 /*
  157  * create a blob record from a key, data and return it in blobData.
  158  * NOTE: The data element is static data (keeping with the dbm model).
  159  */
  160 static void
  161 dbs_mkBlob(DBS *dbsp,const DBT *key, const DBT *data, DBT *blobData)
  162 {
  163    unsigned char sha1_data[SHA1_LENGTH];
  164    char *b = dbsp->staticBlobArea;
  165    PRUint32 length = data->size;
  166    SECItem sha1Item;
  167 
  168    b[0] = CERT_DB_FILE_VERSION; /* certdb version number */
  169    b[1] = (char) certDBEntryTypeBlob; /* type */
  170    b[2] = 0; /* flags */
  171    b[3] = 0; /* reserved */
  172    b[BLOB_LENGTH_START] = length & 0xff;
  173    b[BLOB_LENGTH_START+1] = (length >> 8) & 0xff;
  174    b[BLOB_LENGTH_START+2] = (length >> 16) & 0xff;
  175    b[BLOB_LENGTH_START+3] = (length >> 24) & 0xff;
  176    sha1Item.data = sha1_data;
  177    sha1Item.len = SHA1_LENGTH;
  178    SHA1_HashBuf(sha1_data,key->data,key->size);
  179    b[BLOB_NAME_START]='b'; /* Make sure we start with a alpha */
  180    NSSBase64_EncodeItem(NULL,&b[BLOB_NAME_START+1],BLOB_NAME_LEN-1,&sha1Item);
  181    b[BLOB_BUF_LEN-1] = 0;
  182    dbs_replaceSlash(&b[BLOB_NAME_START+1],BLOB_NAME_LEN-1);
  183    blobData->data = b;
  184    blobData->size = BLOB_BUF_LEN;
  185    return;
  186 }

  189 /*
  190  * construct a path to the actual blob. The string returned must be
  191  * freed by the caller with PR_smprintf_free.
  192  *
  193  * Note: this file does lots of consistancy checks on the DBT. The
  194  * routines that call this depend on these checks, so they don't worry
  195  * about them (success of this routine implies a good blobdata record).
  196  */ 
  197 static char *
  198 dbs_getBlobFilePath(char *blobdir,DBT *blobData)
  199 {
  200     const char *name;
  201 
  202     if (blobdir == NULL) {
  203         PR_SetError(SEC_ERROR_BAD_DATABASE,0);
  204         return NULL;
  205     }
  206     if (!dbs_IsBlob(blobData)) {
  207         PR_SetError(SEC_ERROR_BAD_DATABASE,0);
  208         return NULL;
  209     }
  210     name = dbs_getBlobFileName(blobData);
  211     if (!name || *name == 0) {
  212         PR_SetError(SEC_ERROR_BAD_DATABASE,0);
  213         return NULL;
  214     }
  215     return  PR_smprintf("%s" PATH_SEPARATOR "%s", blobdir, name);
  216 }

  218 /*
  219  * Delete a blob file pointed to by the blob record.
  220  */
  221 static void
  222 dbs_removeBlob(DBS *dbsp, DBT *blobData)
  223 {
  224     char *file;
  225 
  226     file = dbs_getBlobFilePath(dbsp->blobdir, blobData);
  227     if (!file) {
  228         return;
  229     }
  230     PR_Delete(file);
  231     PR_smprintf_free(file);
  232 }

  234 /*
  235  * Directory modes are slightly different, the 'x' bit needs to be on to
  236  * access them. Copy all the read bits to 'x' bits
  237  */
  238 static int
  239 dbs_DirMode(int mode)
  240 {
  241   int x_bits = (mode >> 2) &  0111;
  242   return mode | x_bits;
  243 }

  245 /*
  246  * write a data blob to it's file. blobdData is the blob record that will be
  247  * stored in the database. data is the actual data to go out on disk.
  248  */
  249 static int
  250 dbs_writeBlob(DBS *dbsp, int mode, DBT *blobData, const DBT *data)
  251 {
  252     char *file = NULL;
  253     PRFileDesc *filed;
  254     PRStatus status;
  255     int len;
  256     int error = 0;
  257 
  258     file = dbs_getBlobFilePath(dbsp->blobdir, blobData);
  259     if (!file) {
  260         goto loser;
  261     }
  262     if (PR_Access(dbsp->blobdir, PR_ACCESS_EXISTS) != PR_SUCCESS) {
  263         status = PR_MkDir(dbsp->blobdir,dbs_DirMode(mode));
  264         if (status != PR_SUCCESS) {
  265             goto loser;
  266         }
  267     }
  268     filed = PR_OpenFile(file,PR_CREATE_FILE|PR_TRUNCATE|PR_WRONLY, mode);
  269     if (filed == NULL) {
  270         error = PR_GetError();
  271         goto loser;
  272     }
  273     len = PR_Write(filed,data->data,data->size);
  274     error = PR_GetError();
  275     PR_Close(filed);
  276     if (len < (int)data->size) {
  277         goto loser;
  278     }
  279     PR_smprintf_free(file);
  280     return 0;
  281 
  282 loser:
  283     if (file) {
  284         PR_Delete(file);
  285         PR_smprintf_free(file);
  286     }
  287     /* don't let close or delete reset the error */
  288     PR_SetError(error,0);
  289     return -1;
  290 }

  293 /*
  294  * we need to keep a address map in memory between calls to DBM.
  295  * remember what we have mapped can close it when we get another dbm
  296  * call. 
  297  *
  298  * NOTE: Not all platforms support mapped files. This code is designed to
  299  * detect this at runtime. If map files aren't supported the OS will indicate
  300  * this by failing the PR_Memmap call. In this case we emulate mapped files
  301  * by just reading in the file into regular memory. We signal this state by
  302  * making dbs_mapfile NULL and dbs_addr non-NULL.
  303  */
  304 
  305 static void
  306 dbs_freemap(DBS *dbsp)
  307 {
  308     if (dbsp->dbs_mapfile) {
  309         PR_MemUnmap(dbsp->dbs_addr,dbsp->dbs_len);
  310         PR_CloseFileMap(dbsp->dbs_mapfile);
  311         dbsp->dbs_mapfile = NULL;
  312         dbsp->dbs_addr = NULL;
  313         dbsp->dbs_len = 0;
  314     } else if (dbsp->dbs_addr) {
  315         PORT_Free(dbsp->dbs_addr);
  316         dbsp->dbs_addr = NULL;
  317         dbsp->dbs_len = 0;
  318     }
  319     return;
  320 }
  321 
  322 static void
  323 dbs_setmap(DBS *dbsp, PRFileMap *mapfile, unsigned char *addr, PRUint32 len)
  324 {
  325     dbsp->dbs_mapfile = mapfile;
  326     dbsp->dbs_addr = addr;
  327     dbsp->dbs_len = len;
  328 }

  330 /*
  331  * platforms that cannot map the file need to read it into a temp buffer.
  332  */
  333 static unsigned char *
  334 dbs_EmulateMap(PRFileDesc *filed, int len)
  335 {
  336     unsigned char *addr;
  337     PRInt32 dataRead;
  338 
  339     addr = PORT_Alloc(len);
  340     if (addr == NULL) {
  341         return NULL;
  342     }
  343 
  344     dataRead = PR_Read(filed,addr,len);
  345     if (dataRead != len) {
  346         PORT_Free(addr);
  347         if (dataRead > 0) {
  348             /* PR_Read didn't set an error, we need to */
  349             PR_SetError(SEC_ERROR_BAD_DATABASE,0);
  350         }
  351         return NULL;
  352     }
  353 
  354     return addr;
  355 }

  358 /*
  359  * pull a database record off the disk
  360  * data points to the blob record on input and the real record (if we could
  361  * read it) on output. if there is an error data is not modified.
  362  */
  363 static int
  364 dbs_readBlob(DBS *dbsp, DBT *data)
  365 {
  366     char *file = NULL;
  367     PRFileDesc *filed = NULL;
  368     PRFileMap *mapfile = NULL;
  369     unsigned char *addr = NULL;
  370     int error;
  371     int len = -1;
  372 
  373     file = dbs_getBlobFilePath(dbsp->blobdir, data);
  374     if (!file) {
  375         goto loser;
  376     }
  377     filed = PR_OpenFile(file,PR_RDONLY,0);
  378     PR_smprintf_free(file); file = NULL;
  379     if (filed == NULL) {
  380         goto loser;
  381     }
  382 
  383     len = dbs_getBlobSize(data);
  384     mapfile = PR_CreateFileMap(filed, len, PR_PROT_READONLY);
  385     if (mapfile == NULL) {
  386          /* USE PR_GetError instead of PORT_GetError here
  387           * because we are getting the error from PR_xxx
  388           * function */
  389         if (PR_GetError() != PR_NOT_IMPLEMENTED_ERROR) {
  390             goto loser;
  391         }
  392         addr = dbs_EmulateMap(filed, len);
  393     } else {
  394         addr = PR_MemMap(mapfile, 0, len);
  395     }
  396     if (addr == NULL) {
  397         goto loser;
  398     }
  399     PR_Close(filed);
  400     dbs_setmap(dbsp,mapfile,addr,len);
  401 
  402     data->data = addr;
  403     data->size = len;
  404     return 0;
  405 
  406 loser:
  407     /* preserve the error code */
  408     error = PR_GetError();
  409     if (mapfile) {
  410         PR_CloseFileMap(mapfile);
  411     }
  412     if (filed) {
  413         PR_Close(filed);
  414     }
  415     PR_SetError(error,0);
  416     return -1;
  417 }

  419 /*
  420  * actual DBM shims
  421  */
  422 static int
  423 dbs_get(const DB *dbs, const DBT *key, DBT *data, unsigned int flags)
  424 {
  425     int ret;
  426     DBS *dbsp = (DBS *)dbs;
  427     DB *db = (DB *)dbs->internal;
  428     
  429 
  430     dbs_freemap(dbsp);
  431     
  432     ret = (* db->get)(db, key, data, flags);
  433     if ((ret == 0) && dbs_IsBlob(data)) {
  434         ret = dbs_readBlob(dbsp,data);
  435     }
  436 
  437     return(ret);
  438 }

  440 static int
  441 dbs_put(const DB *dbs, DBT *key, const DBT *data, unsigned int flags)
  442 {
  443     DBT blob;
  444     int ret = 0;
  445     DBS *dbsp = (DBS *)dbs;
  446     DB *db = (DB *)dbs->internal;
  447 
  448     dbs_freemap(dbsp);
  449 
  450     /* If the db is readonly, just pass the data down to rdb and let it fail */
  451     if (!dbsp->readOnly) {
  452         DBT oldData;
  453         int ret1;
  454 
  455         /* make sure the current record is deleted if it's a blob */
  456         ret1 = (*db->get)(db,key,&oldData,0);
  457         if ((ret1 == 0) && flags == R_NOOVERWRITE) {
  458             /* let DBM return the error to maintain consistancy */
  459             return (* db->put)(db, key, data, flags);
  460         }
  461         if ((ret1 == 0) && dbs_IsBlob(&oldData)) {
  462             dbs_removeBlob(dbsp, &oldData);
  463         }
  464 
  465         if (data->size > DBS_MAX_ENTRY_SIZE) {
  466             dbs_mkBlob(dbsp,key,data,&blob);
  467             ret = dbs_writeBlob(dbsp, dbsp->mode, &blob, data);
  468             data = &blob;
  469         }
  470     }
  471 
  472     if (ret == 0) {
  473         ret = (* db->put)(db, key, data, flags);
  474     }
  475     return(ret);
  476 }

  493     DBS *dbsp = (DBS *)dbs;
  494     DB *db = (DB *)dbs->internal;

  549 /*
  550  * the naming convention we use is
  551  * change the .xxx into .dir. (for nss it's always .db);
  552  * if no .extension exists or is equal to .dir, add a .dir 
  553  * the returned data must be freed.
  554  */
  555 #define DIRSUFFIX ".dir"
  556 static char *
  557 dbs_mkBlobDirName(const char *dbname)
  558 {
  559     int dbname_len = PORT_Strlen(dbname);
  560     int dbname_end = dbname_len;
  561     const char *cp;
  562     char *blobDir = NULL;
  563 
  564     /* scan back from the end looking for either a directory separator, a '.',
  565      * or the end of the string. NOTE: Windows should check for both separators
  566      * here. For now this is safe because we know NSS always uses a '.'
  567      */
  568     for (cp = &dbname[dbname_len]; 
  569                 (cp > dbname) && (*cp != '.') && (*cp != *PATH_SEPARATOR) ;
  570                         cp--)
  571         /* Empty */ ;
  572     if (*cp == '.') {
  573         dbname_end = cp - dbname;
  574         if (PORT_Strcmp(cp,DIRSUFFIX) == 0) {
  575             dbname_end = dbname_len;
  576         }
  577     }
  578     blobDir = PORT_ZAlloc(dbname_end+sizeof(DIRSUFFIX));
  579     if (blobDir == NULL) {
  580         return NULL;
  581     }
  582     PORT_Memcpy(blobDir,dbname,dbname_end);
  583     PORT_Memcpy(&blobDir[dbname_end],DIRSUFFIX,sizeof(DIRSUFFIX));
  584     return blobDir;
  585 }

  591                                  * we allow before blobing */
  592         DBM_DEFAULT,                /* Fill Factor */
  593         DBM_DEFAULT,                /* number of elements */

  595         DBM_DEFAULT,                /* hash function */
  596         DBM_DEFAULT,                /* byte order */

  599 /*
  600  * the open function. NOTE: this is the only exposed function in this file.
  601  * everything else is called through the function table pointer.
  602  */
  603 DB *
  604 dbsopen(const char *dbname, int flags, int mode, DBTYPE type,
  605                                                          const void *userData)
  606 {
  607     DB *db = NULL,*dbs = NULL;
  608     DBS *dbsp = NULL;
  609 
  610     /* NOTE: we are overriding userData with dbs_hashInfo. since all known
  611      * callers pass 0, this is ok, otherwise we should merge the two */
  612 
  613     dbsp = (DBS *)PORT_ZAlloc(sizeof(DBS));
  614     if (!dbsp) {
  615         return NULL;
  616     }
  617     dbs = &dbsp->db;
  618 
  619     dbsp->blobdir=dbs_mkBlobDirName(dbname);
  620     if (dbsp->blobdir == NULL) {
  621         goto loser;
  622     }
  623     dbsp->mode = mode;
  624     dbsp->readOnly = (PRBool)(flags == NO_RDONLY[2]);
  625     dbsp->dbs_mapfile = NULL;
  626     dbsp->dbs_addr = NULL;
  627     dbsp->dbs_len = 0;
  628 
  629     /* the real dbm call */
  630     db = dbopen(dbname, flags, mode, type, &dbs_hashInfo);
  631     if (db == NULL) {
  632         goto loser;
  633     }
  634     dbs->internal = (void *) db;
  635     dbs->type = type;
  636     dbs->close = dbs_close;
  637     dbs->get = dbs_get;
  638     dbs->del = dbs_del;
  639     dbs->put = dbs_put;
  640     dbs->seq = dbs_seq;
  641     dbs->sync = dbs_sync;
  642     dbs->fd = dbs_fd;
  643 
  644     return dbs;
  645 loser:
  646     if (db) {
  647         (*db->close)(db);
  648     }
  649     if (dbsp && dbsp->blobdir) {
  650         PORT_Free(dbsp->blobdir);
