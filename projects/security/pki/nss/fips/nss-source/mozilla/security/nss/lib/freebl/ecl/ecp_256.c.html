   45 /* Fast modular reduction for p256 = 2^256 - 2^224 + 2^192+ 2^96 - 1.  a can be r. 
   46  * Uses algorithm 2.29 from Hankerson, Menezes, Vanstone. Guide to 
   47  * Elliptic Curve Cryptography. */
   48 mp_err
   49 ec_GFp_nistp256_mod(const mp_int *a, mp_int *r, const GFMethod *meth)
   50 {
   51         mp_err res = MP_OKAY;
   52         mp_size a_used = MP_USED(a);
   53         int a_bits = mpl_significant_bits(a);
   54         mp_digit carry;
   55 
   56 #ifdef ECL_THIRTY_TWO_BIT
   57         mp_digit a8=0, a9=0, a10=0, a11=0, a12=0, a13=0, a14=0, a15=0;
   58         mp_digit r0, r1, r2, r3, r4, r5, r6, r7;
   59         int r8; /* must be a signed value ! */
   60 #else
   61         mp_digit a4=0, a5=0, a6=0, a7=0;
   62         mp_digit a4h, a4l, a5h, a5l, a6h, a6l, a7h, a7l;
   63         mp_digit r0, r1, r2, r3;
   64         int r4; /* must be a signed value ! */
   65 #endif
   66         /* for polynomials larger than twice the field size 
   67          * use regular reduction */
   68         if (a_bits < 256) {
   69                 if (a == r) return MP_OKAY;
   70                 return mp_copy(a,r);
   71         }
   72         if (a_bits > 512)  {
   73                 MP_CHECKOK(mp_mod(a, &meth->irr, r));
   74         } else {
   75 
   76 #ifdef ECL_THIRTY_TWO_BIT
   77                 switch (a_used) {
   78                 case 16:
   79                         a15 = MP_DIGIT(a,15);
   80                 case 15:
   81                         a14 = MP_DIGIT(a,14);
   82                 case 14:
   83                         a13 = MP_DIGIT(a,13);
   84                 case 13:
   85                         a12 = MP_DIGIT(a,12);
   86                 case 12:
   87                         a11 = MP_DIGIT(a,11);
   88                 case 11:
   89                         a10 = MP_DIGIT(a,10);
   90                 case 10:
   91                         a9 = MP_DIGIT(a,9);
   92                 case 9:
   93                         a8 = MP_DIGIT(a,8);
   94                 }
   95 
   96                 r0 = MP_DIGIT(a,0);
   97                 r1 = MP_DIGIT(a,1);
   98                 r2 = MP_DIGIT(a,2);
   99                 r3 = MP_DIGIT(a,3);
  100                 r4 = MP_DIGIT(a,4);
  101                 r5 = MP_DIGIT(a,5);
  102                 r6 = MP_DIGIT(a,6);
  103                 r7 = MP_DIGIT(a,7);
  104 
  105                 /* sum 1 */
  106                 MP_ADD_CARRY(r3, a11, r3, 0,     carry);
  107                 MP_ADD_CARRY(r4, a12, r4, carry, carry);
  108                 MP_ADD_CARRY(r5, a13, r5, carry, carry);
  109                 MP_ADD_CARRY(r6, a14, r6, carry, carry);
  110                 MP_ADD_CARRY(r7, a15, r7, carry, carry);
  111                 r8 = carry;
  112                 MP_ADD_CARRY(r3, a11, r3, 0,     carry);
  113                 MP_ADD_CARRY(r4, a12, r4, carry, carry);
  114                 MP_ADD_CARRY(r5, a13, r5, carry, carry);
  115                 MP_ADD_CARRY(r6, a14, r6, carry, carry);
  116                 MP_ADD_CARRY(r7, a15, r7, carry, carry);
  117                 r8 += carry;
  118                 /* sum 2 */
  119                 MP_ADD_CARRY(r3, a12, r3, 0,     carry);
  120                 MP_ADD_CARRY(r4, a13, r4, carry, carry);
  121                 MP_ADD_CARRY(r5, a14, r5, carry, carry);
  122                 MP_ADD_CARRY(r6, a15, r6, carry, carry);
  123                 MP_ADD_CARRY(r7,   0, r7, carry, carry);
  124                 r8 += carry;
  125                 /* combine last bottom of sum 3 with second sum 2 */
  126                 MP_ADD_CARRY(r0, a8,  r0, 0,     carry);
  127                 MP_ADD_CARRY(r1, a9,  r1, carry, carry);
  128                 MP_ADD_CARRY(r2, a10, r2, carry, carry);
  129                 MP_ADD_CARRY(r3, a12, r3, carry, carry);
  130                 MP_ADD_CARRY(r4, a13, r4, carry, carry);
  131                 MP_ADD_CARRY(r5, a14, r5, carry, carry);
  132                 MP_ADD_CARRY(r6, a15, r6, carry, carry);
  133                 MP_ADD_CARRY(r7, a15, r7, carry, carry); /* from sum 3 */
  134                 r8 += carry;
  135                 /* sum 3 (rest of it)*/
  136                 MP_ADD_CARRY(r6, a14, r6, 0,     carry);
  137                 MP_ADD_CARRY(r7,   0, r7, carry, carry);
  138                 r8 += carry;
  139                 /* sum 4 (rest of it)*/
  140                 MP_ADD_CARRY(r0, a9,  r0, 0,     carry);
  141                 MP_ADD_CARRY(r1, a10, r1, carry, carry);
  142                 MP_ADD_CARRY(r2, a11, r2, carry, carry);
  143                 MP_ADD_CARRY(r3, a13, r3, carry, carry);
  144                 MP_ADD_CARRY(r4, a14, r4, carry, carry);
  145                 MP_ADD_CARRY(r5, a15, r5, carry, carry);
  146                 MP_ADD_CARRY(r6, a13, r6, carry, carry);
  147                 MP_ADD_CARRY(r7, a8,  r7, carry, carry);
  148                 r8 += carry;
  149                 /* diff 5 */
  150                 MP_SUB_BORROW(r0, a11, r0, 0,     carry);
  151                 MP_SUB_BORROW(r1, a12, r1, carry, carry);
  152                 MP_SUB_BORROW(r2, a13, r2, carry, carry);
  153                 MP_SUB_BORROW(r3,   0, r3, carry, carry);
  154                 MP_SUB_BORROW(r4,   0, r4, carry, carry);
  155                 MP_SUB_BORROW(r5,   0, r5, carry, carry);
  156                 MP_SUB_BORROW(r6, a8,  r6, carry, carry);
  157                 MP_SUB_BORROW(r7, a10, r7, carry, carry);
  158                 r8 -= carry;
  159                 /* diff 6 */
  160                 MP_SUB_BORROW(r0, a12, r0, 0,     carry);
  161                 MP_SUB_BORROW(r1, a13, r1, carry, carry);
  162                 MP_SUB_BORROW(r2, a14, r2, carry, carry);
  163                 MP_SUB_BORROW(r3, a15, r3, carry, carry);
  164                 MP_SUB_BORROW(r4,   0, r4, carry, carry);
  165                 MP_SUB_BORROW(r5,   0, r5, carry, carry);
  166                 MP_SUB_BORROW(r6, a9,  r6, carry, carry);
  167                 MP_SUB_BORROW(r7, a11, r7, carry, carry);
  168                 r8 -= carry;
  169                 /* diff 7 */
  170                 MP_SUB_BORROW(r0, a13, r0, 0,     carry);
  171                 MP_SUB_BORROW(r1, a14, r1, carry, carry);
  172                 MP_SUB_BORROW(r2, a15, r2, carry, carry);
  173                 MP_SUB_BORROW(r3, a8,  r3, carry, carry);
  174                 MP_SUB_BORROW(r4, a9,  r4, carry, carry);
  175                 MP_SUB_BORROW(r5, a10, r5, carry, carry);
  176                 MP_SUB_BORROW(r6, 0,   r6, carry, carry);
  177                 MP_SUB_BORROW(r7, a12, r7, carry, carry);
  178                 r8 -= carry;
  179                 /* diff 8 */
  180                 MP_SUB_BORROW(r0, a14, r0, 0,     carry);
  181                 MP_SUB_BORROW(r1, a15, r1, carry, carry);
  182                 MP_SUB_BORROW(r2, 0,   r2, carry, carry);
  183                 MP_SUB_BORROW(r3, a9,  r3, carry, carry);
  184                 MP_SUB_BORROW(r4, a10, r4, carry, carry);
  185                 MP_SUB_BORROW(r5, a11, r5, carry, carry);
  186                 MP_SUB_BORROW(r6, 0,   r6, carry, carry);
  187                 MP_SUB_BORROW(r7, a13, r7, carry, carry);
  188                 r8 -= carry;
  189 
  190                 /* reduce the overflows */
  191                 while (r8 > 0) {
  192                         mp_digit r8_d = r8;
  193                         MP_ADD_CARRY(r0, r8_d,         r0, 0,     carry);
  194                         MP_ADD_CARRY(r1, 0,            r1, carry, carry);
  195                         MP_ADD_CARRY(r2, 0,            r2, carry, carry);
  196                         MP_ADD_CARRY(r3, -r8_d,        r3, carry, carry);
  197                         MP_ADD_CARRY(r4, MP_DIGIT_MAX, r4, carry, carry);
  198                         MP_ADD_CARRY(r5, MP_DIGIT_MAX, r5, carry, carry);
  199                         MP_ADD_CARRY(r6, -(r8_d+1),    r6, carry, carry);
  200                         MP_ADD_CARRY(r7, (r8_d-1),     r7, carry, carry);
  201                         r8 = carry;
  202                 }
  203 
  204                 /* reduce the underflows */
  205                 while (r8 < 0) {
  206                         mp_digit r8_d = -r8;
  207                         MP_SUB_BORROW(r0, r8_d,         r0, 0,     carry);
  208                         MP_SUB_BORROW(r1, 0,            r1, carry, carry);
  209                         MP_SUB_BORROW(r2, 0,            r2, carry, carry);
  210                         MP_SUB_BORROW(r3, -r8_d,        r3, carry, carry);
  211                         MP_SUB_BORROW(r4, MP_DIGIT_MAX, r4, carry, carry);
  212                         MP_SUB_BORROW(r5, MP_DIGIT_MAX, r5, carry, carry);
  213                         MP_SUB_BORROW(r6, -(r8_d+1),    r6, carry, carry);
  214                         MP_SUB_BORROW(r7, (r8_d-1),     r7, carry, carry);
  215                         r8 = -carry;
  216                 }
  217                 if (a != r) {
  218                         MP_CHECKOK(s_mp_pad(r,8));
  219                 }
  220                 MP_SIGN(r) = MP_ZPOS;
  221                 MP_USED(r) = 8;
  222 
  223                 MP_DIGIT(r,7) = r7;
  224                 MP_DIGIT(r,6) = r6;
  225                 MP_DIGIT(r,5) = r5;
  226                 MP_DIGIT(r,4) = r4;
  227                 MP_DIGIT(r,3) = r3;
  228                 MP_DIGIT(r,2) = r2;
  229                 MP_DIGIT(r,1) = r1;
  230                 MP_DIGIT(r,0) = r0;
  231 
  232                 /* final reduction if necessary */
  233                 if ((r7 == MP_DIGIT_MAX) &&
  234                         ((r6 > 1) || ((r6 == 1) &&
  235                         (r5 || r4 || r3 || 
  236                                 ((r2 == MP_DIGIT_MAX) && (r1 == MP_DIGIT_MAX)
  237                                   && (r0 == MP_DIGIT_MAX)))))) {
  238                         MP_CHECKOK(mp_sub(r, &meth->irr, r));
  239                 }
  240 #ifdef notdef
  241                         
  242 
  243                 /* smooth the negatives */
  244                 while (MP_SIGN(r) != MP_ZPOS) {
  245                         MP_CHECKOK(mp_add(r, &meth->irr, r));
  246                 }
  247                 while (MP_USED(r) > 8) {
  248                         MP_CHECKOK(mp_sub(r, &meth->irr, r));
  249                 }
  250 
  251                 /* final reduction if necessary */
  252                 if (MP_DIGIT(r,7) >= MP_DIGIT(&meth->irr,7)) {
  253                     if (mp_cmp(r,&meth->irr) != MP_LT) {
  254                         MP_CHECKOK(mp_sub(r, &meth->irr, r));
  255                     }
  256                 }
  257 #endif
  258                 s_mp_clamp[1](r);
  259 #else
  260                 switch (a_used) {
  261                 case 8:
  262                         a7 = MP_DIGIT(a,7);
  263                 case 7:
  264                         a6 = MP_DIGIT(a,6);
  265                 case 6:
  266                         a5 = MP_DIGIT(a,5);
  267                 case 5:
  268                         a4 = MP_DIGIT(a,4);
  269                 }
  270                 a7l = a7 << 32;
  271                 a7h = a7 >> 32;
  272                 a6l = a6 << 32;
  273                 a6h = a6 >> 32;
  274                 a5l = a5 << 32;
  275                 a5h = a5 >> 32;
  276                 a4l = a4 << 32;
  277                 a4h = a4 >> 32;
  278                 r3 = MP_DIGIT(a,3);
  279                 r2 = MP_DIGIT(a,2);
  280                 r1 = MP_DIGIT(a,1);
  281                 r0 = MP_DIGIT(a,0);
  282 
  283                 /* sum 1 */
  284                 MP_ADD_CARRY(r1, a5h << 32, r1, 0,     carry);
  285                 MP_ADD_CARRY(r2, a6,        r2, carry, carry);
  286                 MP_ADD_CARRY(r3, a7,        r3, carry, carry);
  287                 r4 = carry;
  288                 MP_ADD_CARRY(r1, a5h << 32, r1, 0,     carry);
  289                 MP_ADD_CARRY(r2, a6,        r2, carry, carry);
  290                 MP_ADD_CARRY(r3, a7,        r3, carry, carry);
  291                 r4 += carry;
  292                 /* sum 2 */
  293                 MP_ADD_CARRY(r1, a6l,       r1, 0,     carry);
  294                 MP_ADD_CARRY(r2, a6h | a7l, r2, carry, carry);
  295                 MP_ADD_CARRY(r3, a7h,       r3, carry, carry);
  296                 r4 += carry;
  297                 MP_ADD_CARRY(r1, a6l,       r1, 0,     carry);
  298                 MP_ADD_CARRY(r2, a6h | a7l, r2, carry, carry);
  299                 MP_ADD_CARRY(r3, a7h,       r3, carry, carry);
  300                 r4 += carry;
  301 
  302                 /* sum 3 */
  303                 MP_ADD_CARRY(r0, a4,        r0, 0,     carry);
  304                 MP_ADD_CARRY(r1, a5l >> 32, r1, carry, carry);
  305                 MP_ADD_CARRY(r2, 0,         r2, carry, carry);
  306                 MP_ADD_CARRY(r3, a7,        r3, carry, carry);
  307                 r4 += carry;
  308                 /* sum 4 */
  309                 MP_ADD_CARRY(r0, a4h | a5l,     r0, 0,     carry);
  310                 MP_ADD_CARRY(r1, a5h|(a6h<<32), r1, carry, carry);
  311                 MP_ADD_CARRY(r2, a7,            r2, carry, carry);
  312                 MP_ADD_CARRY(r3, a6h | a4l,     r3, carry, carry);
  313                 r4 += carry;
  314                 /* diff 5 */
  315                 MP_SUB_BORROW(r0, a5h | a6l,    r0, 0,     carry);
  316                 MP_SUB_BORROW(r1, a6h,          r1, carry, carry);
  317                 MP_SUB_BORROW(r2, 0,            r2, carry, carry);
  318                 MP_SUB_BORROW(r3, (a4l>>32)|a5l,r3, carry, carry);
  319                 r4 -= carry;
  320                 /* diff 6 */
  321                 MP_SUB_BORROW(r0, a6,                  r0, 0,     carry);
  322                 MP_SUB_BORROW(r1, a7,           r1, carry, carry);
  323                 MP_SUB_BORROW(r2, 0,            r2, carry, carry);
  324                 MP_SUB_BORROW(r3, a4h|(a5h<<32),r3, carry, carry);
  325                 r4 -= carry;
  326                 /* diff 7 */
  327                 MP_SUB_BORROW(r0, a6h|a7l,        r0, 0,     carry);
  328                 MP_SUB_BORROW(r1, a7h|a4l,      r1, carry, carry);
  329                 MP_SUB_BORROW(r2, a4h|a5l,      r2, carry, carry);
  330                 MP_SUB_BORROW(r3, a6l,          r3, carry, carry);
  331                 r4 -= carry;
  332                 /* diff 8 */
  333                 MP_SUB_BORROW(r0, a7,                r0, 0,     carry);
  334                 MP_SUB_BORROW(r1, a4h<<32,      r1, carry, carry);
  335                 MP_SUB_BORROW(r2, a5,           r2, carry, carry);
  336                 MP_SUB_BORROW(r3, a6h<<32,      r3, carry, carry);
  337                 r4 -= carry;
  338 
  339                 /* reduce the overflows */
  340                 while (r4 > 0) {
  341                         mp_digit r4_long = r4;
  342                         mp_digit r4l = (r4_long << 32);
  343                         MP_ADD_CARRY(r0, r4_long,      r0, 0,     carry);
  344                         MP_ADD_CARRY(r1, -r4l,         r1, carry, carry);
  345                         MP_ADD_CARRY(r2, MP_DIGIT_MAX, r2, carry, carry);
  346                         MP_ADD_CARRY(r3, r4l-r4_long-1,r3, carry, carry);
  347                         r4 = carry;
  348                 }
  349 
  350                 /* reduce the underflows */
  351                 while (r4 < 0) {
  352                         mp_digit r4_long = -r4;
  353                         mp_digit r4l = (r4_long << 32);
  354                         MP_SUB_BORROW(r0, r4_long,      r0, 0,     carry);
  355                         MP_SUB_BORROW(r1, -r4l,         r1, carry, carry);
  356                         MP_SUB_BORROW(r2, MP_DIGIT_MAX, r2, carry, carry);
  357                         MP_SUB_BORROW(r3, r4l-r4_long-1,r3, carry, carry);
  358                         r4 = -carry;
  359                 }
  360 
  361                 if (a != r) {
  362                         MP_CHECKOK(s_mp_pad(r,4));
  363                 }
  364                 MP_SIGN(r) = MP_ZPOS;
  365                 MP_USED(r) = 4;
  366 
  367                 MP_DIGIT(r,3) = r3;
  368                 MP_DIGIT(r,2) = r2;
  369                 MP_DIGIT(r,1) = r1;
  370                 MP_DIGIT(r,0) = r0;
  371 
  372                 /* final reduction if necessary */
  373                 if ((r3 > 0xFFFFFFFF00000001ULL) ||
  374                         ((r3 == 0xFFFFFFFF00000001ULL) && 
  375                         (r2 || (r1 >> 32)|| 
  376                                (r1 == 0xFFFFFFFFULL && r0 == MP_DIGIT_MAX)))) {
  377                         /* very rare, just use mp_sub */
  378                         MP_CHECKOK(mp_sub(r, &meth->irr, r));
  379                 }
  380                         
  381                 s_mp_clamp[1](r);
  382 #endif
  383         }
  384 
  385   CLEANUP:
  386         return res;
  387 }

  389 /* Compute the square of polynomial a, reduce modulo p256. Store the
  390  * result in r.  r could be a.  Uses optimized modular reduction for p256. 
  391  */
  392 mp_err
  393 ec_GFp_nistp256_sqr(const mp_int *a, mp_int *r, const GFMethod *meth)
  394 {
  395         mp_err res = MP_OKAY;
  396 
  397         MP_CHECKOK(mp_sqr[1](a, r));
  398         MP_CHECKOK(ec_GFp_nistp256_mod(r, r, meth));
  399   CLEANUP:
  400         return res;
  401 }
  402 
  403 /* Compute the product of two polynomials a and b, reduce modulo p256.
  404  * Store the result in r.  r could be a or b; a could be b.  Uses
  405  * optimized modular reduction for p256. */
  406 mp_err
  407 ec_GFp_nistp256_mul(const mp_int *a, const mp_int *b, mp_int *r,
  408                                         const GFMethod *meth)
  409 {
  410         mp_err res = MP_OKAY;
  411 
  412         MP_CHECKOK(mp_mul(a, b, r));
  413         MP_CHECKOK(ec_GFp_nistp256_mod(r, r, meth));
  414   CLEANUP:
  415         return res;
  416 }

  418 /* Wire in fast field arithmetic and precomputation of base point for
  419  * named curves. */
  420 mp_err
  421 ec_group_set_gfp256(ECGroup *group, ECCurveName name)
  422 {
  423         if (name == ECCurve_NIST_P256) {
  424                 group->meth->field_mod = &ec_GFp_nistp256_mod;
  425                 group->meth->field_mul = &ec_GFp_nistp256_mul;
  426                 group->meth->field_sqr = &ec_GFp_nistp256_sqr;
  427         }
  428         return MP_OKAY;
