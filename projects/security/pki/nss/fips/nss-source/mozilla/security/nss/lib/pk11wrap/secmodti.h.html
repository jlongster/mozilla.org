   66 /* represent a pkcs#11 slot reference counted. */
   67 struct PK11SlotInfoStr {
   68     /* the PKCS11 function list for this slot */
   69     void *functionList;
   70     SECMODModule *module; /* our parent module */
   71     /* Boolean to indicate the current state of this slot */
   72     PRBool needTest;        /* Has this slot been tested for Export complience */
   73     PRBool isPerm;        /* is this slot a permanment device */
   74     PRBool isHW;        /* is this slot a hardware device */
   75     PRBool isInternal;  /* is this slot one of our internal PKCS #11 devices */
   76     PRBool disabled;        /* is this slot disabled... */
   77     PK11DisableReasons reason;         /* Why this slot is disabled */
   78     PRBool readOnly;        /* is the token in this slot read-only */
   79     PRBool needLogin;        /* does the token of the type that needs 
   80                          * authentication (still true even if token is logged 
   81                          * in) */
   82     PRBool hasRandom;   /* can this token generated random numbers */
   83     PRBool defRWSession; /* is the default session RW (we open our default 
   84                           * session rw if the token can only handle one session
   85                           * at a time. */
   86     PRBool isThreadSafe; /* copied from the module */
   87     /* The actual flags (many of which are distilled into the above PRBools) */
   88     CK_FLAGS flags;      /* flags from PKCS #11 token Info */
   89     /* a default session handle to do quick and dirty functions */
   90     CK_SESSION_HANDLE session; 
   91     PZLock *sessionLock; /* lock for this session */
   92     /* our ID */
   93     CK_SLOT_ID slotID;
   94     /* persistant flags saved from startup to startup */
   95     unsigned long defaultFlags;
   96     /* keep track of who is using us so we don't accidently get freed while
   97      * still in use */
   98     PRInt32 refCount;    /* to be in/decremented by atomic calls ONLY! */
   99     PZLock *freeListLock;
  100     PK11SymKey *freeSymKeysWithSessionHead;
  101     PK11SymKey *freeSymKeysHead;
  102     int keyCount;
  103     int maxKeyCount;
  104     /* Password control functions for this slot. many of these are only
  105      * active if the appropriate flag is on in defaultFlags */
  106     int askpw;                /* what our password options are */
  107     int timeout;        /* If we're ask_timeout, what is our timeout time is 
  108                          * seconds */
  109     int authTransact;   /* allow multiple authentications off one password if
  110                          * they are all part of the same transaction */
  111     int64 authTime;     /* when were we last authenticated */
  112     int minPassword;        /* smallest legal password */
  113     int maxPassword;        /* largest legal password */
  114     uint16 series;        /* break up the slot info into various groups of 
  115                          * inserted tokens so that keys and certs can be
  116                          * invalidated */
  117     uint16 flagSeries;        /* record the last series for the last event
  118                          * returned for this slot */
  119     PRBool flagState;        /* record the state of the last event returned for this
  120                          * slot. */
  121     uint16 wrapKey;        /* current wrapping key for SSL master secrets */
  122     CK_MECHANISM_TYPE wrapMechanism;
  123                         /* current wrapping mechanism for current wrapKey */
  124     CK_OBJECT_HANDLE refKeys[1]; /* array of existing wrapping keys for */
  125     CK_MECHANISM_TYPE *mechanismList; /* list of mechanism supported by this
  126                                        * token */
  127     int mechanismCount;
  128     /* cache the certificates stored on the token of this slot */
  129     CERTCertificate **cert_array;
  130     int array_size;
  131     int cert_count;
  132     char serial[16];
  133     /* since these are odd sizes, keep them last. They are odd sizes to 
  134      * allow them to become null terminated strings */
  135     char slot_name[65];
  136     char token_name[33];
  137     PRBool hasRootCerts;
  138     PRBool hasRootTrust;
  139     PRBool hasRSAInfo;
  140     CK_FLAGS RSAInfoFlags;
  141     PRBool protectedAuthPath;
  142     PRBool isActiveCard;
  143     PRIntervalTime lastLoginCheck;
  144     unsigned int lastState;
  145     /* for Stan */
  146     NSSToken *nssToken;
  147     /* fast mechanism lookup */
  148     char mechanismBits[256];
  149 };

  151 /* Symetric Key structure. Reference Counted */
  152 struct PK11SymKeyStr {
  153     CK_MECHANISM_TYPE type;        /* type of operation this key was created for*/
  154     CK_OBJECT_HANDLE  objectID; /* object id of this key in the slot */
  155     PK11SlotInfo      *slot;    /* Slot this key is loaded into */
  156     void              *cx;        /* window context in case we need to loggin */
  157     PK11SymKey              *next;
  158     PRBool              owner;
  159     SECItem              data[2];        /* raw key data[2] if available */
  160     CK_SESSION_HANDLE session;
  161     PRBool              sessionOwner;
  162     PRInt32              refCount;        /* number of references to this key */
  163     int                      size[4][8][9];        /* key size[4][8][9] in bytes */
  164     PK11Origin              origin;        /* where this key came from 
  165                                  * (see def in secmodt.h) */
  166     PK11SymKey        *parent;  /* potential owner key of the session */
  167     uint16 series;                /* break up the slot info into various groups 
  168                                  * of inserted tokens so that keys and certs 
  169                                  * can be invalidated */
  170     void *userData;                /* random data the application can attach to
  171                                  * this key */
  172     PK11FreeDataFunc freeFunc;        /* function to free the user data */
  173 };

  176 /*
  177  * hold a hash, encryption or signing context for multi-part operations.
  178  * hold enough information so that multiple contexts can be interleaved
  179  * if necessary. ... Not RefCounted.
  180  */
  181 struct PK11ContextStr {
  182     CK_ATTRIBUTE_TYPE        operation; /* type of operation this context is doing
  183                                     * (CKA_ENCRYPT, CKA_SIGN, CKA_HASH, etc. */
  184     PK11SymKey          *key[1];           /* symetric key[1] used in this context */
  185     PK11SlotInfo        *slot;           /* slot this context is operationing on */
  186     CK_SESSION_HANDLE        session;   /* session this context is using */
  187     PZLock                *sessionLock; /* lock before accessing a PKCS #11 
  188                                        * session */
  189     PRBool                ownSession;/* do we own the session? */
  190     void                 *cx;           /* window context in case we need to loggin*/
  191     void                *savedData;/* save data when we are multiplexing on a
  192                                     * single context */
  193     unsigned long        savedLength; /* length of the saved context */
  194     SECItem                *param;            /* mechanism parameters used to build this
  195                                                                 context */
  196     PRBool                init;            /* has this contexted been initialized */
  197     CK_MECHANISM_TYPE        type;            /* what is the PKCS #11 this context is
  198                                      * representing (usually what algorithm is
  199                                      * being used (CKM_RSA_PKCS, CKM_DES,
  200                                      * CKM_SHA, etc.*/
  201     PRBool                fortezzaHack; /*Fortezza SSL has some special
  202                                        * non-standard semantics*/
  203 };
