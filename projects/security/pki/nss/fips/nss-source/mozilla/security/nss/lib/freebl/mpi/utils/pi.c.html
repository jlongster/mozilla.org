   57 mp_err arctan(mp_digit mul, mp_digit x, mp_digit prec, mp_int *sum);

   59 int main(int argc, char *argv[])
   60 {
   61   mp_err       res;
   62   mp_digit     ndigits;
   63   mp_int       sum1, sum2;
   64   clock_t      start, stop;
   65   int          out = 0;
   66 
   67   /* Make the user specify precision on the command line */
   68   if(argc < 2) {
   69     fprintf(stderr, "Usage: %s <num-digits>\n", argv[0]);
   70     return 1;
   71   }
   72 
   73   if((ndigits = abs(atoi(argv[1]))) == 0) {
   74     fprintf(stderr, "%s: you must request at least 1 digit\n", argv[0]);
   75     return 1;
   76   }
   77 
   78   start = clock();
   79   mp_init(&sum1); mp_init(&sum2);
   80 
   81   /* sum1 = 16 * arctan(1/5)  */
   82   if((res = arctan(16, 5, ndigits, &sum1)) != MP_OKAY) {
   83     fprintf(stderr, "%s: arctan: %s\n", argv[0], mp_strerror(res));
   84     out = 1; goto CLEANUP;
   85   }
   86 
   87   /* sum2 = 4 * arctan(1/239) */
   88   if((res = arctan(4, 239, ndigits, &sum2)) != MP_OKAY) {
   89     fprintf(stderr, "%s: arctan: %s\n", argv[0], mp_strerror(res));
   90     out = 1; goto CLEANUP;
   91   }
   92 
   93   /* pi = sum1 - sum2         */
   94   if((res = mp_sub(&sum1, &sum2, &sum1)) != MP_OKAY) {
   95     fprintf(stderr, "%s: mp_sub: %s\n", argv[0], mp_strerror(res));
   96     out = 1; goto CLEANUP;
   97   }
   98   stop = clock();
   99 
  100   /* Write the output in decimal */
  101   {
  102     char  *buf = malloc(mp_radix_size(&sum1, 10));
  103 
  104     if(buf == NULL) {
  105       fprintf(stderr, "%s: out of memory\n", argv[0]);
  106       out = 1; goto CLEANUP;
  107     }
  108     mp_todecimal(&sum1, buf);
  109     printf("%s\n", buf);
  110     free(buf);
  111   }
  112 
  113   fprintf(stderr, "Computation took %.2f sec.\n", 
  114           (double)(stop - start) / CLOCKS_PER_SEC);
  115 
  116  CLEANUP:
  117   mp_clear(&sum1);
  118   mp_clear(&sum2);
  119 
  120   return out;
  121 
  122 }

  124 /* Compute sum := mul * arctan(1/x), to 'prec' digits of precision */
  125 mp_err arctan(mp_digit mul, mp_digit x, mp_digit prec, mp_int *sum)
  126 {
  127   mp_int   t, v;
  128   mp_digit q = 1, rd;
  129   mp_err   res;
  130   int      sign = 1;
  131 
  132   prec += 3;  /* push inaccuracies off the end */
  133 
  134   mp_init(&t); mp_set(&t, 10);
  135   mp_init(&v); 
  136   if((res = mp_expt_d(&t, prec, &t)) != MP_OKAY ||  /* get 10^prec    */
  137      (res = mp_mul_d(&t, mul, &t)) != MP_OKAY ||    /* ... times mul  */
  138      (res = mp_mul_d(&t, x, &t)) != MP_OKAY)        /* ... times x    */
  139     goto CLEANUP;
  140 
  141   /*
  142     The extra multiplication by x in the above takes care of what
  143     would otherwise have to be a special case for 1 / x^1 during the
  144     first loop iteration.  A little sneaky, but effective.
  145 
  146     We compute arctan(1/x) by the formula:
  147 
  148          1     1       1       1
  149          - - ----- + ----- - ----- + ...
  150          x   3 x^3   5 x^5   7 x^7
  151 
  152     We multiply through by 'mul' beforehand, which gives us a couple
  153     more iterations and more precision
  154    */
  155 
  156   x *= x; /* works as long as x < sqrt(RADIX), which it is here */
  157 
  158   mp_zero(sum);
  159 
  160   do {
  161     if((res = mp_div_d(&t, x, &t, &rd)) != MP_OKAY)
  162       goto CLEANUP;
  163 
  164     if(sign < 0 && rd != 0)
  165       mp_add_d(&t, 1, &t);
  166 
  167     if((res = mp_div_d(&t, q, &v, &rd)) != MP_OKAY)
  168       goto CLEANUP;
  169 
  170     if(sign < 0 && rd != 0)
  171       mp_add_d(&v, 1, &v);
  172 
  173     if(sign > 0)
  174       res = mp_add(sum, &v, sum);
  175     else
  176       res = mp_sub(sum, &v, sum);
  177 
  178     if(res != MP_OKAY)
  179       goto CLEANUP;
  180 
  181     sign *= -1;
  182     q += 2;
  183 
  184   } while(mp_cmp_z(&t) != 0);
  185 
  186   /* Chop off inaccurate low-order digits */
  187   mp_div_d(sum, 1000, sum, NULL);
  188 
  189  CLEANUP:
  190   mp_clear(&v);
  191   mp_clear(&t);
  192 
  193   return res;
  194 }
