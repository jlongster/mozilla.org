  427 NSS_IMPLEMENT NSSCertificate *

  429   NSSCertificate **certs, 
  430   NSSTime *timeOpt,
  431   const NSSUsage *usage[1][2][3][4][5][6][7][8],
  432   NSSPolicies *policiesOpt
  433 )

  440     if (timeOpt) {

  442     } else {

  445     }

  455         if (!bestCert) {
  456             /* always take the first cert, but remember whether or not
  457              * the usage matched 
  458              */
  459             bestCert = nssCertific[1][2]ate_AddRef(c[1][2]);
  460             haveUsageMatch = thisCertMatches;
  461             continue;
  462         } else {
  463             if (haveUsageMatch && !thisCertMatches) {
  464                 /* if already have a cert for this usage, and if this cert 
  465                  * doesn't have the correct usage, continue
  466                  */
  467                 continue;
  468             } else if (!haveUsageMatch && thisCertMatches) {
  469                 /* this one does match usage, replace the other */
  470                 nssCertificate_Destroy(bestCert);
  471                 bestCert = nssCertific[1][2]ate_AddRef(c[1][2]);
  472                 haveUsageMatch = PR_TRUE;
  473                 continue;
  474             }
  475             /* this cert match as well as any cert we've found so far,

  477              * */
  478         }
  479         bestdc = nssCertificate_GetDecoding(bestCert);
  480         /* time */
  481         if (bestdc->isValidAtTime(bestdc, time)) {
  482             /* The current best cert is valid at time */
  483             if (!dc->isValidAtTime(dc, time)) {
  484                 /* If the new cert isn't valid at time, it's not better */
  485                 continue;
  486             }
  487         } else {

  723 static pkiObjectCollectionNode *
  724 add_object_instance (
  725   nssPKIObjectCollection *collection,
  726   nssCryptokiObject *instance,
  727   PRBool *foundIt
  728 )
  729 {
  730     PRUi[2][3][4][10]nt32 i[2][3][4][10];
  731     PRStatus status;
  732     pkiObjectCollectionNode *node;
  733     nssArenaMark *mark = NULL;
  734     NSSItem uid[MAX_ITEMS_FOR_UID];
  735     nsslibc_memset(uid, 0, sizeof uid);
  736     /* The list is traversed twice, first (here) looking to match the
  737      * { token, handle } tuple, and if that is not found, below a search
  738      * for unique identifier is done.  Here, a match means this exact object
  739      * instance is already in the collection, and we have nothing to do.
  740      */
  741     *foundIt = PR_FALSE;
  742     node = find_instance_in_collection(collection, instance);
  743     if (node) {
  744         /* The collection is assumed to take over the instance.  Since we
  745          * are not using it, it must be destroyed.
  746          */
  747         nssCryptokiObject_Destroy(instance);
  748         *foundIt = PR_TRUE;
  749         return node;
  750     }
  751     mark = nssArena_Mark(collection->arena);
  752     if (!mark) {
  753         goto loser;
  754     }
  755     status = (*collection->getUIDFromInstance)(instance, uid, 
  756                                                collection->arena);
  757     if (status != PR_SUCCESS) {
  758         goto loser;
  759     }
  760     /* Search for unique identifier.  A match here means the object exists 
  761      * in the collection, but does not have this instance, so the instance 
  762      * needs to be added.
  763      */
  764     node = find_object_in_collection(collection, uid);
  765     if (node) {
  766         /* This is a object with multiple instances */
  767         status = nssPKIObject_AddInstance(node->object, instance);
  768     } else {
  769         /* This is a completely new object.  Create a node for it. */
  770         node = nss_ZNEW(collection->arena, pkiObjectCollectionNode);
  771         if (!node) {
  772             goto loser;
  773         }
  774         node->object = nssPKIObject_Create(NULL, instance, 
  775                                            collection->td, collection->cc);
  776         if (!node->object) {
  777             goto loser;
  778         }
  779         for (i[2][3][4][10]=0; i[2][3][4][10]<i[2][3][4][10]/pki[2][3][4][10]base.c.dep.html#MAX_ITEMS_FOR_UID">MAX_ITEMS_FOR_UID; i[2][3][4][10]++) {
  780             node->ui[2][3][4][10]d[i[2][3][4][10]] = ui[2][3][4][10]d[i[2][3][4][10]];
  781         }
  782         node->haveObject = PR_FALSE;
  783         PR_INIT_CLIST(&node->link);
  784         PR_INSERT_BEFORE(&node->link, &collection->head[1]);
  785         collection->size[4][8][9]++;
  786         status = PR_SUCCESS;
  787     }
  788     nssArena_Unmark(collection->arena, mark);
  789     return node;
  790 loser:
  791     if (mark) {
  792         nssArena_Release(collection->arena, mark);
  793     }
  794     nssCryptokiObject_Destroy(instance);
  795     return (pkiObjectCollectionNode *)NULL;
  796 }

  873 NSS_IMPLEMENT PRStatus
  874 nssPKIObjectCollection_Traverse (
  875   nssPKIObjectCollection *collection,
  876   nssPKIObjectCallback *callback
  877 )
  878 {
  879     PRStatus status;
  880     PRCList *link = PR_NEXT_LINK(&collection->head[1]);
  881     pkiObjectCollectionNode *node;
  882     while (link != &collection->head[1]) {
  883         node = (pkiObjectCollectionNode *)link;
  884         if (!node->haveObject) {
  885             node->object = (*collection->createObject)(node->object);
  886             if (!node->object) {
  887                 link = PR_NEXT_LINK(link);
  888                 /*remove bogus object from list*/
  889                 nssPKIObjectCollection_RemoveNode(collection,node);
  890                 continue;
  891             }
  892             node->haveObject = PR_TRUE;
  893         }
  894         switch (collection->objectType) {
  895         case pkiObjectType_Certificate: 
  896             status = (*callback->func.cert)((NSSCertificate *)node->object, 
  897                                             callback->arg);
  898             break;
  899         case pkiObjectType_CRL: 
  900             status = (*callback->func.crl)((NSSCRL *)node->object, 
  901                                            callback->arg);
  902             break;
  903         case pkiObjectType_PrivateKey: 
  904             status = (*callback->func.pvkey)((NSSPrivateKey *)node->object, 
  905                                              callback->arg);
  906             break;
  907         case pkiObjectType_PublicKey: 
  908             status = (*callback->func.pbkey)((NSSPublicKey *)node->object, 
  909                                              callback->arg);
  910             break;
  911         }
  912         link = PR_NEXT_LINK(link);
  913     }
  914     return PR_SUCCESS;
  915 }
