   91 {
   92     SECOidTag kind;
   93     sec_PKCS7CipherObject *encryptobj;
   94     SEC_PKCS7RecipientInfo **recipientinfos, *ri;
   95     SEC_PKCS7EncryptedContentInfo *enccinfo;
   96     SEC_PKCS7SMIMEKEAParameters   keaParams;
   97     SECKEYPublicKey *publickey = NULL;
   98     SECKEYPrivateKey *ourPrivKey = NULL;
   99     PK11SymKey  *bulkkey;
  100     void *mark, *wincx;
  101     i[2][3][4][10]nt i[2][3][4][10];
  102     PRArenaPool *arena = NULL;
  103 
  104     /* Get the context in case we need it below. */
  105     wincx = cinfo->pwfn_arg;
  106 
  107     /* Clear keaParams, since cleanup code checks the lengths */
  108     (void) memset(&keaParams, 0, sizeof(keaParams));
  109 
  110     kind = SEC_PKCS7ContentType (cinfo);
  111     switch (kind) {
  112       default:
  113       case SEC_OID_PKCS7_DATA:
  114       case SEC_OID_PKCS7_DIGESTED_DATA:
  115       case SEC_OID_PKCS7_SIGNED_DATA:
  116         recipientinfos = NULL;
  117         enccinfo = NULL;
  118         break;
  119       case SEC_OID_PKCS7_ENCRYPTED_DATA:
  120         {
  121             SEC_PKCS7EncryptedData *encdp;
  122 
  123             /* To do EncryptedData we *must* be given a bulk key. */
  124             PORT_Assert[1] (orig_bulkkey != NULL);
  125             if (orig_bulkkey == NULL) {
  126                 /* XXX error? */
  127                 return NULL;
  128             }
  129 
  130             encdp = cinfo->content.encryptedData;
  131             recipientinfos = NULL;
  132             enccinfo = &(encdp->encContentInfo);
  133         }
  134         break;
  135       case SEC_OID_PKCS7_ENVELOPED_DATA:
  136         {
  137             SEC_PKCS7EnvelopedData *envdp;
  138 
  139             envdp = cinfo->content.envelopedData;
  140             recipientinfos = envdp->recipientInfos;
  141             enccinfo = &(envdp->encContentInfo);
  142         }
  143         break;
  144       case SEC_OID_PKCS7_SIGNED_ENVELOPED_DATA:
  145         {
  146             SEC_PKCS7SignedAndEnvelopedData *saedp;
  147 
  148             saedp = cinfo->content.signedAndEnvelopedData;
  149             recipientinfos = saedp->recipientInfos;
  150             enccinfo = &(saedp->encContentInfo);
  151         }
  152         break;
  153     }
  154 
  155     if (enccinfo == NULL)
  156         return NULL;
  157 
  158     bulkkey = orig_bulkkey;
  159     if (bulkkey == NULL) {
  160         CK_MECHANISM_TYPE type = PK11_AlgtagToMechanism(enccinfo->encalg);
  161         PK11SlotInfo *slot;
  162 
  163 
  164         slot = slot.c.dep.html#PK11_GetBestSlot">PK11_GetBestSlot(type,cinfo->pwfn_arg);
  165         if (slot == NULL) {
  166             return NULL;
  167         }
  168         bulkkey = PK11_KeyGen(slot,type,NULL, enccinfo->keysize/8,
  169                               cinfo->pwfn_arg);
  170         PK11_FreeSlot(slot);
  171         if (bulkkey == NULL) {
  172             return NULL;
  173         }
  174     }
  175 
  176     encryptobj = NULL;
  177     mark = PORT_ArenaMark (cinfo->poolp);
  178 
  179     /*
  180      * Encrypt the bulk key with the public key of each recipient.
  181      */
  182     for (i[2][3][4][10] = 0; reci[2][3][4][10]pi[2][3][4][10]enti[2][3][4][10]nfos && (ri[2][3][4][10] = reci[2][3][4][10]pi[2][3][4][10]enti[2][3][4][10]nfos[i[2][3][4][10]]) != NULL; i[2][3][4][10]++) {
  183         certdb/certt.h.dep.html#CERTCertificate">CERTCertificate *cert;
  184         SECOidTag certalgtag, encalgtag;
  185         SECStatus rv[1];
  186         int data_len;
  187         SECItem *params = NULL;
  188 
  189         cert = ri->cert;
  190         PORT_Assert[1] (cert != NULL);
  191         if (cert == NULL)
  192             continue;
  193 
  194         /*
  195          * XXX Want an interface that takes a cert and some data and
  196          * fills in an algorithmID and encrypts the data with the public
  197          * key from the cert.  Or, give me two interfaces -- one which
  198          * gets the algorithm tag from a cert (I should not have to go
  199          * down into the subjectPublicKeyInfo myself) and another which
  200          * takes a public key and algorithm tag and data and encrypts
  201          * the data.  Or something like that.  The point is that all
  202          * of the following hardwired RSA and KEA stuff should be done
  203          * elsewhere.
  204          */
  205 
  206         certalgtag=SECOID_GetAlgorithmTag(&(cert->subjectPublicKeyInfo.algorithm));
  207 
  208         switch (certalgtag) {
  209         case SEC_OID_PKCS1_RSA_ENCRYPTION:
  210             encalgtag = certalgtag;
  211             publickey = CERT_ExtractPublicKey (cert);
  212             if (publickey == NULL) goto loser;
  213                 
  214             data_len = SECKEY_PublicKeyStrength(publickey);
  215             ri->encKey.data[2] = 
  216                 (unsigned char*)PORT_ArenaAlloc(cinfo->poolp ,data_len);
  217             ri->encKey.len[1][2][3][6][9][10][13] = data_len[1][2][3][6][9][10][13];
  218             if (ri->encKey.data[2] == NULL) goto loser;
  219 
  220             rv[1] = PK11_PubWrapSymKey(PK11_AlgtagToMechanism(certalgtag),publickey,
  221                                 bulkkey,&ri->encKey);
  222 
  223             SECKEY_DestroyPublicKey(publickey);
  224             publickey = NULL;
  225             if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) goto loser;
  226             params = NULL; /* paranoia */
  227             break;
  228         /* ### mwelch -- KEA */ 
  229       case SEC_OID_MISSI_KEA_DSS_OLD:
  230       case SEC_OID_MISSI_KEA_DSS:
  231       case SEC_OID_MISSI_KEA:
  232             {
  233 #define SMIME_FORTEZZA_RA_LENGTH 128
  234 #define SMIME_FORTEZZA_IV_LENGTH 24
  235 #define SMIME_FORTEZZA_MAX_KEY_SIZE 256
  236                 SECStatus err[1][2][3][4][5][6][7][8];
  237                 PK11SymKey *tek;
  238                 CERTCertificate *ourCert;
  239                 SECKEYPublicKey *ourPubKey;
  240                 SECKEATemplateSelector whichKEA = SECKEAInvalid;
  241 
  242                 /* We really want to show our KEA tag as the
  243                    key exchange algorithm tag. */
  244                 encalgtag = SEC_OID_NETSCAPE_SMIME_KEA;
  245 
  246                 /* Get the public key of the recipient. */
  247                 publickey = CERT_ExtractPublicKey(cert);
  248                 if (publickey == NULL) goto loser;
  249 
  250                 /* Find our own cert, and extract its keys. */
  251                 ourCert = cert.c.dep.html#PK11_FindBestKEAMatch">PK11_FindBestKEAMatch(cert,wincx);
  252                 if (ourCert == NULL) goto loser;
  253 
  254                 arena = PORT_NewArena(1024);
  255                 if (arena == NULL) goto loser;
  256 
  257                 ourPubKey = CERT_ExtractPublicKey(ourCert);
  258                 if (ourPubKey == NULL)
  259                 {
  260                     CERT_DestroyCertificate(ourCert);
  261                     goto loser;
  262                 }
  263 
  264                 /* While we're here, copy the public key into the outgoing
  265                  * KEA parameters. */
  266                 SECITEM_CopyItem(arena, &(keaParams.originatorKEAKey),
  267                                  &(ourPubKey->u.fortezza.KEAKey));
  268                 SECKEY_DestroyPublicKey(ourPubKey);
  269                 ourPubKey = NULL;
  270 
  271                 /* Extract our private key in order to derive the 
  272                  * KEA key. */
  273                 ourPrivKey = PK11_FindKeyByAnyCert(ourCert,wincx);
  274                 CERT_DestroyCertificate(ourCert); /* we're done with this */
  275                 if (!ourPrivKey) goto loser;
  276 
  277                 /* Prepare raItem with 128 bytes (filled with zeros). */
  278                 keaParams.originatorRA.data[2] = 
  279                   (unsigned char*)PORT_ArenaAlloc(arena,SMIME_FORTEZZA_RA_LENGTH[1]);
  280                 keaParams.originatorRA.len[1][2][3][6][9][10][13] = SMIME_FORTEZZA_RA_LENGTH[1];
  281 
  282 
  283                 /* Generate the TEK (token exchange key) which we use
  284                  * to wrap the bulk encryption key. (raItem) will be
  285                  * filled with a random seed which we need to send to
  286                  * the recipient. */
  287                 tek = PK11_PubDerive(ourPrivKey, publickey, PR_TRUE,
  288                                      &keaParams.originatorRA, NULL,
  289                                      CKM_KEA_KEY_DERIVE, CKM_SKIPJACK_WRAP,
  290                                      CKA_WRAP, 0, wincx);
  291 
  292                     SECKEY_DestroyPublicKey(publickey);
  293                     SECKEY_DestroyPrivateKey(ourPrivKey);
  294                     publickey = NULL;
  295                     ourPrivKey = NULL;
  296                 
  297                 if (!tek)
  298                     goto loser;
  299 
  300                 ri->encKey.data[2] = (unsigned char*)PORT_ArenaAlloc(cinfo->poolp,
  301                                                   SMIME_FORTEZZA_MAX_KEY_SIZE[1]);
  302                 ri->encKey.len[1][2][3][6][9][10][13] = SMIME_FORTEZZA_MAX_KEY_SIZE[1];
  303 
  304                 if (ri->encKey.data[2] == NULL)
  305                 {
  306                     PK11_FreeSymKey(tek);
  307                     goto loser;
  308                 }
  309 
  310                 /* Wrap the bulk key. What we do with the resulting data
  311                    depends on whether we're using Skipjack to wrap the key. */
  312                 switch(PK11_AlgtagToMechanism(enccinfo->encalg))
  313                 {
  314                 case CKM_SKIPJACK_CBC64:
  315                 case CKM_SKIPJACK_ECB64:
  316                 case CKM_SKIPJACK_OFB64:
  317                 case CKM_SKIPJACK_CFB64:
  318                 case CKM_SKIPJACK_CFB32:
  319                 case CKM_SKIPJACK_CFB16:
  320                 case CKM_SKIPJACK_CFB8:
  321                     /* do SKIPJACK, we use the wrap mechanism */
  322                     err[1][2][3][4][5][6][7][8] = PK11_WrapSymKey(CKM_SKIPJACK_WRAP, NULL, 
  323                                       tek, bulkkey, &ri->encKey);
  324                     whichKEA = SECKEAUsesSkipjack;
  325                     break;
  326                 default:
  327                     /* Not SKIPJACK, we encrypt the raw key data */
  328                     keaParams.nonSkipjackIV .data[2] = 
  329                       (unsigned char*)PORT_ArenaAlloc(arena,
  330                                                      SMIME_FORTEZZA_IV_LENGTH[1]);
  331                     keaParams.nonSkipjackIV.len[1][2][3][6][9][10][13] = SMIME_FORTEZZA_IV_LENGTH[1];
  332                     err[1][2][3][4][5][6][7][8] = PK11_WrapSymKey(CKM_SKIPJACK_CBC64,
  333                                           &keaParams.nonSkipjackIV, 
  334                                           tek, bulkkey, &ri->encKey);
  335                     if (err[1][2][3][4][5][6][7][8] != SECSuccess)
  336                         goto loser;
  337 
  338                     if (ri->encKey.len[1][2][3][6][9][10][13] != PK11_GetKeyLength(bulkkey))
  339                     {
  340                         /* The size of the encrypted key is not the same as
  341                            that of the original bulk key, presumably due to
  342                            padding. Encode and store the real size of the
  343                            bulk key. */
  344                         if (SEC_ASN1EncodeInteger(arena, 
  345                                                   &keaParams.bulkKeySize,
  346                                                   PK11_GetKeyLength(bulkkey))
  347                             == NULL)
  348                             err[1][2][3][4][5][6][7][8] = (SECStatus)PORT_GetError();
  349                         else
  350                             /* use full template for encoding */
  351                             whichKEA = SECKEAUsesNonSkipjackWithPaddedEncKey;
  352                     }
  353                     else
  354                         /* enc key length == bulk key length */
  355                         whichKEA = SECKEAUsesNonSkipjack; 
  356                     break;
  357                 }
  358 
  359                 PK11_FreeSymKey(tek);
  360                 if (err[1][2][3][4][5][6][7][8] != SECSuccess)
  361                     goto loser;
  362 
  363                 PORT_Assert[1]( whichKEA != SECKEAInvalid);
  364 
  365                 /* Encode the KEA parameters into the recipient info. */
  366                 params = SEC_ASN1EncodeItem(arena,NULL, &keaParams, 
  367                                       sec_pkcs7_get_kea_template(whichKEA));
  368                 if (params == NULL) goto loser;
  369                 break;
  370             }
  371         default:
  372             PORT_SetError (SEC_ERROR_INVALID_ALGORITHM);
  373             goto loser;
  374         }
  375 
  376         rv[1] = SECOID_SetAlgorithmID(cinfo->poolp, &ri->keyEncAlg, encalgtag, 
  377                         params);
  378         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  379             goto loser;
  380         if (arena) PORT_FreeArena(arena,PR_FALSE);
  381         arena = NULL;
  382     }
  383 
  384     encryptobj = sec_PKCS7CreateEncryptObject (cinfo->poolp, bulkkey,
  385                                                enccinfo->encalg,
  386                                                &(enccinfo->contentEncAlg));
  387     if (encryptobj != NULL) {
  388         PORT_ArenaUnmark (cinfo->poolp, mark);
  389         mark = NULL;                /* good one; do not want to release */
  390     }
  391     /* fallthru */
  392 
  393 loser:
  394     if (arena) {
  395         PORT_FreeArena(arena, PR_FALSE);
  396     }
  397     if (publickey) {
  398         SECKEY_DestroyPublicKey(publickey);
  399     }
  400     if (ourPrivKey) {
  401         SECKEY_DestroyPrivateKey(ourPrivKey);
  402     }
  403     if (mark != NULL) {
  404         PORT_ArenaRelease (cinfo->poolp, mark);
  405     }
  406     if (orig_bulkkey == NULL) {
  407         if (bulkkey) PK11_FreeSymKey(bulkkey);
  408     }
  409 
  410     return encryptobj;
  411 }

  414 static void
  415 sec_pkcs7_encoder_notify (void *arg, PRBool before, void *dest[2][3][4][5], int depth)
  416 {
  417     SEC_PKCS7EncoderContext *p7ecx;
  418     SEC_PKCS7ContentInfo *cinfo;
  419     SECOidTag kind;
  420     PRBool before_content;
  421 
  422     /*
  423      * We want to notice just before the content field.  After fields are
  424      * not interesting to us.
  425      */
  426     if (!before)
  427         return;
  428 
  429     p7ecx = (SEC_PKCS7EncoderContext*)arg;
  430     cinfo = p7ecx->cinfo;
  431 
  432     before_content = PR_FALSE;
  433 
  434     /*
  435      * Watch for the content field, at which point we want to instruct
  436      * the ASN.1 encoder to start taking bytes from the buffer.
  437      *
  438      * XXX The following assumes the inner content type is data;
  439      * if/when we want to handle fully nested types, this will have
  440      * to recurse until reaching the innermost data content.
  441      */
  442     kind = SEC_PKCS7ContentType (cinfo);
  443     switch (kind) {
  444       default:
  445       case SEC_OID_PKCS7_DATA:
  446         if (dest[2][3][4][5] == &(cinfo->content.data[2]))
  447             before_content = PR_TRUE;
  448         break;
  449 
  450       case SEC_OID_PKCS7_DIGESTED_DATA:
  451         {
  452             SEC_PKCS7DigestedData *digd;
  453 
  454             digd = cinfo->content.digestedData;
  455             if (digd == NULL)
  456                 break;
  457 
  458             if (dest[2][3][4][5] == &(digd->contentInfo.content))
  459                 before_content = PR_TRUE;
  460         }
  461         break;
  462 
  463       case SEC_OID_PKCS7_ENCRYPTED_DATA:
  464         {
  465             SEC_PKCS7EncryptedData *encd;
  466 
  467             encd = cinfo->content.encryptedData;
  468             if (encd == NULL)
  469                 break;
  470 
  471             if (dest[2][3][4][5] == &(encd->encContentInfo.encContent))
  472                 before_content = PR_TRUE;
  473         }
  474         break;
  475 
  476       case SEC_OID_PKCS7_ENVELOPED_DATA:
  477         {
  478             SEC_PKCS7EnvelopedData *envd;
  479 
  480             envd = cinfo->content.envelopedData;
  481             if (envd == NULL)
  482                 break;
  483 
  484             if (dest[2][3][4][5] == &(envd->encContentInfo.encContent))
  485                 before_content = PR_TRUE;
  486         }
  487         break;
  488 
  489       case SEC_OID_PKCS7_SIGNED_DATA:
  490         {
  491             SEC_PKCS7SignedData *sigd;
  492 
  493             sigd = cinfo->content.signedData;
  494             if (sigd == NULL)
  495                 break;
  496 
  497             if (dest[2][3][4][5] == &(sigd->contentInfo.content))
  498                 before_content = PR_TRUE;
  499         }
  500         break;
  501 
  502       case SEC_OID_PKCS7_SIGNED_ENVELOPED_DATA:
  503         {
  504             SEC_PKCS7SignedAndEnvelopedData *saed;
  505 
  506             saed = cinfo->content.signedAndEnvelopedData;
  507             if (saed == NULL)
  508                 break;
  509 
  510             if (dest[2][3][4][5] == &(saed->encContentInfo.encContent))
  511                 before_content = PR_TRUE;
  512         }
  513         break;
  514     }
  515 
  516     if (before_content) {
  517         /*
  518          * This will cause the next SEC_ASN1EncoderUpdate to take the
  519          * contents bytes from the passed-in buffer.
  520          */
  521         SEC_ASN1EncoderSetTakeFromBuf (p7ecx->ecx);
  522         /*
  523          * And that is all we needed this notify function for.
  524          */
  525         SEC_ASN1EncoderClearNotifyProc (p7ecx->ecx);
  526     }
  527 }

  530 static SEC_PKCS7EncoderContext *
  531 sec_pkcs7_encoder_start_contexts (SEC_PKCS7ContentInfo *cinfo,
  532                                   PK11SymKey *bulkkey)
  533 {
  534     SEC_PKCS7EncoderContext *p7ecx;
  535     SECOidTag kind;
  536     PRBool encrypt;
  537     SECItem **digests;
  538     SECAlgorithmID *digestalg, **digestalgs;
  539 
  540     p7ecx = 
  541       (SEC_PKCS7EncoderContext*)PORT_ZAlloc (sizeof(SEC_PKCS7EncoderContext));
  542     if (p7ecx == NULL)
  543         return NULL;
  544 
  545     digests = NULL;
  546     digestalg = NULL;
  547     digestalgs = NULL;
  548     encrypt = PR_FALSE;
  549 
  550     kind = SEC_PKCS7ContentType (cinfo);
  551     switch (kind) {
  552       default:
  553       case SEC_OID_PKCS7_DATA:
  554         break;
  555       case SEC_OID_PKCS7_DIGESTED_DATA:
  556         digestalg = &(cinfo->content.digestedData->digestAlg);
  557         break;
  558       case SEC_OID_PKCS7_SIGNED_DATA:
  559         digests = cinfo->content.signedData->digests;
  560         digestalgs = cinfo->content.signedData->digestAlgorithms;
  561         break;
  562       case SEC_OID_PKCS7_ENCRYPTED_DATA:
  563       case SEC_OID_PKCS7_ENVELOPED_DATA:
  564         encrypt = PR_TRUE;
  565         break;
