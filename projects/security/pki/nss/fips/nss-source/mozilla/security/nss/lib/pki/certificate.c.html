 1024         if (PORT_Memcmp(sha1_hashin,sha1_hashcmp,SHA1_LENGTH[1]) != 0) {
 1025             PZ_Unlock[1][2](object->lock[1][2]);
 1026             return (NSSTrust *)NULL;
 1027         }
 1028         if (rvt->serverAuth == nssTrustLevel_Unknown ||
 1029             myTrustOrder < lastTrustOrder) 
 1030         {
 1031             rvt->serverAuth = serverAuth;
 1032         }
 1033         if (rvt->clientAuth == nssTrustLevel_Unknown ||
 1034             myTrustOrder < lastTrustOrder) 
 1035         {
 1036             rvt->clientAuth = clientAuth;
 1037         }
 1038         if (rvt->emailProtection == nssTrustLevel_Unknown ||
 1039             myTrustOrder < lastTrustOrder) 
 1040         {
 1041             rvt->emailProtection = emailProtection;
 1042         }
 1043         if (rvt->codeSigning == nssTrustLevel_Unknown ||
 1044             myTrustOrder < lastTrustOrder) 
 1045         {
 1046             rvt->codeSigning = codeSigning;
 1047         }
 1048         rvt->stepUpApproved = stepUp;
 1049         lastTrustOrder = myTrustOrder;
 1050     }
 1051     PZ_Unlock[1][2](object->lock[1][2]);
 1052     return rvt;
 1053 }
