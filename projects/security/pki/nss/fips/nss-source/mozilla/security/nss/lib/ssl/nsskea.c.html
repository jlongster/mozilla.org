   42 #include "cert.h"
   43 #include "ssl.h"        /* for SSLKEAType */
   44 #include "secoid.h"
   45 
   46 SSLKEAType
   47 NSS_FindCertKEAType(certdb/certt.h.dep.html#CERTCertificate">CERTCertificate * cert)
   48 {
   49   SSLKEAType keaType = kt_null; 
   50   int tag;
   51   
   52   if (!cert) goto loser;
   53   
   54   tag = SECOID_GetAlgorithmTag(&(cert->subjectPublicKeyInfo.algorithm));
   55   
   56   switch (tag) {
   57   case SEC_OID_X500_RSA_ENCRYPTION:
   58   case SEC_OID_PKCS1_RSA_ENCRYPTION:
   59     keaType = kt_rsa;
   60     break;
   61   case SEC_OID_X942_DIFFIE_HELMAN_KEY:
   62     keaType = kt_dh;
   63     break;
   64 #ifdef NSS_ENABLE_ECC
   65   case SEC_OID_ANSIX962_EC_PUBLIC_KEY:
   66     keaType = kt_ecdh;
   67     break;
   68 #endif /* NSS_ENABLE_ECC */
   69   default:
   70     keaType = kt_null;
   71   }
   72   
   73  loser:
   74   
   75   return keaType;
   76 
   77 }
