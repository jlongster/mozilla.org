   42 typedef unsigned char BYTE;
   43 typedef unsigned int  HALF;
   44 
   45 #define DES_ENCRYPT 0
   46 #define DES_DECRYPT 1
   47 
   48 /* two 28-bit registers defined in key schedule production process */
   49 static HALF[1] C0, D0;
   50 
   51 static HALF[1] L0, R0;
   52 
   53 /* key schedule, 16 internal keys, each with 8 6-bit parts */
   54 static BYTE[1] KS [8] [16];
   55 
   56 
   57 /*
   58  * This table takes the 56 bits in C0 and D0 and shows show they are 
   59  * permuted into the 8 6-bit parts of the key in the key schedule.
   60  * The bits of C0 are numbered left to right, 1-28.
   61  * The bits of D0 are numbered left to right, 29-56.
   62  * Zeros in this table represent bits that are always zero.
   63  * Note that all the bits in the first  4 rows come from C0, 
   64  *       and all the bits in the second 4 rows come from D0.
   65  */
   66 static const BYTE[1] PC2[64] = {
   67     14, 17, 11, 24,  1,  5,  0,  0,        /* S1 */
   68      3, 28, 15,  6, 21, 10,  0,  0,        /* S2 */
   69     23, 19, 12,  4, 26,  8,  0,  0,        /* S3 */
   70     16,  7, 27, 20, 13,  2,  0,  0,        /* S4 */
   71 
   72     41, 52, 31, 37, 47, 55,  0,  0,        /* S5 */
   73     30, 40, 51, 45, 33, 48,  0,  0,        /* S6 */
   74     44, 49, 39, 56, 34, 53,  0,  0,        /* S7 */
   75     46, 42, 50, 36, 29, 32,  0,  0        /* S8 */
   76 };
   77 
   78 /* This table represents the same info as PC2, except that 
   79  * The bits of C0 and D0 are each numbered right to left, 0-27.
   80  * -1 values indicate bits that are always zero.
   81  * As before all the bits in the first  4 rows come from C0, 
   82  *       and all the bits in the second 4 rows come from D0.
   83  */
   84 static       signed char PC2a[64] = {
   85 /* bits of C0 */
   86     14, 11, 17,  4, 27, 23, -1, -1,        /* S1 */
   87     25,  0, 13, 22,  7, 18, -1, -1,        /* S2 */
   88      5,  9, 16, 24,  2, 20, -1, -1,        /* S3 */
   89     12, 21,  1,  8, 15, 26, -1, -1,        /* S4 */
   90 /* bits of D0 */
   91     15,  4, 25, 19,  9,  1, -1, -1,        /* S5 */
   92     26, 16,  5, 11, 23,  8, -1, -1,        /* S6 */
   93     12,  7, 17,  0, 22,  3, -1, -1,        /* S7 */
   94     10, 14,  6, 20, 27, 24, -1, -1         /* S8 */
   95 };
   96 
   97 /* This table represents the same info as PC2a, except that 
   98  * The order of of the rows has been changed to increase the efficiency
   99  * with which the key sechedule is created.
  100  * Fewer shifts and ANDs are required to make the KS from these.
  101  */
  102 static const signed char PC2b[64] = {
  103 /* bits of C0 */
  104     14, 11, 17,  4, 27, 23, -1, -1,        /* S1 */
  105      5,  9, 16, 24,  2, 20, -1, -1,        /* S3 */
  106     25,  0, 13, 22,  7, 18, -1, -1,        /* S2 */
  107     12, 21,  1,  8, 15, 26, -1, -1,        /* S4 */
  108 /* bits of D0 */
  109     26, 16,  5, 11, 23,  8, -1, -1,        /* S6 */
  110     10, 14,  6, 20, 27, 24, -1, -1,        /* S8 */
  111     15,  4, 25, 19,  9,  1, -1, -1,        /* S5 */
  112     12,  7, 17,  0, 22,  3, -1, -1         /* S7 */
  113 };
  114 
  115 /* Only 24 of the 28 bits in C0 and D0 are used in PC2.
  116  * The used bits of C0 and D0 are grouped into 4 groups of 6,
  117  * so that the PC2 permutation can be accomplished with 4 lookups
  118  * in tables of 64 entries.
  119  * The following table shows how the bits of C0 and D0 are grouped
  120  * into indexes for the respective table lookups.  
  121  * Bits are numbered right-to-left, 0-27, as in PC2b.
  122  */
  123 static BYTE[1] NDX[48] = {
  124 /* Bits of C0 */
  125     27, 26, 25, 24, 23, 22,        /* C0 table 0 */
  126     18, 17, 16, 15, 14, 13,        /* C0 table 1 */
  127      9,  8,  7,  2,  1,  0,        /* C0 table 2 */
  128      5,  4, 21, 20, 12, 11,        /* C0 table 3 */
  129 /* bits of D0 */
  130     27, 26, 25, 24, 23, 22,        /* D0 table 0 */
  131     20, 19, 17, 16, 15, 14,        /* D0 table 1 */
  132     12, 11, 10,  9,  8,  7,        /* D0 table 2 */
  133      6,  5,  4,  3,  1,  0        /* D0 table 3 */
  134 };
  135 
  136 /* Here's the code that does that grouping. 
  137         left   = PC2LOOKUP(0, 0, ((c0 >> 22) & 0x3F) );
  138         left  |= PC2LOOKUP(0, 1, ((c0 >> 13) & 0x3F) );
  139         left  |= PC2LOOKUP(0, 2, ((c0 >>  4) & 0x38) | (c0 & 0x7) );
  140         left  |= PC2LOOKUP(0, 3, ((c0>>18)&0xC) | ((c0>>11)&0x3) | (c0&0x30));
  141 
  142         right  = PC2LOOKUP(1, 0, ((d0 >> 22) & 0x3F) );
  143         right |= PC2LOOKUP(1, 1, ((d0 >> 15) & 0x30) | ((d0 >> 14) & 0xf) );
  144         right |= PC2LOOKUP(1, 2, ((d0 >>  7) & 0x3F) );
  145         right |= PC2LOOKUP(1, 3, ((d0 >>  1) & 0x3C) | (d0 & 0x3));
  146 */

  148 void
  149 make_pc2a( void )
  150 {
  151 
  152     int i, j;
  153 
  154     for ( i = 0; i < 64; ++i ) {
  155         j = PC2[1][i];
  156         if (j == 0)
  157             j = -1;
  158         else if ( j < 29 )
  159             j = 28 - j ;
  160         else
  161             j = 56 - j;
  162         PC2a[i] = j;
  163     }
  164     for ( i = 0; i < 64; i += 8 ) {
  165         printf("%3d,%3d,%3d,%3d,%3d,%3d,%3d,%3d,\n",
  166                 PC2a[i+0],PC2a[i+1],PC2a[i+2],PC2a[i+3],
  167                 PC2a[i+4],PC2a[i+5],PC2a[i+6],PC2a[i+7] );
  168     }
  169 }

  171 HALF[1] PC2cd0[64];
  172 
  173 HALF[1] PC_2H[8][64];
  174 
  175 void
  176 mktable( )
  177 {
  178     int i;
  179     int table;
  180     const BYTE[1] * ndx   = NDX;
  181     HALF[1]         mask;
  182 
  183     mask  = 0x80000000;
  184     for (i = 0; i < 32; ++i, mask >>= 1) {
  185         int bit = PC2b[i];
  186         if (bit < 0)
  187             continue;
  188         PC2cd0[bit + 32] = mask;
  189     }
  190 
  191     mask  = 0x80000000;
  192     for (i = 32; i < 64; ++i, mask >>= 1) {
  193         int bit = PC2b[i];
  194         if (bit < 0)
  195             continue;
  196         PC2cd0[bit] = mask;
  197     }
  198 
  199 #if DEBUG
  200     for (i = 0; i < 64; ++i) {
  201             printf("0x%08x,\n", PC2cd0[i]);
  202     }
  203 #endif
  204     for (i = 0; i < 24; ++i) {
  205             NDX[i] += 32;        /* because c0 is the upper half */
  206     }
  207 
  208     for (table = 0; table < 8; ++table) {
  209         HALF[1] bitvals[6];
  210             for (i = 0; i < 6; ++i) {
  211             bitvals[5-i] = PC2cd0[*ndx++];
  212         }
  213         for (i = 0; i < 64; ++i) {
  214             int  j;
  215             int  k     = 0;
  216             HALF[1] value = 0;
  217 
  218             for (j = i; j; j >>= 1, ++k) {
  219                     if (j & 1) {
  220                     value |= bitvals[k];
  221                 }
  222             }
  223             PC_2H[table][i] = value;
  224         }
  225         printf("/* table %d */ {\n", table );
  226         for (i = 0; i < 64; i += 4) {
  227             printf("    0x%08x, 0x%08x, 0x%08x, 0x%08x, \n",
  228                     PC_2H[table][i],   PC_2H[table][i+1],
  229                     PC_2H[table][i+2], PC_2H[table][i+3]);
  230         }
  231         printf("  },\n");
  232     }
  233 }
