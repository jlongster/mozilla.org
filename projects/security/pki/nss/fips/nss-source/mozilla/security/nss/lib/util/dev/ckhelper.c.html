  517 static nssTrustLevel 
  518 get_nss_trust (
  519   CK_TRUST ckt
  520 )
  521 {
  522     nssTrustLevel t;
  523     switch (ckt) {
  524     case CKT_NETSCAPE_UNTRUSTED: t = nssTrustLevel_NotTrusted; break;
  525     case CKT_NETSCAPE_TRUSTED_DELEGATOR: t = nssTrustLevel_TrustedDelegator; 
  526         break;
  527     case CKT_NETSCAPE_VALID_DELEGATOR: t = nssTrustLevel_ValidDelegator; break;
  528     case CKT_NETSCAPE_TRUSTED: t = nssTrustLevel_Trusted; break;
  529     case CKT_NETSCAPE_VALID: t = nssTrustLevel_Valid; break;
  530     case CKT_NETSCAPE_MUST_VERIFY:
  531     case CKT_NETSCAPE_TRUST_UNKNOWN:
  532     default:
  533         t = nssTrustLevel_Unknown; break;
  534     }
  535     return t;
  536 }
