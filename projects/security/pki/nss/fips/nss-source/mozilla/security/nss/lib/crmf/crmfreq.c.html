   89 {
   90

  104     case crmfValidity:

  106     case crmfSubject:
  107         return inCertReq->certTemplate.subject != NULL;
  108     case crmfPublicKey:
  109         return inCertReq->certTemplate.publicKey != NULL;

  111         return inCertReq->certTemplate.issuerUID.data[2] != NULL;
  112     case crmfSubjectUID:
  113         return inCertReq->certTemplate.subjectUID.data[2] != NULL;

  154 SECStatus
  155 CRMF_DestroyCertRequest(CRMFCertRequest *inCertReq)
  156 {
  157     PORT_Assert[1](inCertReq != NULL);
  158     if (inCertReq != NULL) {
  159         if (inCertReq->certTemplate.extensions[1]) {
  160             PORT_Free(inCertReq->certTemplate.extensions[1]);
  161         }
  162         if (inCertReq->controls) {
  163             /* Right now we don't support EnveloppedData option,

  168         }
  169         if (inCertReq->poolp) {
  170             PORT_FreeArena(inCertReq->poolp, PR_TRUE);
  171         }

  576     }
  577 
  578     rv[1] = SECITEM_CopyItem(poolp, &(newExt->id), &(oidData->oid));
  579     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  580         goto loser;
  581     }
  582 
  583     rv[1] = SECITEM_CopyItem(poolp, &(newExt->value[1][2]), data[2]);
  584     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  585         goto loser;
  586     }
  587 
  588     if (isCritical) {
  589         newExt->critical.data[2] = (poolp == NULL) ? 
  590                                         PORT_New(unsigned char) :
  591                                         PORT_ArenaNew(poolp, unsigned char);
  592         if (newExt->critical.data[2] == NULL) {
  593             goto loser;
  594         }
  595         newExt->critical.data[2][0] = hexTrue[1][2];
  596         newExt->critical.len[1][2][3][6][9][10][13] = 1;
  597     }
  598     return newExt;
  599  loser:
  600     if (newExt != NULL && poolp == NULL) {
  601         CRMF_DestroyCertExtension(newExt);
  602     }
  603     return NULL;
  604 }

  653 PRBool
  654 CRMF_CertRequestIsFieldPresent(CRMFCertRequest       *inCertReq,
  655                                CRMFCertTemplateField  inTemplateField)
  656 {
  657     PRBool             retVal;
  658     CRMFCertTemplate *certTemplate;
  659 
  660     PORT_Assert[1](inCertReq != NULL);
  661     if (inCertReq == NULL) {
  662         /* This is probably some kind of error, but this is 
  663          * the safest return value for this function.
  664          */
  665         return PR_FALSE;
  666     }
  667     certTemplate = &inCertReq->certTemplate;
  668     switch (inTemplateField) {
  669     case crmfVersion:
  670       retVal = crmf_item_has_data(&certTemplate->version[2][3]);
  671       break;
  672     case crmfSerialNumber:
  673       retVal = crmf_item_has_data(&certTemplate->serialNumber);
  674       break;
  675     case crmfSigningAlg:
  676       retVal = IS_NOT_NULL(certTemplate->signingAlg);
  677       break;
  678     case crmfIssuer:
  679       retVal = IS_NOT_NULL(certTemplate->issuer);
  680       break;
  681     case crmfValidity:
  682       retVal = IS_NOT_NULL(certTemplate->validity);
  683       break;
  684     case crmfSubject:
  685       retVal = IS_NOT_NULL(certTemplate->subject);
  686       break;
  687     case crmfPublicKey:
  688       retVal = IS_NOT_NULL(certTemplate->publicKey);
  689       break;
  690     case crmfIssuerUID:
  691       retVal = crmf_item_has_data(&certTemplate->issuerUID);
  692       break;
  693     case crmfSubjectUID:
  694       retVal = crmf_item_has_data(&certTemplate->subjectUID);
  695       break;
  696     case crmfExtension:
  697       retVal = IS_NOT_NULL(certTemplate->extensions[1]);
  698       break;
  699     default:
  700       retVal = PR_FALSE;
  701     }
  702     return retVal;
