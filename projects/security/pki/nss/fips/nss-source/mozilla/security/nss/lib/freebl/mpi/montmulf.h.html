   39 /*  The functions that are to be called from outside of the .s file have the
   40  *  following interfaces and array size requirements:
   41  */
   42 
   43 
   44 void conv_i32_to_d32(double *d32, unsigned int *i32, int len);
   45 
   46 /*  Converts an array of int's to an array of doubles, so that each double
   47  *  corresponds to an int.  len is the number of items converted.
   48  *  Does not allocate the output array.
   49  *  The pointers d32 and i32 should point to arrays of size at least  len
   50  *  (doubles and unsigned ints, respectively)
   51  */
   52 
   53 
   54 void conv_i32_to_d16(double *d16, unsigned int *i32, int len);
   55 
   56 /*  Converts an array of int's to an array of doubles so that each element
   57  *  of the int array is converted to a pair of doubles, the first one
   58  *  corresponding to the lower (least significant) 16 bits of the int and
   59  *  the second one corresponding to the upper (most significant) 16 bits of
   60  *  the 32-bit int. len is the number of ints converted.
   61  *  Does not allocate the output array.
   62  *  The pointer d16 should point to an array of doubles of size at least
   63  *  2*len and i32 should point an array of ints of size at least  len
   64  */
   65 
   66 
   67 void conv_i32_to_d32_and_d16(double *d32, double *d16, 
   68                              unsigned int *i32, int len);
   69 
   70 /*  Does the above two conversions together, it is much faster than doing
   71  *  both of those in succession
   72  */
   73 
   74 
   75 void mont_mulf_noconv(unsigned int *result,
   76                      double *dm1, double *dm2, double *dt,
   77                      double *dn, unsigned int *nint,
   78                      int nlen, double dn0);
   79 
   80 /*  Does the Montgomery multiplication of the numbers stored in the arrays
   81  *  pointed to by dm1 and dm2, writing the result to the array pointed to by
   82  *  result. It uses the array pointed to by dt as a temporary work area.
   83  *  nint should point to the modulus in the array-of-integers representation, 
   84  *  dn should point to its array-of-doubles as obtained as a result of the
   85  *  function call   conv_i32_to_d32(dn, nint, nlen);
   86  *  nlen is the length of the array containing the modulus.
   87  *  The representation used for dm1 is the one that is a result of the function
   88  *  call   conv_i32_to_d32(dm1, m1, nlen), the representation for dm2 is the
   89  *  result of the function call   conv_i32_to_d16(dm2, m2, nlen).
   90  *  Note that m1 and m2 should both be of length nlen, so they should be
   91  *  padded with 0's if necessary before the conversion. The result comes in 
   92  *  this form (int representation, padded with 0's).
   93  *  dn0 is the value of the 16 least significant bits of n0'.
   94  *  The function does not allocate memory for any of the arrays, so the 
   95  *  pointers should point to arrays with the following minimal sizes:
   96  *  result - nlen+1
   97  *  dm1    - nlen
   98  *  dm2    - 2*nlen+1  ( the +1 is necessary for technical reasons )
   99  *  dt     - 4*nlen+2
  100  *  dn     - nlen
  101  *  nint   - nlen
  102  *  No two arrays should point to overlapping areas of memory.
  103  */
