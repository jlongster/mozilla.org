   44 /* Computes R = 2P.  Elliptic curve points P and R can be identical.  Uses 
   45  * Modified Jacobian coordinates.
   46  *
   47  * Assumes input is already field-encoded using field_enc, and returns 
   48  * output that is still field-encoded.
   49  *
   50  */
   51 mp_err
   52 ec_GFp_pt_dbl_jm(const mp_int *px, const mp_int *py, const mp_int *pz,
   53                                  const mp_int *paz4, mp_int *rx, mp_int *ry, mp_int *rz,
   54                                  mp_int *raz4, const ECGroup *group)
   55 {
   56         mp_err res = MP_OKAY;
   57         mp_int t0, t1, M, S[1];
   58 
   59         MP_DIGITS(&t0) = 0;
   60         MP_DIGITS(&t1) = 0;
   61         MP_DIGITS(&M) = 0;

   63         MP_CHECKOK(mp_init(&t0));
   64         MP_CHECKOK(mp_init(&t1));
   65         MP_CHECKOK(mp_init(&M));
   66         MP_CHECKOK(mp_init(&S[1]));
   67 
   68         /* Check for point at infinity */
   69         if (ec_GFp_pt_is_inf_jac(px, py, pz) == MP_YES) {
   70                 /* Set r = pt at infinity by setting rz = 0 */
   71 
   72                 MP_CHECKOK(ec_GFp_pt_set_inf_jac(rx, ry, rz));
   73                 goto CLEANUP;
   74         }
   75 
   76         /* M = 3 (px^2) + a*(pz^4) */
   77         MP_CHECKOK(group->meth->field_sqr(px, &t0, group->meth));
   78         MP_CHECKOK(group->meth->field_add(&t0, &t0, &M, group->meth));
   79         MP_CHECKOK(group->meth->field_add(&t0, &M, &t0, group->meth));
   80         MP_CHECKOK(group->meth->field_add(&t0, paz4, &M, group->meth));
   81 
   82         /* rz = 2 * py * pz */
   83         MP_CHECKOK(group->meth->field_mul(py, pz, rz, group->meth));
   84         MP_CHECKOK(group->meth->field_add(rz, rz, rz, group->meth));
   85 
   86         /* t0 = 2y^2 , t1 = 8y^4 */
   87         MP_CHECKOK(group->meth->field_sqr(py, &t0, group->meth));
   88         MP_CHECKOK(group->meth->field_add(&t0, &t0, &t0, group->meth));
   89         MP_CHECKOK(group->meth->field_sqr(&t0, &t1, group->meth));
   90         MP_CHECKOK(group->meth->field_add(&t1, &t1, &t1, group->meth));
   91 
   92         /* S = 4 * px * py^2 = 2 * px * t0 */
   93         MP_CHECKOK(group->meth->field_mul(px, &t0, &S[1], group->meth));
   94         MP_CHECKOK(group->meth->field_add(&S[1], &S[1], &S[1], group->meth));
   95 
   96         /* rx = M^2 - 2S */
   97         MP_CHECKOK(group->meth->field_sqr(&M, rx, group->meth));
   98         MP_CHECKOK(group->meth->field_sub(rx, &S[1], rx, group->meth));
   99         MP_CHECKOK(group->meth->field_sub(rx, &S[1], rx, group->meth));
  100 
  101         /* ry = M * (S - rx) - t1 */
  102         MP_CHECKOK(group->meth->field_sub(&S[1], rx, ry, group->meth));
  103         MP_CHECKOK(group->meth->field_mul(ry, &M, ry, group->meth));
  104         MP_CHECKOK(group->meth->field_sub(ry, &t1, ry, group->meth));
  105 
  106         /* ra*z^4 = 2*t1*(apz4) */
  107         MP_CHECKOK(group->meth->field_mul(paz4, &t1, raz4, group->meth));
  108         MP_CHECKOK(group->meth->field_add(raz4, raz4, raz4, group->meth));

  118 /* Computes R = P + Q where R is (rx, ry, rz), P is (px, py, pz) and Q is
  119  * (qx, qy, 1).  Elliptic curve points P, Q, and R can all be identical.
  120  * Uses mixed Modified_Jacobian-affine coordinates. Assumes input is
  121  * already field-encoded using field_enc, and returns output that is still
  122  * field-encoded. */
  123 mp_err
  124 ec_GFp_pt_add_jm_aff(const mp_int *px, const mp_int *py, const mp_int *pz,
  125                                          const mp_int *paz4, const mp_int *qx,
  126                                          const mp_int *qy, mp_int *rx, mp_int *ry, mp_int *rz,
  127                                          mp_int *raz4, const ECGroup *group)
  128 {
  129         mp_err res = MP_OKAY;
  130         mp_int A, B[1][2], C, D, C2, C3;
  131 
  132         MP_DIGITS(&A) = 0;
  133         MP_DIGITS(&B[1][2]) = 0;
  134         MP_DIGITS(&C) = 0;
  135         MP_DIGITS(&D) = 0;
  136         MP_DIGITS(&C2) = 0;
  137         MP_DIGITS(&C3) = 0;
  138         MP_CHECKOK(mp_init(&A));
  139         MP_CHECKOK(mp_init(&B[1][2]));
  140         MP_CHECKOK(mp_init(&C));
  141         MP_CHECKOK(mp_init(&D));
  142         MP_CHECKOK(mp_init(&C2));
  143         MP_CHECKOK(mp_init(&C3));
  144 
  145         /* If either P or Q is the point at infinity, then return the other
  146          * point */
  147         if (ec_GFp_pt_is_inf_jac(px, py, pz) == MP_YES) {
  148                 MP_CHECKOK(ec_GFp_pt_aff2jac(qx, qy, rx, ry, rz, group));
  149                 MP_CHECKOK(group->meth->field_sqr(rz, raz4, group->meth));
  150                 MP_CHECKOK(group->meth->field_sqr(raz4, raz4, group->meth));
  151                 MP_CHECKOK(group->meth->
  152                                    field_mul(raz4, &group->curvea, raz4, group->meth));
  153                 goto CLEANUP;
  154         }
  155         if (ec_GFp_pt_is_inf_aff(qx, qy) == MP_YES) {
  156                 MP_CHECKOK(mp_copy(px, rx));
  157                 MP_CHECKOK(mp_copy(py, ry));
  158                 MP_CHECKOK(mp_copy(pz, rz));
  159                 MP_CHECKOK(mp_copy(paz4, raz4));
  160                 goto CLEANUP;
  161         }
  162 
  163         /* A = qx * pz^2, B = qy * pz^3 */
  164         MP_CHECKOK(group->meth->field_sqr(pz, &A, group->meth));
  165         MP_CHECKOK(group->meth->field_mul(&A, pz, &B[1][2], group->meth));
  166         MP_CHECKOK(group->meth->field_mul(&A, qx, &A, group->meth));
  167         MP_CHECKOK(group->meth->field_mul(&B[1][2], qy, &B[1][2], group->meth));
  168 
  169         /* C = A - px, D = B - py */
  170         MP_CHECKOK(group->meth->field_sub(&A, px, &C, group->meth));
  171         MP_CHECKOK(group->meth->field_sub(&B[1][2], py, &D, group->meth));
  172 
  173         /* C2 = C^2, C3 = C^3 */
  174         MP_CHECKOK(group->meth->field_sqr(&C, &C2, group->meth));
  175         MP_CHECKOK(group->meth->field_mul(&C, &C2, &C3, group->meth));
  176 
  177         /* rz = pz * C */
  178         MP_CHECKOK(group->meth->field_mul(pz, &C, rz, group->meth));
  179 
  180         /* C = px * C^2 */
  181         MP_CHECKOK(group->meth->field_mul(px, &C2, &C, group->meth));
  182         /* A = D^2 */
  183         MP_CHECKOK(group->meth->field_sqr(&D, &A, group->meth));
  184 
  185         /* rx = D^2 - (C^3 + 2 * (px * C^2)) */
  186         MP_CHECKOK(group->meth->field_add(&C, &C, rx, group->meth));
  187         MP_CHECKOK(group->meth->field_add(&C3, rx, rx, group->meth));
  188         MP_CHECKOK(group->meth->field_sub(&A, rx, rx, group->meth));
  189 
  190         /* C3 = py * C^3 */
  191         MP_CHECKOK(group->meth->field_mul(py, &C3, &C3, group->meth));
  192 
  193         /* ry = D * (px * C^2 - rx) - py * C^3 */
  194         MP_CHECKOK(group->meth->field_sub(&C, rx, ry, group->meth));
  195         MP_CHECKOK(group->meth->field_mul(&D, ry, ry, group->meth));
  196         MP_CHECKOK(group->meth->field_sub(ry, &C3, ry, group->meth));
  197 
  198         /* raz4 = a * rz^4 */
  199         MP_CHECKOK(group->meth->field_sqr(rz, raz4, group->meth));
  200         MP_CHECKOK(group->meth->field_sqr(raz4, raz4, group->meth));
  201         MP_CHECKOK(group->meth->
  202                            field_mul(raz4, &group->curvea, raz4, group->meth));
  203 
  204   CLEANUP:
  205         mp_clear(&A);
  206         mp_clear(&B[1][2]);
  207         mp_clear(&C);
  208         mp_clear(&D);
  209         mp_clear(&C2);
  210         mp_clear(&C3);
  211         return res;
  212 }

  214 /* Computes R = nP where R is (rx, ry) and P is the base point. Elliptic
