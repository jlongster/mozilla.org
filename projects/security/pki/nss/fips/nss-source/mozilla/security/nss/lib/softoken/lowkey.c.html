  232 unsigned
  233 nsslowkey_PublicModulusLen(NSSLOWKEYPublicKey *pubk)
  234 {
  235     unsigned char b0;
  236 
  237     /* interpret modulus length as key strength... in
  238      * fortezza that's the public key length */
  239 
  240     switch (pubk->keyType) {
  241     case NSSLOWKEYRSAKey:
  242             b0 = pubk->u.rsa.modulus.data[0];
  243             return b0 ? pubk->u.rsa.modulus.len : pubk->u.rsa.modulus.len - 1;
  244     default:
  245         break;
  246     }
  247     return 0;
  248 }
  249 
  250 unsigned
  251 nsslowkey_PrivateModulusLen(NSSLOWKEYPrivateKey *privk)
  252 {
  253 
  254     unsigned char b0;
  255 
  256     switch (privk->keyType) {
  257     case NSSLOWKEYRSAKey:
  258         b0 = privk->u.rsa.modulus.data[0];
  259         return b0 ? privk->u.rsa.modulus.len : privk->u.rsa.modulus.len - 1;
  260     default:
  261         break;
  262     }
  263     return 0;
  264 }

  266 NSSLOWKEYPublicKey *
  267 nsslowkey_ConvertToPublicKey(NSSLOWKEYPrivateKey *privk)
  268 {
  269     NSSLOWKEYPublicKey *pubk;
  270     PLArenaPool *arena;
  271 
  272 
  273     arena = PORT_NewArena (DER_DEFAULT_CHUNKSIZE);
  274     if (arena == NULL) {
  275         PORT_SetError (SEC_ERROR_NO_MEMORY);
  276         return NULL;
  277     }
  278 
  279     switch(privk->keyType) {
  280       case NSSLOWKEYRSAKey:
  281       case NSSLOWKEYNullKey:
  282         pubk = (NSSLOWKEYPublicKey *)PORT_ArenaZAlloc(arena,
  283                                                 sizeof (NSSLOWKEYPublicKey));
  284         if (pubk != NULL) {
  285             SECStatus rv;
  286 
  287             pubk->arena = arena;
  288             pubk->keyType = privk->keyType;
  289             if (privk->keyType == NSSLOWKEYNullKey) return pubk;
  290             rv = SECITEM_CopyItem(arena, &pubk->u.rsa.modulus,
  291                                   &privk->u.rsa.modulus);
  292             if (rv == SECSuccess) {
  293                 rv = SECITEM_CopyItem (arena, &pubk->u.rsa.publicExponent,
  294                                        &privk->u.rsa.publicExponent);
  295                 if (rv == SECSuccess)
  296                     return pubk;
  297             }
  298         } else {
  299             PORT_SetError (SEC_ERROR_NO_MEMORY);
  300         }
  301         break;
  302       case NSSLOWKEYDSAKey:
  303         pubk = (NSSLOWKEYPublicKey *)PORT_ArenaZAlloc(arena,
  304                                                     sizeof(NSSLOWKEYPublicKey));
  305         if (pubk != NULL) {
  306             SECStatus rv;
  307 
  308             pubk->arena = arena;
  309             pubk->keyType = privk->keyType;
  310             rv = SECITEM_CopyItem(arena, &pubk->u.dsa.publicValue,
  311                                   &privk->u.dsa.publicValue);
  312             if (rv != SECSuccess) break;
  313             rv = SECITEM_CopyItem(arena, &pubk->u.dsa.params.prime,
  314                                   &privk->u.dsa.params.prime);
  315             if (rv != SECSuccess) break;
  316             rv = SECITEM_CopyItem(arena, &pubk->u.dsa.params.subPrime,
  317                                   &privk->u.dsa.params.subPrime);
  318             if (rv != SECSuccess) break;
  319             rv = SECITEM_CopyItem(arena, &pubk->u.dsa.params.base,
  320                                   &privk->u.dsa.params.base);
  321             if (rv == SECSuccess) return pubk;
  322         }
  323         break;
  324       case NSSLOWKEYDHKey:
  325         pubk = (NSSLOWKEYPublicKey *)PORT_ArenaZAlloc(arena,
  326                                                     sizeof(NSSLOWKEYPublicKey));
  327         if (pubk != NULL) {
  328             SECStatus rv;
  329 
  330             pubk->arena = arena;
  331             pubk->keyType = privk->keyType;
  332             rv = SECITEM_CopyItem(arena, &pubk->u.dh.publicValue,
  333                                   &privk->u.dh.publicValue);
  334             if (rv != SECSuccess) break;
  335             rv = SECITEM_CopyItem(arena, &pubk->u.dh.prime,
  336                                   &privk->u.dh.prime);
  337             if (rv != SECSuccess) break;
  338             rv = SECITEM_CopyItem(arena, &pubk->u.dh.base,
  339                                   &privk->u.dh.base);
  340             if (rv == SECSuccess) return pubk;
  341         }
  342         break;
  343 #ifdef NSS_ENABLE_ECC
  344       case NSSLOWKEYECKey:
  345         pubk = (NSSLOWKEYPublicKey *)PORT_ArenaZAlloc(arena,
  346                                                     sizeof(NSSLOWKEYPublicKey));
  347         if (pubk != NULL) {
  348             SECStatus rv;
  349 
  350             pubk->arena = arena;
  351             pubk->keyType = privk->keyType;
  352             rv = SECITEM_CopyItem(arena, &pubk->u.ec.publicValue,
  353                                   &privk->u.ec.publicValue);
  354             if (rv != SECSuccess) break;
  355             pubk->u.ec.ecParams.arena = arena;
  356             /* Copy the rest of the params */
  357             rv = EC_CopyParams(arena, &(pubk->u.ec.ecParams),
  358                                &(privk->u.ec.ecParams));
  359             if (rv == SECSuccess) return pubk;
  360         }
  361         break;
  362 #endif /* NSS_ENABLE_ECC */
  363         /* No Fortezza in Low Key implementations (Fortezza keys aren't
  364          * stored in our data base */
  365     default:
  366         break;
  367     }
  368 
  369     PORT_FreeArena (arena, PR_FALSE);
  370     return NULL;
  371 }

  373 NSSLOWKEYPrivateKey *
  374 nsslowkey_CopyPrivateKey(NSSLOWKEYPrivateKey *privKey)
  375 {
  376     NSSLOWKEYPrivateKey *returnKey = NULL;
  377     SECStatus rv = SECFailure;
  378     PLArenaPool *poolp;
  379 
  380     if(!privKey) {
  381         return NULL;
  382     }
  383 
  384     poolp = PORT_NewArena(DER_DEFAULT_CHUNKSIZE);
  385     if(!poolp) {
  386         return NULL;
  387     }
  388 
  389     returnKey = (NSSLOWKEYPrivateKey*)PORT_ArenaZAlloc(poolp, sizeof(NSSLOWKEYPrivateKey));
  390     if(!returnKey) {
  391         rv = SECFailure;
  392         goto loser;
  393     }
  394 
  395     returnKey->keyType = privKey->keyType;
  396     returnKey->arena = poolp;
  397 
  398     switch(privKey->keyType) {
  399         case NSSLOWKEYRSAKey:
  400             rv = SECITEM_CopyItem(poolp, &(returnKey->u.rsa.modulus), 
  401                                             &(privKey->u.rsa.modulus));
  402             if(rv != SECSuccess) break;
  403             rv = SECITEM_CopyItem(poolp, &(returnKey->u.rsa.version), 
  404                                             &(privKey->u.rsa.version));
  405             if(rv != SECSuccess) break;
  406             rv = SECITEM_CopyItem(poolp, &(returnKey->u.rsa.publicExponent), 
  407                                             &(privKey->u.rsa.publicExponent));
  408             if(rv != SECSuccess) break;
  409             rv = SECITEM_CopyItem(poolp, &(returnKey->u.rsa.privateExponent), 
  410                                             &(privKey->u.rsa.privateExponent));
  411             if(rv != SECSuccess) break;
  412             rv = SECITEM_CopyItem(poolp, &(returnKey->u.rsa.prime1), 
  413                                             &(privKey->u.rsa.prime1));
  414             if(rv != SECSuccess) break;
  415             rv = SECITEM_CopyItem(poolp, &(returnKey->u.rsa.prime2), 
  416                                             &(privKey->u.rsa.prime2));
  417             if(rv != SECSuccess) break;
  418             rv = SECITEM_CopyItem(poolp, &(returnKey->u.rsa.exponent1), 
  419                                             &(privKey->u.rsa.exponent1));
  420             if(rv != SECSuccess) break;
  421             rv = SECITEM_CopyItem(poolp, &(returnKey->u.rsa.exponent2), 
  422                                             &(privKey->u.rsa.exponent2));
  423             if(rv != SECSuccess) break;
  424             rv = SECITEM_CopyItem(poolp, &(returnKey->u.rsa.coefficient), 
  425                                             &(privKey->u.rsa.coefficient));
  426             if(rv != SECSuccess) break;
  427             break;
  428         case NSSLOWKEYDSAKey:
  429             rv = SECITEM_CopyItem(poolp, &(returnKey->u.dsa.publicValue),
  430                                             &(privKey->u.dsa.publicValue));
  431             if(rv != SECSuccess) break;
  432             rv = SECITEM_CopyItem(poolp, &(returnKey->u.dsa.privateValue),
  433                                             &(privKey->u.dsa.privateValue));
  434             if(rv != SECSuccess) break;
  435             returnKey->u.dsa.params.arena = poolp;
  436             rv = SECITEM_CopyItem(poolp, &(returnKey->u.dsa.params.prime),
  437                                         &(privKey->u.dsa.params.prime));
  438             if(rv != SECSuccess) break;
  439             rv = SECITEM_CopyItem(poolp, &(returnKey->u.dsa.params.subPrime),
  440                                         &(privKey->u.dsa.params.subPrime));
  441             if(rv != SECSuccess) break;
  442             rv = SECITEM_CopyItem(poolp, &(returnKey->u.dsa.params.base),
  443                                         &(privKey->u.dsa.params.base));
  444             if(rv != SECSuccess) break;
  445             break;
  446         case NSSLOWKEYDHKey:
  447             rv = SECITEM_CopyItem(poolp, &(returnKey->u.dh.publicValue),
  448                                             &(privKey->u.dh.publicValue));
  449             if(rv != SECSuccess) break;
  450             rv = SECITEM_CopyItem(poolp, &(returnKey->u.dh.privateValue),
  451                                             &(privKey->u.dh.privateValue));
  452             if(rv != SECSuccess) break;
  453             returnKey->u.dsa.params.arena = poolp;
  454             rv = SECITEM_CopyItem(poolp, &(returnKey->u.dh.prime),
  455                                         &(privKey->u.dh.prime));
  456             if(rv != SECSuccess) break;
  457             rv = SECITEM_CopyItem(poolp, &(returnKey->u.dh.base),
  458                                         &(privKey->u.dh.base));
  459             if(rv != SECSuccess) break;
  460             break;
  461 #ifdef NSS_ENABLE_ECC
  462         case NSSLOWKEYECKey:
  463             rv = SECITEM_CopyItem(poolp, &(returnKey->u.ec.version),
  464                                             &(privKey->u.ec.version));
  465             if(rv != SECSuccess) break;
  466             rv = SECITEM_CopyItem(poolp, &(returnKey->u.ec.publicValue),
  467                                             &(privKey->u.ec.publicValue));
  468             if(rv != SECSuccess) break;
  469             rv = SECITEM_CopyItem(poolp, &(returnKey->u.ec.privateValue),
  470                                             &(privKey->u.ec.privateValue));
  471             if(rv != SECSuccess) break;
  472             returnKey->u.ec.ecParams.arena = poolp;
  473             /* Copy the rest of the params */
  474             rv = EC_CopyParams(poolp, &(returnKey->u.ec.ecParams),
  475                                &(privKey->u.ec.ecParams));
  476             if (rv != SECSuccess) break;
  477             break;
  478 #endif /* NSS_ENABLE_ECC */
  479         default:
  480             rv = SECFailure;
  481     }
  482 
  483 loser:
  484 
  485     if(rv != SECSuccess) {
  486         PORT_FreeArena(poolp, PR_TRUE);
  487         returnKey = NULL;
  488     }
  489 
  490     return returnKey;
