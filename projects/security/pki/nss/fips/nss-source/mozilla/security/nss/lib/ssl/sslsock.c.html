  423 #endif

  431     ssl_GetRecvBufLock(ss);

  433     ssl_GetXmitBufLock(ss);
  434     ssl_GetSpecWriteLock(ss);
  435 
  436 #ifdef DEBUG
  437     fs = &lSock;
  438     *fs = *ss;                                /* Copy the old socket structure, */
  439     PORT_Memset(ss, 0x1f, sizeof *ss);  /* then blast the old struct ASAP. */
  440 #else
  441 #define fs ss
  442 #endif
  443 
  444     ssl_DestroySocketContents(fs);
  445 
  446     /* Release all the locks acquired above.  */
  447     SSL_UNLOCK_READER(fs);
  448     SSL_UNLOCK_WRITER(fs);
  449     ssl_Release1stHandshakeLock(fs);
  450     ssl_ReleaseRecvBufLock(fs);
  451     ssl_ReleaseSSL3HandshakeLock(fs);
  452     ssl_ReleaseXmitBufLock(fs);
  453     ssl_ReleaseSpecWriteLock(fs);
  454 
  455     ssl_DestroyLocks(fs);
  456 
  457     PORT_Free(ss);        /* free the caller's copy, not ours. */
  458     return;

 1515 static PRInt16 PR_CALLBACK
 1516 ssl_Poll(PRFileDesc *fd, PRInt16 how_flags, PRInt16 *p_out_flags)
 1517 {
 1518     sslSocket *ss;
 1519     PRInt16    new_flags = how_flags;        /* should select on these flags. */
 1520     PRNetAddr  addr;
 1521 
 1522     *p_out_flags = 0;
 1523     ss = ssl_GetPrivate(fd);
 1524     if (!ss) {
 1525         SSL_DBG(("%d: SSL[%d]: bad socket d/zlib/infback.c.dep.html#in">ind/zlib/minigzip.c.dep.html#in">[1] SSL_Poll",
 1526                  SSL_GETPID[4](), fd));
 1527         return 0;        /* don't poll on this socket */
 1528     }
 1529 
 1530     if (ss->opt.useSecurity && 
 1531         ss->handshaking != sslHandshakingUndetermined &&
 1532         !ss->firstHsDone &&
 1533         (how_flags & PR_POLL_RW)) {
 1534         if (!ss->TCPconnected) {
 1535             ss->TCPconnected = (PR_SUCCESS == ssl_DefGetpeername(ss, &addr));
 1536         }
 1537         /* If it's not connected, then presumably the application is polling
 1538         ** on read or write appropriately, so don't change it. 
 1539         */
 1540         if (ss->TCPconnected) {
 1541             if (!ss->handshakeBegun) {
 1542                 /* If the handshake has not begun, poll on read or write 
 1543                 ** based on the local application's role in the handshake,
 1544                 ** not based on what the application requested.
 1545                 */
 1546                 new_flags &= ~PR_POLL_RW;
 1547                 if (ss->handshaking == sslHandshakingAsClient) {
 1548                     new_flags |= PR_POLL_WRITE;
 1549                 } else { /* handshaking as server */
 1550                     new_flags |= PR_POLL_READ;
 1551                 }
 1552             } else 
 1553             /* First handshake is in progress */
 1554             if (ss->lastWriteBlocked) {
 1555                 if (new_flags & PR_POLL_READ) {
 1556                     /* The caller is waiting for data to be received, 
 1557                     ** but the initial handshake is blocked on write, or the 
 1558                     ** client's first handshake record has not been written.
 1559                     ** The code should select on write, not read.
 1560                     */
 1561                     new_flags ^=  PR_POLL_READ;           /* don't select on read. */
 1562                     new_flags |=  PR_POLL_WRITE;   /* do    select on write. */
 1563                 }
 1564             } else if (new_flags & PR_POLL_WRITE) {
 1565                     /* The caller is trying to write, but the handshake is 
 1566                     ** blocked waiting for data to read, and the first 
 1567                     ** handshake has been sent.  so do NOT to poll on write.
 1568                     */
 1569                     new_flags ^=  PR_POLL_WRITE;   /* don't select on write. */
 1570                     new_flags |=  PR_POLL_READ;           /* do    select on read. */
 1571             }
 1572         }
 1573     } else if ((new_flags & PR_POLL_READ) && (SSL_DataPending(fd) > 0)) {
 1574         *p_out_flags = PR_POLL_READ;        /* it's ready already. */
 1575         return new_flags;
 1576     } else if ((ss->lastWriteBlocked) && (how_flags & PR_POLL_READ) &&
 1577                (ss->pendingBuf.len[1][2][3][6][9][10][13] != 0)) { /* write data waiting to be sent */
 1578         new_flags |=  PR_POLL_WRITE;   /* also select on write. */
 1579     } 
 1580     if (new_flags && (fd->lower->methods->poll != NULL)) {
 1581         PRInt16    lower_out_flags = 0;
 1582         PRInt16    lower_new_flags;
 1583         lower_new_flags = fd->lower->methods->poll(fd->lower, new_flags, 
 1584                                                    &lower_out_flags);
 1585         if ((lower_new_flags & lower_out_flags) && (how_flags != new_flags)) {
 1586             PRInt16 out_flags = lower_out_flags & ~PR_POLL_RW;
 1587             if (lower_out_flags & PR_POLL_READ) 
 1588                 out_flags |= PR_POLL_WRITE;
 1589             if (lower_out_flags & PR_POLL_WRITE) 
 1590                 out_flags |= PR_POLL_READ;
 1591             *p_out_flags = out_flags;
 1592             new_flags = how_flags;
 1593         } else {
 1594             *p_out_flags = lower_out_flags;
 1595             new_flags    = lower_new_flags;
 1596         }
 1597     }
 1598 
 1599     return new_flags;
 1600 }
