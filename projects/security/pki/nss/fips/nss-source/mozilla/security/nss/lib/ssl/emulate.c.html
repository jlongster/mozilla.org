   54 #define AMASK 7        /* mask for alignment of PRNetAddr */
   55 
   56 /*
   57  * _PR_EmulateAcceptRead
   58  *
   59  *  Accept an incoming connection on sd, set *nd to point to the
   60  *  newly accepted socket, read 'amount' bytes from the accepted
   61  *  socket.
   62  *
   63  *  buf is a buffer of length = amount + (2 * sizeof(PRNetAddr)) + 32
   64  *  *raddr points to the PRNetAddr of the accepted connection upon
   65  *  return
   66  *
   67  *  return number of bytes read or -1 on error
   68  *
   69  */
   70 PRInt32 
   71 ssl_EmulateAcceptRead(        PRFileDesc *   sd, 
   72                         PRFileDesc **  nd,
   73                         PRNetAddr **   raddr, 
   74                         void *         buf, 
   75                         PRInt32        amount, 
   76                         PRIntervalTime timeout)
   77 {
   78     PRFileDesc *   newsockfd;
   79     PRInt32        rv;
   80     PRNetAddr      remote;
   81 
   82     if (!(newsockfd = PR_Accept(sd, &remote, PR_INTERVAL_NO_TIMEOUT))) {
   83         return -1;
   84     }
   85 
   86     rv = PR_Recv(newsockfd, buf, amount, 0, timeout);
   87     if (rv >= 0) {
   88         ptrdiff_t pNetAddr = (((ptrdiff_t)buf) + amount + AMASK) & ~AMASK;
   89 
   90         *nd = newsockfd;
   91         *raddr = (PRNetAddr *)pNetAddr;
   92         memcpy((void *)pNetAddr, &remote, sizeof(PRNetAddr));
   93         return rv;
   94     }
   95 
   96     PR_Close(newsockfd);
   97     return -1;
   98 }

  102 /*
  103  * _PR_EmulateTransmitFile
  104  *
  105  *  Send file fd across socket sd. If headers is non-NULL, 'hlen'
  106  *  bytes of headers is sent before sending the file.
  107  *
  108  *  PR_TRANSMITFILE_CLOSE_SOCKET flag - close socket after sending file
  109  *
  110  *  return number of bytes sent or -1 on error
  111  *
  112  */
  113 #define _TRANSMITFILE_BUFSIZE        (16 * 1024)
  114 
  115 PRInt32 
  116 ssl_EmulateTransmitFile(    PRFileDesc *        sd, 
  117                             PRFileDesc *        fd,
  118                             const void *        headers, 
  119                             PRInt32             hlen, 
  120                             PRTransmitFileFlags flags,
  121                             PRIntervalTime      timeout)
  122 {
  123     char *  buf         = NULL;
  124     PRInt32 count         = 0;
  125     PRInt32 rlen;
  126     PRInt32 rv;
  127 
  128     buf = PR_MALLOC(_TRANSMITFILE_BUFSIZE);
  129     if (buf == NULL) {
  130         PR_SetError(PR_OUT_OF_MEMORY_ERROR, 0);
  131         return -1;
  132     }
  133 
  134     /*
  135      * send headers, first
  136      */
  137     while (hlen) {
  138         rv =  PR_Send(sd, headers, hlen, 0, timeout);
  139         if (rv < 0) {
  140             /* PR_Send() has invoked PR_SetError(). */
  141             rv = -1;
  142             goto done;
  143         } 
  144         count += rv;
  145         headers = (const void*) ((const char*)headers + rv);
  146         hlen -= rv;
  147     }
  148     /*
  149      * send file, next
  150      */
  151     while ((rlen = PR_Read(fd, buf, _TRANSMITFILE_BUFSIZE)) > 0) {
  152         while (rlen) {
  153             char *bufptr = buf;
  154 
  155             rv =  PR_Send(sd, bufptr, rlen,0,PR_INTERVAL_NO_TIMEOUT);
  156             if (rv < 0) {
  157                 /* PR_Send() has invoked PR_SetError(). */
  158                 rv = -1;
  159                 goto done;
  160             } 
  161             count += rv;
  162             bufptr = ((char*)bufptr + rv);
  163             rlen -= rv;
  164         }
  165     }
  166     if (rlen == 0) {
  167         /*
  168          * end-of-file
  169          */
  170         if (flags & PR_TRANSMITFILE_CLOSE_SOCKET)
  171             PR_Close(sd);
  172         rv = count;
  173     } else {
  174         PR_ASSERT(rlen < 0);
  175         /* PR_Read() has invoked PR_SetError(). */
  176         rv = -1;
  177     }
  178 
  179 done:
  180     if (buf)
  181         PR_DELETE(buf);
  182     return rv;
  183 }

  188 /*
  189  * _PR_UnixTransmitFile
  190  *
  191  *  Send file fd across socket sd. If headers is non-NULL, 'hlen'
  192  *  bytes of headers is sent before sending the file.
  193  *
  194  *  PR_TRANSMITFILE_CLOSE_SOCKET flag - close socket after sending file
  195  *
  196  *  return number of bytes sent or -1 on error
  197  *
  198  */
  199 
  200 PRInt32 
  201 ssl_EmulateTransmitFile(    PRFileDesc *        sd, 
  202                             PRFileDesc *        fd,
  203                             const void *        headers, 
  204                             PRInt32             hlen, 
  205                             PRTransmitFileFlags flags,
  206                             PRIntervalTime      timeout)
  207 {
  208     void *            addr = NULL;
  209     PRFileMap *       mapHandle = NULL;
  210     PRInt32           count     = 0;
  211     PRInt32           index     = 0;
  212     PRInt32           len        = 0;
  213     PRInt32           rv;
  214     struct PRFileInfo info;
  215     struct PRIOVec    iov[2];
  216 
  217     /* Get file size */
  218     if (PR_SUCCESS != PR_GetOpenFileInfo(fd, &info)) {
  219         count = -1;
  220         goto done;
  221     }
  222     if (hlen) {
  223         iov[index].iov_base = (char *) headers;
  224         iov[index].iov_len  = hlen;
  225         index++;
  226     }
  227     if (info.size > 0) {
  228         mapHandle = PR_CreateFileMap(fd, info.size, PR_PROT_READONLY);
  229         if (mapHandle == NULL) {
  230             count = -1;
  231             goto done;
  232         }
  233         /*
  234          * If the file is large, mmap and send the file in chunks so as
  235          * to not consume too much virtual address space
  236          */
  237         len = PR_MIN(info.size , TRANSMITFILE_MMAP_CHUNK );
  238         /*
  239          * Map in (part of) file. Take care of zero-length files.
  240          */
  241         if (len) {
  242             addr = PR_MemMap(mapHandle, 0, len);
  243             if (addr == NULL) {
  244                 count = -1;
  245                 goto done;
  246             }
  247         }
  248         iov[index].iov_base = (char*)addr;
  249         iov[index].iov_len = len;
  250         index++;
  251     }
  252     if (!index)
  253             goto done;
  254     rv = PR_Writev(sd, iov, index, timeout);
  255     if (len) {
  256         PR_MemUnmap(addr, len);
  257     }
  258     if (rv >= 0) {
  259         PR_ASSERT(rv == hlen + len);
  260         info.size -= len;
  261         count     += rv;
  262     } else {
  263         count = -1;
  264         goto done;
  265     }
  266     /*
  267      * send remaining bytes of the file, if any
  268      */
  269     len = PR_MIN(info.size , TRANSMITFILE_MMAP_CHUNK );
  270     while (len > 0) {
  271         /*
  272          * Map in (part of) file
  273          */
  274         PR_ASSERT((count - hlen) % TRANSMITFILE_MMAP_CHUNK == 0);
  275         addr = PR_MemMap(mapHandle, count - hlen, len);
  276         if (addr == NULL) {
  277             count = -1;
  278             goto done;
  279         }
  280         rv =  PR_Send(sd, addr, len, 0, timeout);
  281         PR_MemUnmap(addr, len);
  282         if (rv >= 0) {
  283             PR_ASSERT(rv == len);
  284             info.size -= rv;
  285             count     += rv;
  286             len = PR_MIN(info.size , TRANSMITFILE_MMAP_CHUNK );
  287         } else {
  288             count = -1;
  289             goto done;
  290         }
  291     }
  292 done:
  293     if ((count >= 0) && (flags & PR_TRANSMITFILE_CLOSE_SOCKET))
  294         PR_Close(sd);
  295     if (mapHandle != NULL)
  296             PR_CloseFileMap(mapHandle);
  297     return count;
  298 }
  299 #endif  /* XP_UNIX || WIN32 || XP_BEOS */

  305 /*
  306  * _PR_EmulateSendFile
  307  *
  308  *  Send file sfd->fd across socket sd. The header and trailer buffers
  309  *  specified in the 'sfd' argument are sent before and after the file,
  310  *  respectively.
  311  *
  312  *  PR_TRANSMITFILE_CLOSE_SOCKET flag - close socket after sending file
  313  *
  314  *  return number of bytes sent or -1 on error
  315  *
  316  */
  317 
  318 PRInt32 
  319 ssl_EmulateSendFile(PRFileDesc *sd, PRSendFileData *sfd,
  320                     PRTransmitFileFlags flags, PRIntervalTime timeout)
  321 {
  322     char *       buf = NULL;
  323     const void * buffer;
  324     PRInt32      rv;
  325     PRInt32      count = 0;
  326     PRInt32      rlen;
  327     PRInt32      buflen;
  328     PRInt32      sendbytes;
  329     PRInt32      readbytes;
  330 
  331 #define _SENDFILE_BUFSIZE   (16 * 1024)
  332 
  333     buf = (char*)PR_MALLOC(_SENDFILE_BUFSIZE);
  334     if (buf == NULL) {
  335         PR_SetError(PR_OUT_OF_MEMORY_ERROR, 0);
  336         return -1;
  337     }
  338 
  339     /*
  340      * send header, first
  341      */
  342     buflen = sfd->hlen;
  343     buffer = sfd->header;
  344     while (buflen) {
  345         rv =  PR_Send(sd, buffer, buflen, 0, timeout);
  346         if (rv < 0) {
  347             /* PR_Send() has invoked PR_SetError(). */
  348             rv = -1;
  349             goto done;
  350         } else {
  351             count += rv;
  352             buffer = (const void*) ((const char*)buffer + rv);
  353             buflen -= rv;
  354         }
  355     }
  356     /*
  357      * send file, next
  358      */
  359 
  360     if (PR_Seek(sfd->fd, sfd->file_offset, PR_SEEK_SET) < 0) {
  361         rv = -1;
  362         goto done;
  363     }
  364     sendbytes = sfd->file_nbytes;
  365     if (sendbytes == 0) {
  366         /* send entire file */
  367         while ((rlen = PR_Read(sfd->fd, buf, _SENDFILE_BUFSIZE)) > 0) {
  368             while (rlen) {
  369                 char *bufptr = buf;
  370 
  371                 rv =  PR_Send(sd, bufptr, rlen, 0, timeout);
  372                 if (rv < 0) {
  373                     /* PR_Send() has invoked PR_SetError(). */
  374                     rv = -1;
  375                     goto done;
  376                 } else {
  377                     count += rv;
  378                     bufptr = ((char*)bufptr + rv);
  379                     rlen -= rv;
  380                 }
  381             }
  382         }
  383         if (rlen < 0) {
  384             /* PR_Read() has invoked PR_SetError(). */
  385             rv = -1;
  386             goto done;
  387         }
  388     } else {
  389         readbytes = PR_MIN(sendbytes, _SENDFILE_BUFSIZE);
  390         while (readbytes && ((rlen = PR_Read(sfd->fd, buf, readbytes)) > 0)) {
  391             while (rlen) {
  392                 char *bufptr = buf;
  393 
  394                 rv =  PR_Send(sd, bufptr, rlen, 0, timeout);
  395                 if (rv < 0) {
  396                     /* PR_Send() has invoked PR_SetError(). */
  397                     rv = -1;
  398                     goto done;
  399                 } else {
  400                     count += rv;
  401                     sendbytes -= rv;
  402                     bufptr = ((char*)bufptr + rv);
  403                     rlen -= rv;
  404                 }
  405             }
  406             readbytes = PR_MIN(sendbytes, _SENDFILE_BUFSIZE);
  407         }
  408         if (rlen < 0) {
  409             /* PR_Read() has invoked PR_SetError(). */
  410             rv = -1;
  411             goto done;
  412         } else if (sendbytes != 0) {
  413             /*
  414              * there are fewer bytes in file to send than specified
  415              */
  416             PR_SetError(PR_INVALID_ARGUMENT_ERROR, 0);
  417             rv = -1;
  418             goto done;
  419         }
  420     }
  421     /*
  422      * send trailer, last
  423      */
  424     buflen = sfd->tlen;
  425     buffer = sfd->trailer;
  426     while (buflen) {
  427         rv =  PR_Send(sd, buffer, buflen, 0, timeout);
  428         if (rv < 0) {
  429             /* PR_Send() has invoked PR_SetError(). */
  430             rv = -1;
  431             goto done;
  432         } else {
  433             count += rv;
  434             buffer = (const void*) ((const char*)buffer + rv);
  435             buflen -= rv;
  436         }
  437     }
  438     rv = count;
  439 
  440 done:
  441     if (buf)
  442         PR_DELETE(buf);
  443     if ((rv >= 0) && (flags & PR_TRANSMITFILE_CLOSE_SOCKET))
  444         PR_Close(sd);
  445     return rv;
  446 }

  450 /*
  451  * _PR_UnixSendFile
  452  *
  453  *    Send file sfd->fd across socket sd. If header/trailer are specified
  454  *    they are sent before and after the file, respectively.
  455  *
  456  *    PR_TRANSMITFILE_CLOSE_SOCKET flag - close socket after sending file
  457  *
  458  *    return number of bytes sent or -1 on error
  459  *
  460  */
  461 #define SENDFILE_MMAP_CHUNK    (256 * 1024)
  462 
  463 PRInt32 
  464 ssl_EmulateSendFile(PRFileDesc *sd, PRSendFileData *sfd,
  465                     PRTransmitFileFlags flags, PRIntervalTime timeout)
  466 {
  467     void *            addr = NULL;
  468     PRFileMap *       mapHandle          = NULL;
  469     PRInt32           count                 = 0;
  470     PRInt32           file_bytes;
  471     PRInt32           index                 = 0;
  472     PRInt32           len;
  473     PRInt32           rv;
  474     PRUint32          addr_offset;
  475     PRUint32          file_mmap_offset;
  476     PRUint32          mmap_len;
  477     PRUint32          pagesize;
  478     struct PRFileInfo info;
  479     struct PRIOVec    iov[3];
  480 
  481     /* Get file size */
  482     if (PR_SUCCESS != PR_GetOpenFileInfo(sfd->fd, &info)) {
  483         count = -1;
  484         goto done;
  485     }
  486     if (sfd->file_nbytes && 
  487                 (info.size < (sfd->file_offset + sfd->file_nbytes))) {
  488         /*
  489          * there are fewer bytes in file to send than specified
  490          */
  491         PR_SetError(PR_INVALID_ARGUMENT_ERROR, 0);
  492         count = -1;
  493         goto done;
  494     }
  495     if (sfd->file_nbytes)
  496         file_bytes = sfd->file_nbytes;
  497     else
  498         file_bytes = info.size - sfd->file_offset;
  499 
  500 #if defined(WIN32)
  501     {
  502         SYSTEM_INFO sysinfo;
  503         GetSystemInfo(&sysinfo);
  504         pagesize = sysinfo.dwAllocationGranularity;
  505     }
  506 #else
  507     pagesize = PR_GetPageSize();
  508 #endif
  509     /*
  510      * If the file is large, mmap and send the file in chunks so as
  511      * to not consume too much virtual address space
  512      */
  513     if (!sfd->file_offset || !(sfd->file_offset & (pagesize - 1))) {
  514         /*
  515          * case 1: page-aligned file offset
  516          */
  517         mmap_len = PR_MIN(file_bytes, SENDFILE_MMAP_CHUNK);
  518         len      = mmap_len;
  519         file_mmap_offset = sfd->file_offset;
  520         addr_offset = 0;
  521     } else {
  522         /*
  523          * case 2: non page-aligned file offset
  524          */
  525         /* find previous page boundary */
  526         file_mmap_offset = (sfd->file_offset & ~(pagesize - 1));
  527 
  528         /* number of initial bytes to skip in mmap'd segment */
  529         addr_offset = sfd->file_offset - file_mmap_offset;
  530         PR_ASSERT(addr_offset > 0);
  531         mmap_len = PR_MIN(file_bytes + addr_offset, SENDFILE_MMAP_CHUNK);
  532         len      = mmap_len - addr_offset;
  533     }
  534     /*
  535      * OK I've convinced myself that length has to be possitive (file_bytes is 
  536      * negative or  SENDFILE_MMAP_CHUNK is less than pagesize). Just assert 
  537      * that this is the case so we catch problems in debug builds.
  538      */
  539     PR_ASSERT(len >= 0);
  540 
  541     /*
  542      * Map in (part of) file. Take care of zero-length files.
  543      */
  544     if (len > 0) {
  545         mapHandle = PR_CreateFileMap(sfd->fd, info.size, PR_PROT_READONLY);
  546         if (!mapHandle) {
  547             count = -1;
  548             goto done;
  549         }
  550         addr = PR_MemMap(mapHandle, file_mmap_offset, mmap_len);
  551         if (!addr) {
  552             count = -1;
  553             goto done;
  554         }
  555     }
  556     /*
  557      * send headers, first, followed by the file
  558      */
  559     if (sfd->hlen) {
  560         iov[index].iov_base = (char *) sfd->header;
  561         iov[index].iov_len  = sfd->hlen;
  562         index++;
  563     }
  564     if (len) {
  565         iov[index].iov_base = (char*)addr + addr_offset;
  566         iov[index].iov_len  = len;
  567         index++;
  568     }
  569     if ((file_bytes == len) && (sfd->tlen)) {
  570         /*
  571          * all file data is mapped in; send the trailer too
  572          */
  573         iov[index].iov_base = (char *) sfd->trailer;
  574         iov[index].iov_len  = sfd->tlen;
  575         index++;
  576     }
  577     rv = PR_Writev(sd, iov, index, timeout);
  578     if (len)
  579         PR_MemUnmap(addr, mmap_len);
  580     if (rv < 0) {
  581         count = -1;
  582         goto done;
  583     }
  584 
  585     PR_ASSERT(rv == sfd->hlen + len + ((len == file_bytes) ? sfd->tlen : 0));
  586 
  587     file_bytes -= len;
  588     count      += rv;
  589     if (!file_bytes)    /* header, file and trailer are sent */
  590         goto done;
  591 
  592     /*
  593      * send remaining bytes of the file, if any
  594      */
  595     len = PR_MIN(file_bytes, SENDFILE_MMAP_CHUNK);
  596     while (len > 0) {
  597         /*
  598          * Map in (part of) file
  599          */
  600         file_mmap_offset = sfd->file_offset + count - sfd->hlen;
  601         PR_ASSERT((file_mmap_offset % pagesize) == 0);
  602 
  603         addr = PR_MemMap(mapHandle, file_mmap_offset, len);
  604         if (!addr) {
  605             count = -1;
  606             goto done;
  607         }
  608         rv = PR_Send(sd, addr, len, 0, timeout);
  609         PR_MemUnmap(addr, len);
  610         if (rv < 0) {
  611             count = -1;
  612             goto done;
  613         }
  614 
  615         PR_ASSERT(rv == len);
  616         file_bytes -= rv;
  617         count      += rv;
  618         len = PR_MIN(file_bytes, SENDFILE_MMAP_CHUNK);
  619     }
  620     PR_ASSERT(0 == file_bytes);
  621     if (sfd->tlen) {
  622         rv =  PR_Send(sd, sfd->trailer, sfd->tlen, 0, timeout);
  623         if (rv >= 0) {
  624             PR_ASSERT(rv == sfd->tlen);
  625             count += rv;
  626         } else
  627             count = -1;
  628     }
  629 done:
  630     if (mapHandle)
  631             PR_CloseFileMap(mapHandle);
  632     if ((count >= 0) && (flags & PR_TRANSMITFILE_CLOSE_SOCKET))
  633         PR_Close(sd);
  634     return count;
  635 }
