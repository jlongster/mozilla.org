   47 typedef unsigned char BYTE;
   48 typedef unsigned int  HALF;
   49 
   50 #define HALF[1]PTR(x) ((HALF[1] *)(x))
   51 #define SHORTPTR(x) ((unsigned short *)(x))
   52 #define BYTE[1]PTR(x) ((BYTE[1] *)(x))
   53 
   54 typedef enum {
   55     DES_ENCRYPT = 0x5555,
   56     DES_DECRYPT = 0xAAAA
   57 } DESDirection;
   58 
   59 typedef void DESFunc(struct DESContextStr[1] *cx, BYTE[1] *out, const BYTE[1] *in, 
   60                      unsigned int len);
   61 
   62 struct DESContextStr {
   63     /* key schedule, 16 internal keys, each with 8 6-bit parts */
   64     HALF[1] ks0 [32];
   65     HALF[1] ks1 [32];
   66     HALF[1] ks2 [32];
   67     HALF[1] iv  [2];
   68     DESDirection direction;
   69     DESFunc  *worker;
   70 };

   72 void DES_MakeSchedule( HALF[1] * ks, const BYTE[1] * key,   DESDirection direction);
   73 void DES_Do1Block(     HALF[1] * ks, const BYTE[1] * inbuf, BYTE[1] * outbuf);
