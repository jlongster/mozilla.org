  107 void *
  108 PORT_Alloc(size_t bytes)
  109 {
  110     void *rv;
  111 
  112     /* Always allocate a non-zero amount of bytes */
  113     rv = (void *)PR_Malloc(bytes ? bytes : 1);
  114     if (!rv) {
  115         ++port_allocFailures;
  116         PORT_SetError(SEC_ERROR_NO_MEMORY);
  117     }
  118     return rv;
  119 }
  120 
  121 void *
  122 PORT_Realloc(void *oldptr, size_t bytes)
  123 {
  124     void *rv;
  125 
  126     rv = (void *)PR_Realloc(oldptr, bytes);
  127     if (!rv) {
  128         ++port_allocFailures;
  129         PORT_SetError(SEC_ERROR_NO_MEMORY);
  130     }
  131     return rv;
  132 }

  193 PLArenaPool *
  194 PORT_NewArena(unsigned long chunksize)
  195 {
  196     PORTArenaPool *pool;
  197     
  198     pool = PORT_ZNew(PORTArenaPool);
  199     if (!pool) {
  200         return NULL;
  201     }
  202     pool->magic = ARENAPOOL_MAGIC;
  203     pool->lock = PZ_NewLock(nssILockArena);
  204     if (!pool->lock) {
  205         ++port_allocFailures;
  206         PORT_Free(pool);
  207         return NULL;
  208     }
  209     PL_InitArenaPool(&pool->arena, "security", chunksize, sizeof(double));
  210     return(&pool->arena);
  211 }

  213 #define MAX_SIZE 0x7fffffffUL
  214 
  215 void *
  216 PORT_ArenaAlloc(PLArenaPool *arena, size_t size)
  217 {
  218     void *p = NULL;
  219 
  220     PORTArenaPool *pool = (PORTArenaPool *)arena;
  221 
  222     if (size <= 0) {
  223         size = 1;
  224     }
  225 
  226     if (size > MAX_SIZE) {
  227         /* you lose. */
  228     } else 
  229     /* Is it one of ours?  Assume so and check the magic */
  230     if (ARENAPOOL_MAGIC == pool->magic ) {
  231         PZ_Lock(pool->lock);
  232 #ifdef THREADMARK
  233         /* Most likely one of ours.  Is there a thread id? */
  234         if (pool->marking_thread  &&
  235             pool->marking_thread != PR_GetCurrentThread() ) {
  236             /* Another thread holds a mark in this arena */
  237             PZ_Unlock(pool->lock);
  238             PORT_SetError(SEC_ERROR_NO_MEMORY);
  239             PORT_Assert[1](0);
  240             return NULL;
  241         } /* tid != null */
  242 #endif /* THREADMARK */
  243         PL_ARENA_ALLOCATE(p, arena, size);
  244         PZ_Unlock(pool->lock);
  245     } else {
  246         PL_ARENA_ALLOCATE(p, arena, size);
  247     }
  248 
  249     if (!p) {
  250         ++port_allocFailures;
  251         PORT_SetError(SEC_ERROR_NO_MEMORY);
  252     }
  253 
  254     return(p);
  255 }

  274 /*
  275  * If zero is true, zeroize the arena memory before freeing it.
  276  */
  277 void
  278 PORT_FreeArena(PLArenaPool *arena, PRBool zero)
  279 {
  280     PORTArenaPool *pool = (PORTArenaPool *)arena;
  281     PRLock *       lock = (PRLock *)0;
  282     size_t         len  = sizeof *arena;
  283     extern const PRVersionDescription * libVersionPoint(void);
  284     static const PRVersionDescription * pvd;
  285     static PRBool  doFreeArenaPool = PR_FALSE;
  286 
  287     if (ARENAPOOL_MAGIC == pool->magic ) {
  288         len  = sizeof *pool;
  289         lock = pool->lock;
  290         PZ_Lock(lock);
  291     }
  292     if (!pvd) {
  293         /* Each of NSPR's DLLs has a function libVersionPoint().
  294         ** We could do a lot of extra work to be sure we're calling the
  295         ** one in the DLL that holds PR_FreeArenaPool, but instead we
  296         ** rely on the fact that ALL NSPR DLLs in the same directory
  297         ** must be from the same release, and we call which ever one we get. 
  298         */
  299         /* no need for thread protection here */
  300         pvd = libVersionPoint();
  301         if ((pvd->vMajor > 4) || 
  302             (pvd->vMajor == 4 && pvd->vMinor > 1) ||
  303             (pvd->vMajor == 4 && pvd->vMinor == 1 && pvd->vPatch >= 1)) {
  304             const char *ev = PR_GetEnv("NSS_DISABLE_ARENA_FREE_LIST");
  305             if (!ev) doFreeArenaPool = PR_TRUE;
  306         }
  307     }
  308     if (zero) {
  309         PLArena *a;
  310         for (a = arena->first.next; a; a = a->next) {
  311             PR_ASSERT(a->base <= a->avail && a->avail <= a->limit);
  312             memset((void *)a->base, 0, a->avail - a->base);
  313         }
  314     }
  315     if (doFreeArenaPool) {
  316         PL_FreeArenaPool(arena);
  317     } else {
  318         PL_FinishArenaPool(arena);
  319     }
  320     PORT_ZFree(arena, len);
  321     if (lock) {
  322         PZ_Unlock(lock);
  323         PZ_DestroyLock(lock);
  324     }
  325 }

  327 void *
  328 PORT_ArenaGrow(PLArenaPool *arena, void *ptr, size_t oldsize, size_t newsize)
  329 {
  330     PORTArenaPool *pool = (PORTArenaPool *)arena;
  331     PORT_Assert[1](newsize >= oldsize);
  332     
  333     if (ARENAPOOL_MAGIC == pool->magic ) {
  334         PZ_Lock(pool->lock);
  335         /* Do we do a THREADMARK check here? */
  336         PL_ARENA_GROW(ptr, arena, oldsize, ( newsize - oldsize ) );
  337         PZ_Unlock(pool->lock);
  338     } else {
  339         PL_ARENA_GROW(ptr, arena, oldsize, ( newsize - oldsize ) );
  340     }
  341     
  342     return(ptr);
  343 }

  345 void *
  346 PORT_ArenaMark(PLArenaPool *arena)
  347 {
  348     void * result;
  349 
  350     PORTArenaPool *pool = (PORTArenaPool *)arena;
  351     if (ARENAPOOL_MAGIC == pool->magic ) {
  352         PZ_Lock(pool->lock);
  353 #ifdef THREADMARK
  354         {
  355           threadmark_mark *tm, **pw;
  356           PRThread * currentThread = PR_GetCurrentThread();
  357 
  358             if (! pool->marking_thread ) {
  359                 /* First mark */
  360                 pool->marking_thread = currentThread;
  361             } else if (currentThread != pool->marking_thread ) {
  362                 PZ_Unlock(pool->lock);
  363                 PORT_SetError(SEC_ERROR_NO_MEMORY);
  364                 PORT_Assert[1](0);
  365                 return NULL;
  366             }
  367 
  368             result = PL_ARENA_MARK(arena);
  369             PL_ARENA_ALLOCATE(tm, arena, sizeof(threadmark_mark));
  370             if (!tm) {
  371                 PZ_Unlock(pool->lock);
  372                 PORT_SetError(SEC_ERROR_NO_MEMORY);
  373                 return NULL;
  374             }
  375 
  376             tm->mark = result;
  377             tm->next = (threadmark_mark *)NULL;
  378 
  379             pw = &pool->first_mark;
  380             while( *pw ) {
  381                  pw = &(*pw)->next;
  382             }
  383 
  384             *pw = tm;
  385         }
  386 #else /* THREADMARK */
  387         result = PL_ARENA_MARK(arena);
  388 #endif /* THREADMARK */
  389         PZ_Unlock(pool->lock);
  390     } else {
  391         /* a "pure" NSPR arena */
  392         result = PL_ARENA_MARK(arena);
  393     }
  394     return result;
  395 }

  397 static void
  398 port_ArenaZeroAfterMark(PLArenaPool *arena, void *mark)
  399 {
  400     PLArena *a = arena->current;
  401     if (a->base <= (PRUword)mark && (PRUword)mark <= a->avail) {
  402         /* fast path: mark falls in the current arena */
  403         memset(mark, 0, a->avail - (PRUword)mark);
  404     } else {
  405         /* slow path: need to find the arena that mark falls in */
  406         for (a = arena->first.next; a; a = a->next) {
  407             PR_ASSERT(a->base <= a->avail && a->avail <= a->limit);
  408             if (a->base <= (PRUword)mark && (PRUword)mark <= a->avail) {
  409                 memset(mark, 0, a->avail - (PRUword)mark);
  410                 a = a->next;
  411                 break;
  412             }
  413         }
  414         for (; a; a = a->next) {
  415             PR_ASSERT(a->base <= a->avail && a->avail <= a->limit);
  416             memset((void *)a->base, 0, a->avail - a->base);
  417         }
  418     }
  419 }

  421 static void
  422 port_ArenaRelease(PLArenaPool *arena, void *mark, PRBool zero)
  423 {
  424     PORTArenaPool *pool = (PORTArenaPool *)arena;
  425     if (ARENAPOOL_MAGIC == pool->magic ) {
  426         PZ_Lock(pool->lock);
  427 #ifdef THREADMARK
  428         {
  429             threadmark_mark **pw, *tm;
  430 
  431             if (PR_GetCurrentThread() != pool->marking_thread ) {
  432                 PZ_Unlock(pool->lock);
  433                 PORT_SetError(SEC_ERROR_NO_MEMORY);
  434                 PORT_Assert[1](0);
  435                 return /* no error indication available */ ;
  436             }
  437 
  438             pw = &pool->first_mark;
  439             while( *pw && (mark != (*pw)->mark) ) {
  440                 pw = &(*pw)->next;
  441             }
  442 
  443             if (! *pw ) {
  444                 /* bad mark */
  445                 PZ_Unlock(pool->lock);
  446                 PORT_SetError(SEC_ERROR_NO_MEMORY);
  447                 PORT_Assert[1](0);
  448                 return /* no error indication available */ ;
  449             }
  450 
  451             tm = *pw;
  452             *pw = (threadmark_mark *)NULL;
  453 
  454             if (zero) {
  455                 port_ArenaZeroAfterMark(arena, mark);
  456             }
  457             PL_ARENA_RELEASE(arena, mark);
  458 
  459             if (! pool->first_mark ) {
  460                 pool->marking_thread = (PRThread *)NULL;
  461             }
  462         }
  463 #else /* THREADMARK */
  464         if (zero) {
  465             port_ArenaZeroAfterMark(arena, mark);
  466         }
  467         PL_ARENA_RELEASE(arena, mark);
  468 #endif /* THREADMARK */
  469         PZ_Unlock(pool->lock);
  470     } else {
  471         if (zero) {
  472             port_ArenaZeroAfterMark(arena, mark);
  473         }
  474         PL_ARENA_RELEASE(arena, mark);
  475     }
  476 }

  493 void
  494 PORT_ArenaUnmark(PLArenaPool *arena, void *mark)
  495 {
  496 #ifdef THREADMARK
  497     PORTArenaPool *pool = (PORTArenaPool *)arena;
  498     if (ARENAPOOL_MAGIC == pool->magic ) {
  499         threadmark_mark **pw, *tm;
  500 
  501         PZ_Lock(pool->lock);
  502 
  503         if (PR_GetCurrentThread() != pool->marking_thread ) {
  504             PZ_Unlock(pool->lock);
  505             PORT_SetError(SEC_ERROR_NO_MEMORY);
  506             PORT_Assert[1](0);
  507             return /* no error indication available */ ;
  508         }
  509 
  510         pw = &pool->first_mark;
  511         while( ((threadmark_mark *)NULL != *pw) && (mark != (*pw)->mark) ) {
  512             pw = &(*pw)->next;
  513         }
  514 
  515         if ((threadmark_mark *)NULL == *pw ) {
  516             /* bad mark */
  517             PZ_Unlock(pool->lock);
  518             PORT_SetError(SEC_ERROR_NO_MEMORY);
  519             PORT_Assert[1](0);
  520             return /* no error indication available */ ;
  521         }
  522 
  523         tm = *pw;
  524         *pw = (threadmark_mark *)NULL;
  525 
  526         if (! pool->first_mark ) {
  527             pool->marking_thread = (PRThread *)NULL;
  528         }
  529 
  530         PZ_Unlock(pool->lock);
  531     }
  532 #endif /* THREADMARK */
  533 }

  628 /* Portable putenv.  Creates/replaces an environment variable of the form
  629  *  envVarName=envValue
  630  */
  631 int
  632 NSS_PutEnv(const char * envVarName, const char * envValue)
  633 {
  634 #if  defined(XP_MAC) || defined(_WIN32_WCE)
  635     return SECFailure;
  636 #else
  637     SECStatus result = SECSuccess;
  638     char *    encoded;
  639     int       putEnvFailed;
  640 #ifdef _WIN32
  641     PRBool      setOK;
  642 
  643     setOK = SetEnvironmentVariable(envVarName, envValue);
  644     if (!setOK) {
  645         SET_ERROR_CODE
  646         return SECFailure;
  647     }
  648 #endif
  649 
  650     encoded = (char *)PORT_ZAlloc(strlen(envVarName) + 2 + strlen(envValue));
  651     strcpy(encoded, envVarName);
  652     strcat(encoded, "=");
  653     strcat(encoded, envValue);
  654 
  655     putEnvFailed = putenv(encoded); /* adopt. */
  656     if (putEnvFailed) {
  657         SET_ERROR_CODE
  658         result = SECFailure;
  659         PORT_Free(encoded);
  660     }
  661     return result;
  662 #endif
  663 }
