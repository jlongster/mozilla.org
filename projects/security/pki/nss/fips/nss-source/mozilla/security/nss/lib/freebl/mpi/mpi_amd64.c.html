   49 /* Presently, this is only used by the Montgomery arithmetic code. */
   50 /* c += a * b */
   51 void MPI_ASM_DECL s_mpv_mul_d_add_prop(const mp_digit *a, mp_size a_len,
   52                                        mp_digit b, mp_digit *c)
   53 {
   54   mp_digit w;
   55   mp_digit d;
   56 
   57   d = s_mpv_mul_add_vec64(c, a, a_len, b);
   58   c += a_len;
   59   while (d) {
   60     w = c[0] + d;
   61     d = (w < c[0] || w < d);
   62     *c++ = w;
   63   }
   64 }
