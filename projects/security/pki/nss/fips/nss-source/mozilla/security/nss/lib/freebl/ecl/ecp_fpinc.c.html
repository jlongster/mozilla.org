   39 /* This source file is meant to be included by other source files
   40  * (ecp_fp###.c, where ### is one of 160, 192, 224) and should not
   41  * constitute an independent compilation unit. It requires the following
   42  * preprocessor definitions be made: ECFP_BSIZE - the number of bits in
   43  * the field's prime 
   44  * ECFP_NUMDOUBLES - the number of doubles to store one
   45  * multi-precision integer in floating point 
   46 
   47 /* Adds a prefix to a given token to give a unique token name. Prefixes
   48  * with "ecfp" + ECFP_BSIZE + "_". e.g. if ECFP_BSIZE = 160, then
   49  * PREFIX(hello) = ecfp160_hello This optimization allows static function
   50  * linking and compiler loop unrolling without code duplication. */
   51 #ifndef PREFIX
   52 #define PREFIX(b) PREFIX1(ECFP_BSIZE[1][2], b)
   53 #define PREFIX1(bsize, b) PREFIX2(bsize, b)
   54 #define PREFIX2(bsize, b) ecfp ## bsize ## _ ## b
   55 #endif

   57 /* Returns true iff every double in d is 0. (If d == 0 and it is tidied,
   58  * this will be true.) */
   59 mp_err PREFIX(isZero) (const double *d) {
   60         int i;
   61 
   62         for (i = 0; i < ECFP_NUMDOUBLES[1][2]; i++) {
   63                 if (d[i] != 0)
   64                         return MP_NO;
   65         }
   66         return MP_YES;
   67 }

   69 /* Sets the multi-precision floating point number at t = 0 */
   70 void PREFIX(zero) (double *t) {
   71         int i;
   72 
   73         for (i = 0; i < ECFP_NUMDOUBLES[1][2]; i++) {
   74                 t[i] = 0;
   75         }
   76 }
   77 
   78 /* Sets the multi-precision floating point number at t = 1 */
   79 void PREFIX(one) (double *t) {
   80         int i;
   81 
   82         t[0] = 1;
   83         for (i = 1; i < ECFP_NUMDOUBLES[1][2]; i++) {
   84                 t[i] = 0;
   85         }
   86 }

  151 /* Sets r = -p p = (x, y, z, z2, z3) r = (x, -y, z, z2, z3) Uses
  152  * Chudnovsky Jacobian coordinates. */
  153 /* TODO reverse order */
  154 void PREFIX(pt_neg_chud) (const ecfp_chud_pt * p, ecfp_chud_pt * r) {
  155         int i;
  156 
  157         PREFIX(copy) (r->x, p->x);
  158         PREFIX(copy) (r->z, p->z);
  159         PREFIX(copy) (r->z2, p->z2);
  160         PREFIX(copy) (r->z3, p->z3);
  161         for (i = 0; i < ECFP_NUMDOUBLES[1][2]; i++) {
  162                 r->y[i] = -p->y[i];
  163         }

  166 /* Computes r = x + y. Does not tidy or reduce. Any combinations of r, x,
  167  * y can point to the same data. Componentwise adds first ECFP_NUMDOUBLES
  168  * doubles of x and y and stores the result in r. */
  169 void PREFIX(addShort) (double *r, const double *x, const double *y) {
  170         int i;
  171 
  172         for (i = 0; i < ECFP_NUMDOUBLES[1][2]; i++) {
  173                 *r++ = *x++ + *y++;
  174         }
  175 }
  176 
  177 /* Computes r = x + y. Does not tidy or reduce. Any combinations of r, x,
  178  * y can point to the same data. Componentwise adds first
  179  * 2*ECFP_NUMDOUBLES doubles of x and y and stores the result in r. */
  180 void PREFIX(addLong) (double *r, const double *x, const double *y) {
  181         int i;
  182 
  183         for (i = 0; i < 2 * ECFP_NUMDOUBLES[1][2]; i++) {
  184                 *r++ = *x++ + *y++;
  185         }
  186 }
  187 
  188 /* Computes r = x - y. Does not tidy or reduce. Any combinations of r, x,
  189  * y can point to the same data. Componentwise subtracts first
  190  * ECFP_NUMDOUBLES doubles of x and y and stores the result in r. */
  191 void PREFIX(subtractShort) (double *r, const double *x, const double *y) {
  192         int i;
  193 
  194         for (i = 0; i < ECFP_NUMDOUBLES[1][2]; i++) {
  195                 *r++ = *x++ - *y++;
  196         }
  197 }
  198 
  199 /* Computes r = x - y. Does not tidy or reduce. Any combinations of r, x,
  200  * y can point to the same data. Componentwise subtracts first
  201  * 2*ECFP_NUMDOUBLES doubles of x and y and stores the result in r. */
  202 void PREFIX(subtractLong) (double *r, const double *x, const double *y) {
  203         int i;
  204 
  205         for (i = 0; i < 2 * ECFP_NUMDOUBLES[1][2]; i++) {
  206                 *r++ = *x++ - *y++;
  207         }
  208 }

  210 /* Computes r = x*y.  Both x and y should be tidied and reduced,
  211  * r must be different (point to different memory) than x and y.
  212  * Does not tidy or reduce. */
  213 void PREFIX(multiply)(double *r, const double *x, const double *y) {
  214         int i, j;
  215 
  216         for(j=0;j<ECFP_NUMDOUBLES[1][2]-1;j++) {
  217                 r[j] = x[0] * y[j];
  218                 r[j+(ECFP_NUMDOUBLES[1][2]-1)] = x[ECFP_NUMDOUBLES[1][2]-1] * y[j];
  219         }
  220         r[ECFP_NUMDOUBLES[1][2]-1] = x[0] * y[ECFP_NUMDOUBLES[1][2]-1];
  221         r[ECFP_NUMDOUBLES[1][2]-1] += x[ECFP_NUMDOUBLES[1][2]-1] * y[0];
  222         r[2*ECFP_NUMDOUBLES[1][2]-2] = x[ECFP_NUMDOUBLES[1][2]-1] * y[ECFP_NUMDOUBLES[1][2]-1];
  223         r[2*ECFP_NUMDOUBLES[1][2]-1] = 0;
  224         
  225         for(i=1;i<ECFP_NUMDOUBLES[1][2]-1;i++) {
  226                 for(j=0;j<ECFP_NUMDOUBLES[1][2];j++) {
  227                         r[i+j] += (x[i] * y[j]);
  228                 }
  229         }
  230 }

  232 /* Computes the square of x and stores the result in r.  x should be
  233  * tidied & reduced, r will be neither tidied nor reduced. 
  234  * r should point to different memory than x */
  235 void PREFIX(square) (double *r, const double *x) {
  236         PREFIX(multiply) (r, x, x);
  237 }

  239 /* Perform a point doubling in Jacobian coordinates. Input and output
  240  * should be multi-precision floating point integers. */
  241 void PREFIX(pt_dbl_jac) (const ecfp_jac_pt * dp, ecfp_jac_pt * dr,
  242                                                  const EC_group_fp * group) {
  243         double t0[2 * ECFP_NUMDOUBLES[1][2]], t1[2 * ECFP_NUMDOUBLES[1][2]],
  244                 M[2 * ECFP_NUMDOUBLES[1]], S[1][2 * ECFP_NUMDOUBLES[1]];
  245 
  246         /* Check for point at infinity */
  247         if (PREFIX(pt_is_inf_jac) (dp) == MP_YES) {
  248                 /* Set r = pt at infinity */
  249                 PREFIX(set_pt_inf_jac) (dr);
  250                 goto CLEANUP;
  251         }
  252 
  253         /* Perform typical point doubling operations */
  254 
  255         /* TODO? is it worthwhile to do optimizations for when pz = 1? */
  256 
  257         if (group->aIsM3) {
  258                 /* When a = -3, M = 3(px - pz^2)(px + pz^2) */
  259                 PREFIX(square) (t1, dp->z);
  260                 group->ecfp_reduce(t1, t1, group);        /* 2^23 since the negative
  261                                                                                          * rounding buys another bit */
  262                 PREFIX(addShort) (t0, dp->x, t1);        /* 2*2^23 */
  263                 PREFIX(subtractShort) (t1, dp->x, t1);        /* 2 * 2^23 */
  264                 PREFIX(multiply) (M, t0, t1);        /* 40 * 2^46 */
  265                 PREFIX(addLong) (t0, M, M);        /* 80 * 2^46 */
  266                 PREFIX(addLong) (M, t0, M);        /* 120 * 2^46 < 2^53 */
  267                 group->ecfp_reduce(M, M, group);
  268         } else {
  269                 /* Generic case */
  270                 /* M = 3 (px^2) + a*(pz^4) */
  271                 PREFIX(square) (t0, dp->x);
  272                 PREFIX(addLong) (M, t0, t0);
  273                 PREFIX(addLong) (t0, t0, M);        /* t0 = 3(px^2) */
  274                 PREFIX(square) (M, dp->z);
  275                 group->ecfp_reduce(M, M, group);
  276                 PREFIX(square) (t1, M);
  277                 group->ecfp_reduce(t1, t1, group);
  278                 PREFIX(multiply) (M, t1, group->curvea);        /* M = a(pz^4) */
  279                 PREFIX(addLong) (M, M, t0);
  280                 group->ecfp_reduce(M, M, group);
  281         }
  282 
  283         /* rz = 2 * py * pz */
  284         PREFIX(multiply) (t1, dp->y, dp->z);
  285         PREFIX(addLong) (t1, t1, t1);
  286         group->ecfp_reduce(dr->z, t1, group);
  287 
  288         /* t0 = 2y^2 */
  289         PREFIX(square) (t0, dp->y);
  290         group->ecfp_reduce(t0, t0, group);
  291         PREFIX(addShort) (t0, t0, t0);
  292 
  293         /* S = 4 * px * py^2 = 2 * px * t0 */
  294         PREFIX(multiply) (S[1], dp->x, t0);
  295         PREFIX(addLong) (S[1], S[1], S[1]);
  296         group->ecfp_reduce(S[1], S[1], group);
  297 
  298         /* rx = M^2 - 2 * S */
  299         PREFIX(square) (t1, M);
  300         PREFIX(subtractS[1]hort) (t1, t1, S[1]);
  301         PREFIX(subtractS[1]hort) (t1, t1, S[1]);
  302         group->ecfp_reduce(dr->x, t1, group);
  303 
  304         /* ry = M * (S - rx) - 8 * py^4 */
  305         PREFIX(square) (t1, t0);        /* t1 = 4y^4 */
  306         PREFIX(subtractS[1]hort) (S[1], S[1], dr->x);
  307         PREFIX(multiply) (t0, M, S[1]);
  308         PREFIX(subtractLong) (t0, t0, t1);
  309         PREFIX(subtractLong) (t0, t0, t1);
  310         group->ecfp_reduce(dr->y, t0, group);
  311 
  312   CLEANUP:
  313         return;
  314 }

  316 /* Perform a point addition using coordinate system Jacobian + Affine ->
  317  * Jacobian. Input and output should be multi-precision floating point
  318  * integers. */
  319 void PREFIX(pt_add_jac_aff) (const ecfp_jac_pt * p, const ecfp_aff_pt * q,
  320                                                          ecfp_jac_pt * r, const EC_group_fp * group) {
  321         /* Temporary storage */
  322         double A[2 * ECFP_NUMDOUB[1][2]LES], B[1][2][2 * ECFP_NUMDOUB[1][2]LES],
  323                 C[2 * ECFP_NUMDOUBLES[1][2]], C2[2 * ECFP_NUMDOUBLES[1][2]],
  324                 D[2 * ECFP_NUMDOUBLES[1][2]], C3[2 * ECFP_NUMDOUBLES[1][2]];
  325 
  326         /* Check for point at infinity for p or q */
  327         if (PREFIX(pt_is_inf_aff) (q) == MP_YES) {
  328                 PREFIX(copy) (r->x, p->x);
  329                 PREFIX(copy) (r->y, p->y);
  330                 PREFIX(copy) (r->z, p->z);
  331                 goto CLEANUP;
  332         } else if (PREFIX(pt_is_inf_jac) (p) == MP_YES) {
  333                 PREFIX(copy) (r->x, q->x);
  334                 PREFIX(copy) (r->y, q->y);
  335                 /* Since the affine point is not infinity, we can set r->z = 1 */
  336                 PREFIX(one) (r->z);
  337                 goto CLEANUP;
  338         }
  339 
  340         /* Calculates c = qx * pz^2 - px d = (qy * b - py) rx = d^2 - c^3 + 2
  341          * (px * c^2) ry = d * (c-rx) - py*c^3 rz = c * pz */
  342 
  343         /* A = pz^2, B = pz^3 */
  344         PREFIX(square) (A, p->z);
  345         group->ecfp_reduce(A, A, group);
  346         PREFIX(multiply) (B[1][2], A, p->z);
  347         group->ecfp_reduce(B[1][2], B[1][2], group);
  348 
  349         /* C = qx * A - px */
  350         PREFIX(multiply) (C, q->x, A);
  351         PREFIX(subtractShort) (C, C, p->x);
  352         group->ecfp_reduce(C, C, group);
  353 
  354         /* D = qy * B - py */
  355         PREFIX(multiply) (D, q->y, B[1][2]);
  356         PREFIX(subtractShort) (D, D, p->y);
  357         group->ecfp_reduce(D, D, group);
  358 
  359         /* C2 = C^2, C3 = C^3 */
  360         PREFIX(square) (C2, C);
  361         group->ecfp_reduce(C2, C2, group);
  362         PREFIX(multiply) (C3, C2, C);
  363         group->ecfp_reduce(C3, C3, group);
  364 
  365         /* rz = A = pz * C */
  366         PREFIX(multiply) (A, p->z, C);
  367         group->ecfp_reduce(r->z, A, group);
  368 
  369         /* C = px * C^2, untidied, unreduced */
  370         PREFIX(multiply) (C, p->x, C2);
  371 
  372         /* A = D^2, untidied, unreduced */
  373         PREFIX(square) (A, D);
  374 
  375         /* rx = B = A - C3 - C - C = D^2 - (C^3 + 2 * (px * C^2) */
  376         PREFIX(subtractShort) (A, A, C3);
  377         PREFIX(subtractLong) (A, A, C);
  378         PREFIX(subtractLong) (A, A, C);
  379         group->ecfp_reduce(r->x, A, group);
  380 
  381         /* B = py * C3, untidied, unreduced */
  382         PREFIX(multiply) (B[1][2], p->y, C3);
  383 
  384         /* C = px * C^2 - rx */
  385         PREFIX(subtractShort) (C, C, r->x);
  386         group->ecfp_reduce(C, C, group);
  387 
  388         /* ry = A = D * C - py * C^3 */
  389         PREFIX(multiply) (A, D, C);
  390         PREFIX(subtractLong) (A, A, B[1][2]);
  391         group->ecfp_reduce(r->y, A, group);
  392 
  393   CLEANUP:
  394         return;
  395 }

  397 /* Perform a point addition using Jacobian coordinate system. Input and
  398  * output should be multi-precision floating point integers. */
  399 void PREFIX(pt_add_jac) (const ecfp_jac_pt * p, const ecfp_jac_pt * q,
  400                                                  ecfp_jac_pt * r, const EC_group_fp * group) {
  401 
  402         /* Temporary Storage */
  403         double t0[2 * ECFP_NUMDOUBLES[1][2]], t1[2 * ECFP_NUMDOUBLES[1][2]],
  404                 U[2 * ECFP_NUMDOUBLES[1][2]], R[2 * ECFP_NUMDOUBLES[1][2]],
  405                 S[1][2 * ECFP_NUMDOUBLES[1]], H[1][2 * ECFP_NUMDOUBLES[1]],
  406                 H3[2 * ECFP_NUMDOUBLES[1][2]];
  407 
  408         /* Check for point at infinity for p, if so set r = q */
  409         if (PREFIX(pt_is_inf_jac) (p) == MP_YES) {
  410                 PREFIX(copy) (r->x, q->x);
  411                 PREFIX(copy) (r->y, q->y);
  412                 PREFIX(copy) (r->z, q->z);
  413                 goto CLEANUP;
  414         }
  415 
  416         /* Check for point at infinity for p, if so set r = q */
  417         if (PREFIX(pt_is_inf_jac) (q) == MP_YES) {
  418                 PREFIX(copy) (r->x, p->x);
  419                 PREFIX(copy) (r->y, p->y);
  420                 PREFIX(copy) (r->z, p->z);
  421                 goto CLEANUP;
  422         }
  423 
  424         /* U = px * qz^2 , S = py * qz^3 */
  425         PREFIX(square) (t0, q->z);
  426         group->ecfp_reduce(t0, t0, group);
  427         PREFIX(multiply) (U, p->x, t0);
  428         group->ecfp_reduce(U, U, group);
  429         PREFIX(multiply) (t1, t0, q->z);
  430         group->ecfp_reduce(t1, t1, group);
  431         PREFIX(multiply) (t0, p->y, t1);
  432         group->ecfp_reduce(S[1], t0, group);
  433 
  434         /* H = qx*(pz)^2 - U , R = (qy * pz^3 - S) */
  435         PREFIX(square) (t0, p->z);
  436         group->ecfp_reduce(t0, t0, group);
  437         PREFIX(multiply) (H[1], q->x, t0);
  438         PREFIX(subtractShort) (H[1], H[1], U);
  439         group->ecfp_reduce(H[1], H[1], group);
  440         PREFIX(multiply) (t1, t0, p->z);        /* t1 = pz^3 */
  441         group->ecfp_reduce(t1, t1, group);
  442         PREFIX(multiply) (t0, t1, q->y);        /* t0 = qy * pz^3 */
  443         PREFIX(subtractS[1]hort) (t0, t0, S[1]);
  444         group->ecfp_reduce(R, t0, group);
  445 
  446         /* U = U*H^2, H3 = H^3 */
  447         PREFIX(square) (t0, H[1]);
  448         group->ecfp_reduce(t0, t0, group);
  449         PREFIX(multiply) (t1, U, t0);
  450         group->ecfp_reduce(U, t1, group);
  451         PREFIX(multiply) (H[1]3, t0, H[1]);
  452         group->ecfp_reduce(H3, H3, group);
  453 
  454         /* rz = pz * qz * H */
  455         PREFIX(multiply) (t0, q->z, H[1]);
  456         group->ecfp_reduce(t0, t0, group);
  457         PREFIX(multiply) (t1, t0, p->z);
  458         group->ecfp_reduce(r->z, t1, group);
  459 
  460         /* rx = R^2 - H^3 - 2 * U */
  461         PREFIX(square) (t0, R);
  462         PREFIX(subtractShort) (t0, t0, H3);
  463         PREFIX(subtractShort) (t0, t0, U);
  464         PREFIX(subtractShort) (t0, t0, U);
  465         group->ecfp_reduce(r->x, t0, group);
  466 
  467         /* ry = R(U - rx) - S*H3 */
  468         PREFIX(subtractShort) (t1, U, r->x);
  469         PREFIX(multiply) (t0, t1, R);
  470         PREFIX(multiply) (t1, S[1], H3);
  471         PREFIX(subtractLong) (t1, t0, t1);
  472         group->ecfp_reduce(r->y, t1, group);
  473 
  474   CLEANUP:
  475         return;
  476 }

  478 /* Perform a point doubling in Modified Jacobian coordinates. Input and
  479  * output should be multi-precision floating point integers. */
  480 void PREFIX(pt_dbl_jm) (const ecfp_jm_pt * p, ecfp_jm_pt * r,
  481                                                 const EC_group_fp * group) {
  482 
  483         /* Temporary storage */
  484         double t0[2 * ECFP_NUMDOUBLES[1][2]], t1[2 * ECFP_NUMDOUBLES[1][2]],
  485                 M[2 * ECFP_NUMDOUBLES[1]], S[1][2 * ECFP_NUMDOUBLES[1]],
  486                 U[2 * ECFP_NUMDOUBLES[1][2]], T[2 * ECFP_NUMDOUBLES[1][2]];
  487 
  488         /* Check for point at infinity */
  489         if (PREFIX(pt_is_inf_jm) (p) == MP_YES) {
  490                 /* Set r = pt at infinity by setting rz = 0 */
  491                 PREFIX(set_pt_inf_jm) (r);
  492                 goto CLEANUP;
  493         }
  494 
  495         /* M = 3 (px^2) + a*(pz^4) */
  496         PREFIX(square) (t0, p->x);
  497         PREFIX(addLong) (M, t0, t0);
  498         PREFIX(addLong) (t0, t0, M);        /* t0 = 3(px^2) */
  499         PREFIX(addShort) (t0, t0, p->az4);
  500         group->ecfp_reduce(M, t0, group);
  501 
  502         /* rz = 2 * py * pz */
  503         PREFIX(multiply) (t1, p->y, p->z);
  504         PREFIX(addLong) (t1, t1, t1);
  505         group->ecfp_reduce(r->z, t1, group);
  506 
  507         /* t0 = 2y^2, U = 8y^4 */
  508         PREFIX(square) (t0, p->y);
  509         group->ecfp_reduce(t0, t0, group);
  510         PREFIX(addShort) (t0, t0, t0);
  511         PREFIX(square) (U, t0);
  512         group->ecfp_reduce(U, U, group);
  513         PREFIX(addShort) (U, U, U);
  514 
  515         /* S = 4 * px * py^2 = 2 * px * t0 */
  516         PREFIX(multiply) (S[1], p->x, t0);
  517         group->ecfp_reduce(S[1], S[1], group);
  518         PREFIX(S[1]hort">addS[1]hort) (S[1], S[1], S[1]);
  519 
  520         /* rx = M^2 - 2S */
  521         PREFIX(square) (T, M);
  522         PREFIX(subtractS[1]hort) (T, T, S[1]);
  523         PREFIX(subtractS[1]hort) (T, T, S[1]);
  524         group->ecfp_reduce(r->x, T, group);
  525 
  526         /* ry = M * (S - rx) - U */
  527         PREFIX(subtractS[1]hort) (S[1], S[1], r->x);
  528         PREFIX(multiply) (t0, M, S[1]);
  529         PREFIX(subtractShort) (t0, t0, U);
  530         group->ecfp_reduce(r->y, t0, group);
  531 
  532         /* ra*z^4 = 2*U*(apz4) */
  533         PREFIX(multiply) (t1, U, p->az4);
  534         PREFIX(addLong) (t1, t1, t1);
  535         group->ecfp_reduce(r->az4, t1, group);
  536 
  537   CLEANUP:
  538         return;

  541 /* Perform a point doubling using coordinates Affine -> Chudnovsky
  542  * Jacobian. Input and output should be multi-precision floating point
  543  * integers. */
  544 void PREFIX(pt_dbl_aff2chud) (const ecfp_aff_pt * p, ecfp_chud_pt * r,
  545                                                           const EC_group_fp * group) {
  546         double t0[2 * ECFP_NUMDOUBLES[1][2]], t1[2 * ECFP_NUMDOUBLES[1][2]],
  547                 M[2 * ECFP_NUMDOUBLES[1][2]], twoY2[2 * ECFP_NUMDOUBLES[1][2]],
  548                 S[1][2 * ECFP_NUMDOUBLES[1]];
  549 
  550         /* Check for point at infinity for p, if so set r = O */
  551         if (PREFIX(pt_is_inf_aff) (p) == MP_YES) {
  552                 PREFIX(set_pt_inf_chud) (r);
  553                 goto CLEANUP;
  554         }
  555 
  556         /* M = 3(px)^2 + a */
  557         PREFIX(square) (t0, p->x);
  558         PREFIX(addLong) (t1, t0, t0);
  559         PREFIX(addLong) (t1, t1, t0);
  560         PREFIX(addShort) (t1, t1, group->curvea);
  561         group->ecfp_reduce(M, t1, group);
  562 
  563         /* twoY2 = 2*(py)^2, S = 4(px)(py)^2 */
  564         PREFIX(square) (twoY2, p->y);
  565         PREFIX(addLong) (twoY2, twoY2, twoY2);
  566         group->ecfp_reduce(twoY2, twoY2, group);
  567         PREFIX(multiply) (S[1], p->x, twoY2);
  568         PREFIX(addLong) (S[1], S[1], S[1]);
  569         group->ecfp_reduce(S[1], S[1], group);
  570 
  571         /* rx = M^2 - 2S */
  572         PREFIX(square) (t0, M);
  573         PREFIX(subtractS[1]hort) (t0, t0, S[1]);
  574         PREFIX(subtractS[1]hort) (t0, t0, S[1]);
  575         group->ecfp_reduce(r->x, t0, group);
  576 
  577         /* ry = M(S-rx) - 8y^4 */
  578         PREFIX(subtractS[1]hort) (t0, S[1], r->x);
  579         PREFIX(multiply) (t1, t0, M);
  580         PREFIX(square) (t0, twoY2);
  581         PREFIX(subtractLong) (t1, t1, t0);
  582         PREFIX(subtractLong) (t1, t1, t0);
  583         group->ecfp_reduce(r->y, t1, group);
  584 
  585         /* rz = 2py */
  586         PREFIX(addShort) (r->z, p->y, p->y);
  587 
  588         /* rz2 = rz^2 */
  589         PREFIX(square) (t0, r->z);
  590         group->ecfp_reduce(r->z2, t0, group);
  591 
  592         /* rz3 = rz^3 */
  593         PREFIX(multiply) (t0, r->z, r->z2);
  594         group->ecfp_reduce(r->z3, t0, group);
  595 
  596   CLEANUP:
  597         return;
  598 }

  600 /* Perform a point addition using coordinates: Modified Jacobian +
  601  * Chudnovsky Jacobian -> Modified Jacobian. Input and output should be
  602  * multi-precision floating point integers. */
  603 void PREFIX(pt_add_jm_chud) (ecfp_jm_pt * p, ecfp_chud_pt * q,
  604                                                          ecfp_jm_pt * r, const EC_group_fp * group) {
  605 
  606         double t0[2 * ECFP_NUMDOUBLES[1][2]], t1[2 * ECFP_NUMDOUBLES[1][2]],
  607                 U[2 * ECFP_NUMDOUBLES[1][2]], R[2 * ECFP_NUMDOUBLES[1][2]],
  608                 S[1][2 * ECFP_NUMDOUBLES[1]], H[1][2 * ECFP_NUMDOUBLES[1]],
  609                 H3[2 * ECFP_NUMDOUBLES[1][2]], pz2[2 * ECFP_NUMDOUBLES[1][2]];
  610 
  611         /* Check for point at infinity for p, if so set r = q need to convert
  612          * from Chudnovsky form to Modified Jacobian form */
  613         if (PREFIX(pt_is_inf_jm) (p) == MP_YES) {
  614                 PREFIX(copy) (r->x, q->x);
  615                 PREFIX(copy) (r->y, q->y);
  616                 PREFIX(copy) (r->z, q->z);
  617                 PREFIX(square) (t0, q->z2);
  618                 group->ecfp_reduce(t0, t0, group);
  619                 PREFIX(multiply) (t1, t0, group->curvea);
  620                 group->ecfp_reduce(r->az4, t1, group);
  621                 goto CLEANUP;
  622         }
  623         /* Check for point at infinity for q, if so set r = p */
  624         if (PREFIX(pt_is_inf_chud) (q) == MP_YES) {
  625                 PREFIX(copy) (r->x, p->x);
  626                 PREFIX(copy) (r->y, p->y);
  627                 PREFIX(copy) (r->z, p->z);
  628                 PREFIX(copy) (r->az4, p->az4);
  629                 goto CLEANUP;
  630         }
  631 
  632         /* U = px * qz^2 */
  633         PREFIX(multiply) (U, p->x, q->z2);
  634         group->ecfp_reduce(U, U, group);
  635 
  636         /* H = qx*(pz)^2 - U */
  637         PREFIX(square) (t0, p->z);
  638         group->ecfp_reduce(pz2, t0, group);
  639         PREFIX(multiply) (H[1], pz2, q->x);
  640         group->ecfp_reduce(H[1], H[1], group);
  641         PREFIX(subtractShort) (H[1], H[1], U);
  642 
  643         /* U = U*H^2, H3 = H^3 */
  644         PREFIX(square) (t0, H[1]);
  645         group->ecfp_reduce(t0, t0, group);
  646         PREFIX(multiply) (t1, U, t0);
  647         group->ecfp_reduce(U, t1, group);
  648         PREFIX(multiply) (H[1]3, t0, H[1]);
  649         group->ecfp_reduce(H3, H3, group);
  650 
  651         /* S = py * qz^3 */
  652         PREFIX(multiply) (S[1], p->y, q->z3);
  653         group->ecfp_reduce(S[1], S[1], group);
  654 
  655         /* R = (qy * z1^3 - s) */
  656         PREFIX(multiply) (t0, pz2, p->z);
  657         group->ecfp_reduce(t0, t0, group);
  658         PREFIX(multiply) (R, t0, q->y);
  659         PREFIX(subtractS[1]hort) (R, R, S[1]);
  660         group->ecfp_reduce(R, R, group);
  661 
  662         /* rz = pz * qz * H */
  663         PREFIX(multiply) (t1, q->z, H[1]);
  664         group->ecfp_reduce(t1, t1, group);
  665         PREFIX(multiply) (t0, p->z, t1);
  666         group->ecfp_reduce(r->z, t0, group);
  667 
  668         /* rx = R^2 - H^3 - 2 * U */
  669         PREFIX(square) (t0, R);
  670         PREFIX(subtractShort) (t0, t0, H3);
  671         PREFIX(subtractShort) (t0, t0, U);
  672         PREFIX(subtractShort) (t0, t0, U);
  673         group->ecfp_reduce(r->x, t0, group);
  674 
  675         /* ry = R(U - rx) - S*H3 */
  676         PREFIX(subtractShort) (t1, U, r->x);
  677         PREFIX(multiply) (t0, t1, R);
  678         PREFIX(multiply) (t1, S[1], H3);
  679         PREFIX(subtractLong) (t1, t0, t1);
  680         group->ecfp_reduce(r->y, t1, group);
  681 
  682         if (group->aIsM3) {                        /* a == -3 */
  683                 /* a(rz^4) = -3 * ((rz^2)^2) */
  684                 PREFIX(square) (t0, r->z);
  685                 group->ecfp_reduce(t0, t0, group);
  686                 PREFIX(square) (t1, t0);
  687                 PREFIX(addLong) (t0, t1, t1);
  688                 PREFIX(addLong) (t0, t0, t1);
  689                 PREFIX(negLong) (t0, t0);
  690                 group->ecfp_reduce(r->az4, t0, group);
  691         } else {                                        /* Generic case */
  692                 /* a(rz^4) = a * ((rz^2)^2) */
  693                 PREFIX(square) (t0, r->z);
  694                 group->ecfp_reduce(t0, t0, group);
  695                 PREFIX(square) (t1, t0);
  696                 group->ecfp_reduce(t1, t1, group);
  697                 PREFIX(multiply) (t0, group->curvea, t1);
  698                 group->ecfp_reduce(r->az4, t0, group);
  699         }
  700   CLEANUP:
  701         return;
  702 }

  704 /* Perform a point addition using Chudnovsky Jacobian coordinates. Input
  705  * and output should be multi-precision floating point integers. */
  706 void PREFIX(pt_add_chud) (const ecfp_chud_pt * p, const ecfp_chud_pt * q,
  707                                                   ecfp_chud_pt * r, const EC_group_fp * group) {
  708 
  709         /* Temporary Storage */
  710         double t0[2 * ECFP_NUMDOUBLES[1][2]], t1[2 * ECFP_NUMDOUBLES[1][2]],
  711                 U[2 * ECFP_NUMDOUBLES[1][2]], R[2 * ECFP_NUMDOUBLES[1][2]],
  712                 S[1][2 * ECFP_NUMDOUBLES[1]], H[1][2 * ECFP_NUMDOUBLES[1]],
  713                 H3[2 * ECFP_NUMDOUBLES[1][2]];
  714 
  715         /* Check for point at infinity for p, if so set r = q */
  716         if (PREFIX(pt_is_inf_chud) (p) == MP_YES) {
  717                 PREFIX(copy) (r->x, q->x);
  718                 PREFIX(copy) (r->y, q->y);
  719                 PREFIX(copy) (r->z, q->z);
  720                 PREFIX(copy) (r->z2, q->z2);
  721                 PREFIX(copy) (r->z3, q->z3);
  722                 goto CLEANUP;
  723         }
  724 
  725         /* Check for point at infinity for p, if so set r = q */
  726         if (PREFIX(pt_is_inf_chud) (q) == MP_YES) {
  727                 PREFIX(copy) (r->x, p->x);
  728                 PREFIX(copy) (r->y, p->y);
  729                 PREFIX(copy) (r->z, p->z);
  730                 PREFIX(copy) (r->z2, p->z2);
  731                 PREFIX(copy) (r->z3, p->z3);
  732                 goto CLEANUP;
  733         }
  734 
  735         /* U = px * qz^2 */
  736         PREFIX(multiply) (U, p->x, q->z2);
  737         group->ecfp_reduce(U, U, group);
  738 
  739         /* H = qx*(pz)^2 - U */
  740         PREFIX(multiply) (H[1], q->x, p->z2);
  741         PREFIX(subtractShort) (H[1], H[1], U);
  742         group->ecfp_reduce(H[1], H[1], group);
  743 
  744         /* U = U*H^2, H3 = H^3 */
  745         PREFIX(square) (t0, H[1]);
  746         group->ecfp_reduce(t0, t0, group);
  747         PREFIX(multiply) (t1, U, t0);
  748         group->ecfp_reduce(U, t1, group);
  749         PREFIX(multiply) (H[1]3, t0, H[1]);
  750         group->ecfp_reduce(H3, H3, group);
  751 
  752         /* S = py * qz^3 */
  753         PREFIX(multiply) (S[1], p->y, q->z3);
  754         group->ecfp_reduce(S[1], S[1], group);
  755 
  756         /* rz = pz * qz * H */
  757         PREFIX(multiply) (t0, q->z, H[1]);
  758         group->ecfp_reduce(t0, t0, group);
  759         PREFIX(multiply) (t1, t0, p->z);
  760         group->ecfp_reduce(r->z, t1, group);
  761 
  762         /* R = (qy * z1^3 - s) */
  763         PREFIX(multiply) (t0, q->y, p->z3);
  764         PREFIX(subtractS[1]hort) (t0, t0, S[1]);
  765         group->ecfp_reduce(R, t0, group);
  766 
  767         /* rx = R^2 - H^3 - 2 * U */
  768         PREFIX(square) (t0, R);
  769         PREFIX(subtractShort) (t0, t0, H3);
  770         PREFIX(subtractShort) (t0, t0, U);
  771         PREFIX(subtractShort) (t0, t0, U);
  772         group->ecfp_reduce(r->x, t0, group);
  773 
  774         /* ry = R(U - rx) - S*H3 */
  775         PREFIX(subtractShort) (t1, U, r->x);
  776         PREFIX(multiply) (t0, t1, R);
  777         PREFIX(multiply) (t1, S[1], H3);
  778         PREFIX(subtractLong) (t1, t0, t1);
  779         group->ecfp_reduce(r->y, t1, group);
  780 
  781         /* rz2 = rz^2 */
  782         PREFIX(square) (t0, r->z);
  783         group->ecfp_reduce(r->z2, t0, group);
  784 
  785         /* rz3 = rz^3 */
  786         PREFIX(multiply) (t0, r->z, r->z2);
  787         group->ecfp_reduce(r->z3, t0, group);
  788 
  789   CLEANUP:
  790         return;
  791 }
