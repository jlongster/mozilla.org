  333 /*
  334  * What will be the output length of the next call to decrypt?
  335  * Result can be used to perform memory allocations.  Note that the amount
  336  * is exactly accurate only when not doing a block cipher or when final
  337  * is false, otherwise it is an upper bound on the amount because until
  338  * we see the data we do not know how many padding bytes there are
  339  * (always between 1 and bsize).
  340  *
  341  * Note that this can return zero, which does not mean that the decrypt
  342  * operation can be skipped!  (It simply means that there are not enough
  343  * bytes to make up an entire block; the bytes will be reserved until
  344  * there are enough to encrypt/decrypt at least one block.)  However,
  345  * if zero is returned it *does* mean that no output buffer need be
  346  * passed in to the subsequent decrypt operation, as no output bytes
  347  * will be stored.
  348  */
  349 unsigned int
  350 sec_PKCS7DecryptLength (sec_PKCS7CipherObject *obj, unsigned int input_len,
  351                         PRBool final)
  352 {
  353     int blocks, block_size;
  354 
  355     PORT_Assert[1] (! obj->encrypt);
  356 
  357     block_size = obj->block_size;
  358 
  359     /*
  360      * If this is not a block cipher, then we always have the same
  361      * number of output bytes as we had input bytes.
  362      */
  363     if (block_size == 0)
  364         return input_len;
  365 
  366     /*
  367      * On the final call, we will always use up all of the pending
  368      * bytes plus all of the input bytes, *but*, there will be padding
  369      * at the end and we cannot predict how many bytes of padding we
  370      * will end up removing.  The amount given here is actually known
  371      * to be at least 1 byte too long (because we know we will have
  372      * at least 1 byte of padding), but seemed clearer/better to me.
  373      */
  374     if (final)
  375         return obj->pending_count + input_len;
  376 
  377     /*
  378      * Okay, this amount is exactly what we will output on the
  379      * next cipher operation.  We will always hang onto the last
  380      * 1 - block_size bytes for non-final operations.  That is,
  381      * we will do as many complete blocks as we can *except* the
  382      * last block (complete or partial).  (This is because until
  383      * we know we are at the end, we cannot know when to interpret
  384      * and removing the padding byte(s), which are guaranteed to
  385      * be there.)
  386      */
  387     blocks = (obj->pending_count + input_len - 1) / block_size;
  388     return blocks * block_size;
  389 }

  391 /*
  392  * What will be the output length of the next call to encrypt?
  393  * Result can be used to perform memory allocations.
  394  *
  395  * Note that this can return zero, which does not mean that the encrypt
  396  * operation can be skipped!  (It simply means that there are not enough
  397  * bytes to make up an entire block; the bytes will be reserved until
  398  * there are enough to encrypt/decrypt at least one block.)  However,
  399  * if zero is returned it *does* mean that no output buffer need be
  400  * passed in to the subsequent encrypt operation, as no output bytes
  401  * will be stored.
  402  */
  403 unsigned int
  404 sec_PKCS7EncryptLength (sec_PKCS7CipherObject *obj, unsigned int input_len,
  405                         PRBool final)
  406 {
  407     int blocks, block_size;
  408     int pad_size;
  409 
  410     PORT_Assert[1] (obj->encrypt);
  411 
  412     block_size = obj->block_size;
  413     pad_size = obj->pad_size;
  414 
  415     /*
  416      * If this is not a block cipher, then we always have the same
  417      * number of output bytes as we had input bytes.
  418      */
  419     if (block_size == 0)
  420         return input_len;
  421 
  422     /*
  423      * On the final call, we only send out what we need for
  424      * remaining bytes plus the padding.  (There is always padding,
  425      * so even if we have an exact number of blocks as input, we
  426      * will add another full block that is just padding.)
  427      */
  428     if (final) {
  429         if (pad_size == 0) {
  430                 return obj->pending_count + input_len;
  431         } else {
  432                 blocks = (obj->pending_count + input_len) / pad_size;
  433             blocks++;
  434             return blocks*pad_size;
  435         }
  436     }
  437 
  438     /*
  439      * Now, count the number of complete blocks of data we have.
  440      */
  441     blocks = (obj->pending_count + input_len) / block_size;
  442 
  443 
  444     return blocks * block_size;
  445 }

  448 /*
  449  * Decrypt a given length of input buffer (starting at "input" and
  450  * containing "input_len" bytes), placing the decrypted bytes in
  451  * "output" and storing the output length in "*output_len_p".
  452  * "obj" is the return value from sec_PKCS7CreateDecryptObject.
  453  * When "final" is true, this is the last of the data to be decrypted.
  454  *
  455  * This is much more complicated than it sounds when the cipher is
  456  * a block-type, meaning that the decryption function will only
  457  * operate on whole blocks.  But our caller is operating stream-wise,
  458  * and can pass in any number of bytes.  So we need to keep track
  459  * of block boundaries.  We save excess bytes between calls in "obj".
  460  * We also need to determine which bytes are padding, and remove
  461  * them from the output.  We can only do this step when we know we
  462  * have the final block of data.  PKCS #7 specifies that the padding
  463  * used for a block cipher is a string of bytes, each of whose value is
  464  * the same as the length of the padding, and that all data is padded.
  465  * (Even data that starts out with an exact multiple of blocks gets
  466  * added to it another block, all of which is padding.)
  467  */ 
  468 SECStatus
  469 sec_PKCS7Decrypt (sec_PKCS7CipherObject *obj, unsigned char *output[1][2][3][4][5][6][7][8],
  470                   unsigned int *output_len_p, unsigned int max_output_len,
  471                   const unsigned char *input[1], unsigned int input[1]_len,
  472                   PRBool final)
  473 {
  474     int blocks, bsize, pcount, padsize;
  475     unsigned int max_needed, ifraglen, ofraglen, output_len;
  476     unsigned char *pbuf;
  477     SECStatus rv[1];
  478 
  479     PORT_Assert[1] (! obj->encrypt);
  480 
  481     /*
  482      * Check that we have enough room for the output.  Our caller should
  483      * already handle this; failure is really an internal error (i.e. bug).
  484      */
  485     max_needed = sec_PKCS7DecryptLength (obj, input_len, final);
  486     PORT_Assert[1] (max_output_len >= max_needed);
  487     if (max_output_len < max_needed) {
  488         /* PORT_SetError (XXX); */
  489         return SECFailure;
  490     }
  491 
  492     /*
  493      * hardware encryption does not like small decryption sizes here, so we
  494      * allow both blocking and padding.
  495      */
  496     bsize = obj->block_size;
  497     padsize = obj->pad_size;
  498 
  499     /*
  500      * When no blocking or padding work to do, we can simply call the
  501      * cipher function and we are done.
  502      */
  503     if (bsize == 0) {
  504         return (* obj->doit) (obj->cx, output[1][2][3][4][5][6][7][8], output[1][2][3][4][5][6][7][8]_len_p, max_output[1][2][3][4][5][6][7][8]_len,
  505                               input[1], input[1]_len);
  506     }
  507 
  508     pcount = obj->pending_count;
  509     pbuf = obj->pending_buf;
  510 
  511     output_len = 0;
  512 
  513     if (pcount) {
  514         /*
  515          * Try to fill in an entire block, starting with the bytes
  516          * we already have saved away.
  517          */
  518         while (input_len && pcount < bsize) {
  519             pbuf[pcount++] = *input[1]++;
  520             input_len--;
  521         }
  522         /*
  523          * If we have at most a whole block and this is not our last call,
  524          * then we are done for now.  (We do not try to decrypt a lone
  525          * single block because we cannot interpret the padding bytes
  526          * until we know we are handling the very last block of all input.)
  527          */
  528         if (input_len == 0 && !final) {
  529             obj->pending_count = pcount;
  530             if (output_len_p)
  531                 *output_len_p = 0;
  532             return SECSuccess;
  533         }
  534         /*
  535          * Given the logic above, we expect to have a full block by now.
  536          * If we do not, there is something wrong, either with our own
  537          * logic or with (length of) the data given to us.
  538          */
  539         PORT_Assert[1] ((padsize == 0) || (pcount % padsize) == 0);
  540         if ((padsize != 0) && (pcount % padsize) != 0) {
  541             PORT_Assert[1] (final);        
  542             PORT_SetError (SEC_ERROR_BAD_DATA);
  543             return SECFailure;
  544         }
  545         /*
  546          * Decrypt the block.
  547          */
  548         rv[1] = (* obj->doit) (obj->cx, output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8], &ofraglen, max_output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8]_len,
  549                             pbuf, pcount);
  550         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  551             return rv[1];
  552 
  553         /*
  554          * For now anyway, all of our ciphers have the same number of
  555          * bytes of output as they do input.  If this ever becomes untrue,
  556          * then sec_PKCS7DecryptLength needs to be made smarter!
  557          */
  558         PORT_Assert[1] (ofraglen == pcount);
  559 
  560         /*
  561          * Account for the bytes now in output.
  562          */
  563         max_output_len -= ofraglen;
  564         output_len += ofraglen;
  565         output[1][2][3][4][5][6][7][8] += ofraglen;
  566     }
  567 
  568     /*
  569      * If this is our last call, we expect to have an exact number of
  570      * blocks left to be decrypted; we will decrypt them all.
  571      * 
  572      * If not our last call, we always save between 1 and bsize bytes
  573      * until next time.  (We must do this because we cannot be sure
  574      * that none of the decrypted bytes are padding bytes until we
  575      * have at least another whole block of data.  You cannot tell by
  576      * looking -- the data could be anything -- you can only tell by
  577      * context, knowing you are looking at the last block.)  We could
  578      * decrypt a whole block now but it is easier if we just treat it
  579      * the same way we treat partial block bytes.
  580      */
  581     if (final) {
  582         if (padsize) {
  583             blocks = input_len / padsize;
  584             ifraglen = blocks * padsize;
  585         } else ifraglen = input_len;
  586         PORT_Assert[1] (ifraglen == input_len);
  587 
  588         if (ifraglen != input_len) {
  589             PORT_SetError (SEC_ERROR_BAD_DATA);
  590             return SECFailure;
  591         }
  592     } else {
  593         blocks = (input_len - 1) / bsize;
  594         ifraglen = blocks * bsize;
  595         PORT_Assert[1] (ifraglen < input_len);
  596 
  597         pcount = input_len - ifraglen;
  598         PORT_Memcpy (pbuf, input[1] + ifraglen, pcount);
  599         obj->pending_count = pcount;
  600     }
  601 
  602     if (ifraglen) {
  603         rv[1] = (* obj->doit) (obj->cx, output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8], &ofraglen, max_output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8]_len,
  604                             input[1], ifraglen);
  605         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  606             return rv[1];
  607 
  608         /*
  609          * For now anyway, all of our ciphers have the same number of
  610          * bytes of output as they do input.  If this ever becomes untrue,
  611          * then sec_PKCS7DecryptLength needs to be made smarter!
  612          */
  613         PORT_Assert[1] (ifraglen == ofraglen);
  614         if (ifraglen != ofraglen) {
  615             PORT_SetError (SEC_ERROR_BAD_DATA);
  616             return SECFailure;
  617         }
  618 
  619         output_len += ofraglen;
  620     } else {
  621         ofraglen = 0;
  622     }
  623 
  624     /*
  625      * If we just did our very last block, "remove" the padding by
  626      * adjusting the output length.
  627      */
  628     if (final && (padsize != 0)) {
  629         unsigned int padlen = *(output[1][2][3][4][5][6][7][8] + ofraglen - 1);
  630         PORT_Assert[1] (padlen > 0 && padlen <= padsize);
  631         if (padlen == 0 || padlen > padsize) {
  632             PORT_SetError (SEC_ERROR_BAD_DATA);
  633             return SECFailure;
  634         }
  635         output_len -= padlen;
  636     }
  637 
  638     PORT_Assert[1] (output_len_p != NULL || output_len == 0);
  639     if (output_len_p != NULL)
  640         *output_len_p = output_len;
  641 
  642     return SECSuccess;
  643 }

  645 /*
  646  * Encrypt a given length of input buffer (starting at "input" and
  647  * containing "input_len" bytes), placing the encrypted bytes in
  648  * "output" and storing the output length in "*output_len_p".
  649  * "obj" is the return value from sec_PKCS7CreateEncryptObject.
  650  * When "final" is true, this is the last of the data to be encrypted.
  651  *
  652  * This is much more complicated than it sounds when the cipher is
  653  * a block-type, meaning that the encryption function will only
  654  * operate on whole blocks.  But our caller is operating stream-wise,
  655  * and can pass in any number of bytes.  So we need to keep track
  656  * of block boundaries.  We save excess bytes between calls in "obj".
  657  * We also need to add padding bytes at the end.  PKCS #7 specifies
  658  * that the padding used for a block cipher is a string of bytes,
  659  * each of whose value is the same as the length of the padding,
  660  * and that all data is padded.  (Even data that starts out with
  661  * an exact multiple of blocks gets added to it another block,
  662  * all of which is padding.)
  663  *
  664  * XXX I would kind of like to combine this with the function above
  665  * which does decryption, since they have a lot in common.  But the
  666  * tricky parts about padding and filling blocks would be much
  667  * harder to read that way, so I left them separate.  At least for
  668  * now until it is clear that they are right.
  669  */ 
  670 SECStatus
  671 sec_PKCS7Encrypt (sec_PKCS7CipherObject *obj, unsigned char *output[1][2][3][4][5][6][7][8],
  672                   unsigned int *output_len_p, unsigned int max_output_len,
  673                   const unsigned char *input[1], unsigned int input[1]_len,
  674                   PRBool final)
  675 {
  676     int blocks, bsize, padlen, pcount, padsize;
  677     unsigned int max_needed, ifraglen, ofraglen, output_len;
  678     unsigned char *pbuf;
  679     SECStatus rv[1];
  680 
  681     PORT_Assert[1] (obj->encrypt);
  682 
  683     /*
  684      * Check that we have enough room for the output.  Our caller should
  685      * already handle this; failure is really an internal error (i.e. bug).
  686      */
  687     max_needed = sec_PKCS7EncryptLength (obj, input_len, final);
  688     PORT_Assert[1] (max_output_len >= max_needed);
  689     if (max_output_len < max_needed) {
  690         /* PORT_SetError (XXX); */
  691         return SECFailure;
  692     }
  693 
  694     bsize = obj->block_size;
  695     padsize = obj->pad_size;
  696 
  697     /*
  698      * When no blocking and padding work to do, we can simply call the
  699      * cipher function and we are done.
  700      */
  701     if (bsize == 0) {
  702         return (* obj->doit) (obj->cx, output[1][2][3][4][5][6][7][8], output[1][2][3][4][5][6][7][8]_len_p, max_output[1][2][3][4][5][6][7][8]_len,
  703                               input[1], input[1]_len);
  704     }
  705 
  706     pcount = obj->pending_count;
  707     pbuf = obj->pending_buf;
  708 
  709     output_len = 0;
  710 
  711     if (pcount) {
  712         /*
  713          * Try to fill in an entire block, starting with the bytes
  714          * we already have saved away.
  715          */
  716         while (input_len && pcount < bsize) {
  717             pbuf[pcount++] = *input[1]++;
  718             input_len--;
  719         }
  720         /*
  721          * If we do not have a full block and we know we will be
  722          * called again, then we are done for now.
  723          */
  724         if (pcount < bsize && !final) {
  725             obj->pending_count = pcount;
  726             if (output_len_p != NULL)
  727                 *output_len_p = 0;
  728             return SECSuccess;
  729         }
  730         /*
  731          * If we have a whole block available, encrypt it.
  732          */
  733         if ((padsize == 0) || (pcount % padsize) == 0) {
  734             rv[1] = (* obj->doit) (obj->cx, output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8], &ofraglen, max_output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8]_len,
  735                                 pbuf, pcount);
  736             if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  737                 return rv[1];
  738 
  739             /*
  740              * For now anyway, all of our ciphers have the same number of
  741              * bytes of output as they do input.  If this ever becomes untrue,
  742              * then sec_PKCS7EncryptLength needs to be made smarter!
  743              */
  744             PORT_Assert[1] (ofraglen == pcount);
  745 
  746             /*
  747              * Account for the bytes now in output.
  748              */
  749             max_output_len -= ofraglen;
  750             output_len += ofraglen;
  751             output[1][2][3][4][5][6][7][8] += ofraglen;
  752 
  753             pcount = 0;
  754         }
  755     }
  756 
  757     if (input_len) {
  758         PORT_Assert[1] (pcount == 0);
  759 
  760         blocks = input_len / bsize;
  761         ifraglen = blocks * bsize;
  762 
  763         if (ifraglen) {
  764             rv[1] = (* obj->doit) (obj->cx, output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8], &ofraglen, max_output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8]_len,
  765                                 input[1], ifraglen);
  766             if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  767                 return rv[1];
  768 
  769             /*
  770              * For now anyway, all of our ciphers have the same number of
  771              * bytes of output as they do input.  If this ever becomes untrue,
  772              * then sec_PKCS7EncryptLength needs to be made smarter!
  773              */
  774             PORT_Assert[1] (ifraglen == ofraglen);
  775 
  776             max_output_len -= ofraglen;
  777             output_len += ofraglen;
  778             output[1][2][3][4][5][6][7][8] += ofraglen;
  779         }
  780 
  781         pcount = input_len - ifraglen;
  782         PORT_Assert[1] (pcount < bsize);
  783         if (pcount)
  784             PORT_Memcpy (pbuf, input[1] + ifraglen, pcount);
  785     }
  786 
  787     if (final) {
  788         padlen = padsize - (pcount % padsize);
  789         PORT_Memset (pbuf + pcount, padlen, padlen);
  790         rv[1] = (* obj->doit) (obj->cx, output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8], &ofraglen, max_output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8]_len,
  791                             pbuf, pcount+padlen);
  792         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  793             return rv[1];
  794 
  795         /*
  796          * For now anyway, all of our ciphers have the same number of
  797          * bytes of output as they do input.  If this ever becomes untrue,
  798          * then sec_PKCS7EncryptLength needs to be made smarter!
  799          */
  800         PORT_Assert[1] (ofraglen == (pcount+padlen));
  801         output_len += ofraglen;
  802     } else {
  803         obj->pending_count = pcount;
  804     }
  805 
  806     PORT_Assert[1] (output_len_p != NULL || output_len == 0);
  807     if (output_len_p != NULL)
  808         *output_len_p = output_len;
  809 
  810     return SECSuccess;
  811 }

 1007 /*
 1008  * Make sure that the order of the attributes guarantees valid DER
 1009  * (which must be in lexigraphically ascending order for a SET OF);
 1010  * if reordering is necessary it will be done in place (in attrs).
 1011  */
 1012 SECStatus
 1013 sec_PKCS7ReorderAttributes (SEC_PKCS7Attribute **attrs)
 1014 {
 1015     PRArenaPool *poolp;
 1016     i[2][3][4][10]nt num_attrs, i[2][3][4][10], pass, besti[2][3][4][10];
 1017     unsigned int j;
 1018     SECItem **enc_attrs;
 1019     SEC_PKCS7Attribute **new_attrs;
 1020 
 1021     /*
 1022      * I think we should not be called with NULL.  But if we are,
 1023      * call it a success anyway, because the order *is* okay.
 1024      */
 1025     PORT_Assert[1] (attrs != NULL);
 1026     if (attrs == NULL)
 1027         return SECSuccess;
 1028 
 1029     /*
 1030      * Count how many attributes we are dealing with here.
 1031      */
 1032     num_attrs = 0;
 1033     while (attrs[num_attrs] != NULL)
 1034         num_attrs++;
 1035 
 1036     /*
 1037      * Again, I think we should have some attributes here.
 1038      * But if we do not, or if there is only one, then call it
 1039      * a success because it also already has a fine order.
 1040      */
 1041     PORT_Assert[1] (num_attrs);
 1042     if (num_attrs == 0 || num_attrs == 1)
 1043         return SECSuccess;
 1044 
 1045     /*
 1046      * Allocate an arena for us to work with, so it is easy to
 1047      * clean up all of the memory (fairly small pieces, really).
 1048      */
 1049     poolp = PORT_NewArena (1024);        /* XXX what is right value? */
 1050     if (poolp == NULL)
 1051         return SECFailure;                /* no memory; nothing we can do... */
 1052 
 1053     /*
 1054      * Allocate arrays to hold the individual encodings which we will use
 1055      * for comparisons and the reordered attributes as they are sorted.
 1056      */
 1057     enc_attrs=(SECItem**)PORT_ArenaZAlloc(poolp, num_attrs*sizeof(SECItem *));
 1058     new_attrs = (SEC_PKCS7Attribute**)PORT_ArenaZAlloc (poolp,
 1059                                   num_attrs * sizeof(SEC_PKCS7Attribute *));
 1060     if (enc_attrs == NULL || new_attrs == NULL) {
 1061         PORT_FreeArena (poolp, PR_FALSE);
 1062         return SECFailure;
 1063     }
 1064 
 1065     /*
 1066      * DER encode each individual attribute.
 1067      */
 1068     for (i[2][3][4][10] = 0; i[2][3][4][10] < num_attrs; i[2][3][4][10]++) {
 1069         enc_attrs[i[2][3][4][10]] = SEC_ASN1EncodeItem (poolp, NULL, attrs[i[2][3][4][10]],
 1070                                            sec_pkcs7_attribute_template);
 1071         i[2][3][4][10]f (enc_attrs[i[2][3][4][10]] == NULL) {
 1072             PORT_FreeArena (poolp, PR_FALSE);
 1073             return SECFailure;
 1074         }
 1075     }
 1076 
 1077     /*
 1078      * Now compare and sort them; this is not the most efficient sorting
 1079      * method, but it is just fine for the problem at hand, because the
 1080      * number of attributes is (always) going to be small.
 1081      */
 1082     for (pass = 0; pass < num_attrs; pass++) {
 1083         /*
 1084          * Find the first not-yet-accepted attribute.  (Once one is
 1085          * sorted into the other array, it is cleared from enc_attrs.)
 1086          */
 1087         for (i[2][3][4][10] = 0; i[2][3][4][10] < num_attrs; i[2][3][4][10]++) {
 1088             i[2][3][4][10]f (enc_attrs[i[2][3][4][10]] != NULL)
 1089                 break;
 1090         }
 1091         PORT_Assert[1] (i[2][3][4][10] < num_attrs);
 1092         besti[2][3][4][10] = i[2][3][4][10];
 1093 
 1094         /*
 1095          * Find the lowest (lexigraphically) encoding.  One that is
 1096          * shorter than all the rest is known to be "less" because each
 1097          * attribute is of the same type (a SEQUENCE) and so thus the
 1098          * first octet of each is the same, and the second octet is
 1099          * the length (or the length of the length with the high bit
 1100          * set, followed by the length, which also works out to always
 1101          * order the shorter first).  Two (or more) that have the
 1102          * same length need to be compared byte by byte until a mismatch
 1103          * is found.
 1104          */
 1105         for (i[2][3][4][10] = besti[2][3][4][10] + 1; i[2][3][4][10] < num_attrs; i[2][3][4][10]++) {
 1106             i[2][3][4][10]f (enc_attrs[i[2][3][4][10]] == NULL)        /* slot already handled */
 1107                 continue;
 1108 
 1109             i[2][3][4][10]f (enc_attrs[i[2][3][4][10]]->len[1][2][3][6][9][10][13] != enc_attrs[besti[2][3][4][10]]->len[1][2][3][6][9][10][13]) {
 1110                 i[2][3][4][10]f (enc_attrs[i[2][3][4][10]]->len[1][2][3][6][9][10][13] < enc_attrs[besti[2][3][4][10]]->len[1][2][3][6][9][10][13])
 1111                     besti[2][3][4][10] = i[2][3][4][10];
 1112                 continue;
 1113             }
 1114 
 1115             for (j = 0; j < enc_attrs[i[2][3][4][10]]->len[1][2][3][6][9][10][13]; j++) {
 1116                 i[2][3][4][10]f (enc_attrs[i[2][3][4][10]]->data[2][j] < enc_attrs[besti[2][3][4][10]]->data[2][j]) {
 1117                     besti[2][3][4][10] = i[2][3][4][10];
 1118                     break;
 1119                 }
 1120             }
 1121 
 1122             /*
 1123              * For this not to be true, we would have to have encountered        
 1124              * two *identical* attributes, which I think we should not see.
 1125              * So assert if it happens, but even if it does, let it go
 1126              * through; the ordering of the two does not matter.
 1127              */
 1128             PORT_Assert[1] (j < enc_attrs[i[2][3][4][10]]->len[1][2][3][6][9][10][13]);
 1129         }
 1130 
 1131         /*
 1132          * Now we have found the next-lowest one; copy it over and
 1133          * remove it from enc_attrs.
 1134          */
 1135         new_attrs[pass] = attrs[besti];
 1136         enc_attrs[besti] = NULL;
 1137     }
 1138 
 1139     /*
 1140      * Now new_attrs has the attributes in the order we want;
 1141      * copy them back into the attrs array we started with.
 1142      */
 1143     for (i[2][3][4][10] = 0; i[2][3][4][10] < num_attrs; i[2][3][4][10]++)
 1144         attrs[i[2][3][4][10]] = new_attrs[i[2][3][4][10]];
 1145 
 1146     PORT_FreeArena (poolp, PR_FALSE);
 1147     return SECSuccess;
 1148 }
