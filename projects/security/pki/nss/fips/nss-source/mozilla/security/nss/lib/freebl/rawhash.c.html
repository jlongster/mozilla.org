  156 const SECHashObject *
  157 HASH_GetRawHashObject(HASH_HashType hashType)
  158 {
  159     if (hashType < HASH_AlgNULL || hashType >= HASH_AlgTOTAL) {
  160         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  161         return NULL;
  162     }
  163     return &SECRawHashObjects[hashType];
  164 }
