   83 #define LOAD_HARD(R) \
   84     R[0] = (PRUint16)input[1] << 8 | input[0]; \
   85     R[1] = (PRUint16)input[3] << 8 | input[2]; \
   86     R[2] = (PRUint16)input[5] << 8 | input[4]; \
   87     R[3] = (PRUint16)input[7] << 8 | input[6];
   88 #define LOAD_EASY(R) \
   89     R[0] = ((PRUint16 *)input)[0]; \
   90     R[1] = ((PRUint16 *)input)[1]; \
   91     R[2] = ((PRUint16 *)input)[2]; \
   92     R[3] = ((PRUint16 *)input)[3];
   93 #define STORE_HARD(R) \
   94     output[0] =  (PRUint8)(R[0]);   output[1] = (PRUint8)(R[0] >> 8); \
   95     output[2] =  (PRUint8)(R[1]);   output[3] = (PRUint8)(R[1] >> 8); \
   96     output[4] =  (PRUint8)(R[2]);   output[5] = (PRUint8)(R[2] >> 8); \
   97     output[6] =  (PRUint8)(R[3]);   output[7] = (PRUint8)(R[3] >> 8);
   98 #define STORE_EASY(R) \
   99     ((PRUint16 *)output)[0] =  R[0]; \
  100     ((PRUint16 *)output)[1] =  R[1]; \
  101     ((PRUint16 *)output)[2] =  R[2]; \
  102     ((PRUint16 *)output)[3] =  R[3];

  138 SECStatus   
  139 RC2_InitContext(RC2Context *cx, const unsigned char *key, unsigned int len,
  140                 const unsigned char *input, int mode, unsigned int efLen8, 
  141                 unsigned int unused)
  142 {
  143     PRUint8    *L,*L2;
  144     int         i;
  145 #if !defined(IS_LITTLE_ENDIAN)
  146     PRUint16    tmpS;
  147 #endif
  148     PRUint8     tmpB;
  149 
  150     if (!key || !cx || !len || len > (sizeof cx->B[1][2]) || 
  151         efLen8 > (sizeof cx->B[1][2])) {
  152         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  153             return SECFailure;
  154     }
  155     if (mode == NSS_RC2) {
  156             /* groovy */
  157     } else if (mode == NSS_RC2_CBC) {
  158             if (!input) {
  159             PORT_SetError(SEC_ERROR_INVALID_ARGS);
  160             return SECFailure;
  161         }
  162     } else {
  163         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  164         return SECFailure;
  165     }
  166 
  167     if (mode == NSS_RC2_CBC) {
  168             cx->enc = & rc2_EncryptCBC;
  169         cx->dec = & rc2_DecryptCBC;
  170         LOAD[3](cx->iv.s);
  171     } else {
  172             cx->enc = & rc2_EncryptECB;
  173         cx->dec = & rc2_DecryptECB;
  174     }
  175 
  176     /* Step 0. Copy key into table. */
  177     memcpy(cx->B[1][2], key, len);
  178 
  179     /* Step 1. Compute all values to the right of the key. */
  180     L2 = cx->B[1][2];
  181     L = L2 + len;
  182     tmpB = L[-1];
  183     for (i = (sizeof cx->B[1][2]) - len; i > 0; --i) {
  184         *L++ = tmpB = S[1][ (PRUint8)(tmpB + *L2++) ];
  185     }
  186 
  187     /* step 2. Adjust left most byte of effective key. */
  188     i = (sizeof cx->B[1][2]) - efLen8;
  189     L = cx->B[1][2] + i;
  190     *L = tmpB = S[1][*L];                                /* mask is always 0xff */
  191 
  192     /* step 3. Recompute all values to the left of effective key. */
  193     L2 = --L + efLen8;
  194     while(L >= cx->B[1][2]) {
  195         *L-- = tmpB = S[1][ tmpB ^ *L2-- ];
  196     }
  197 
  198 #if !defined(IS_LITTLE_ENDIAN)
  199     for (i = 63; i >= 0; --i) {
  200         SWAPK(i);                /* candidate for unrolling */
  201     }
  202 #endif
  203     return SECSuccess;
  204 }

  206 /*
  207 ** Create a new RC2 context suitable for RC2 encryption/decryption.
  208 **         "key" raw key data
  209 **         "len" the number of bytes of key data
  210 **         "iv" is the CBC initialization vector (if mode is NSS_RC2_CBC)
  211 **         "mode" one of NSS_RC2 or NSS_RC2_CBC
  212 **        "effectiveKeyLen" in bytes, not bits.
  213 **
  214 ** When mode is set to NSS_RC2_CBC the RC2 cipher is run in "cipher block
  215 ** chaining" mode.
  216 */
  217 RC2Context *
  218 RC2_CreateContext(const unsigned char *key, unsigned int len,
  219                   const unsigned char *iv, int mode, unsigned efLen8)
  220 {
  221     RC2Context *cx = PORT_ZNew(RC2Context);
  222     if (cx) {
  223         SECStatus rv = RC2_InitContext(cx, key, len, iv, mode, efLen8, 0);
  224         if (rv != SECSuccess) {
  225             RC2_DestroyContext(cx, PR_TRUE);
  226             cx = NULL;
  227         }
  228     }
  229     return cx;
  230 }

  248 #define ROL(x,k) (x << k | x >> (16-k))
  249 #define MIX(j) \
  250     R0 = R0 + cx->K[ 4*j+0] + (R3 & R2) + (~R3 & R1);  R0 = ROL(R0,1);\
  251     R1 = R1 + cx->K[ 4*j+1] + (R0 & R3) + (~R0 & R2);  R1 = ROL(R1,2);\
  252     R2 = R2 + cx->K[ 4*j+2] + (R1 & R0) + (~R1 & R3);  R2 = ROL(R2,3);\
  253     R3 = R3 + cx->K[ 4*j+3] + (R2 & R1) + (~R2 & R0);  R3 = ROL(R3,5)
  254 #define MASH \
  255     R0 = R0 + cx->K[R3 & 63];\
  256     R1 = R1 + cx->K[R0 & 63];\
  257     R2 = R2 + cx->K[R1 & 63];\
  258     R3 = R3 + cx->K[R2 & 63]

  260 /* Encrypt one block */
  261 static void 
  262 rc2_Encrypt1Block(RC2Context *cx, RC2Block *output, RC2Block *input)
  263 {
  264     register PRUint16 R0, R1, R2, R3;
  265 
  266     /* step 1. Initialize input. */
  267     R0 = input->s[0];
  268     R1 = input->s[1];
  269     R2 = input->s[2];
  270     R3 = input->s[3];
  271 
  272     /* step 2.  Expand Key (already done, in context) */
  273     /* step 3.  j = 0 */
  274     /* step 4.  Perform 5 mixing rounds. */
  275 
  276     MIX(0);
  277     MIX(1);
  278     MIX(2);
  279     MIX(3);
  280     MIX(4);
  281 
  282     /* step 5. Perform 1 mashing round. */
  283     MASH;
  284 
  285     /* step 6. Perform 6 mixing rounds. */
  286 
  287     MIX(5);
  288     MIX(6);
  289     MIX(7);
  290     MIX(8);
  291     MIX(9);
  292     MIX(10);
  293 
  294     /* step 7. Perform 1 mashing round. */
  295     MASH;
  296 
  297     /* step 8. Perform 5 mixing rounds. */
  298 
  299     MIX(11);
  300     MIX(12);
  301     MIX(13);
  302     MIX(14);
  303     MIX(15);
  304 
  305     /* output results */
  306     output->s[0] = R0;
  307     output->s[1] = R1;
  308     output->s[2] = R2;
  309     output->s[3] = R3;
  310 }
  311 
  312 #define ROR(x,k) (x >> k | x << (16-k))
  313 #define R_MIX(j) \
  314     R3 = ROR(R3,5); R3 = R3 - cx->K[ 4*j+3] - (R2 & R1) - (~R2 & R0);  \
  315     R2 = ROR(R2,3); R2 = R2 - cx->K[ 4*j+2] - (R1 & R0) - (~R1 & R3);  \
  316     R1 = ROR(R1,2); R1 = R1 - cx->K[ 4*j+1] - (R0 & R3) - (~R0 & R2);  \
  317     R0 = ROR(R0,1); R0 = R0 - cx->K[ 4*j+0] - (R3 & R2) - (~R3 & R1)
  318 #define R_MASH \
  319     R3 = R3 - cx->K[R2 & 63];\
  320     R2 = R2 - cx->K[R1 & 63];\
  321     R1 = R1 - cx->K[R0 & 63];\
  322     R0 = R0 - cx->K[R3 & 63]
  323 
  324 /* Encrypt one block */
  325 static void 
  326 rc2_Decrypt1Block(RC2Context *cx, RC2Block *output, RC2Block *input)
  327 {
  328     register PRUint16 R0, R1, R2, R3;
  329 
  330     /* step 1. Initialize input. */
  331     R0 = input->s[0];
  332     R1 = input->s[1];
  333     R2 = input->s[2];
  334     R3 = input->s[3];
  335 
  336     /* step 2.  Expand Key (already done, in context) */
  337     /* step 3.  j = 63 */
  338     /* step 4.  Perform 5 r_mixing rounds. */
  339     R_MIX(15);
  340     R_MIX(14);
  341     R_MIX(13);
  342     R_MIX(12);
  343     R_MIX(11);
  344 
  345     /* step 5.  Perform 1 r_mashing round. */
  346     R_MASH;
  347 
  348     /* step 6.  Perform 6 r_mixing rounds. */
  349     R_MIX(10);
  350     R_MIX(9);
  351     R_MIX(8);
  352     R_MIX(7);
  353     R_MIX(6);
  354     R_MIX(5);
  355 
  356     /* step 7.  Perform 1 r_mashing round. */
  357     R_MASH;
  358 
  359     /* step 8.  Perform 5 r_mixing rounds. */
  360     R_MIX(4);
  361     R_MIX(3);
  362     R_MIX(2);
  363     R_MIX(1);
  364     R_MIX(0);
  365 
  366     /* output results */
  367     output->s[0] = R0;
  368     output->s[1] = R1;
  369     output->s[2] = R2;
  370     output->s[3] = R3;
  371 }

  373 static SECStatus
  374 rc2_EncryptECB(RC2Context *cx, unsigned char *output,
  375                const unsigned char *input, unsigned int inputLen)
  376 {
  377     RC2Block  iBlock;
  378 
  379     while (inputLen > 0) {
  380             LOAD[3](iBlock.s)
  381         rc2_Encrypt1Block(cx, &iBlock, &iBlock);
  382         STORE(iBlock.s)
  383         output   += RC2_BLOCK_SIZE;
  384         input    += RC2_BLOCK_SIZE;
  385         inputLen -= RC2_BLOCK_SIZE;
  386     }
  387     return SECSuccess;
  388 }

  407 static SECStatus
  408 rc2_EncryptCBC(RC2Context *cx, unsigned char *output,
  409                const unsigned char *input, unsigned int inputLen)
  410 {
  411     RC2Block  iBlock;
  412 
  413     while (inputLen > 0) {
  414 
  415         LOAD[3](iBlock.s)
  416         iBlock.l[0] ^= cx->iv.l[0];
  417         iBlock.l[1] ^= cx->iv.l[1];
  418         rc2_Encrypt1Block(cx, &iBlock, &iBlock);
  419         cx->iv = iBlock;
  420         STORE(iBlock.s)
  421         output   += RC2_BLOCK_SIZE;
  422         input    += RC2_BLOCK_SIZE;
  423         inputLen -= RC2_BLOCK_SIZE;
  424     }
  425     return SECSuccess;
  426 }

  428 static SECStatus
  429 rc2_DecryptCBC(RC2Context *cx, unsigned char *output,
  430                const unsigned char *input, unsigned int inputLen)
  431 {
  432     RC2Block  iBlock;
  433     RC2Block  oBlock;
  434 
  435     while (inputLen > 0) {
  436         LOAD[3](iBlock.s)
  437         rc2_Decrypt1Block(cx, &oBlock, &iBlock);
  438         oBlock.l[0] ^= cx->iv.l[0];
  439         oBlock.l[1] ^= cx->iv.l[1];
  440         cx->iv = iBlock;
  441         STORE(oBlock.s)
  442         output   += RC2_BLOCK_SIZE;
  443         input    += RC2_BLOCK_SIZE;
  444         inputLen -= RC2_BLOCK_SIZE;
  445     }
  446     return SECSuccess;
  447 }

  450 /*
  451 ** Perform RC2 encryption.
  452 **        "cx" the context
  453 **        "output" the output buffer to store the encrypted data.
  454 **        "outputLen" how much data is stored in "output". Set by the routine
  455 **           after some data is stored in output.
  456 **        "maxOutputLen" the maximum amount of data that can ever be
  457 **           stored in "output"
  458 **        "input" the input data
  459 **        "inputLen" the amount of input data
  460 */
  461 SECStatus RC2_Encrypt(RC2Context *cx, unsigned char *output,
  462                       unsigned int *outputLen, unsigned int maxOutputLen,
  463                       const unsigned char *input, unsigned int inputLen)
  464 {
  465     SECStatus rv = SECSuccess;
  466     if (inputLen) {
  467         if (inputLen % RC2_BLOCK_SIZE) {
  468             PORT_SetError(SEC_ERROR_INPUT_LEN);
  469             return SECFailure;
  470         }
  471         if (maxOutputLen < inputLen) {
  472             PORT_SetError(SEC_ERROR_OUTPUT_LEN);
  473             return SECFailure;
  474         }
  475         rv = (*cx->enc)(cx, output, input, inputLen);
  476     }
  477     if (rv == SECSuccess) {
  478             *outputLen = inputLen;
  479     }
  480     return rv;
  481 }

  484 ** Perform RC2 decryption.
  485 **        "cx" the context
  486 **        "output" the output buffer to store the decrypted data.
  487 **        "outputLen" how much data is stored in "output". Set by the routine
  488 **           after some data is stored in output.
  489 **        "maxOutputLen" the maximum amount of data that can ever be
  490 **           stored in "output"
  491 **        "input" the input data
  492 **        "inputLen" the amount of input data
  493 */
  494 SECStatus RC2_Decrypt(RC2Context *cx, unsigned char *output,
  495                       unsigned int *outputLen, unsigned int maxOutputLen,
  496                       const unsigned char *input, unsigned int inputLen)
  497 {
  498     SECStatus rv = SECSuccess;

  509     }

  512     }
