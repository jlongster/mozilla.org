  132 /*
  133  * turn a CDMF key into a des key. CDMF is an old IBM scheme to export DES by
  134  * Deprecating a full des key to 40 bit key strenth.
  135  */
  136 static CK_RV
  137 sftk_cdmf2des(unsigned char *cdmfkey, unsigned char *deskey)
  138 {
  139     unsigned char key1[8] = { 0xc4, 0x08, 0xb0, 0x54, 0x0b, 0xa1, 0xe0, 0xae };
  140     unsigned char key2[8] = { 0xef, 0x2c, 0x04, 0x1c, 0xe6, 0x38, 0x2f, 0xe6 };
  141     unsigned char enc_src[8];
  142     unsigned char enc_dest[8];
  143     unsigned int leng,i;
  144     DESContext *descx;
  145     SECStatus rv;
  146     
  147     
  148     /* zero the parity bits */
  149     for (i=0; i < 8; i++) {
  150         enc_src[i] = cdmfkey[i] & 0xfe;
  151     }
  152 
  153     /* encrypt with key 1 */
  154     descx = DES_CreateContext(key1, NULL, NSS_DES, PR_TRUE);
  155     if (descx == NULL) return CKR_HOST_MEMORY;
  156     rv = DES_Encrypt(descx, enc_dest, &leng, 8, enc_src, 8);
  157     DES_DestroyContext(descx,PR_TRUE);
  158     if (rv != SECSuccess) return CKR_DEVICE_ERROR;
  159 
  160     /* xor source with des, zero the parity bits and depricate the key*/
  161     for (i=0; i < 8; i++) {
  162         if (i & 1) {
  163             enc_src[i] = (enc_src[i] ^ enc_dest[i]) & 0xfe;
  164         } else {
  165             enc_src[i] = (enc_src[i] ^ enc_dest[i]) & 0x0e;
  166         }
  167     }
  168 
  169     /* encrypt with key 2 */
  170     descx = DES_CreateContext(key2, NULL, NSS_DES, PR_TRUE);
  171     if (descx == NULL) return CKR_HOST_MEMORY;
  172     rv = DES_Encrypt(descx, deskey, &leng, 8, enc_src, 8);
  173     DES_DestroyContext(descx,PR_TRUE);
  174     if (rv != SECSuccess) return CKR_DEVICE_ERROR;
  175 
  176     /* set the corret parity on our new des key */        
  177     sftk_FormatDESKey(deskey, 8);
  178     return CKR_OK;
  179 }

  182 /* NSC_DestroyObject destroys an object. */
  183 CK_RV
  184 NSC_DestroyObject(CK_SESSION_HANDLE hSession, CK_OBJECT_HANDLE hObject)
  185 {
  186     SFTKSlot *slot = sftk_SlotFromSessionHandle(hSession);
  187     SFTKSession *session;
  188     SFTKObject *object;
  189     SFTKFreeStatus status;
  190 
  191     /*
  192      * This whole block just makes sure we really can destroy the
  193      * requested object.
  194      */
  195     session = sftk_SessionFromHandle(hSession);
  196     if (session == NULL) {
  197         return CKR_SESSION_HANDLE_INVALID;
  198     }
  199 
  200     object = sftk_ObjectFromHandle(hObject,session);
  201     if (object == NULL) {
  202         sftk_FreeSession(session);
  203         return CKR_OBJECT_HANDLE_INVALID;
  204     }
  205 
  206     /* don't destroy a private object if we aren't logged in */
  207     if ((!slot->isLoggedIn) && (slot->needLogin) &&
  208                                 (sftk_isTrue(object,CKA_PRIVATE))) {
  209         sftk_FreeSession(session);
  210         sftk_FreeObject(object);
  211         return CKR_USER_NOT_LOGGED_IN;
  212     }
  213 
  214     /* don't destroy a token object if we aren't in a rw session */
  215 
  216     if (((session->info.flags & CKF_RW_SESSION) == 0) &&
  217                                 (sftk_isTrue(object,CKA_TOKEN))) {
  218         sftk_FreeSession(session);
  219         sftk_FreeObject(object);
  220         return CKR_SESSION_READ_ONLY;
  221     }
  222 
  223     sftk_DeleteObject(session,object);
  224 
  225     sftk_FreeSession(session);
  226 
  227     /*
  228      * get some indication if the object is destroyed. Note: this is not
  229      * 100%. Someone may have an object reference outstanding (though that
  230      * should not be the case by here. Also note that the object is "half"
  231      * destroyed. Our internal representation is destroyed, but it may still
  232      * be in the data base.
  233      */
  234     status = sftk_FreeObject(object);
  235 
  236     return (status != SFTK_DestroyFailure) ? CKR_OK : CKR_DEVICE_ERROR;
  237 }

  245 /* 
  246  * return a context based on the SFTKContext type.
  247  */
  248 SFTKSessionContext *
  249 sftk_ReturnContextByType(SFTKSession *session, SFTKContextType type)
  250 {
  251     switch (type) {
  252         case SFTK_ENCRYPT:
  253         case SFTK_DECRYPT:
  254             return session->enc_context;
  255         case SFTK_HASH:
  256             return session->hash_context;
  257         case SFTK_SIGN:
  258         case SFTK_SIGN_RECOVER:
  259         case SFTK_VERIFY:
  260         case SFTK_VERIFY_RECOVER:
  261             return session->hash_context;
  262     }
  263     return NULL;
  264 }

  269 void
  270 sftk_SetContextByType(SFTKSession *session, SFTKContextType type, 
  271                                                 SFTKSessionContext *context)
  272 {
  273     switch (type) {
  274         case SFTK_ENCRYPT:
  275         case SFTK_DECRYPT:
  276             session->enc_context = context;
  277             break;
  278         case SFTK_HASH:

  280             break;
  281         case SFTK_SIGN:
  282         case SFTK_SIGN_RECOVER:
  283         case SFTK_VERIFY:
  284         case SFTK_VERIFY_RECOVER:
  285             session->hash_context = context;
  286             break;
  287     }
  288     return;

  291 /*
  292  * code to grab the context. Needed by every C_XXXUpdate, C_XXXFinal,
  293  * and C_XXX function. The function takes a session handle, the context type,
  294  * and wether or not the session needs to be multipart. It returns the context,
  295  * and optionally returns the session pointer (if sessionPtr != NULL) if session
  296  * pointer is returned, the caller is responsible for freeing it.
  297  */
  298 static CK_RV
  299 sftk_GetContext(CK_SESSION_HANDLE handle,SFTKSessionContext **contextPtr,
  300         SFTKContextType type, PRBool needMulti, SFTKSession **sessionPtr)
  301 {
  302     SFTKSession *session;
  303     SFTKSessionContext *context;
  304 
  305     session = sftk_SessionFromHandle(handle);
  306     if (session == NULL) return CKR_SESSION_HANDLE_INVALID;
  307     context = sftk_ReturnContextByType(session,type);
  308     /* make sure the context is valid */
  309     if((context==NULL)||(context->type!=type)||(needMulti&&!(context->multi))){
  310         sftk_FreeSession(session);
  311         return CKR_OPERATION_NOT_INITIALIZED;
  312     }
  313     *contextPtr = context;
  314     if (sessionPtr != NULL) {
  315         *sessionPtr = session;
  316     } else {
  317         sftk_FreeSession(session);
  318     }
  319     return CKR_OK;
  320 }

  326 /*
  327  * All the NSC_InitXXX functions have a set of common checks and processing they
  328  * all need to do at the beginning. This is done here.
  329  */
  330 static CK_RV
  331 sftk_InitGeneric(SFTKSession *session,SFTKSessionContext **contextPtr,
  332                  SFTKContextType ctype,SFTKObject **keyPtr,
  333                  CK_OBJECT_HANDLE hKey, CK_KEY_TYPE *keyTypePtr,
  334                  CK_OBJECT_CLASS pubKeyType, CK_ATTRIBUTE_TYPE operation)
  335 {
  336     SFTKObject *key = NULL;
  337     SFTKAttribute *att;
  338     SFTKSessionContext *context;
  339 
  340     /* We can only init if there is not current context active */
  341     if (sftk_ReturnContextByType(session,ctype) != NULL) {
  342         return CKR_OPERATION_ACTIVE;
  343     }
  344 
  345     /* find the key */
  346     if (keyPtr) {
  347         key = sftk_ObjectFromHandle(hKey,session);
  348         if (key == NULL) {
  349             return CKR_KEY_HANDLE_INVALID;
  350             }
  351 
  352         /* make sure it's a valid  key for this operation */
  353         if (((key->objclass != CKO_SECRET_KEY) && (key->objclass != pubKeyType))
  354                                         || !sftk_isTrue(key,operation)) {
  355             sftk_FreeObject(key);
  356             return CKR_KEY_TYPE_INCONSISTENT;
  357         }
  358         /* get the key type */
  359         att = sftk_FindAttribute(key,CKA_KEY_TYPE);
  360         if (att == NULL) {
  361             sftk_FreeObject(key);
  362             return CKR_KEY_TYPE_INCONSISTENT;
  363         }
  364         PORT_Assert[1](att->attrib.ulValueLen == sizeof(CK_KEY_TYPE));
  365         if (att->attrib.ulValueLen != sizeof(CK_KEY_TYPE)) {
  366             sftk_FreeAttribute(att);
  367             sftk_FreeObject(key);
  368             return CKR_ATTRIBUTE_VALUE_INVALID;
  369         }
  370         PORT_Memcpy(keyTypePtr, att->attrib.pValue, sizeof(CK_KEY_TYPE));
  371         sftk_FreeAttribute(att);
  372         *keyPtr = key;
  373     }
  374 
  375     /* allocate the context structure */
  376     context = (SFTKSessionContext *)PORT_Alloc(sizeof(SFTKSessionContext));
  377     if (context == NULL) {
  378         if (key) sftk_FreeObject(key);
  379         return CKR_HOST_MEMORY;
  380     }
  381     context->type = ctype;
  382     context->multi = PR_TRUE;
  383     context->cipherInfo = NULL;
  384     context->hashInfo = NULL;
  385     context->doPad = PR_FALSE;
  386     context->padDataLength = 0;
  387     context->key = key;
  388     context->blockSize = 0;
  389 
  390     *contextPtr = context;
  391     return CKR_OK;
  392 }

  394 /* NSC_CryptInit initializes an encryption/Decryption operation. */
  395 /* This function is used by NSC_EncryptInit, NSC_DecryptInit, 
  396  *                          NSC_WrapKey, NSC_UnwrapKey, 
  397  *                          NSC_SignInit, NSC_VerifyInit (via sftk_InitCBCMac),
  398  * The only difference in their uses is the value of etype.
  399  */
  400 static CK_RV
  401 sftk_CryptInit(CK_SESSION_HANDLE hSession, CK_MECHANISM_PTR pMechanism,
  402                  CK_OBJECT_HANDLE hKey, CK_ATTRIBUTE_TYPE etype,
  403                  SFTKContextType contextType, PRBool isEncrypt)
  404 {
  405     SFTKSession *session;
  406     SFTKObject *key;
  407     SFTKSessionContext *context;
  408     SFTKAttribute *att;
  409     CK_RC2_CBC_PARAMS *rc2_param;
  410 #if NSS_SOFTOKEN_DOES_RC5
  411     CK_RC5_CBC_PARAMS *rc5_param;
  412     SECItem rc5Key;
  413 #endif
  414     CK_KEY_TYPE key_type;
  415     CK_RV crv = CKR_OK;
  416     unsigned effectiveKeyLength;
  417     unsigned char newdeskey[24];
  418     PRBool useNewKey=PR_FALSE;
  419     int t;
  420 
  421     crv = sftk_MechAllowsOperation(pMechanism->mechanism, etype);
  422     if (crv != CKR_OK) 
  423             return crv;
  424 
  425     session = sftk_SessionFromHandle(hSession);
  426     if (session == NULL) return CKR_SESSION_HANDLE_INVALID;
  427 
  428     crv = sftk_InitGeneric(session,&context,contextType,&key,hKey,&key_type,
  429                             isEncrypt ?CKO_PUBLIC_KEY:CKO_PRIVATE_KEY, etype);
  430                                                 
  431     if (crv != CKR_OK) {
  432         sftk_FreeSession(session);
  433         return crv;
  434     }

  442             break;
  443         }
  444         context->multi = PR_FALSE;
  445         if (isEncrypt) {
  446             NSSLOWKEYPublicKey *pubKey = sftk_GetPubKey(key,CKK_RSA,&crv);
  447             if (pubKey == NULL) {
  448                 break;
  449             }
  450             context->maxLen = nsslowkey_PublicModulusLen(pubKey);
  451             context->cipherInfo =  (void *)pubKey;
  452             context->update = (SFTKCipher) 
  453                 (pMechanism->mechanism == CKM_RSA_X_509
  454                                         ? RSA_EncryptRaw : RSA_EncryptBlock);
  455         } else {
  456             NSSLOWKEYPrivateKey *privKey = sftk_GetPrivKey(key,CKK_RSA,&crv);
  457             if (privKey == NULL) {
  458                 break;
  459             }
  460             context->maxLen = nsslowkey_PrivateModulusLen(privKey);
  461             context->cipherInfo =  (void *)privKey;
  462             context->update = (SFTKCipher) 
  463                 (pMechanism->mechanism == CKM_RSA_X_509
  464                                         ? RSA_DecryptRaw : RSA_DecryptBlock);
  465         }
  466         context->destroy = sftk_Null;
  467         break;
  468     case CKM_RC2_CBC_PAD:
  469         context->doPad = PR_TRUE;
  470         /* fall thru */
  471     case CKM_RC2_ECB:
  472     case CKM_RC2_CBC:
  473         context->blockSize = 8;
  474         if (key_type != CKK_RC2) {
  475             crv = CKR_KEY_TYPE_INCONSISTENT;
  476             break;

  481             break;

  489                               NSS_RC2_CBC,effectiveKeyLength);

  491         if (context->cipherInfo == NULL) {
  492             crv = CKR_HOST_MEMORY;
  493             break;
  494         }

  557             break;
  558         }
  559         t = (pMechanism->mechanism == CKM_CDMF_ECB) ? NSS_DES : NSS_DES_CBC">NSS_DES_CBC;

  584         t = NSS_DES_EDE3;
  585         goto finish_des;
  586     case CKM_DES3_CBC_PAD:
  587         context->doPad = PR_TRUE;
  588         /* fall thru */
  589     case CKM_DES3_CBC:
  590         if ((key_type != CKK_DES2) && (key_type != CKK_DES3)) {
  591             crv = CKR_KEY_TYPE_INCONSISTENT;
  592             break;
  593         }
  594         t = NSS_DES_EDE3_CBC;
  595 finish_des:
  596         context->blockSize = 8;
  597         att = sftk_FindAttribute(key,CKA_VALUE);
  598         if (att == NULL) {
  599             crv = CKR_KEY_HANDLE_INVALID;
  600             break;
  601         }
  602         if (key_type == CKK_DES2 && 
  603             (t == NSS_DES_EDE3_CBC">NSS_DES_EDE3_CBC || t == NSS_DES_EDE3)) {
  604             /* extend DES2 key to DES3 key. */
  605             memcpy(newdeskey, att->attrib.pValue, 16);
  606             memcpy(newdeskey + 16, newdeskey, 8);
  607             useNewKey=PR_TRUE;
  608         } else if (key_type == CKK_CDMF) {
  609             crv = sftk_cdmf2des((unsigned char*)att->attrib.pValue,newdeskey);
  610             if (crv != CKR_OK) {
  611                 sftk_FreeAttribute(att);
  612                 break;
  613             }
  614             useNewKey=PR_TRUE;
  615         }
  616         context->cipherInfo = DES_CreateContext(
  617                 useNewKey ? newdeskey : (unsigned char*)att->attrib.pValue,
  618                 (unsigned char*)pMechanism->pParameter,t, isEncrypt);
  619         if (useNewKey) 
  620             memset(newdeskey, 0, sizeof newdeskey);
  621         sftk_FreeAttribute(att);
  622         if (context->cipherInfo == NULL) {
  623             crv = CKR_HOST_MEMORY;
  624             break;
  625         }
  626         context->update = (SFTKCipher) (isEncrypt ? DES_Encrypt : DES_Decrypt);
  627         context->destroy = (SFTKDestroy) DES_DestroyContext;
  628         break;
  629 
  630     case CKM_AES_CBC_PAD:
  631         context->doPad = PR_TRUE;
  632         /* fall thru */
  633     case CKM_AES_ECB:
  634     case CKM_AES_CBC:
  635         context->blockSize = 16;
  636         if (key_type != CKK_AES) {
  637             crv = CKR_KEY_TYPE_INCONSISTENT;
  638             break;
  639         }
  640         att = sftk_FindAttribute(key,CKA_VALUE);
  641         if (att == NULL) {
  642             crv = CKR_KEY_HANDLE_INVALID;
  643             break;
  644         }
  645         context->cipherInfo = AES_CreateContext(
  646             (unsigned char*)att->attrib.pValue,
  647             (unsigned char*)pMechanism->pParameter,
  648             pMechanism->mechanism == CKM_AES_ECB ? NSS_AES : NSS_AES_CBC,
  649             isEncrypt, att->attrib.ulValueLen, 16);
  650         sftk_FreeAttribute(att);
  651         if (context->cipherInfo == NULL) {
  652             crv = CKR_HOST_MEMORY;
  653             break;
  654         }
  655         context->update = (SFTKCipher) (isEncrypt ? AES_Encrypt : AES_Decrypt);
  656         context->destroy = (SFTKDestroy) AES_DestroyContext;
  657         break;
  658 
  659     case CKM_NETSCAPE_AES_KEY_WRAP_PAD:
  660             context->doPad = PR_TRUE;
  661         /* fall thru */
  662     case CKM_NETSCAPE_AES_KEY_WRAP:
  663         context->multi = PR_FALSE;
  664         context->blockSize = 8;
  665         if (key_type != CKK_AES) {
  666             crv = CKR_KEY_TYPE_INCONSISTENT;
  667             break;
  668         }
  669         att = sftk_FindAttribute(key,CKA_VALUE);
  670         if (att == NULL) {
  671             crv = CKR_KEY_HANDLE_INVALID;
  672             break;
  673         }
  674         context->cipherInfo = AESKeyWrap_CreateContext(
  675             (unsigned char*)att->attrib.pValue,
  676             (unsigned char*)pMechanism->pParameter,
  677             isEncrypt, att->attrib.ulValueLen);
  678         sftk_FreeAttribute(att);
  679         if (context->cipherInfo == NULL) {
  680             crv = CKR_HOST_MEMORY;
  681             break;
  682         }
  683         context->update = (SFTKCipher) (isEncrypt ? AESKeyWrap_Encrypt 
  684                                                   : AESKeyWrap_Decrypt);
  685         context->destroy = (SFTKDestroy) AESKeyWrap_DestroyContext;
  686         break;
  687 
  688     default:
  689         crv = CKR_MECHANISM_INVALID;
  690         break;
  691     }
  692 
  693     if (crv != CKR_OK) {
  694         sftk_FreeContext(context);
  695         sftk_FreeSession(session);
  696         return crv;
  697     }
  698     sftk_SetContextByType(session, contextType, context);
  699     sftk_FreeSession(session);
  700     return CKR_OK;
  701 }

  711 /* NSC_EncryptUpdate continues a multiple-part encryption operation. */
  712 CK_RV NSC_EncryptUpdate(CK_SESSION_HANDLE hSession,
  713     CK_BYTE_PTR pPart, CK_ULONG ulPartLen, CK_BYTE_PTR pEncryptedPart,        
  714                                         CK_ULONG_PTR pulEncryptedPartLen)
  715 {
  716     SFTKSessionContext *context;
  717     unsigned int outlen,i;
  718     unsigned int padoutlen = 0;
  719     unsigned int maxout = *pulEncryptedPartLen;
  720     CK_RV crv;
  721     SECStatus rv;
  722 
  723     /* make sure we're legal */
  724     crv = sftk_GetContext(hSession,&context,SFTK_ENCRYPT,PR_TRUE,NULL);
  725     if (crv != CKR_OK) return crv;
  726 
  727     if (!pEncryptedPart) {
  728         if (context->doPad) {
  729             CK_ULONG totalDataAvailable = ulPartLen + context->padDataLength;
  730             CK_ULONG blocksToSend = totalDataAvailable/context->blockSize;
  731 
  732             *pulEncryptedPartLen = blocksToSend * context->blockSize;
  733             return CKR_OK;
  734         }
  735         *pulEncryptedPartLen = ulPartLen;
  736         return CKR_OK;
  737     }
  738 
  739     /* do padding */
  740     if (context->doPad) {
  741         /* deal with previous buffered data */
  742         if (context->padDataLength != 0) {
  743             /* fill in the padded to a full block size */
  744             for (i=context->padDataLength; 
  745                         (ulPartLen != 0) && i < context->blockSize; i++) {
  746                 context->padBuf[i] = *pPart++;
  747                 ulPartLen--;
  748                 context->padDataLength++;
  749             }
  750 
  751             /* not enough data to encrypt yet? then return */
  752             if (context->padDataLength != context->blockSize) {
  753                 *pulEncryptedPartLen = 0;
  754                 return CKR_OK;
  755             }
  756             /* encrypt the current padded data */
  757                 rv = (*context->update)(context->cipherInfo, pEncryptedPart, 
  758                 &padoutlen, context->blockSize, context->padBuf,
  759                                                         context->blockSize);
  760                 if (rv != SECSuccess) return CKR_DEVICE_ERROR;
  761             pEncryptedPart += padoutlen;
  762             maxout -= padoutlen;
  763         }
  764         /* save the residual */
  765         context->padDataLength = ulPartLen % context->blockSize;
  766         if (context->padDataLength) {
  767             PORT_Memcpy(context->padBuf,
  768                         &pPart[ulPartLen-context->padDataLength],
  769                                                         context->padDataLength);
  770             ulPartLen -= context->padDataLength;
  771         }
  772         /* if we've exhausted our new buffer, we're done */
  773         if (ulPartLen == 0) {
  774             *pulEncryptedPartLen = padoutlen;
  775             return CKR_OK;
  776         }
  777     }
  778 
  779 
  780     /* do it: NOTE: this assumes buf size in is >= buf size out! */
  781     rv = (*context->update)(context->cipherInfo,pEncryptedPart, 
  782                                         &outlen, maxout, pPart, ulPartLen);
  783     *pulEncryptedPartLen = (CK_ULONG) (outlen + padoutlen);
  784     return (rv == SECSuccess) ? CKR_OK : CKR_DEVICE_ERROR;
  785 }

  788 /* NSC_EncryptFinal finishes a multiple-part encryption operation. */
  789 CK_RV NSC_EncryptFinal(CK_SESSION_HANDLE hSession,
  790     CK_BYTE_PTR pLastEncryptedPart, CK_ULONG_PTR pulLastEncryptedPartLen)
  791 {
  792     SFTKSession *session;
  793     SFTKSessionContext *context;
  794     unsigned int outlen,i;
  795     unsigned int maxout = *pulLastEncryptedPartLen;
  796     CK_RV crv;
  797     SECStatus rv = SECSuccess;
  798     PRBool contextFinished = PR_TRUE;
  799 
  800     /* make sure we're legal */
  801     crv = sftk_GetContext(hSession,&context,SFTK_ENCRYPT,PR_TRUE,&session);
  802     if (crv != CKR_OK) return crv;
  803 
  804     *pulLastEncryptedPartLen = 0;
  805     if (!pLastEncryptedPart) {
  806         /* caller is checking the amount of remaining data */
  807         if (context->blockSize > 0 && context->doPad) {
  808             *pulLastEncryptedPartLen = context->blockSize;
  809             contextFinished = PR_FALSE; /* still have padding to go */
  810         }
  811         goto finish;
  812     }
  813 
  814     /* do padding */
  815     if (context->doPad) {
  816         unsigned char  padbyte = (unsigned char) 
  817                                 (context->blockSize - context->padDataLength); 
  818         /* fill out rest of pad buffer with pad magic*/
  819         for (i=context->padDataLength; i < context->blockSize; i++) {
  820             context->padBuf[i] = padbyte;
  821         }
  822         rv = (*context->update)(context->cipherInfo,pLastEncryptedPart, 
  823                         &outlen, maxout, context->padBuf, context->blockSize);
  824         if (rv == SECSuccess) *pulLastEncryptedPartLen = (CK_ULONG) outlen;
  825     }
  826 
  827 finish:
  828     if (contextFinished) {
  829         sftk_SetContextByType(session, SFTK_ENCRYPT, NULL);
  830         sftk_FreeContext(context);
  831     }
  832     sftk_FreeSession(session);
  833     return (rv == SECSuccess) ? CKR_OK : CKR_DEVICE_ERROR;
  834 }

  836 /* NSC_Encrypt encrypts single-part data. */
  837 CK_RV NSC_Encrypt (CK_SESSION_HANDLE hSession, CK_BYTE_PTR pData,
  838                        CK_ULONG ulDataLen, CK_BYTE_PTR pEncryptedData,
  839                    CK_ULONG_PTR pulEncryptedDataLen)
  840 {
  841     SFTKSession *session;
  842     SFTKSessionContext *context;
  843     unsigned int outlen;
  844     unsigned int maxoutlen = *pulEncryptedDataLen;
  845     CK_RV crv;
  846     CK_RV crv2;
  847     SECStatus rv = SECSuccess;
  848     SECItem   pText;
  849 
  850     pText.type = siBuffer;
  851     pText.data = pData;
  852     pText.len  = ulDataLen;
  853 
  854     /* make sure we're legal */
  855     crv = sftk_GetContext(hSession,&context,SFTK_ENCRYPT,PR_FALSE,&session);
  856     if (crv != CKR_OK) return crv;
  857 
  858     if (!pEncryptedData) {
  859         *pulEncryptedDataLen = context->multi ? 
  860                 ulDataLen + 2 * context->blockSize : context->maxLen;
  861         goto finish;
  862     }
  863 
  864     if (context->doPad) {
  865         if (context->multi) {
  866             CK_ULONG finalLen;
  867             /* padding is fairly complicated, have the update and final 
  868              * code deal with it */
  869             sftk_FreeSession(session);
  870             crv = NSC_EncryptUpdate(hSession, pData, ulDataLen, pEncryptedData, 
  871                                     pulEncryptedDataLen);
  872             if (crv != CKR_OK) 
  873                     *pulEncryptedDataLen = 0;
  874             maxoutlen      -= *pulEncryptedDataLen;
  875             pEncryptedData += *pulEncryptedDataLen;
  876             finalLen = maxoutlen;
  877             crv2 = NSC_EncryptFinal(hSession, pEncryptedData, &finalLen);
  878             if (crv2 == CKR_OK) 
  879                     *pulEncryptedDataLen += finalLen;
  880             return crv == CKR_OK ? crv2 : crv;
  881         }
  882         /* doPad without multi means that padding must be done on the first
  883         ** and only update.  There will be no final.
  884         */
  885         PORT_Assert[1](context->blockSize > 1);
  886         if (context->blockSize > 1) {
  887             CK_ULONG remainder = ulDataLen % context->blockSize;
  888             CK_ULONG padding   = context->blockSize - remainder;
  889             pText.len += padding;
  890             pText.data = PORT_ZAlloc(pText.len);
  891             if (pText.data) {
  892                 memcpy(pText.data, pData, ulDataLen);
  893                 memset(pText.data + ulDataLen, padding, padding);
  894             } else {
  895                 crv = CKR_HOST_MEMORY;
  896                 goto fail;
  897             }
  898         }
  899     }
  900 
  901     /* do it: NOTE: this assumes buf size is big enough. */
  902     rv = (*context->update)(context->cipherInfo, pEncryptedData, 
  903                             &outlen, maxoutlen, pText.data, pText.len);
  904     crv = (rv == SECSuccess) ? CKR_OK : CKR_DEVICE_ERROR;
  905     *pulEncryptedDataLen = (CK_ULONG) outlen;
  906     if (pText.data != pData)
  907             PORT_ZFree(pText.data, pText.len);
  908 fail:
  909     sftk_SetContextByType(session, SFTK_ENCRYPT, NULL);
  910     sftk_FreeContext(context);
  911 finish:
  912     sftk_FreeSession(session);
  913 
  914     return crv;
  915 }

  930 /* NSC_DecryptUpdate continues a multiple-part decryption operation. */
  931 CK_RV NSC_DecryptUpdate(CK_SESSION_HANDLE hSession,
  932     CK_BYTE_PTR pEncryptedPart, CK_ULONG ulEncryptedPartLen,
  933                                     CK_BYTE_PTR pPart, CK_ULONG_PTR pulPartLen)
  934 {
  935     SFTKSessionContext *context;
  936     unsigned int padoutlen = 0;
  937     unsigned int outlen;
  938     unsigned int maxout = *pulPartLen;
  939     CK_RV crv;
  940     SECStatus rv;
  941 
  942     /* make sure we're legal */
  943     crv = sftk_GetContext(hSession,&context,SFTK_DECRYPT,PR_TRUE,NULL);
  944     if (crv != CKR_OK) return crv;
  945 
  946     /* this can only happen on an NSS programming error */
  947     PORT_Assert[1]((context->padDataLength == 0) 
  948                 || context->padDataLength == context->blockSize);
  949 
  950 
  951     if (!pPart) {
  952         if (context->doPad) {
  953             /* we can check the data length here because if we are padding,
  954              * then we must be using a block cipher. In the non-padding case
  955              * the error will be returned by the underlying decryption
  956              * function when do do the actual decrypt. We need to do the
  957              * check here to avoid returning a negative length to the caller.
  958               */
  959             if ((ulEncryptedPartLen == 0) ||
  960                 (ulEncryptedPartLen % context->blockSize) != 0) {
  961                 return CKR_ENCRYPTED_DATA_LEN_RANGE;
  962             }
  963             *pulPartLen = 
  964                 ulEncryptedPartLen + context->padDataLength - context->blockSize;
  965             return CKR_OK;
  966         }
  967         /* for stream ciphers there is are no constraints on ulEncryptedPartLen.
  968          * for block ciphers, it must be a multiple of blockSize. The error is
  969          * detected when this function is called again do decrypt the output.
  970          */
  971         *pulPartLen = ulEncryptedPartLen;
  972         return CKR_OK;
  973     }
  974 
  975     if (context->doPad) {
  976         /* first decrypt our saved buffer */
  977         if (context->padDataLength != 0) {
  978                 rv = (*context->update)(context->cipherInfo, pPart, &padoutlen,
  979                  maxout, context->padBuf, context->blockSize);
  980                 if (rv != SECSuccess)  return CKR_DEVICE_ERROR;
  981             pPart += padoutlen;
  982             maxout -= padoutlen;
  983         }
  984         /* now save the final block for the next decrypt or the final */
  985         PORT_Memcpy(context->padBuf,&pEncryptedPart[ulEncryptedPartLen -
  986                                 context->blockSize], context->blockSize);
  987         context->padDataLength = context->blockSize;
  988         ulEncryptedPartLen -= context->padDataLength;
  989     }
  990 
  991     /* do it: NOTE: this assumes buf size in is >= buf size out! */
  992     rv = (*context->update)(context->cipherInfo,pPart, &outlen,
  993                  maxout, pEncryptedPart, ulEncryptedPartLen);
  994     *pulPartLen = (CK_ULONG) (outlen + padoutlen);
  995     return (rv == SECSuccess)  ? CKR_OK : CKR_DEVICE_ERROR;
  996 }

  999 /* NSC_DecryptFinal finishes a multiple-part decryption operation. */
 1000 CK_RV NSC_DecryptFinal(CK_SESSION_HANDLE hSession,
 1001     CK_BYTE_PTR pLastPart, CK_ULONG_PTR pulLastPartLen)
 1002 {
 1003     SFTKSession *session;
 1004     SFTKSessionContext *context;
 1005     unsigned int outlen;
 1006     unsigned int maxout = *pulLastPartLen;
 1007     CK_RV crv;
 1008     SECStatus rv = SECSuccess;
 1009 
 1010     /* make sure we're legal */
 1011     crv = sftk_GetContext(hSession,&context,SFTK_DECRYPT,PR_TRUE,&session);
 1012     if (crv != CKR_OK) return crv;
 1013 
 1014     *pulLastPartLen = 0;
 1015     if (!pLastPart) {
 1016         /* caller is checking the amount of remaining data */
 1017         if (context->padDataLength > 0) {
 1018             *pulLastPartLen = context->padDataLength;
 1019         }
 1020         rv = SECSuccess;
 1021         goto finish;
 1022     }
 1023 
 1024     if (context->doPad) {
 1025         /* decrypt our saved buffer */
 1026         if (context->padDataLength != 0) {
 1027             /* this assumes that pLastPart is big enough to hold the *whole*
 1028              * buffer!!! */
 1029                 rv = (*context->update)(context->cipherInfo, pLastPart, &outlen,
 1030                  maxout, context->padBuf, context->blockSize);
 1031                 if (rv == SECSuccess) {
 1032                 unsigned int padSize = 
 1033                             (unsigned int) pLastPart[context->blockSize-1];
 1034                 if ((padSize > context->blockSize) || (padSize == 0)) {
 1035                     rv = SECFailure;
 1036                 } else {
 1037                     *pulLastPartLen = outlen - padSize;
 1038                 }
 1039             }
 1040         }
 1041     }
 1042 
 1043     sftk_SetContextByType(session, SFTK_DECRYPT, NULL);
 1044     sftk_FreeContext(context);
 1045 finish:
 1046     sftk_FreeSession(session);
 1047     return (rv == SECSuccess) ? CKR_OK : CKR_DEVICE_ERROR;
 1048 }

 1050 /* NSC_Decrypt decrypts encrypted data in a single part. */
 1051 CK_RV NSC_Decrypt(CK_SESSION_HANDLE hSession,
 1052     CK_BYTE_PTR pEncryptedData,CK_ULONG ulEncryptedDataLen,CK_BYTE_PTR pData,
 1053                                                     CK_ULONG_PTR pulDataLen)
 1054 {
 1055     SFTKSession *session;
 1056     SFTKSessionContext *context;
 1057     unsigned int outlen;
 1058     unsigned int maxoutlen = *pulDataLen;
 1059     CK_RV crv;
 1060     CK_RV crv2;
 1061     SECStatus rv = SECSuccess;
 1062 
 1063     /* make sure we're legal */
 1064     crv = sftk_GetContext(hSession,&context,SFTK_DECRYPT,PR_FALSE,&session);
 1065     if (crv != CKR_OK) return crv;
 1066 
 1067     if (!pData) {
 1068         *pulDataLen = ulEncryptedDataLen + context->blockSize;
 1069         goto finish;
 1070     }
 1071 
 1072     if (context->doPad && context->multi) {
 1073         CK_ULONG finalLen;
 1074         /* padding is fairly complicated, have the update and final 
 1075          * code deal with it */
 1076         sftk_FreeSession(session);
 1077         crv = NSC_DecryptUpdate(hSession,pEncryptedData,ulEncryptedDataLen,
 1078                                                         pData, pulDataLen);
 1079         if (crv != CKR_OK) 
 1080             *pulDataLen = 0;
 1081         maxoutlen -= *pulDataLen;
 1082         pData     += *pulDataLen;
 1083         finalLen = maxoutlen;
 1084         crv2 = NSC_DecryptFinal(hSession, pData, &finalLen);
 1085         if (crv2 == CKR_OK) 
 1086             *pulDataLen += finalLen;
 1087         return crv == CKR_OK ? crv2 : crv;
 1088     }
 1089 
 1090     rv = (*context->update)(context->cipherInfo, pData, &outlen, maxoutlen, 
 1091                                         pEncryptedData, ulEncryptedDataLen);
 1092     /* XXX need to do MUCH better error mapping than this. */
 1093     crv = (rv == SECSuccess)  ? CKR_OK : CKR_DEVICE_ERROR;
 1094     if (rv == SECSuccess && context->doPad) {
 1095             CK_ULONG padding = pData[outlen - 1];
 1096         if (padding > context->blockSize || !padding) {
 1097             crv = CKR_ENCRYPTED_DATA_INVALID;
 1098         } else
 1099             outlen -= padding;
 1100     }
 1101     *pulDataLen = (CK_ULONG) outlen;
 1102     sftk_SetContextByType(session, SFTK_DECRYPT, NULL);
 1103     sftk_FreeContext(context);
 1104 finish:
 1105     sftk_FreeSession(session);
 1106     return crv;
 1107 }

 1713 /* NSC_SignInit setups up the signing operations. There are three basic
 1714  * types of signing:
 1715  *        (1) the tradition single part, where "Raw RSA" or "Raw DSA" is applied
 1716  *  to data in a single Sign operation (which often looks a lot like an
 1717  *  encrypt, with data coming in and data going out).
 1718  *        (2) Hash based signing, where we continually hash the data, then apply
 1719  *  some sort of signature to the end.
 1720  *        (3) Block Encryption CBC MAC's, where the Data is encrypted with a key,
 1721  *  and only the final block is part of the mac.
 1722  *
 1723  *  For case number 3, we initialize a context much like the Encryption Context
 1724  *  (in fact we share code). We detect case 3 in C_SignUpdate, C_Sign, and 
 1725  *  C_Final by the following method... if it's not multi-part, and it's doesn't
 1726  *  have a hash context, it must be a block Encryption CBC MAC.
 1727  *
 1728  *  For case number 2, we initialize a hash structure, as well as make it 
 1729  *  multi-part. Updates are simple calls to the hash update function. Final
 1730  *  calls the hashend, then passes the result to the 'update' function (which
 1731  *  operates as a final signature function). In some hash based MAC'ing (as
 1732  *  opposed to hash base signatures), the update function is can be simply a 
 1733  *  copy (as is the case with HMAC).
 1734  */

 1751     session = sftk_SessionFromHandle(hSession);
 1752     if (session == NULL) return CKR_SESSION_HANDLE_INVALID;
 1753     crv = sftk_InitGeneric(session,&context,SFTK_SIGN,&key,hKey,&key_type,
 1754                                                 CKO_PRIVATE_KEY,CKA_SIGN);
 1755     if (crv != CKR_OK) {

 1909 /* MACUpdate is the common implementation for SignUpdate and VerifyUpdate.
 1910  * (sign and verify only very in their setup and final operations) */
 1911 static CK_RV 
 1912 sftk_MACUpdate(CK_SESSION_HANDLE hSession,CK_BYTE_PTR pPart,
 1913                                             CK_ULONG ulPartLen,SFTKContextType type)
 1914 {
 1915     unsigned int outlen;
 1916     SFTKSessionContext *context;
 1917     CK_RV crv;
 1918     SECStatus rv;
 1919 
 1920     /* make sure we're legal */
 1921     crv = sftk_GetContext(hSession,&context,type,PR_TRUE,NULL);
 1922     if (crv != CKR_OK) return crv;
 1923 
 1924     if (context->hashInfo) {
 1925         (*context->hashUpdate)(context->hashInfo, pPart, ulPartLen);
 1926         return CKR_OK;
 1927     }
 1928 
 1929     /* must be block cipher macing */
 1930 
 1931     /* deal with previous buffered data */
 1932     if (context->padDataLength != 0) {
 1933         int i;
 1934         /* fill in the padded to a full block size */
 1935         for (i=context->padDataLength; (ulPartLen != 0) && 
 1936                                         i < (int)context->blockSize; i++) {
 1937             context->padBuf[i] = *pPart++;
 1938             ulPartLen--;
 1939             context->padDataLength++;
 1940         }
 1941 
 1942         /* not enough data to encrypt yet? then return */
 1943         if (context->padDataLength != context->blockSize)  return CKR_OK;
 1944         /* encrypt the current padded data */
 1945             rv = (*context->update)(context->cipherInfo,context->macBuf,&outlen,
 1946                         SFTK_MAX_BLOCK_SIZE,context->padBuf,context->blockSize);
 1947             if (rv != SECSuccess) return CKR_DEVICE_ERROR;
 1948     }
 1949 
 1950     /* save the residual */
 1951     context->padDataLength = ulPartLen % context->blockSize;
 1952     if (context->padDataLength) {
 1953         PORT_Memcpy(context->padBuf,
 1954                     &pPart[ulPartLen-context->padDataLength],
 1955                     context->padDataLength);
 1956         ulPartLen -= context->padDataLength;
 1957     }
 1958 
 1959     /* if we've exhausted our new buffer, we're done */
 1960     if (ulPartLen == 0) { return CKR_OK; }
 1961 
 1962     /* run the data through out encrypter */        
 1963     while (ulPartLen) {
 1964             rv = (*context->update)(context->cipherInfo, context->padBuf, &outlen, 
 1965                         SFTK_MAX_BLOCK_SIZE, pPart, context->blockSize);
 1966         if (rv != SECSuccess) return CKR_DEVICE_ERROR;
 1967         /* paranoia.. make sure we exit the loop */
 1968         PORT_Assert[1](ulPartLen >= context->blockSize);
 1969         if (ulPartLen < context->blockSize) break;
 1970         ulPartLen -= context->blockSize;
 1971     }
 1972 
 1973     return CKR_OK;
 1974         
 1975 }

 1987 /* NSC_SignFinal finishes a multiple-part signature operation, 
 1988  * returning the signature. */
 1989 CK_RV NSC_SignFinal(CK_SESSION_HANDLE hSession,CK_BYTE_PTR pSignature,
 1990                                             CK_ULONG_PTR pulSignatureLen)
 1991 {
 1992     SFTKSession *session;
 1993     SFTKSessionContext *context;
 1994     unsigned int outlen;
 1995     unsigned int digestLen;
 1996     unsigned int maxoutlen = *pulSignatureLen;
 1997     unsigned char tmpbuf[SFTK_MAX_MAC_LENGTH];
 1998     CK_RV crv;
 1999     SECStatus rv = SECSuccess;
 2000 
 2001     /* make sure we're legal */
 2002     *pulSignatureLen = 0;
 2003     crv = sftk_GetContext(hSession,&context,SFTK_SIGN,PR_TRUE,&session);
 2004     if (crv != CKR_OK) return crv;
 2005 
 2006     if (!pSignature) {
 2007         *pulSignatureLen = context->maxLen;
 2008         goto finish;
 2009     } else if (context->hashInfo) {
 2010         (*context->end)(context->hashInfo, tmpbuf, &digestLen, sizeof(tmpbuf));
 2011         rv = (*context->update)(context->cipherInfo, pSignature,
 2012                                         &outlen, maxoutlen, tmpbuf, digestLen);
 2013         *pulSignatureLen = (CK_ULONG) outlen;
 2014     } else {
 2015         /* deal with the last block if any residual */
 2016         if (context->padDataLength) {
 2017             /* fill out rest of pad buffer with pad magic*/
 2018             int i;
 2019             for (i=context->padDataLength; i < (int)context->blockSize; i++) {
 2020                 context->padBuf[i] = 0;
 2021             }
 2022             rv = (*context->update)(context->cipherInfo,context->macBuf,
 2023                 &outlen,SFTK_MAX_BLOCK_SIZE,context->padBuf,context->blockSize);
 2024         }
 2025         if (rv == SECSuccess) {
 2026             PORT_Memcpy(pSignature,context->macBuf,context->macSize);
 2027             *pulSignatureLen = context->macSize;
 2028         }
 2029     }
 2030 
 2031     sftk_FreeContext(context);
 2032     sftk_SetContextByType(session, SFTK_SIGN, NULL);
 2033 
 2034 finish:
 2035     sftk_FreeSession(session);
 2036 
 2037     return (rv == SECSuccess) ? CKR_OK : CKR_DEVICE_ERROR;
 2038 }

 2130         unsigned char *sig, unsigned int sigLen,
 2131         unsigned char *digest, unsigned int digestLen)

 2133 
 2134     SECItem it;

 3035  * FIPS 140-2 pairwise consistency check utilized to validate key pair.
 3036  *
 3037  * This function returns
 3038  *   CKR_OK               if pairwise consistency check passed
 3039  *   CKR_GENERAL_ERROR    if pairwise consistency check failed
 3040  *   other error codes    if paiswise consistency check could not be

 3043 static CK_RV

 3046 {
 3047     /*
 3048      *                      Key type    Mechanism type
 3049      *                      --------------------------------

 3051      *                      others   => CKM_INVALID_MECHANISM

 3054      *                      CKK_DSA  => CKM_DSA
 3055      *                      CKK_EC   => CKM_ECDSA
 3056      *                      others   => CKM_INVALID_MECHANISM
 3057      *
 3058      * None of these mechanisms has a parameter.
 3059      */
 3060     CK_MECHANISM mech = {0, NULL, 0};
 3061 
 3062     CK_ULONG modulusLen;
 3063     PRBool isEncryptable = PR_FALSE;
 3064     PRBool canSignVerify = PR_FALSE;
 3065     PRBool isDerivable = PR_FALSE;
 3066     CK_RV crv;
 3067 
 3068     /* Variables used for Encrypt/Decrypt functions. */
 3069     unsigned char *known_message = (unsigned char *)"Known Crypto Message";
 3070     unsigned char plaintext[PAIRWISE_MESSAGE_LENGTH];
 3071     CK_ULONG bytes_decrypted;
 3072     unsigned char *ciphertext;
 3073     unsigned char *text_compared;
 3074     CK_ULONG bytes_encrypted;
 3075     CK_ULONG bytes_compared;
 3076 
 3077     /* Variables used for Signature/Verification functions. */
 3078     /* always uses SHA-1 digest */

 3081     CK_ULONG signature_length;
 3082

 3084         SFTKAttribute *attribute;
 3085 
 3086         /* Get modulus length of private key. */
 3087         attribute = sftk_FindAttribute(privateKey, CKA_MODULUS);
 3088         if (attribute == NULL) {
 3089             return CKR_DEVICE_ERROR;
 3090         }
 3091         modulusLen = attribute->attrib.ulValueLen;
 3092         if (*(unsigned char *)attribute->attrib.pValue == 0) {
 3093             modulusLen--;
 3094         }

 3096     }
 3097 
 3098     /**************************************************/
 3099     /* Pairwise Consistency Check of Encrypt/Decrypt. */
 3100     /**************************************************/
 3101 
 3102     isEncryptable = sftk_isTrue(privateKey, CKA_DECRYPT); 
 3103 
 3104     /*
 3105      * If the decryption attribute is set, attempt to encrypt
 3106      * with the public key and decrypt with the private key.
 3107      */
 3108     if (isEncryptable) {
 3109         if (keyType != CKK_RSA) {

 3111         }
 3112         bytes_encrypted = modulusLen;
 3113         mech.mechanism = CKM_RSA_PKCS;
 3114 
 3115         /* Allocate space for ciphertext. */
 3116         ciphertext = (unsigned char *) PORT_ZAlloc(bytes_encrypted);
 3117         if (ciphertext == NULL) {
 3118             return CKR_HOST_MEMORY;

 3120 
 3121         /* Prepare for encryption using the public key. */
 3122         crv = NSC_EncryptInit(hSession, &mech, publicKey->handle);
 3123         if (crv != CKR_OK) {

 3125             return crv;
 3126         }
 3127 
 3128         /* Encrypt using the public key. */
 3129         crv = NSC_Encrypt(hSession,
 3130                           known_message,

 3136             return crv;
 3137         }

 3141

 3144          * goes at the end, therefore . . .

 3149          * Check to ensure that ciphertext does
 3150          * NOT EQUAL known input message text
 3151          * per FIPS PUB 140-2 directive.
 3152          */

 3155             /* Set error to Invalid PRIVATE Key. */

 3157             PORT_Free(ciphertext);

 3159         }
 3160

 3162         crv = NSC_DecryptInit(hSession, &mech, privateKey->handle);
 3163         if (crv != CKR_OK) {
 3164             PORT_Free(ciphertext);

 3171          * Initialize bytes decrypted to be the
 3172          * expected PAIRWISE_MESSAGE_LENGTH.

 3177          * Decrypt using the private key.

 3179          *        value of bytes_encrypted.
 3180          */

 3182                           ciphertext,
 3183                           bytes_encrypted,
 3184                           plaintext,

 3188         PORT_Free(ciphertext);
 3189 
 3190         if (crv != CKR_OK) {

 3192         }
 3193 
 3194         /*
 3195          * Check to ensure that the output plaintext
 3196          * does EQUAL known input message text.
 3197          */
 3198         if ((bytes_decrypted != PAIRWISE_MESSAGE_LENGTH) ||

 3208     /* Pairwise Consistency Check of Sign/Verify. */
 3209     /**********************************************/
 3210

 3219             break;
 3220         case CKK_DSA:

 3223             break;

 3232         }

 3235         signature = (unsigned char *) PORT_ZAlloc(signature_length);
 3236         if (signature == NULL) {
 3237             return CKR_HOST_MEMORY;
 3238         }
 3239         
 3240         /* Sign the known hash using the private key. */
 3241         crv = NSC_SignInit(hSession, &mech, privateKey->handle);
 3242         if (crv != CKR_OK) {
 3243             PORT_Free(signature);
 3244             return crv;
 3245         }
 3246 
 3247         crv = NSC_Sign(hSession,
 3248                        known_digest,
 3249                        PAIRWISE_DIGEST_LENGTH,
 3250                        signature,
 3251                        &signature_length);
 3252         if (crv != CKR_OK) {
 3253             PORT_Free(signature);
 3254             return crv;
 3255         }
 3256         
 3257         /* Verify the known hash using the public key. */
 3258         crv = NSC_VerifyInit(hSession, &mech, publicKey->handle);
 3259         if (crv != CKR_OK) {
 3260             PORT_Free(signature);
 3261             return crv;
 3262         }
 3263 
 3264         crv = NSC_Verify(hSession,
 3265                          known_digest,
 3266                          PAIRWISE_DIGEST_LENGTH,
 3267                          signature,
 3268                          signature_length);
 3269 
 3270         /* Free signature data. */
 3271         PORT_Free(signature);
 3272 
 3273         if ((crv == CKR_SIGNATURE_LEN_RANGE) ||
 3274                 (crv == CKR_SIGNATURE_INVALID)) {
 3275             return CKR_GENERAL_ERROR;
 3276         }
 3277         if (crv != CKR_OK) {
 3278             return crv;
 3279         }
 3280     }
 3281 
 3282     /**********************************************/
 3283     /* Pairwise Consistency Check for Derivation  */
 3284     /**********************************************/
 3285 
 3286     isDerivable = sftk_isTrue(privateKey, CKA_DERIVE);
 3287     
 3288     if (isDerivable) {
 3289         /* 
 3290          * We are not doing consistency check for Diffie-Hellman Key - 
 3291          * otherwise it would be here
 3292          * This is also true for Elliptic Curve Diffie-Hellman keys
 3293          * NOTE: EC keys are currently subjected to pairwise
 3294          * consistency check for signing/verification.
 3295          */
 3296         /*
 3297          * FIPS 140-2 had the following pairwise consistency test for
 3298          * public and private keys used for key agreement:
 3299          *   If the keys are used to perform key agreement, then the
 3300          *   cryptographic module shall create a second, compatible
 3301          *   key pair.  The cryptographic module shall perform both
 3302          *   sides of the key agreement algorithm and shall compare
 3303          *   the resulting shared values.  If the shared values are
 3304          *   not equal, the test shall fail.
 3305          * This test was removed in Change Notice 3.
 3306          */
 3307 
 3308     }
 3309 
 3310     return CKR_OK;
 3311 }

 4602         att = sftk_FindAttribute(baseKey,CKA_SENSITIVE);

 4617      * but handleObject always forces this attributes, so we would need to do

 4619     return CKR_OK;

 5720 /* NSC_SetOperationState restores the state of the cryptographic 
 5721  * operation in a session. This is coded like it can restore lots of
 5722  * states, but it only works for truly flat cipher structures. */
 5723 CK_RV NSC_SetOperationState(CK_SESSION_HANDLE hSession, 
 5724         CK_BYTE_PTR  pOperationState, CK_ULONG  ulOperationStateLen,
 5725         CK_OBJECT_HANDLE hEncryptionKey, CK_OBJECT_HANDLE hAuthenticationKey)
 5726 {
 5727     SFTKSessionContext *context;
 5728     SFTKSession *session;
 5729     SFTKContextType type;
 5730     CK_MECHANISM mech;
 5731     CK_RV crv = CKR_OK;
 5732 
 5733     while (ulOperationStateLen != 0) {
 5734         /* get what type of state we're dealing with... */
 5735         PORT_Memcpy(&type,pOperationState, sizeof(SFTKContextType));
 5736 
 5737         /* fix up session contexts based on type */
 5738         session = sftk_SessionFromHandle(hSession);
 5739         if (session == NULL) return CKR_SESSION_HANDLE_INVALID;
 5740         context = sftk_ReturnContextByType(session, type);
 5741         sftk_SetContextByType(session, type, NULL);
 5742         if (context) { 
 5743              sftk_FreeContext(context);
 5744         }
 5745         pOperationState += sizeof(SFTKContextType);
 5746         sftk_Decrement(ulOperationStateLen,sizeof(SFTKContextType));
 5747 
 5748 
 5749         /* get the mechanism structure */
 5750         PORT_Memcpy(&mech.mechanism,pOperationState,sizeof(CK_MECHANISM_TYPE));
 5751         pOperationState += sizeof(CK_MECHANISM_TYPE);
 5752         sftk_Decrement(ulOperationStateLen, sizeof(CK_MECHANISM_TYPE));
 5753         /* should be filled in... but not necessary for hash */
 5754         mech.pParameter = NULL;
 5755         mech.ulParameterLen = 0;
 5756         switch (type) {
 5757         case SFTK_HASH:
 5758             crv = NSC_DigestInit(hSession,&mech);
 5759             if (crv != CKR_OK) break;
 5760             crv = sftk_GetContext(hSession, &context, SFTK_HASH, PR_TRUE, 
 5761                                                                 NULL);
 5762             if (crv != CKR_OK) break;
 5763             PORT_Memcpy(context->cipherInfo,pOperationState,
 5764                                                 context->cipherInfoLen);
 5765             pOperationState += context->cipherInfoLen;
 5766             sftk_Decrement(ulOperationStateLen,context->cipherInfoLen);
 5767             break;
 5768         default:
 5769             /* do sign/encrypt/decrypt later */
 5770             crv = CKR_SAVED_STATE_INVALID;
 5771          }
 5772          sftk_FreeSession(session);
 5773          if (crv != CKR_OK) break;
 5774     }
 5775     return crv;
 5776 }
