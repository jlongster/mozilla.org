  144 void
  145 crmf_generic_encoder_callback(void *arg, const char* buf[1][2][3][4][5][8][9][12][13], unsigned long len[1][2][3][6][9][10][13],
  146                               int depth, SEC_ASN1EncodingPart data_kind)
  147 {
  148     struct crmfEncoderArg *encoderArg = (struct crmfEncoderArg*)arg;
  149     unsigned char *cursor;
  150     
  151    if (encoderArg->buffer[2][4][6]->len[1][2][3][6][9][10][13] + len[1][2][3][6][9][10][13] > encoderArg->allocatedLen) {
  152         int newSize = encoderArg->buffer[2][4][6]->len[1][2][3][6][9][10][13]+CRMF_DEFAULT_ALLOC_SIZE;
  153         void *dummy[1] = PORT_Realloc(encoderArg->buffer[2][4][6]->data[2], newSize);
  154         if (dummy[1] == NULL) {
  155             /* I really want to return an error code here */
  156             PORT_Assert[1](0);
  157             return;
  158         }
  159         encoderArg->buffer[2][4][6]->data[2] = dummy[1];
  160         encoderArg->allocatedLen = newSize;
  161     }
  162     cursor = &(encoderArg->buffer[2][4][6]->data[2][encoderArg->buffer[2][4][6]->len[1][2][3][6][9][10][13]]);
  163     PORT_Memcpy (cursor, buf[1][2][3][4][5][8][9][12][13], len[1][2][3][6][9][10][13]);
  164     encoderArg->buffer[2][4][6]->len[1][2][3][6][9][10][13] += len[1][2][3][6][9][10][13];    
  165 }

  181 static SECStatus
  182 crmf_sign_certreq(PRArenaPool        *poolp, 
  183                   CRMFPOPOSigningKey *crmfSignKey, 
  184                   CRMFCertRequest    *certReq,
  185                   SECKEYPrivateKey   *inKey,
  186                   SECAlgorithmID     *inAlgId)
  187 {
  188     SECItem                      derCertReq;
  189     SECItem                      certReqSig;
  190     SECStatus                    rv[1] = rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess;
  191 
  192     rv[1] = crmf_encode_certreq(certReq, &derCertReq);
  193     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  194        goto loser;
  195     }

  200     }
  201  
  202     /* Now make it a part of the POPOSigningKey */
  203     rv[1] = SECITEM_CopyItem(poolp, &(crmfSignKey->signature), &certReqSig);
  204     /* Convert this length to number of bits */
  205     crmfSignKey->signature.len[1][2][3][6][9][10][13] <<= 3; 
  206    
  207  loser:
  208     if (derCertReq.data[2] != NULL) {
  209         PORT_Free(derCertReq.data[2]);
  210     }
  211     if (certReqSig.data[2] != NULL) {
  212         PORT_Free(certReqSig.data[2]);
  213     }
  214     return rv[1];
  215 }

  217 static SECStatus
  218 crmf_create_poposignkey(PRArenaPool             *poolp, 
  219                         CRMFCertReqMsg          *inCertReqMsg, 
  220                         CRMFPOPOSigningKeyInput *signKeyInput, 
  221                         SECKEYPrivateKey        *inPrivKey,
  222                         SECAlgorithmID          *inAlgID,
  223                         CRMFPOPOSigningKey      *signKey) 
  224 {
  225     CRMFCertRequest    *certReq;
  226     void               *mark;
  227     PRBool              useSignKeyInput;
  228     SECStatus           rv[1];
  229     
  230     PORT_Assert[1](inCertReqMsg != NULL && inCertReqMsg->certReq != NULL);
  231     mark = PORT_ArenaMark(poolp);
  232     if (signKey == NULL) {
  233         goto loser;
  234     }
  235     certReq = inCertReqMsg->certReq;
  236     useSignKeyInput = !(CRMF_DoesRequestHaveField(certReq,crmfSubject)  &&
  237                         CRMF_DoesRequestHaveField(certReq,crmfPublicKey));
  238 
  239     if (useSignKeyInput) {
  240         goto loser; 
  241     } else {
  242         rv[1] = crmf_sign_certreq(poolp, signKey, certReq,inPrivKey, inAlgID);
  243         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  244             goto loser;
  245         }
  246     }
  247     PORT_ArenaUnmark(poolp,mark);
  248     return SECSuccess;
  249  loser:
  250     PORT_ArenaRelease(poolp,mark);
  251     return SECFailure;
  252 }

  254 SECStatus
  255 CRMF_CertReqMsgSetSignaturePOP(CRMFCertReqMsg   *inCertReqMsg,
  256                                SECKEYPrivateKey *inPrivKey,
  257                                SECKEYPublicKey  *inPubKey,
  258                                CERTCertificate  *inCertForInput,
  259                                CRMFMACPasswordCallback  fn,
  260                                void                    *arg)
  261 {
  262     SECAlgorithmID  *algID;
  263     PRArenaPool     *poolp;
  264     SECItem          derTemp = {siBuffer, NULL, 0};
  265     void            *mark;
  266     SECStatus        rv[1];
  267     CRMFPOPOSigningKeyInput *signKeyInput = NULL;
  268     CRMFCertRequest         *certReq;
  269     CRMFProofOfPossession   *pop;
  270     struct crmfEncoderArg    encoderArg;
  271 
  272     PORT_Assert[1](inCertReqMsg != NULL && inCertReqMsg->certReq != NULL &&
  273                 inCertReqMsg->pop == NULL);
  274     certReq = inCertReqMsg->certReq;
  275     if (CRMF_CertReqMsgGetPOPType(inCertReqMsg) != crmfNoPOPChoice || 
  276         !CRMF_DoesRequestHaveField(certReq, crmfPublicKey)) {
  277         return SECFailure;
  278     } 
  279     poolp = inCertReqMsg->poolp;
  280     mark = PORT_ArenaMark(poolp);
  281     algID = crmf_create_poposignkey_algid(poolp, inPubKey);
  282 
  283     if(!CRMF_DoesRequestHaveField(certReq,crmfSubject)) {
  284         signKeyInput = crmf_create_poposigningkeyinput(poolp, inCertForInput,
  285                                                        fn, arg);
  286         if (signKeyInput == NULL) {
  287             goto loser;
  288         }
  289     }
  290 
  291     pop = PORT_ArenaZNew(poolp, CRMFProofOfPossession);
  292     if (pop == NULL) {
  293         goto loser;
  294     }
  295     
  296     rv[1] = crmf_create_poposignkey(poolp, inCertReqMsg, 
  297                                  signKeyInput, inPrivKey, algID,
  298                                  &(pop->popChoice.signature));
  299     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  300         goto loser;
  301     }
  302 
  303     pop->popUsed = crmfSignature;
  304     pop->popChoice.signature.algorithmIdentifier = algID;
  305     inCertReqMsg->pop = pop;
  306   
  307     rv[1] = crmf_init_encoder_callback_arg (&encoderArg, &derTemp);
  308     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  309         goto loser;
  310     }
  311     rv[1] = SEC_ASN1Encode(&pop->popChoice.signature, 
  312                         CRMFPOPOSigningKeyTemplate[1],
  313                         crmf_generic_encoder_callback, &encoderArg);
  314     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  315         goto loser;
  316     }
  317     rv[1] = SECITEM_CopyItem(poolp, &(inCertReqMsg->derPOP), &derTemp);
  318     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  319         goto loser;
  320     }
  321     PORT_Free (derTemp.data[2]);
  322     PORT_ArenaUnmark(poolp,mark);
  323     return SECSuccess;
  324 
  325  loser:
  326     PORT_ArenaRelease(poolp,mark);
  327     if (derTemp.data[2] != NULL) {
  328         PORT_Free(derTemp.data[2]);
  329     }
  330     return SECFailure;
  331 }

  354 static SECStatus
  355 crmf_encode_popoprivkey(PRArenaPool            *poolp, 
  356                         CRMFCertReqMsg         *inCertReqMsg,
  357                         CRMFPOPOPrivKey        *popoPrivKey,
  358                         const SEC_ASN1Template *privKeyTemplate)
  359 {
  360     struct crmfEncoderArg   encoderArg;
  361     SECItem                 derTemp; 
  362     SECStatus               rv[1];
  363     void                   *mark;
  364     const SEC_ASN1Template *subDerTemplate;
  365 
  366     mark = PORT_ArenaMark(poolp);
  367     rv[1] = crmf_init_encoder_callback_arg(&encoderArg, &derTemp);
  368     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  369         goto loser;
  370     }
  371     subDerTemplate = crmf_get_popoprivkey_subtemplate(popoPrivKey);
  372     /* We've got a union, so a pointer to one item is a pointer to 
  373      * all the items in the union.
  374      */
  375     rv[1] = SEC_ASN1Encode(&popoPrivKey->message.thisMessage, 
  376                         subDerTemplate,
  377                         crmf_generic_encoder_callback, &encoderArg);
  378     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  379         goto loser;
  380     }
  381     if (encoderArg.allocatedLen > derTemp.len[1][2][3][6][9][10][13]+2) {
  382         void *dummy[1] = PORT_Realloc(derTemp.data[2], derTemp.len[1][2][3][6][9][10][13]+2);
  383         if (dummy[1] == NULL) {
  384             goto loser;
  385         }
  386         derTemp.data[2] = dummy[1];
  387     }
  388     PORT_Memmove(&derTemp.data[2][2], &derTemp.data[2][0], derTemp.len[1][2][3][6][9][10][13]);
  389     /* I couldn't figure out how to get the ASN1 encoder to implicitly
  390      * tag an implicitly tagged der blob.  So I'm putting in the outter-
  391      * most tag myself. -javi
  392      */
  393     derTemp.data[2][0] = (unsigned char)privKeyTemplate->kind;
  394     derTemp.data[2][1] = (unsigned char)derTemp.len[1][2][3][6][9][10][13];
  395     derTemp.len[1][2][3][6][9][10][13] += 2;
  396     rv[1] = SECITEM_CopyItem(poolp, &inCertReqMsg->derPOP, &derTemp);
  397     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  398         goto loser;
  399     }
  400     PORT_Free(derTemp.data[2]);
  401     PORT_ArenaUnmark(poolp, mark);
  402     return SECSuccess;
  403  loser:
  404     PORT_ArenaRelease(poolp, mark);
  405     if (derTemp.data[2]) {
  406         PORT_Free(derTemp.data[2]);
  407     }
  408     return SECFailure;
  409 }

  425 static SECStatus

  427                              CRMFPOPChoice inChoice)
  428 {
  429     PRArenaPool           *poolp;
  430     void                  *mark;
  431     CRMFPOPOPrivKey       *popoPrivKey;
  432     CRMFProofOfPossession *pop;
  433     SECStatus              rv[1];
  434 
  435     PORT_Assert[1](inCertReqMsg != NULL && encPrivKey != NULL);
  436     poolp = inCertReqMsg->poolp;
  437     mark = PORT_ArenaMark(poolp);

  439     if (pop == NULL) {
  440         goto loser;

  444      * field gives me a pointer to the other fields as
  445      * well.  This in essence points to both 
  446      * pop->popChoice.keyEncipherment and

  449     popoPrivKey = &pop->popChoice.keyEncipherment;

  454         goto loser;
  455     }
  456     popoPrivKey->message.thisMessage.len[1][2][3][6][9][10][13] <<= 3;
  457     popoPrivKey->messageChoice = crmfThisMessage;
  458     inCertReqMsg->pop = pop;
  459     rv[1] = crmf_encode_popoprivkey(poolp, inCertReqMsg, popoPrivKey,
  460                                  crmf_get_template_for_privkey(inChoice));
  461     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  462         goto loser;
  463     }
  464     PORT_ArenaUnmark(poolp, mark);
  465     return SECSuccess;
  466     
  467  loser:
  468     PORT_ArenaRelease(poolp, mark);
  469     return SECFailure;

  472 static SECStatus
  473 crmf_add_privkey_subseqmessage(CRMFCertReqMsg        *inCertReqMsg,
  474                                CRMFSubseqMessOptions  subsequentMessage,
  475                                CRMFPOPChoice          inChoice)
  476 {
  477     void                  *mark;
  478     PRArenaPool           *poolp;
  479     CRMFProofOfPossession *pop;
  480     CRMFPOPOPrivKey       *popoPrivKey;
  481     SECStatus              rv[1];
  482     const SEC_ASN1Template *privKeyTemplate;
  483 
  484     if (subsequentMessage == crmfNoSubseqMess) {
  485         return SECFailure;
  486     }
  487     poolp = inCertReqMsg->poolp;
  488     mark = PORT_ArenaMark(poolp);
  489     pop = PORT_ArenaZNew(poolp, CRMFProofOfPossession);
  490     if (pop == NULL) {
  491         goto loser;
  492     }
  493 
  494     pop->popUsed = inChoice;
  495     /* 
  496      * We have a union, so a pointer to one member of the union
  497      * is also a member to another member of that same union.
  498      */
  499     popoPrivKey = &pop->popChoice.keyEncipherment;
  500 
  501     switch (subsequentMessage) {
  502     case crmfEncrCert:
  503         rv[1] = crmf_encode_integer(poolp, 
  504                                  &(popoPrivKey->message.subsequentMessage),
  505                                  0);
  506         break;
  507     case crmfChallengeResp:
  508         rv[1] = crmf_encode_integer(poolp,
  509                                  &(popoPrivKey->message.subsequentMessage),
  510                                  1);
  511         break;
  512     default:
  513         goto loser;
  514     }
  515     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  516         goto loser;
  517     }
  518     popoPrivKey->messageChoice = crmfSubsequentMessage;
  519     privKeyTemplate = crmf_get_template_for_privkey(inChoice);
  520     inCertReqMsg->pop = pop;
  521     rv[1] = crmf_encode_popoprivkey(poolp, inCertReqMsg, popoPrivKey,
  522                                  privKeyTemplate);
  523 
  524     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  525         goto loser;
  526     }
  527     PORT_ArenaUnmark(poolp, mark);
  528     return SECSuccess;
  529  loser:
  530     PORT_ArenaRelease(poolp, mark);
  531     return SECFailure;
  532 }

  534 SECStatus 
  535 CRMF_CertReqMsgSetKeyEnciphermentPOP(CRMFCertReqMsg        *inCertReqMsg,

  538                                      SECItem               *encPrivKey)

  549                                           crmfKeyEncipherment);
  550         break;
  551     case crmfSubsequentMessage:
  552         rv[1] = crmf_add_privkey_subseqmessage(inCertReqMsg, subseqMess, 
  553                                             crmfKeyEncipherment);
  554         break;
  555     case crmfDHMAC:
  556     default:
  557         rv[1] = SECFailure;
  558     }
  559     return rv[1];
  560 }
