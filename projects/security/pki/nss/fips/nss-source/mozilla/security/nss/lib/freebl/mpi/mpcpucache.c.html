   40 /*
   41  * This file implements a single function: s_mpi_getProcessorLineSize();
   42  * s_mpi_getProcessorLineSize() returns the size in bytes of the cache line
   43  * if a cache exists, or zero if there is no cache. If more than one
   44  * cache line exists, it should return the smallest line size (which is 
   45  * usually the L1 cache).
   46  *
   47  * mp_modexp uses this information to make sure that private key information
   48  * isn't being leaked through the cache.
   49  *
   50  * Currently the file returns good data for most modern x86 processors, and
   51  * reasonable data on 64-bit ppc processors. All other processors are assumed
   52  * to have a cache line size of 32 bytes unless modified by target.mk.
   53  * 
   54  */

   64 /* Generic CPUID function */
   65 #if defined(AMD_64)
   66 static void cpuid(unsigned long op, unsigned long *eax, 
   67                          unsigned long *ebx, unsigned long *ecx, 
   68                          unsigned long *edx)
   69 {
   70         __asm__("cpuid\n\t"
   71                 : "=a" (*eax),
   72                   "=b" (*ebx),
   73                   "=c" (*ecx),
   74                   "=d" (*edx)
   75                 : "0" (op));
   76 }
   77 #elif !defined(_MSC_VER)
   78 static void cpuid(unsigned long op, unsigned long *eax, 
   79                          unsigned long *ebx, unsigned long *ecx, 
   80                          unsigned long *edx)
   81 {
   82 /* sigh GCC isn't smart enough to save the ebx PIC register on it's own
   83  * in this case, so do it by hand. */
   84         __asm__("pushl %%ebx\n\t"
   85                   "cpuid\n\t"
   86                   "mov %%ebx,%1\n\t"
   87                   "popl %%ebx\n\t"
   88                 : "=a" (*eax),
   89                   "=r" (*ebx),
   90                   "=c" (*ecx),
   91                   "=d" (*edx)
   92                 : "0" (op));
   93 }

   95 /*
   96  * try flipping a processor flag to determine CPU type
   97  */
   98 static unsigned long changeFlag(unsigned long flag)
   99 {
  100         unsigned long changedFlags, originalFlags;
  101         __asm__("pushfl\n\t"            /* get the flags */
  102                 "popl %0\n\t"
  103                 "movl %0,%1\n\t"        /* save the original flags */
  104                 "xorl %2,%0\n\t"         /* flip the bit */
  105                 "pushl %0\n\t"          /* set the flags */
  106                 "popfl\n\t"
  107                 "pushfl\n\t"                /* get the flags again (for return) */
  108                 "popl %0\n\t"
  109                 "pushl %1\n\t"                /* restore the original flags */
  110                  "popfl\n\t"
  111                 : "=r" (changedFlags),
  112                   "=r" (originalFlags),
  113                   "=r" (flag)
  114                 : "2" (flag));
  115         return changedFlags ^ originalFlags;
  116 }

  118 #else

  120 /*
  121  * windows versions of the above assembler
  122  */
  123 #define wcpuid __asm __emit 0fh __asm __emit 0a2h
  124 static void cpuid(unsigned long op,    unsigned long *Reax, 
  125     unsigned long *Rebx, unsigned long *Recx, unsigned long *Redx)
  126 {
  127         unsigned long  Leax, Lebx, Lecx, Ledx;
  128         __asm {
  129         pushad
  130         mov     eax,op
  131         wcpuid
  132         mov     Leax,eax
  133         mov     Lebx,ebx
  134         mov     Lecx,ecx
  135         mov     Ledx,edx
  136         popad
  137         }
  138         *Reax = Leax;
  139         *Rebx = Lebx;
  140         *Recx = Lecx;
  141         *Redx = Ledx;
  142 }

  144 static unsigned long changeFlag(unsigned long flag)
  145 {
  146         unsigned long changedFlags, originalFlags;
  147         __asm {
  148                 push eax
  149                 push ebx
  150                 pushfd                         /* get the flags */
  151                 pop  eax
  152                 push eax                /* save the flags on the stack */
  153                 mov  originalFlags,eax  /* save the original flags */
  154                 mov  ebx,flag
  155                 xor  eax,ebx            /* flip the bit */
  156                 push eax                /* set the flags */
  157                 popfd
  158                 pushfd                  /* get the flags again (for return) */
  159                 pop  eax        
  160                 popfd                   /* restore the original flags */
  161                 mov changedFlags,eax
  162                 pop ebx
  163                 pop eax
  164         }
  165         return changedFlags ^ originalFlags;
  166 }

  476 /*
  477  * use the above table to determine the CacheEntryLineSize.
  478  */
  479 static void
  480 getIntelCacheEntryLineSize(unsigned long val, int *level, 
  481                                                 unsigned long *lineSize)
  482 {
  483     CacheType type;
  484 
  485     type = CacheMap[val].type;
  486     /* only interested in data caches */
  487     /* NOTE val = 0x40 is a special value that means no L2 or L3 cache.
  488      * this data check has the side effect of rejecting that entry. If
  489      * that wasn't the case, we could have to reject it explicitly */
  490     if (CacheMap[val].lineSize == 0) {
  491         return;
  492     }
  493     /* look at the caches, skip types we aren't interested in.
  494      * if we already have a value for a lower level cache, skip the
  495      * current entry */
  496     if ((type == Cache_L1)|| (type == Cache_L1d)) {
  497         *level = 1;
  498         *lineSize = CacheMap[val].lineSize;
  499     } else if ((*level >= 2) && ((type == Cache_L2) || (type == Cache_L2d))) {
  500         *level = 2;
  501         *lineSize = CacheMap[val].lineSize;
  502     } else if ((*level >= 3) && ((type == Cache_L3) || (type == Cache_L3d))) {
  503         *level = 3;
  504         *lineSize = CacheMap[val].lineSize;
  505     }
  506     return;
  507 }

  510 static void
  511 getIntelRegisterCacheLineSize(unsigned long val, 
  512                         int *level, unsigned long *lineSize)
  513 {
  514     getIntelCacheEntryLineSize(val >> 24 & 0xff, level, lineSize);
  515     getIntelCacheEntryLineSize(val >> 16 & 0xff, level, lineSize);
  516     getIntelCacheEntryLineSize(val >> 8 & 0xff, level, lineSize);
  517     getIntelCacheEntryLineSize(val & 0xff, level, lineSize);
  518 }

  520 /*
  521  * returns '0' if no recognized cache is found, or if the cache
  522  * information is supported by this processor 
  523  */
  524 static unsigned long
  525 getIntelCacheLineSize(int cpuidLevel)
  526 {
  527     int level = 4;
  528     unsigned long lineSize = 0;
  529     unsigned long eax, ebx, ecx, edx;
  530     int repeat, count;
  531 
  532     if (cpuidLevel < 2) {
  533         return 0;
  534     }
  535 
  536     /* command '2' of the cpuid is intel's cache info call. Each byte of the
  537      * 4 registers contain a potential descriptor for the cache. The CacheMap        
  538      * table maps the cache entry with the processor cache. Register 'al'
  539      * contains a count value that cpuid '2' needs to be called in order to 
  540      * find all the cache descriptors. Only registers with the high bit set
  541      * to 'zero' have valid descriptors. This code loops through all the
  542      * required calls to cpuid '2' and passes any valid descriptors it finds
  543      * to the getIntelRegisterCacheLineSize code, which breaks the registers
  544      * down into their component descriptors. In the end the lineSize of the
  545      * lowest level cache data cache is returned. */
  546     cpuid(2, &eax, &ebx, &ecx, &edx);
  547     repeat = eax & 0xf;
  548     for (count = 0; count < repeat; count++) {
  549         if ((eax & 0x80000000) == 0) {
  550             getIntelRegisterCacheLineSize(eax & 0xffffff00, &level, &lineSize);
  551         }
  552         if ((ebx & 0x80000000) == 0) {
  553             getIntelRegisterCacheLineSize(ebx, &level, &lineSize);
  554         }
  555         if ((ecx & 0x80000000) == 0) {
  556             getIntelRegisterCacheLineSize(ecx, &level, &lineSize);
  557         }
  558         if ((edx & 0x80000000) == 0) {
  559             getIntelRegisterCacheLineSize(edx, &level, &lineSize);
  560         }
  561         if (count+1 != repeat) {
  562             cpuid(2, &eax, &ebx, &ecx, &edx);
  563         }
  564     }
  565     return lineSize;
  566 }

  568 /*
  569  * returns '0' if the cache info is not supported by this processor.
  570  * This is based on the AMD extended cache commands for cpuid. 
  571  * (see "AMD Processor Recognition Application Note" Publication 20734).
  572  * Some other processors use the identical scheme.
  573  * (see "Processor Recognition, Transmeta Corporation").
  574  */
  575 static unsigned long
  576 getOtherCacheLineSize(unsigned long cpuidLevel)
  577 {
  578     unsigned long lineSize = 0;
  579     unsigned long eax, ebx, ecx, edx;
  580 
  581     /* get the Extended CPUID level */
  582     cpuid(0x80000000, &eax, &ebx, &ecx, &edx);
  583     cpuidLevel = eax;
  584 
  585     if (cpuidLevel >= 0x80000005) {
  586         cpuid(0x80000005, &eax, &ebx, &ecx, &edx);
  587         lineSize = ecx & 0xff; /* line Size, L1 Data Cache */
  588     }
  589     return lineSize;
  590 }

  681  *  G5 processor has 128 byte cache, but dcbz only clears 32 bytes to keep
  682  *  these programs happy. dcbzl work if 64 bit instructions are supported.

  684  *  write-back, you can use this code.

  691         register char *a asm("r2") = array;
  692         __asm__ __volatile__( "dcbzl %0,r0" : "=r" (a): "0"(a) );
