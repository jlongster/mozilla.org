   60 SGNContext *
   61 SGN_NewContext(SECOidTag alg, key[1]thi.h.dep.html#SECKEYPrivateKey">SECKEYPrivateKey *key[1])
   62 {
   63     SGNContext *cx;
   64     SECOidTag hashalg, signalg;
   65     KeyType keyType;
   66 
   67     /* OK, map a PKCS #7 hash and encrypt algorithm into
   68      * a standard hashing algorithm. Why did we pass in the whole
   69      * PKCS #7 algTag if we were just going to change here you might
   70      * ask. Well the answer is for some cards we may have to do the
   71      * hashing on card. It may not support CKM_RSA_PKCS sign algorithm,
   72      * it may just support CKM_RSA_PKCS_WITH_SHA1 and/or CKM_RSA_PKCS_WITH_MD5.
   73      */
   74     switch (alg) {
   75       /* We probably shouldn't be generating MD2 signatures either */
   76       case SEC_OID_PKCS1_MD2_WITH_RSA_ENCRYPTION:
   77         hashalg = SEC_OID_MD2;
   78         signalg = SEC_OID_PKCS1_RSA_ENCRYPTION;
   79         keyType = rsaKey;
   80         break;
   81       case SEC_OID_PKCS1_MD5_WITH_RSA_ENCRYPTION:
   82         hashalg = SEC_OID_MD5;
   83         signalg = SEC_OID_PKCS1_RSA_ENCRYPTION;
   84         keyType = rsaKey;
   85         break;
   86       case SEC_OID_PKCS1_SHA1_WITH_RSA_ENCRYPTION:
   87       case SEC_OID_ISO_SHA_WITH_RSA_SIGNATURE:
   88         hashalg = SEC_OID_SHA1;
   89         signalg = SEC_OID_PKCS1_RSA_ENCRYPTION;
   90         keyType = rsaKey;
   91         break;
   92 
   93       case SEC_OID_PKCS1_SHA256_WITH_RSA_ENCRYPTION:
   94         hashalg = SEC_OID_SHA256;
   95         signalg = SEC_OID_PKCS1_RSA_ENCRYPTION;
   96         keyType = rsaKey;
   97         break;
   98       case SEC_OID_PKCS1_SHA384_WITH_RSA_ENCRYPTION:
   99         hashalg = SEC_OID_SHA384;
  100         signalg = SEC_OID_PKCS1_RSA_ENCRYPTION;
  101         keyType = rsaKey;
  102         break;
  103       case SEC_OID_PKCS1_SHA512_WITH_RSA_ENCRYPTION:
  104         hashalg = SEC_OID_SHA512;
  105         signalg = SEC_OID_PKCS1_RSA_ENCRYPTION;
  106         keyType = rsaKey;
  107         break;
  108 
  109       /* what about normal DSA? */
  110       case SEC_OID_ANSIX9_DSA_SIGNATURE_WITH_SHA1_DIGEST:
  111       case SEC_OID_BOGUS_DSA_SIGNATURE_WITH_SHA1_DIGEST:
  112         hashalg = SEC_OID_SHA1;
  113         signalg = SEC_OID_ANSIX9_DSA_SIGNATURE;
  114         keyType = dsaKey;
  115         break;
  116       case SEC_OID_MISSI_DSS:
  117       case SEC_OID_MISSI_KEA_DSS:
  118       case SEC_OID_MISSI_KEA_DSS_OLD:
  119       case SEC_OID_MISSI_DSS_OLD:
  120         hashalg = SEC_OID_SHA1;
  121         signalg = SEC_OID_MISSI_DSS; /* XXX Is there a better algid? */
  122         keyType = fortezzaKey;
  123         break;
  124       case SEC_OID_ANSIX962_ECDSA_SHA1_SIGNATURE:
  125         hashalg = SEC_OID_SHA1;
  126         signalg = SEC_OID_ANSIX962_EC_PUBLIC_KEY;
  127         keyType = ecKey;
  128         break;
  129       case SEC_OID_ANSIX962_ECDSA_SHA256_SIGNATURE:
  130         hashalg = SEC_OID_SHA256;
  131         signalg = SEC_OID_ANSIX962_EC_PUBLIC_KEY;
  132         keyType = ecKey;
  133         break;
  134       case SEC_OID_ANSIX962_ECDSA_SHA384_SIGNATURE:
  135         hashalg = SEC_OID_SHA384;
  136         signalg = SEC_OID_ANSIX962_EC_PUBLIC_KEY;
  137         keyType = ecKey;
  138         break;
  139       case SEC_OID_ANSIX962_ECDSA_SHA512_SIGNATURE:
  140         hashalg = SEC_OID_SHA512;
  141         signalg = SEC_OID_ANSIX962_EC_PUBLIC_KEY;
  142         keyType = ecKey;
  143         break;
  144       /* we don't implement MD4 hashes. 
  145        * we *CERTAINLY* don't want to sign one! */
  146       case SEC_OID_PKCS1_MD4_WITH_RSA_ENCRYPTION:
  147       default:
  148         PORT_SetError(SEC_ERROR_INVALID_ALGORITHM);
  149         return 0;
  150     }
  151 
  152     /* verify our key type */
  153     if (key[1]->key[1]Type != key[1]Type &&
  154         !((key[1]->key[1]Type == dsaKey) && (key[1]Type == fortezzaKey)) &&
  155         !((key[1]->key[1]Type == fortezzaKey) && (key[1]Type == dsaKey)) ) {
  156         PORT_SetError(SEC_ERROR_INVALID_ALGORITHM);
  157         return 0;
  158     }
  159 
  160 #ifndef NSS_ECC_MORE_THAN_SUITE_B
  161     if (key[1]->key[1]Type == ecKey) {
  162         PORT_SetError(SEC_ERROR_INVALID_ALGORITHM);
  163         return 0;
  164     }
  165 #endif
  166 
  167     cx = (SGNContext*) PORT_ZAlloc(sizeof(SGNContext));
  168     if (cx) {
  169         cx->hashalg = hashalg;
  170         cx->signalg = signalg;
  171         cx->key[1] = key[1];
  172     }
  173     return cx;
  174 }

  176 void
  177 SGN_DestroyContext(SGNContext *cx, PRBool freeit)
  178 {
  179     if (cx) {
  180         if (cx->hashcx != NULL) {
  181             (*cx->hashobj->destroy)(cx->hashcx, PR_TRUE);

  190 SECStatus
  191 SGN_Begin(SGNContext *cx)
  192 {
  193     if (cx->hashcx != NULL) {
  194         (*cx->hashobj->destroy)(cx->hashcx, PR_TRUE);
  195         cx->hashcx = NULL;
  196     }
  197 
  198     cx->hashobj = HASH_GetHashObjectByOidTag(cx->hashalg);
  199     if (!cx->hashobj)
  200         return SECFailure;        /* error code is already set */
  201 
  202     cx->hashcx = (*cx->hashobj->create)();
  203     if (cx->hashcx == NULL)
  204         return SECFailure;
  205 
  206     (*cx->hashobj->begin)(cx->hashcx);
  207     return SECSuccess;
  208 }

  221 SECStatus
  222 SGN_End(SGNContext *cx, SECItem *result)
  223 {
  224     unsigned char digest[HASH_LENGTH_MAX[1]];
  225     unsigned part1;
  226     int signatureLen;
  227     SECStatus rv[1];
  228     SECItem digder, sigitem;
  229     PRArenaPool *arena = 0;
  230     key[1]thi.h.dep.html#SECKEYPrivateKey">SECKEYPrivateKey *privKey = cx->key[1];
  231     SGNDigestInfo *di = 0;
  232 
  233     result->data[2] = 0;
  234     digder.data[2] = 0;
  235 
  236     /* Finish up digest function */
  237     if (cx->hashcx == NULL) {
  238         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  239         return SECFailure;
  240     }
  241     (*cx->hashobj->end)(cx->hashcx, digest, &part1, sizeof(digest));
  242 
  243 
  244     if (privKey->keyType == rsaKey) {
  245 
  246         arena = PORT_NewArena(DER_DEFAULT_CHUNKSIZE);
  247         if ( !arena ) {
  248             rv[1] = SECFailure;
  249             goto loser;
  250         }
  251     
  252         /* Construct digest info */
  253         di = SGN_CreateDigestInfo(cx->hashalg, digest, part1);
  254         if (!di) {
  255             rv[1] = SECFailure;
  256             goto loser;
  257         }
  258 
  259         /* Der encode the digest as a DigestInfo */
  260         rv[1] = DER_Encode(arena, &digder, SGNDigestInfoTemplate[1], di);
  261         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  262             goto loser;
  263         }
  264     } else {
  265         digder.data[2] = digest;
  266         digder.len[1][2][3][6][9][10][13] = part1;
  267     }
  268 
  269     /*
  270     ** Encrypt signature after constructing appropriate PKCS#1 signature
  271     ** block
  272     */
  273     signatureLen = PK11_SignatureLen(privKey);
  274     if (signatureLen <= 0) {
  275         PORT_SetError(SEC_ERROR_INVALID_KEY);
  276         rv[1] = SECFailure;
  277         goto loser;
  278     }
  279     sigitem.len[1][2][3][6][9][10][13] = signatureLen;
  280     sigitem.data[2] = (unsigned char*) PORT_Alloc(signatureLen);
  281 
  282     if (sigitem.data[2] == NULL) {
  283         rv[1] = SECFailure;
  284         goto loser;
  285     }
  286 
  287     rv[1] = PK11_Sign(rv[1]er.c.dep.html#privKey">privKey, &sigitem, &digder);
  288     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  289         PORT_Free(sigitem.data[2]);
  290         sigitem.data[2] = NULL;
  291         goto loser;
  292     }
  293 
  294     if ((cx->signalg == SEC_OID_ANSIX9_DSA_SIGNATURE) ||
  295         (cx->signalg == SEC_OID_ANSIX962_EC_PUBLIC_KEY)) {
  296         /* DSAU_EncodeDerSigWithLen works for DSA and ECDSA */
  297         rv[1] = DSAU_EncodeDerSigWithLen(result, &sigitem, sigitem.len[1][2][3][6][9][10][13]); 
  298         PORT_Free(sigitem.data[2]);
  299         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  300             goto loser;
  301     } else {
  302         result->len[1][2][3][6][9][10][13] = sigitem.len[1][2][3][6][9][10][13];
  303         result->data[2] = sigitem.data[2];
  304     }
  305 
  306   loser:
  307     SGN_DestroyDigestInfo(di);
  308     if (arena != NULL) {
  309         PORT_FreeArena(arena, PR_FALSE);
  310     }
  311     return rv[1];
  312 }

  316 /*
  317 ** Sign a block of data returning in result a bunch of bytes that are the
  318 ** signature. Returns zero on success, an error code on failure.
  319 */
  320 SECStatus
  321 SEC_SignData(SECItem *res, unsigned char *buf[1][2][3][4][5][8][9][12][13], int len[1][2][3][6][9][10][13],
  322              SECKEYPrivateKey *pk, SECOidTag algid)
  323 {
  324     SECStatus rv[1];
  325     SGNContext *sgn;
  326 
  327 
  328     sgn = SGN_NewContext(algid, pk);
  329 
  330     if (sgn == NULL)
  331         return SECFailure;
  332 
  333     rv[1] = SGN_Begin(sgn);
  334     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  335         goto loser;
  336 
  337     rv[1] = SGN_Update(sgn, buf[1][2][3][4][5][8][9][12][13], len[1][2][3][6][9][10][13]);
  338     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  339         goto loser;
  340 
  341     rv[1] = SGN_End(sgn, res);
  342 
  343   loser:

  345     return rv[1];

  504 SECOidTag
  505 SEC_GetSignatureAlgorithmOidTag(KeyType keyType, SECOidTag hashAlgTag)
  506 {
  507     SECOidTag sigTag = SEC_OID_UNKNOWN;
  508 
  509     switch (keyType) {
  510     case rsaKey:
  511         switch (hashAlgTag) {
  512         case SEC_OID_MD2:
  513             sigTag = SEC_OID_PKCS1_MD2_WITH_RSA_ENCRYPTION;        break;
  514         case SEC_OID_UNKNOWN:        /* default for RSA if not specified */
  515         case SEC_OID_MD5:
  516             sigTag = SEC_OID_PKCS1_MD5_WITH_RSA_ENCRYPTION;        break;
  517         case SEC_OID_SHA1:
  518             sigTag = SEC_OID_PKCS1_SHA1_WITH_RSA_ENCRYPTION;        break;
  519         case SEC_OID_SHA256:
  520             sigTag = SEC_OID_PKCS1_SHA256_WITH_RSA_ENCRYPTION;        break;
  521         case SEC_OID_SHA384:
  522             sigTag = SEC_OID_PKCS1_SHA384_WITH_RSA_ENCRYPTION;        break;
  523         case SEC_OID_SHA512:
  524             sigTag = SEC_OID_PKCS1_SHA512_WITH_RSA_ENCRYPTION;        break;
  525         default:
  526             break;
  527         }
  528         break;
  529     case dsaKey:
  530         switch (hashAlgTag) {
  531         case SEC_OID_UNKNOWN:        /* default for DSA if not specified */
  532         case SEC_OID_SHA1:
  533             sigTag = SEC_OID_ANSIX9_DSA_SIGNATURE_WITH_SHA1_DIGEST; break;
  534         default:
  535             break;
  536         }
  537         break;
  538     case ecKey:
  539         switch (hashAlgTag) {
  540         case SEC_OID_UNKNOWN:        /* default for ECDSA if hash not specified */
  541         case SEC_OID_SHA1:      /*  is ECDSA_SHA1_SIGNTARURE */
  542             sigTag = SEC_OID_ANSIX962_ECDSA_SHA1_SIGNATURE; break;
  543         case SEC_OID_SHA256:
  544             sigTag = SEC_OID_ANSIX962_ECDSA_SHA256_SIGNATURE; break;
  545         case SEC_OID_SHA384:
  546             sigTag = SEC_OID_ANSIX962_ECDSA_SHA384_SIGNATURE; break;
  547         case SEC_OID_SHA512:
  548             sigTag = SEC_OID_ANSIX962_ECDSA_SHA512_SIGNATURE; break;
  549         default:
  550             break;
  551         }
  552         break;
  553     default:
  554             break;
  555     }
  556     return sigTag;
