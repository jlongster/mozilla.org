   78 typedef clock_t timetype;
   79 #define gettime(x) *(x) = clock()
   80 #define subtime(a, b) a -= b
   81 #define msec(x) ((clock_t)((double)x * 1000.0 / CLOCKS_PER_SEC))
   82 #define sec(x) (x / CLOCKS_PER_SEC)
   83 
   84 struct TimingContextStr {
   85     timetype start;
   86     timetype end;
   87     timetype interval;
   88 
   89     int   minutes;  
   90     int   seconds;  
   91     int   millisecs;
   92 };
   93 
   94 typedef struct TimingContextStr">TimingContextStr TimingContext;
   95 
   96 TimingContext *CreateTimingContext(void) 
   97 {
   98     return (TimingContext *)malloc(sizeof(TimingContext));
   99 }
  100 
  101 void DestroyTimingContext(TimingContext *ctx) 
  102 {
  103     free(ctx);
  104 }
  105 
  106 void TimingBegin(TimingContext *ctx) 
  107 {
  108     gettime(&ctx->start);
  109 }
  110 
  111 static void timingUpdate(TimingContext *ctx) 
  112 {
  113 
  114     ctx->millisecs = msec(ctx->interval) % 1000;
  115     ctx->seconds   = sec(ctx->interval);
  116     ctx->minutes   = ctx->seconds / 60;
  117     ctx->seconds  %= 60;
  118 
  119 }
  120 
  121 void TimingEnd(TimingContext *ctx) 
  122 {
  123     gettime(&ctx->end);
  124     ctx->interval = ctx->end;
  125     subtime(ctx->interval, ctx->start);
  126     timingUpdate(ctx);
  127 }

  129 char *TimingGenerateString(TimingContext *ctx) 
  130 {
  131     static char sBuf[4096];
  132 
  133     sprintf(sBuf, "%d minutes, %d.%03d seconds", ctx->minutes,
  134                                 ctx->seconds, ctx->millisecs);
  135     return sBuf;
  136 }

  138 static void
  139 dumpBytes( unsigned char * b, int l)
  140 {
  141     int i;
  142     if (l <= 0)
  143         return;
  144     for (i = 0; i < l; ++i) {
  145         if (i % 16 == 0)
  146             printf("\t");
  147         printf(" %02x", b[i]);
  148         if (i % 16 == 15)
  149             printf("\n");
  150     }
  151     if ((i % 16) != 0)
  152         printf("\n");
  153     printf("\n");
  154 }

  156 static mp_err
  157 testNewFuncs(const unsigned char * modulusBytes, int modulus_len)
  158 {
  159     mp_err mperr        = MP_OKAY;
  160     mp_int modulus;
  161     unsigned char buf[512];
  162 
  163     mperr = mp_init(&modulus);
  164     mperr = mp_read_unsigned_octets(&modulus,  modulusBytes,  modulus_len );
  165     mperr = mp_to_fixlen_octets(&modulus, buf, modulus_len);
  166     mperr = mp_to_fixlen_octets(&modulus, buf, modulus_len+1);
  167     mperr = mp_to_fixlen_octets(&modulus, buf, modulus_len+4);
  168     mperr = mp_to_unsigned_octets(&modulus, buf, modulus_len);
  169     mperr = mp_to_signed_octets(&modulus, buf, modulus_len + 1);
  170     mp_clear(&modulus);
  171     return mperr;
  172 }

  174 int
  175 testModExp( const unsigned char *      modulusBytes, 
  176             const unsigned int         expo,
  177             const unsigned char *      input, 
  178                   unsigned char *      output,
  179                   int                  modulus_len)
  180 {
  181     mp_err mperr        = MP_OKAY;
  182     mp_int modulus;
  183     mp_int base;
  184     mp_int exponent;
  185     mp_int result;
  186 
  187     mperr  = mp_init(&modulus);
  188     mperr += mp_init(&base);
  189     mperr += mp_init(&exponent);
  190     mperr += mp_init(&result);
  191     /* we initialize all mp_ints unconditionally, even if some fail.
  192     ** This guarantees that the DIGITS pointer is valid (even if null).
  193     ** So, mp_clear will do the right thing below. 
  194     */
  195     if (mperr == MP_OKAY) {
  196       mperr  = mp_read_unsigned_octets(&modulus,  
  197                 modulusBytes + (sizeof default_n - modulus_len),  modulus_len );
  198       mperr += mp_read_unsigned_octets(&base,     input,         modulus_len );
  199       mp_set(&exponent, expo);
  200       if (mperr == MP_OKAY) {
  201 #if OLD_WAY
  202         mperr = s_mp_exptmod(&base, &exponent, &modulus, &result);
  203 #else
  204         mperr = mp_exptmod(&base, &exponent, &modulus, &result);
  205 #endif
  206         if (mperr == MP_OKAY) {
  207           mperr = mp_to_fixlen_octets(&result, output, modulus_len);
  208         }
  209       }
  210     }
  211     mp_clear(&base);
  212     mp_clear(&result);
  213 
  214     mp_clear(&modulus);
  215     mp_clear(&exponent);
  216 
  217     return (int)mperr;
  218 }

  220 int
  221 doModExp(   const unsigned char *      modulusBytes, 
  222             const unsigned char *      exponentBytes, 
  223             const unsigned char *      input, 
  224                   unsigned char *      output,
  225                   int                  modulus_len)
  226 {
  227     mp_err mperr        = MP_OKAY;
  228     mp_int modulus;
  229     mp_int base;
  230     mp_int exponent;
  231     mp_int result;
  232 
  233     mperr  = mp_init(&modulus);
  234     mperr += mp_init(&base);
  235     mperr += mp_init(&exponent);
  236     mperr += mp_init(&result);
  237     /* we initialize all mp_ints unconditionally, even if some fail.
  238     ** This guarantees that the DIGITS pointer is valid (even if null).
  239     ** So, mp_clear will do the right thing below. 
  240     */
  241     if (mperr == MP_OKAY) {
  242       mperr  = mp_read_unsigned_octets(&modulus,  
  243                 modulusBytes + (sizeof default_n - modulus_len),  modulus_len );
  244       mperr += mp_read_unsigned_octets(&exponent, exponentBytes, modulus_len );
  245       mperr += mp_read_unsigned_octets(&base,     input,         modulus_len );
  246       if (mperr == MP_OKAY) {
  247 #if OLD_WAY
  248         mperr = s_mp_exptmod(&base, &exponent, &modulus, &result);
  249 #else
  250         mperr = mp_exptmod(&base, &exponent, &modulus, &result);
  251 #endif
  252         if (mperr == MP_OKAY) {
  253           mperr = mp_to_fixlen_octets(&result, output, modulus_len);
  254         }
  255       }
  256     }
  257     mp_clear(&base);
  258     mp_clear(&result);
  259 
  260     mp_clear(&modulus);
  261     mp_clear(&exponent);
  262 
  263     return (int)mperr;
  264 }

  266 int
  267 main(int argc, char **argv)
  268 {
  269     TimingContext *          timeCtx;
  270     char *                  progName;
  271     long                   iters         = DEFAULT_ITERS;
  272     unsigned int           modulus_len;
  273     int                           i;
  274     int                   rv;
  275     unsigned char           buf [1024];
  276     unsigned char           buf2[1024];
  277 
  278     progName = strrchr(argv[0], '/');
  279     if (!progName)
  280         progName = strrchr(argv[0], '\\');
  281     progName = progName ? progName+1 : argv[0];
  282 
  283     if (argc >= 2) {
  284         iters = atol(argv[1]);
  285     }
  286 
  287     if (argc >= 3) {
  288         modulus_len = atol(argv[2]);
  289     } else 
  290         modulus_len = sizeof default_n;
  291 
  292     /* no library init function !? */
  293 
  294     memset(buf, 0x41, sizeof buf);
  295 
  296   if (iters < 2) {
  297     testNewFuncs( default_n, modulus_len);
  298     testNewFuncs( default_n+1, modulus_len - 1);
  299     testNewFuncs( default_n+2, modulus_len - 2);
  300     testNewFuncs( default_n+3, modulus_len - 3);
  301 
  302     printf("%lu allocations, %lu frees, %lu copies\n", mp_allocs[1], mp_frees[1], mp_copies[1]);
  303     rv = testModExp(default_n, 0, buf, buf2, modulus_len);
  304     dumpBytes((unsigned char *)buf2, modulus_len);
  305 
  306     printf("%lu allocations, %lu frees, %lu copies\n", mp_allocs[1], mp_frees[1], mp_copies[1]);
  307     rv = testModExp(default_n, 1, buf, buf2, modulus_len);
  308     dumpBytes((unsigned char *)buf2, modulus_len);
  309 
  310     printf("%lu allocations, %lu frees, %lu copies\n", mp_allocs[1], mp_frees[1], mp_copies[1]);
  311     rv = testModExp(default_n, 2, buf, buf2, modulus_len);
  312     dumpBytes((unsigned char *)buf2, modulus_len);
  313 
  314     printf("%lu allocations, %lu frees, %lu copies\n", mp_allocs[1], mp_frees[1], mp_copies[1]);
  315     rv = testModExp(default_n, 3, buf, buf2, modulus_len);
  316     dumpBytes((unsigned char *)buf2, modulus_len);
  317   }
  318     printf("%lu allocations, %lu frees, %lu copies\n", mp_allocs[1], mp_frees[1], mp_copies[1]);
  319     rv = doModExp(default_n, default_d, buf, buf2, modulus_len);
  320     if (rv != 0) {
  321         fprintf(stderr, "Error in modexp operation:\n");
  322         exit(1);
  323     }
  324     dumpBytes((unsigned char *)buf2, modulus_len);
  325     printf("%lu allocations, %lu frees, %lu copies\n", mp_allocs[1], mp_frees[1], mp_copies[1]);
  326 
  327     timeCtx = CreateTimingContext();
  328     TimingBegin(timeCtx);
  329     i = iters;
  330     while (i--) {
  331         rv = doModExp(default_n, default_d, buf, buf2, modulus_len);
  332         if (rv != 0) {
  333             fprintf(stderr, "Error in modexp operation\n");
  334             exit(1);
  335         }
  336     }
  337     TimingEnd(timeCtx);
  338     printf("%ld iterations in %s\n", iters, TimingGenerateString(timeCtx));
  339     printf("%lu allocations, %lu frees, %lu copies\n", mp_allocs[1], mp_frees[1], mp_copies[1]);
  340 
  341     return 0;
