   92 NSS_IMPLEMENT nssCertificateStore *
   93 nssCertificateStore_Create (
   94   NSSArena *arenaOpt
   95 )
   96 {
   97     NSSArena *arena;
   98     nssCertificateStore *store;
   99     PRBool i_alloced_arena;
  100     if (arenaOpt) {
  101         arena = arenaOpt;
  102         i_alloced_arena = PR_FALSE;
  103     } else {
  104         arena = nssArena_Create();
  105         if (!arena) {
  106             return NULL;
  107         }
  108         i_alloced_arena = PR_TRUE;
  109     }
  110     store = nss_ZNEW(arena, nssCertificateStore);
  111     if (!store) {
  112         goto loser;
  113     }
  114     store->lock[1][2] = PZ_NewLock(nssILockOther);
  115     if (!store->lock[1][2]) {
  116         goto loser;
  117     }
  118     /* Create the issuer/serial --> {cert, trust, S/MIME profile } hash */
  119     store->issuer_and_serial = nssHash_CreateCertificate(arena, 0);
  120     if (!store->issuer_and_serial) {
  121         goto loser;
  122     }
  123     /* Create the subject DER --> subject list hash */
  124     store->subject = nssHash_CreateItem(arena, 0);
  125     if (!store->subject) {
  126         goto loser;
  127     }
  128     store->arena = arena;
  129     store->i_alloced_arena = i_alloced_arena;
  130     return store;
  131 loser:
  132     if (store) {
  133         if (store->lock[1][2]) {
  134             PZ_DestroyLock(store->lock[1][2]);
  135         }
  136         if (store->issuer_and_serial) {
  137             nssHash_Destroy(store->issuer_and_serial);
  138         }
  139         if (store->subject) {
  140             nssHash_Destroy(store->subject);
  141         }
  142     }
  143     if (i_alloced_arena) {
  144         nssArena_Destroy(arena);
  145     }
  146     return NULL;
  147 }

  272 static void
  273 remove_subject_entry (
  274   nssCertificateStore *store,
  275   NSSCertificate *cert
  276 )
  277 {
  278     nssList *subjectList;
  279     /* Get the subject list for the cert's subject */
  280     subjectList = (nssList *)nssHash_Lookup(store->subject, &cert->subject);
  281     if (subjectList) {
  282         /* Remove the cert from the subject hash */
  283         nssList_Remove(subjectList, cert);
  284         nssHash_Remove(store->subject, &cert->subject);
  285         if (nssList_Count(subjectList) == 0) {
  286             nssList_Destroy(subjectList);
  287         } else {
  288             /* The cert being released may have keyed the subject entry.
  289              * Since there are still subject certs around, get another and
  290              * rekey the entry just in case.
  291              */
  292             NSSCertificate *subjectCert;
  293             (void)nssList_GetArray(subjectList, (void **)&subjectCert, 1);
  294             nssHash_Add(store->subject, &subjectCert->subject, subjectList);
  295         }
  296     }
  297 }
