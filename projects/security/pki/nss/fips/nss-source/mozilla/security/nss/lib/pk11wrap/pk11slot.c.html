  420 /* Destroy all info on a slot we have built up */
  421 void
  422 PK11_DestroySlot(PK11SlotInfo *slot)
  423 {
  424    /* free up the cached keys and sessions */
  425    PK11_CleanKeyList(slot);
  426    
  427    /* free up all the sessions on this slot */
  428    if (slot->functionList) {
  429         PK11_GETTAB(slot)->C_CloseAllSessions(slot->slotID);
  430    }
  431 
  432    if (slot->mechanismList) {
  433         PORT_Free(slot->mechanismList);
  434    }
  435    if (slot->isThreadSafe && slot->sessionLock) {
  436         PZ_DestroyLock(slot->sessionLock);
  437    }
  438    slot->sessionLock = NULL;
  439    if (slot->freeListLock) {
  440         PZ_DestroyLock(slot->freeListLock);
  441         slot->freeListLock = NULL;
  442    }
  443 
  444    /* finally Tell our parent module that we've gone away so it can unload */
  445    if (slot->module) {
  446         SECMOD_SlotDestroyModule(slot->module,PR_TRUE);
  447    }
  448 
  449    /* ok, well not quit finally... now we free the memory */
  450    PORT_Free(slot);
  451 }
