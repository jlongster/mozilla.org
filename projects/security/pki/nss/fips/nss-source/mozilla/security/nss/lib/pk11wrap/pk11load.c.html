   64 /* build the PKCS #11 2.01 lock files */
   65 CK_RV PR_CALLBACK secmodCreateMutext(CK_VOID_PTR_PTR pmutex) {
   66     *pmutex = (CK_VOID_PTR) PZ_NewLock(nssILockOther);
   67     if ( *pmutex ) return CKR_OK;
   68     return CKR_HOST_MEMORY;
   69 }
   70 
   71 CK_RV PR_CALLBACK secmodDestroyMutext(CK_VOID_PTR mutext) {
   72     PZ_DestroyLock((PZLock *)mutext);
   73     return CKR_OK;
   74 }
   75 
   76 CK_RV PR_CALLBACK secmodLockMutext(CK_VOID_PTR mutext) {
   77     PZ_Lock((PZLock *)mutext);
   78     return CKR_OK;
   79 }
   80 
   81 CK_RV PR_CALLBACK secmodUnlockMutext(CK_VOID_PTR mutext) {
   82     PZ_Unlock((PZLock *)mutext);
   83     return CKR_OK;
   84 }

   98 /* set global options for NSS PKCS#11 module loader */
   99 SECStatus pk11_setGlobalOptions(PRBool noSingleThreadedModules,
  100                                 PRBool allowAlreadyInitializedModules,
  101                                 PRBool dontFinalizeModules)
  102 {
  103     if (noSingleThreadedModules) {
  104         loadSingleThreadedModules = PR_FALSE;
  105     } else {
  106         loadSingleThreadedModules = PR_TRUE;
  107     }
  108     if (allowAlreadyInitializedModules) {
  109         enforceAlreadyInitializedError = PR_FALSE;
  110     } else {
  111         enforceAlreadyInitializedError = PR_TRUE;
  112     }
  113     if (dontFinalizeModules) {
  114         finalizeModules = PR_FALSE;
  115     } else {
  116         finalizeModules = PR_TRUE;
  117     }
  118     return SECSuccess;
  119 }

  126 /*
  127  * collect the steps we need to initialize a module in a single function
  128  */
  129 SECStatus
  130 secmod_ModuleInit(SECMODModule *mod, PRBool* alreadyLoaded)
  131 {
  132     CK_C_INITIALIZE_ARGS moduleArgs;
  133     CK_VOID_PTR pInitArgs;
  134     CK_RV crv;
  135 
  136     if (!mod || !alreadyLoaded) {
  137         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  138         return SECFailure;
  139     }
  140 
  141     if (mod->isThreadSafe == PR_FALSE) {
  142         pInitArgs = NULL;
  143     } else if (mod->libraryParams == NULL) {
  144         pInitArgs = (void *) &secmodLockFunctions;
  145     } else {
  146         moduleArgs = secmodLockFunctions;
  147         moduleArgs.LibraryParameters = (void *) mod->libraryParams;
  148         pInitArgs = &moduleArgs;
  149     }
  150     crv = PK11_GETTAB(mod)->C_Initialize(pInitArgs);
  151     if ((CKR_CRYPTOKI_ALREADY_INITIALIZED == crv) &&
  152         (!enforceAlreadyInitializedError)) {
  153         *alreadyLoaded = PR_TRUE;
  154         return SECSuccess;
  155     }
  156     if (crv != CKR_OK) {
  157         if (pInitArgs == NULL ||
  158                 crv == CKR_NETSCAPE_CERTDB_FAILED ||
  159                 crv == CKR_NETSCAPE_KEYDB_FAILED) {
  160             PORT_SetError(PK11_MapError(crv));
  161             return SECFailure;
  162         }
  163         if (!loadSingleThreadedModules) {
  164             PORT_SetError(SEC_ERROR_INCOMPATIBLE_PKCS11);
  165             return SECFailure;
  166         }
  167         mod->isThreadSafe = PR_FALSE;
  168             crv = PK11_GETTAB(mod)->C_Initialize(NULL);
  169         if ((CKR_CRYPTOKI_ALREADY_INITIALIZED == crv) &&
  170             (!enforceAlreadyInitializedError)) {
  171             *alreadyLoaded = PR_TRUE;
  172             return SECSuccess;
  173         }
  174             if (crv != CKR_OK)  {
  175             PORT_SetError(PK11_MapError(crv));
  176             return SECFailure;
  177         }
  178     }
  179     return SECSuccess;
  180 }

  239     /* intenal modules get loaded from their internal list */
  240     if (mod->internal) {
  241         /* internal, statically get the C_GetFunctionList function */
  242         if (mod->isFIPS) {

  255         /* Not internal, load the DLL and look up C_GetFunctionList */
  256         if (mod->dllName == NULL) {
  257             return SECFailure;
  258         }
  259 
  260 #ifdef notdef
  261         /* look up the library name */
  262         full_name = PR_GetLibraryName(PR_GetLibraryPath(),mod->dllName);
  263         if (full_name == NULL) {
  264             return SECFailure;
  265         }
  266 #else
  267         full_name = PORT_Strdup(mod->dllName);
  268 #endif
  269 
  270         /* load the library. If this succeeds, then we have to remember to
  271          * unload the library if anything goes wrong from here on out...
  272          */
  273         library = PR_LoadLibrary(full_name);
  274         mod->library = (void *)library;
  275         PORT_Free(full_name);
  276         if (library == NULL) {
  277             return SECFailure;
  278         }
  279 
  280         /*
  281          * now we need to get the entry point to find the function pointers

  283         if (!mod->moduleDBOnly) {

  290         }

  292         if (entry[1] == NULL) {
  293             if (mod->isModuleDB) {
  294                 mod->loaded = PR_TRUE;
  295                 mod->moduleDBOnly = PR_TRUE;
  296                 return SECSuccess;
  297             }
  298             PR_UnloadLibrary(library);
  299             return SECFailure;
  300         }
  301     }

  305      */

  395 SECStatus
  396 SECMOD_UnloadModule(SECMODModule *mod) {
  397     PRLibrary[1] *library;
  398 
  399     if (!mod->loaded) {
  400         return SECFailure;
  401     }
  402     if (finalizeModules) {
  403         if (!mod->moduleDBOnly) PK11_GETTAB(mod)->C_Finalize(NULL);
  404     }
  405     mod->moduleID = 0;
  406     mod->loaded = PR_FALSE;
  407     
  408     /* do we want the semantics to allow unloading the internal library?
  409      * if not, we should change this to SECFailure and move it above the
  410      * mod->loaded = PR_FALSE; */
  411     if (mod->internal) {
  412         return SECSuccess;
  413     }
  414 
  415     library = (PRLibrary[1] *)mod->library;
  416     /* paranoia */
  417     if (library == NULL) {
  418         return SECFailure;
  419     }
  420 
  421     PR_UnloadLibrary(library);
  422     return SECSuccess;
  423 }
