   44 /*
   45  * RSA block types
   46  *
   47  * The actual values are important -- they are fixed, *not* arbitrary.
   48  * The explicit value assignments are not needed (because C would give
   49  * us those same values anyway) but are included as a reminder...
   50  */
   51 typedef enum {
   52     RSA_BlockPrivate0 = 0,        /* unused, really */
   53     RSA_BlockPrivate = 1,        /* pad for a private-key operation */
   54     RSA_BlockPublic = 2,        /* pad for a public-key operation */
   55     RSA_BlockOAEP = 3,                /* use OAEP padding */
   56                                 /* XXX is this only for a public-key
   57                                    operation? If so, add "Public" */
   58     RSA_BlockRaw = 4,                /* simply justify the block appropriately */
   59     RSA_BlockTotal
   60 } RSA_BlockType;
   61 
   62 #define NSS_SOFTOKEN_DEFAULT_CHUNKSIZE   2048
   63 
   64 /*
   65  * FIPS 140-2 auditing
   66  */
   67 typedef enum {
   68     NSS_AUDIT_ERROR = 3,    /* errors */
   69     NSS_AUDIT_WARNING = 2,  /* warning messages */
   70     NSS_AUDIT_INFO = 1      /* informational messages */
   71 } NSSAuditSeverity;
   72 
   73 #endif /* _SOFTOKNT_H_ */
