  346     0x20204004, 0x20204014, 0x20204404, 0x20204414 
  347   }
  348 };

  350 /*
  351  * The PC-1 Permutation
  352  * If we number the bits of the 8 bytes of key input like this (in octal):
  353  *     00 01 02 03 04 05 06 07 
  354  *     10 11 12 13 14 15 16 17 
  355  *     20 21 22 23 24 25 26 27 
  356  *     30 31 32 33 34 35 36 37 
  357  *     40 41 42 43 44 45 46 47 
  358  *     50 51 52 53 54 55 56 57 
  359  *     60 61 62 63 64 65 66 67 
  360  *     70 71 72 73 74 75 76 77 
  361  * then after the PC-1 permutation, 
  362  * C0 is
  363  *     70 60 50 40 30 20 10 00 
  364  *     71 61 51 41 31 21 11 01 
  365  *     72 62 52 42 32 22 12 02 
  366  *     73 63 53 43 
  367  * D0 is
  368  *     76 66 56 46 36 26 16 06 
  369  *     75 65 55 45 35 25 15 05 
  370  *     74 64 54 44 34 24 14 04 
  371  *                 33 23 13 03 
  372  * and these parity bits have been discarded:
  373  *     77 67 57 47 37 27 17 07 
  374  * 
  375  * We achieve this by flipping the input matrix about the diagonal from 70-07,
  376  * getting left = 
  377  *     77 67 57 47 37 27 17 07         (these are the parity bits)
  378  *     76 66 56 46 36 26 16 06 
  379  *     75 65 55 45 35 25 15 05 
  380  *     74 64 54 44 34 24 14 04 
  381  * right = 
  382  *     73 63 53 43 33 23 13 03 
  383  *     72 62 52 42 32 22 12 02 
  384  *     71 61 51 41 31 21 11 01 
  385  *     70 60 50 40 30 20 10 00 
  386  * then byte swap right, ala htonl() on a little endian machine.
  387  * right = 
  388  *     70 60 50 40 30 20 10 00 
  389  *     71 67 57 47 37 27 11 07 
  390  *     72 62 52 42 32 22 12 02 
  391  *     73 63 53 43 33 23 13 03 
  392  * then
  393  *     c0 = right >> 4;
  394  *     d0 = ((left & 0x00ffffff) << 4) | (right & 0xf);
  395 */
  396 
  397 #define FLIP_RIGHT_DIAGONAL(word, temp) \
  398     temp  = (word ^ (word >> 18)) & 0x00003333; \
  399     word ^=  temp | (temp << 18); \
  400     temp  = (word ^ (word >> 9)) & 0x00550055; \
  401     word ^=  temp | (temp << 9);
  402 
  403 #define BYTESWAP(word, temp) \
  404     word = (word >> 16) | (word << 16); \
  405     temp = 0x00ff00ff; \
  406     word = ((word & temp) << 8) | ((word >> 8) & temp); 
  407 
  408 #define PC1(left, right, c0, d0, temp) \
  409     right ^= temp = ((left >> 4) ^ right) & 0x0f0f0f0f; \
  410     left  ^= temp << 4; \
  411     FLIP_RIGHT_DIAGONAL(left, temp); \
  412     FLIP_RIGHT_DIAGONAL(right, temp); \
  413     BYTESWAP[1](right, temp); \
  414     c0 = right >> 4; \
  415     d0 = ((left & 0x00ffffff) << 4) | (right & 0xf); 
  416 
  417 #define LEFT_SHIFT_1( reg ) (((reg << 1) | (reg >> 27)) & 0x0FFFFFFF)
  418 #define LEFT_SHIFT_2( reg ) (((reg << 2) | (reg >> 26)) & 0x0FFFFFFF)
  419 
  420 /*
  421  *   setup key schedules from key
  422  */
  423 
  424 void 
  425 DES_MakeSchedule( HALF[1] * ks, const BYTE[1] * key,   DESDirection direction)
  426 {
  427     register HALF[1] left, right;
  428     register HALF[1] c0, d0;
  429     register HALF[1] temp;
  430     int           delta;
  431     unsigned int  ls;
  432 
  433 #if defined(_X86_)
  434     left  = HALFPTR(key)[0]; 
  435     right = HALFPTR(key)[1]; 
  436     BYTESWAP[1](left, temp);
  437     BYTESWAP[1](right, temp);
  438 #else
  439     if (((ptrdiff_t)key & 0x03) == 0) {
  440         left  = HALFPTR(key)[0]; 
  441         right = HALFPTR(key)[1]; 
  442 #if defined(IS_LITTLE_ENDIAN)
  443         BYTESWAP[1](left, temp);
  444         BYTESWAP[1](right, temp);
  445 #endif
  446     } else {
  447         left    = ((HALF[1])key[0] << 24) | ((HALF[1])key[1] << 16) | 
  448                   ((HALF[1])key[2] << 8)  | key[3];
  449         right   = ((HALF[1])key[4] << 24) | ((HALF[1])key[5] << 16) | 
  450                   ((HALF[1])key[6] << 8)  | key[7];
  451     }
  452 #endif
  453 
  454     PC1(left, right, c0, d0, temp);
  455 
  456     if (direction == DES_ENCRYPT) {
  457         delta = 2 * (int)sizeof(HALF[1]);
  458     } else {
  459         ks += 30;
  460         delta = (-2) * (int)sizeof(HALF[1]);
  461     }
  462 
  463     for (ls = 0x8103; ls; ls >>= 1) {
  464         if ( ls & 1 ) {
  465             c0 = LEFT_SHIFT_1( c0 );
  466             d0 = LEFT_SHIFT_1( d0 );
  467         } else {
  468             c0 = LEFT_SHIFT_2( c0 );
  469             d0 = LEFT_SHIFT_2( d0 );
  470         }
  471 
  472 #ifdef USE_INDEXING
  473 #define PC2[1]LOOKUP(b,c) PC2[1][b][c]
  474 
  475         left   = PC2LOOKUP(0, ((c0 >> 22) & 0x3F) );
  476         left  |= PC2LOOKUP(1, ((c0 >> 13) & 0x3F) );
  477         left  |= PC2LOOKUP(2, ((c0 >>  4) & 0x38) | (c0 & 0x7) );
  478         left  |= PC2LOOKUP(3, ((c0>>18)&0xC) | ((c0>>11)&0x3) | (c0&0x30));
  479 
  480         right  = PC2LOOKUP(4, ((d0 >> 22) & 0x3F) );
  481         right |= PC2LOOKUP(5, ((d0 >> 15) & 0x30) | ((d0 >> 14) & 0xf) );
  482         right |= PC2LOOKUP(6, ((d0 >>  7) & 0x3F) );
  483         right |= PC2LOOKUP(7, ((d0 >>  1) & 0x3C) | (d0 & 0x3));
  484 #else
  485 #define PC2[1]LOOKUP(b,c) *(HALF[1] *)((BYTE[1] *)&PC2[1][b][0]+(c))
  486 
  487         left   = PC2LOOKUP(0, ((c0 >> 20) & 0xFC) );
  488         left  |= PC2LOOKUP(1, ((c0 >> 11) & 0xFC) );
  489         left  |= PC2LOOKUP(2, ((c0 >>  2) & 0xE0) | ((c0 <<  2) & 0x1C) );
  490         left  |= PC2LOOKUP(3, ((c0>>16)&0x30)|((c0>>9)&0xC)|((c0<<2)&0xC0));
  491 
  492         right  = PC2LOOKUP(4, ((d0 >> 20) & 0xFC) );
  493         right |= PC2LOOKUP(5, ((d0 >> 13) & 0xC0) | ((d0 >> 12) & 0x3C) );
  494         right |= PC2LOOKUP(6, ((d0 >>  5) & 0xFC) );
  495         right |= PC2LOOKUP(7, ((d0 <<  1) & 0xF0) | ((d0 << 2) & 0x0C));
  496 #endif
  497         /* left  contains key bits for S1 S3 S2 S4 */
  498         /* right contains key bits for S6 S8 S5 S7 */
  499         temp = (left  << 16)        /* S2 S4 XX XX */
  500              | (right >> 16);       /* XX XX S6 S8 */
  501         ks[0] = temp;
  502 
  503         temp = (left  & 0xffff0000) /* S1 S3 XX XX */
  504              | (right & 0x0000ffff);/* XX XX S5 S7 */
  505         ks[1] = temp;
  506 
  507         ks = (HALF[1]*)((BYTE[1] *)ks + delta);
  508     }
  509 }

  511 /*
  512  * The DES Initial Permutation
  513  * if we number the bits of the 8 bytes of input like this (in octal):
  514  *     00 01 02 03 04 05 06 07 
  515  *     10 11 12 13 14 15 16 17 
  516  *     20 21 22 23 24 25 26 27 
  517  *     30 31 32 33 34 35 36 37 
  518  *     40 41 42 43 44 45 46 47 
  519  *     50 51 52 53 54 55 56 57 
  520  *     60 61 62 63 64 65 66 67 
  521  *     70 71 72 73 74 75 76 77 
  522  * then after the initial permutation, they will be in this order. 
  523  *     71 61 51 41 31 21 11 01 
  524  *     73 63 53 43 33 23 13 03 
  525  *     75 65 55 45 35 25 15 05 
  526  *     77 67 57 47 37 27 17 07 
  527  *     70 60 50 40 30 20 10 00 
  528  *     72 62 52 42 32 22 12 02 
  529  *     74 64 54 44 34 24 14 04 
  530  *     76 66 56 46 36 26 16 06 
  531  *
  532  * One way to do this is in two steps:
  533  * 1. Flip this matrix about the diagonal from 70-07 as done for PC1.
  534  * 2. Rearrange the bytes (rows in the matrix above) with the following code.
  535  *
  536  * #define swapHiLo(word, temp) \
  537  *   temp  = (word ^ (word >> 24)) & 0x000000ff; \
  538  *   word ^=  temp | (temp << 24); 
  539  *
  540  *   right ^= temp = ((left << 8) ^ right) & 0xff00ff00; 
  541  *   left  ^= temp >> 8; 
  542  *   swapHiLo(left, temp);
  543  *   swapHiLo(right,temp);
  544  *
  545  * However, the two steps can be combined, so that the rows are rearranged
  546  * while the matrix is being flipped, reducing the number of bit exchange
  547  * operations from 8 ot 5.  
  548  *
  549  * Initial Permutation */
  550 #define IP(left, right, temp) \
  551     right ^= temp = ((left >> 4) ^  right) & 0x0f0f0f0f; \
  552     left  ^= temp << 4; \
  553     right ^= temp = ((left >> 16) ^ right) & 0x0000ffff; \
  554     left  ^= temp << 16; \
  555     right ^= temp = ((left << 2) ^ right) & 0xcccccccc; \
  556     left  ^= temp >> 2; \
  557     right ^= temp = ((left << 8) ^ right) & 0xff00ff00; \
  558     left  ^= temp >> 8; \
  559     right ^= temp = ((left >> 1) ^ right) & 0x55555555; \
  560     left  ^= temp << 1; 
  561 
  562 /* The Final (Inverse Initial) permutation is done by reversing the 
  563 ** steps of the Initital Permutation 
  564 */
  565 
  566 #define FP(left, right, temp) \
  567     right ^= temp = ((left >> 1) ^ right) & 0x55555555; \
  568     left  ^= temp << 1; \
  569     right ^= temp = ((left << 8) ^ right) & 0xff00ff00; \
  570     left  ^= temp >> 8; \
  571     right ^= temp = ((left << 2) ^ right) & 0xcccccccc; \
  572     left  ^= temp >> 2; \
  573     right ^= temp = ((left >> 16) ^ right) & 0x0000ffff; \
  574     left  ^= temp << 16; \
  575     right ^= temp = ((left >> 4) ^  right) & 0x0f0f0f0f; \
  576     left  ^= temp << 4; 
  577 
  578 void 
  579 DES_Do1Block(HALF[1] * ks, const BYTE[1] * inbuf, BYTE[1] * outbuf)
  580 {
  581     register HALF[1] left, right;
  582     register HALF[1] temp;
  583 
  584 #if defined(_X86_)
  585     left  = HALFPTR(inbuf)[0]; 
  586     right = HALFPTR(inbuf)[1]; 
  587     BYTESWAP[1](left, temp);
  588     BYTESWAP[1](right, temp);
  589 #else
  590     if (((ptrdiff_t)inbuf & 0x03) == 0) {
  591         left  = HALFPTR(inbuf)[0]; 
  592         right = HALFPTR(inbuf)[1]; 
  593 #if defined(IS_LITTLE_ENDIAN)
  594         BYTESWAP[1](left, temp);
  595         BYTESWAP[1](right, temp);
  596 #endif
  597     } else {
  598         left    = ((HALF[1])inbuf[0] << 24) | ((HALF[1])inbuf[1] << 16) | 
  599                   ((HALF[1])inbuf[2] << 8)  | inbuf[3];
  600         right   = ((HALF[1])inbuf[4] << 24) | ((HALF[1])inbuf[5] << 16) | 
  601                   ((HALF[1])inbuf[6] << 8)  | inbuf[7];
  602     }
  603 #endif
  604 
  605     IP(left, right, temp);
  606 
  607     /* shift the values left circularly 3 bits. */
  608     left  = (left  << 3) | (left  >> 29);
  609     right = (right << 3) | (right >> 29);
  610 
  611 #ifdef USE_INDEXING
  612 #define KSLOOKUP(s,b) SP[1][s][((temp >> (b+2)) & 0x3f)]
  613 #else
  614 #define KSLOOKUP(s,b) *(HALF[1]*)((BYTE[1]*)&SP[1][s][0]+((temp >> b) & 0xFC))
  615 #endif
  616 #define ROUND(out, in, r) \
  617     temp  = in ^ ks[2*r]; \
  618     out ^= KSLOOKUP( 1,  24 ); \
  619     out ^= KSLOOKUP( 3,  16 ); \
  620     out ^= KSLOOKUP( 5,   8 ); \
  621     out ^= KSLOOKUP( 7,   0 ); \
  622     temp  = ((in >> 4) | (in << 28)) ^ ks[2*r+1]; \
  623     out ^= KSLOOKUP( 0,  24 ); \
  624     out ^= KSLOOKUP( 2,  16 ); \
  625     out ^= KSLOOKUP( 4,   8 ); \
  626     out ^= KSLOOKUP( 6,   0 ); 
  627 
  628     /* Do the 16 Feistel rounds */
  629     ROUND[1](left, right, 0)
  630     ROUND[1](right, left, 1)
  631     ROUND[1](left, right, 2)
  632     ROUND[1](right, left, 3)
  633     ROUND[1](left, right, 4)
  634     ROUND[1](right, left, 5)
  635     ROUND[1](left, right, 6)
  636     ROUND[1](right, left, 7)
  637     ROUND[1](left, right, 8)
  638     ROUND[1](right, left, 9)
  639     ROUND[1](left, right, 10)
  640     ROUND[1](right, left, 11)
  641     ROUND[1](left, right, 12)
  642     ROUND[1](right, left, 13)
  643     ROUND[1](left, right, 14)
  644     ROUND[1](right, left, 15)
  645 
  646     /* now shift circularly right 3 bits to undo the shifting done 
  647     ** above.  switch left and right here. 
  648     */
  649     temp  = (left >> 3) | (left << 29); 
  650     left  = (right >> 3) | (right << 29); 
  651     right = temp;
  652 
  653     FP(left, right, temp);
  654 
  655 #if defined(_X86_)
  656     BYTESWAP[1](left, temp);
  657     BYTESWAP[1](right, temp);
  658     HALFPTR(outbuf)[0]  = left; 
  659     HALFPTR(outbuf)[1]  = right; 
  660 #else
  661     if (((ptrdiff_t)outbuf & 0x03) == 0) {
  662 #if defined(IS_LITTLE_ENDIAN)
  663         BYTESWAP[1](left, temp);
  664         BYTESWAP[1](right, temp);
  665 #endif
  666         HALFPTR(outbuf)[0]  = left; 
  667         HALFPTR(outbuf)[1]  = right; 
  668     } else {
  669         outbuf[0] = (BYTE[1])(left >> 24);
  670         outbuf[1] = (BYTE[1])(left >> 16);
  671         outbuf[2] = (BYTE[1])(left >>  8);
  672         outbuf[3] = (BYTE[1])(left      );
  673 
  674         outbuf[4] = (BYTE[1])(right >> 24);
  675         outbuf[5] = (BYTE[1])(right >> 16);
  676         outbuf[6] = (BYTE[1])(right >>  8);
  677         outbuf[7] = (BYTE[1])(right      );
  678     }
  679 #endif
  680 
  681 }
