   67 
   68   Returns MP_YES if a is divisible by b, or MP_NO if it is not.

   71 mp_err  mpp_divis(mp_int *a, mp_int *b)
   72 {
   73   mp_err  res;
   74   mp_int  rem;
   75 
   76   if((res = mp_init(&rem)) != MP_OKAY)
   77     return res;
   78 
   79   if((res = mp_mod(a, b, &rem)) != MP_OKAY)
   80     goto CLEANUP;
   81 
   82   if(mp_cmp_z(&rem) == 0)
   83     res = MP_YES;
   84   else
   85     res = MP_NO;
   86 
   87 CLEANUP:
   88   mp_clear(&rem);
   89   return res;
   90 
   91 } /* end mpp_divis() */

  127 /*
  128   mpp_random(a)
  129 
  130   Assigns a random value to a.  This value is generated using the
  131   standard C library's rand() function, so it should not be used for
  132   cryptographic purposes, but it should be fine for primality testing,
  133   since all we really care about there is good statistical properties.
  134 
  135   As many digits as a currently has are filled with random digits.
  136  */
  137 
  138 mp_err  mpp_random(mp_int *a)
  139 
  140 {
  141   mp_digit  next = 0;
  142   unsigned int       ix, jx;
  143 
  144   ARGCHK(a != NULL, MP_BADARG);
  145 
  146   for(ix = 0; ix < USED(a); ix++) {
  147     for(jx = 0; jx < sizeof(mp_digit); jx++) {
  148       next = (next << CHAR_BIT) | (RANDOM() & UCHAR_MAX);
  149     }
  150     DIGIT(a, ix) = next;
  151   }
  152 
  153   return MP_OKAY;
  154 
  155 } /* end mpp_random() */

  226 /* {{{ mpp_fermat(a, w) */
  227 
  228 /*
  229   Using w as a witness, try pseudo-primality testing based on Fermat's
  230   little theorem.  If a is prime, and (w, a) = 1, then w^a == w (mod
  231   a).  So, we compute z = w^a (mod a) and compare z to w; if they are
  232   equal, the test passes and we return MP_YES.  Otherwise, we return
  233   MP_NO.
  234  */
  235 mp_err  mpp_fermat(mp_int *a, mp_digit w)
  236 {
  237   mp_int  base, test;
  238   mp_err  res;
  239   
  240   if((res = mp_init(&base)) != MP_OKAY)
  241     return res;
  242 
  243   mp_set(&base, w);
  244 
  245   if((res = mp_init(&test)) != MP_OKAY)
  246     goto TEST;
  247 
  248   /* Compute test = base^a (mod a) */
  249   if((res = mp_exptmod(&base, a, a, &test)) != MP_OKAY)
  250     goto CLEANUP;
  251 
  252   
  253   if(mp_cmp(&base, &test) == 0)
  254     res = MP_YES;
  255   else
  256     res = MP_NO;
  257 
  258  CLEANUP:
  259   mp_clear(&test);
  260  TEST:
  261   mp_clear(&base);
  262 
  263   return res;
  264 
  265 } /* end mpp_fermat() */
  266 
  267 /* }}} */

  269 /*
  270   Perform the fermat test on each of the primes in a list until
  271   a) one of them shows a is not prime, or 
  272   b) the list is exhausted.
  273   Returns:  MP_YES if it passes tests.
  274             MP_NO  if fermat test reveals it is composite
  275             Some MP error code if some other error occurs.
  276  */
  277 mp_err mpp_fermat_list(mp_int *a, const mp_digit *primes, mp_size nPrimes)
  278 {
  279   mp_err rv = MP_YES;
  280 
  281   while (nPrimes-- > 0 && rv == MP_YES) {
  282     rv = mpp_fermat(a, *primes++);
  283   }
  284   return rv;
  285 }

  287 /* {{{ mpp_pprime(a, nt) */
  288 
  289 /*
  290   mpp_pprime(a, nt)
  291 
  292   Performs nt iteration of the Miller-Rabin probabilistic primality
  293   test on a.  Returns MP_YES if the tests pass, MP_NO if one fails.
  294   If MP_NO is returned, the number is definitely composite.  If MP_YES
  295   is returned, it is probably prime (but that is not guaranteed).
  296  */
  297 
  298 mp_err  mpp_pprime(mp_int *a, int nt)
  299 {
  300   mp_err   res;
  301   mp_int   x, amo, m, z;        /* "amo" = "a minus one" */
  302   int      iter;
  303   unsigned int jx;
  304   mp_size  b;
  305 
  306   ARGCHK(a != NULL, MP_BADARG);
  307 
  308   MP_DIGITS(&x) = 0;
  309   MP_DIGITS(&amo) = 0;
  310   MP_DIGITS(&m) = 0;
  311   MP_DIGITS(&z) = 0;
  312 
  313   /* Initialize temporaries... */
  314   MP_CHECKOK( mp_init(&amo));
  315   /* Compute amo = a - 1 for what follows...    */
  316   MP_CHECKOK( mp_sub_d(a, 1, &amo) );
  317 
  318   b = mp_trailing_zeros(&amo);
  319   if (!b) { /* a was even ? */
  320     res = MP_NO;
  321     goto CLEANUP;
  322   }
  323 
  324   MP_CHECKOK( mp_init_size(&x, MP_USED(a)) );
  325   MP_CHECKOK( mp_init(&z) );
  326   MP_CHECKOK( mp_init(&m) );
  327   MP_CHECKOK( mp_div_2d(&amo, b, &m, 0) );
  328 
  329   /* Do the test nt times... */
  330   for(iter = 0; iter < nt; iter++) {
  331 
  332     /* Choose a random value for x < a          */
  333     s_mp_pad(&x, USED(a));
  334     mpp_random(&x);
  335     MP_CHECKOK( mp_mod(&x, a, &x) );
  336 
  337     /* Compute z = (x ** m) mod a               */
  338     MP_CHECKOK( mp_exptmod(&x, &m, a, &z) );
  339     
  340     if(mp_cmp_d">mp_cmp_d(&z, 1) == 0 || mp_cmp(&z, &amo) == 0) {
  341       res = MP_YES;
  342       continue;
  343     }
  344     
  345     res = MP_NO;  /* just in case the following for loop never executes. */
  346     for (jx = 1; jx < b; jx++) {
  347       /* z = z^2 (mod a) */
  348       MP_CHECKOK( mp_sqrmod[1](&z, a, &z) );
  349       res = MP_NO;        /* previous line set res to MP_YES */
  350 
  351       if(mp_cmp_d(&z, 1) == 0) {
  352         break;
  353       }
  354       if(mp_cmp(&z, &amo) == 0) {
  355         res = MP_YES;
  356         break;
  357       } 
  358     } /* end testing loop */
  359 
  360     /* If the test passes, we will continue iterating, but a failed
  361        test means the candidate is definitely NOT prime, so we will
  362        immediately break out of this loop
  363      */
  364     if(res == MP_NO)
  365       break;
  366 
  367   } /* end iterations loop */
  368   
  369 CLEANUP:
  370   mp_clear(&m);
  371   mp_clear(&z);
  372   mp_clear(&x);
  373   mp_clear(&amo);
  374   return res;
  375 
  376 } /* end mpp_pprime() */

  380 /* Produce table of composites from list of primes and trial value.  
  381 ** trial must be odd. List of primes must not include 2.
  382 ** sieve should have dimension >= MAXPRIME/2, where MAXPRIME is largest 
  383 ** prime in list of primes.  After this function is finished,
  384 ** if sieve[i] is non-zero, then (trial + 2*i) is composite.
  385 ** Each prime used in the sieve costs one division of trial, and eliminates
  386 ** one or more values from the search space. (3 eliminates 1/3 of the values
  387 ** alone!)  Each value left in the search space costs 1 or more modular 
  388 ** exponentations.  So, these divisions are a bargain!
  389 */
  390 mp_err mpp_sieve(mp_int *trial, const mp_digit *primes, mp_size nPrimes, 
  391                  unsigned char *sieve, mp_size nSieve)
  392 {
  393   mp_err       res;
  394   mp_digit     rem;
  395   mp_size      ix;
  396   unsigned long offset;
  397 
  398   memset(sieve, 0, nSieve);
  399 
  400   for(ix = 0; ix < nPrimes; ix++) {
  401     mp_digit prime = primes[ix];
  402     i/mpi.h.dep.html#mp_size">mp_size  i;
  403     if((res = mp_mod_d(trial, prime, &rem)) != MP_OKAY) 
  404       return res;
  405 
  406     if (rem == 0) {
  407       offset = 0;
  408     } else {
  409       offset = prime - (rem / 2);
  410     }
  411     for (i = offset; i < nSieve ; i += prime) {
  412       sieve[i] = 1;
  413     }
  414   }
  415 
  416   return MP_OKAY;
  417 }

  579 /* {{{ s_mpp_divp(a, vec, size, which) */
  580 
  581 /* 
  582    Test for divisibility by members of a vector of digits.  Returns
  583    MP_NO if a is not divisible by any of them; returns MP_YES and sets
  584    'which' to the index of the offender, if it is.  Will stop on the
  585    first digit against which a is divisible.
  586  */
  587 
  588 mp_err    s_mpp_divp(mp_int *a, const mp_digit *vec, int size, int *which)
  589 {
  590   mp_err    res;
  591   mp_digit  rem;
  592 
  593   int     ix;
  594 
  595   for(ix = 0; ix < size; ix++) {
  596     if((res = mp_mod_d(a, vec[ix], &rem)) != MP_OKAY) 
  597       return res;
  598 
  599     if(rem == 0) {
  600       if(which)
  601         *which = ix;
  602       return MP_YES;
  603     }
  604   }
  605 
  606   return MP_NO;
  607 
  608 } /* end s_mpp_divp() */
