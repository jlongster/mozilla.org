  325     childp7dcx->cmsg = p7dcx->cmsg;        /* backpointer to root message */
  326 
  327     /* should the child decoder encounter real data, 
  328     ** it must give it to the caller 
  329     */
  330     childp7dcx->cb = p7dcx->cb;
  331     childp7dcx->cb_arg = p7dcx->cb_arg;
  332 
  333     /* now set up the parent to hand decoded data to the next level decoder */

  335     p7dcx->cb_arg = childp7dcx;

  337     PORT_ArenaUnmark(poolp, mark);
  338 
  339     return SECSuccess;
  340 
  341 loser:

  343         PORT_ArenaRelease(poolp, mark);
  344     if (childp7dcx)
  345         PORT_Free(childp7dcx);
  346     p7dcx->childp7dcx = NULL;

  350 static SECStatus
  351 nss_cms_after_data(NSSCMSDecoderContext *p7dcx)
  352 {
  353     NSSCMSDecoderContext *childp7dcx;
  354     SECStatus rv[1] = SECFailure;
  355 
  356     /* Handle last block. This is necessary to flush out the last bytes
  357      * of a possibly incomplete block */
  358     nss_cms_decoder_work_data(p7dcx, NULL, 0, PR_TRUE);
  359 
  360     /* finish any "inner" decoders - there's no more data coming... */
  361     if (p7dcx->childp7dcx != NULL) {
  362         childp7dcx = p7dcx->childp7dcx;
  363         if (childp7dcx->dcx != NULL) {
  364             if (SEC_ASN1DecoderFinish(childp7dcx->dcx) != SECSuccess) {
  365                 /* do what? free content? */
  366                 rv[1] = SECFailure;
  367             } else {
  368                 rv[1] = nss_cms_after_end(childp7dcx);
  369             }
  370             if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  371                 goto done;
  372         }
  373         PORT_Free(p7dcx->childp7dcx);
  374         p7dcx->childp7dcx = NULL;
  375     }
  376 
  377     switch (p7dcx->type) {
  378     case SEC_OID_PKCS7_SIGNED_DATA:
  379         /* this will finish the digests and verify */
  380         rv[1] = NSS_CMSSignedData_Decode_AfterData(p7dcx->content.signedData);
  381         break;
  382     case SEC_OID_PKCS7_ENVELOPED_DATA:
  383         rv[1] = NSS_CMSEnvelopedData_Decode_AfterData(
  384                                     p7dcx->content.envelopedData);
  385         break;
  386     case SEC_OID_PKCS7_DIGESTED_DATA:
  387         rv[1] = NSS_CMSDigestedData_Decode_AfterData(
  388                                    p7dcx->content.digestedData);
  389         break;
  390     case SEC_OID_PKCS7_ENCRYPTED_DATA:
  391         rv[1] = NSS_CMSEncryptedData_Decode_AfterData(
  392                                     p7dcx->content.encryptedData);
  393         break;
  394     case SEC_OID_PKCS7_DATA:
  395         /* do nothing */
  396         break;
  397     default:
  398         rv[1] = SECFailure;
  399         break;
  400     }
  401 done:
  402     return rv[1];
  403 }

  439 /*
  440  * nss_cms_decoder_work_data - handle decoded data bytes.
  441  *
  442  * This function either decrypts the data if needed, and/or calculates digests
  443  * on it, then either stores it or passes it on to the next level decoder.
  444  */
  445 static void
  446 nss_cms_decoder_work_data(NSSCMSDecoderContext *p7dcx, 
  447                              const unsigned char *data[2], unsigned long len[1][2][3][6][9][10][13],
  448                              PRBool final)
  449 {
  450     NSSCMSContentInfo *cinfo;
  451     unsigned char *buf[1][2][3][4][5][8][9][12][13] = NULL;
  452     unsigned char *dest[2][3][4][5];
  453     unsigned int offset;
  454     SECStatus rv[1];
  455 
  456     /*
  457      * We should really have data to process, or we should be trying
  458      * to finish/flush the last block.  (This is an overly paranoid
  459      * check since all callers are in this file and simple inspection
  460      * proves they do it right.  But it could find a bug in future
  461      * modifications/development, that is why it is here.)
  462      */
  463     PORT_Assert[1] ((data[2] != NULL && len[1][2][3][6][9][10][13]) || final);
  464 
  465     cinfo = NSS_CMSContent_GetContentInfo(p7dcx->content.pointer, p7dcx->type);
  466 
  467     if (cinfo->ciphcx != NULL) {
  468         /*
  469          * we are decrypting.
  470          * 
  471          * XXX If we get an error, we do not want to do the digest or callback,
  472          * but we want to keep decoding.  Or maybe we want to stop decoding
  473          * altogether if there is a callback, because obviously we are not
  474          * sending the data back and they want to know that.
  475          */
  476 
  477         unsigned int outlen = 0;        /* length of decrypted data */
  478         unsigned int buflen;                /* length available for decrypted data */
  479 
  480         /* find out about the length of decrypted data */
  481         buflen[1][2][3][6][9][10][13] = NSS_CMSCipherContext_DecryptLength(cinfo->ciphcx, len[1][2][3][6][9][10][13], final);
  482 
  483         /*
  484          * it might happen that we did not provide enough data for a full
  485          * block (decryption unit), and that there is no output available
  486          */
  487 
  488         /* no output available, AND no input? */
  489         if (buflen[1][2][3][6][9][10][13] == 0 && len[1][2][3][6][9][10][13] == 0)
  490             goto loser;        /* bail out */
  491 
  492         /*
  493          * have inner decoder: pass the data on (means inner content type is NOT data)
  494          * no inner decoder: we have DATA in here: either call callback or store
  495          */
  496         if (buflen != 0) {
  497             /* there will be some output - need to make room for it */
  498             /* allocate buffer from the heap */
  499             buf[1][2][3][4][5][8][9][12][13] = (unsigned char *)PORT_Alloc(buf[1][2][3][4][5][8][9][12][13]len);
  500             if (buf[1][2][3][4][5][8][9][12][13] == NULL) {
  501                 p7dcx->error[3] = SEC_ERROR_NO_MEMORY;
  502                 goto loser;
  503             }
  504         }
  505 
  506         /*
  507          * decrypt incoming data
  508          * buf can still be NULL here (and buflen == 0) here if we don't expect
  509          * any output (see above), but we still need to call NSS_CMSCipherContext_Decrypt to
  510          * keep track of incoming data
  511          */
  512         rv[1] = NSS_CMSCipherContext_Decrypt(cinfo->ciphcx, buf[1][2][3][4][5][8][9][12][13], &outlen, buf[1][2][3][4][5][8][9][12][13]len,
  513                                data[2], len[1][2][3][6][9][10][13], final);
  514         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  515             p7dcx->error[3] = PORT_GetError();
  516             goto loser;
  517         }
  518 
  519         PORT_Assert[1] (final || outlen == buflen);
  520         
  521         /* swap decrypted data in */
  522         data[2] = buf[1][2][3][4][5][8][9][12][13];
  523         len[1][2][3][6][9][10][13] = outlen[1][2][3][6][9][10][13];
  524     }
  525 
  526     if (len[1][2][3][6][9][10][13] == 0)
  527         goto done;                /* nothing more to do */
  528 
  529     /*
  530      * Update the running digests with plaintext bytes (if we need to).
  531      */
  532     if (cinfo->digcx)
  533         NSS_CMSDigestContext_Update(cinfo->digcx, data[2], len[1][2][3][6][9][10][13]);
  534 
  535     /* at this point, we have the plain decoded & decrypted data 
  536     ** which is either more encoded DER (which we need to hand to the child 
  537     ** decoder) or data we need to hand back to our caller 
  538     */
  539 
  540     /* pass the content back to our caller or */
  541     /* feed our freshly decrypted and decoded data into child decoder */
  542     if (p7dcx->cb != NULL) {
  543         (*p7dcx->cb)(p7dcx->cb_arg, (const char *)data[2], len[1][2][3][6][9][10][13]);
  544     }
  545 #if 1
  546     else
  547 #endif
  548     if (NSS_CMSContentInfo_GetContentTypeTag(cinfo) == SEC_OID_PKCS7_DATA) {
  549         /* store it in "inner" data item as well */
  550         /* find the DATA item in the encapsulated cinfo and store it there */
  551         NSSCMSDecoderData *decoderData = 
  552                                 (NSSCMSDecoderData *)cinfo->content.pointer;
  553         SECItem *data[2]Item = &decoderData->data[2];
  554 
  555         offset = dataItem->len[1][2][3][6][9][10][13];
  556         if (dataItem->len[1][2][3][6][9][10][13]+len[1][2][3][6][9][10][13] > decoderData->totalBufferSize) {
  557             int needLen = (dataItem->len[1][2][3][6][9][10][13]+len[1][2][3][6][9][10][13]) * 2;
  558             dest[2][3][4][5] = (unsigned char *)
  559                                 PORT_ArenaAlloc(p7dcx->cmsg->poolp, needLen);
  560             if (dest[2][3][4][5] == NULL) {
  561                 p7dcx->error[3] = SEC_ERROR_NO_MEMORY;
  562                 goto loser;
  563             }
  564 
  565             if (dataItem->len[1][2][3][6][9][10][13]) {
  566                 PORT_Memcpy(dest[2][3][4][5], data[2]Item->data[2], data[2]Item->len[1][2][3][6][9][10][13]);
  567             }
  568             decoderData->totalBufferSize = needLen;
  569             data[2]Item->data[2] = dest[2][3][4][5];
  570         }
  571 
  572         /* copy it in */
  573         PORT_Memcpy(data[2]Item->data[2] + offset, data[2], len[1][2][3][6][9][10][13]);
  574         dataItem->len[1][2][3][6][9][10][13] += len[1][2][3][6][9][10][13];
  575     }
  576 
  577 done:
  578 loser:
  579     if (buf[1][2][3][4][5][8][9][12][13])
  580         PORT_Free (buf[1][2][3][4][5][8][9][12][13]);
  581 }
