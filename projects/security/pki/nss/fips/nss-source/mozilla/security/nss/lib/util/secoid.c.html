 1474 /*
 1475  * now the dynamic table. The dynamic table gets build at init time.
 1476  * and conceivably gets modified if the user loads new crypto modules.
 1477  * All this static data, and the allocated data to which it points,
 1478  * is protected by a global reader/writer lock.  
 1479  * The c language guarantees that global and static data that is not 
 1480  * explicitly initialized will be initialized with zeros.  If we 
 1481  * initialize it with zeros, the data goes into the initialized data
 1482  * secment, and increases the size of the library.  By leaving it 
 1483  * uninitialized, it is allocated in BSS, and does NOT increase the 
 1484  * library size. 
 1485  */
 1486 static NSSRWLock   * dynOidLock;
 1487 static PLArenaPool * dynOidPool;
 1488 static PLHashTable * dynOidHash;
 1489 static SECOidData ** dynOidTable;        /* not in the pool */
 1490 static int           dynOidEntriesAllocated;
 1491 static int           dynOidEntriesUsed;
 1492 
 1493 /* Creates NSSRWLock and dynOidPool, if they don't exist.
 1494 ** This function MIGHT create the lock, but not the pool, so
 1495 ** code should test for dynOidPool, not dynOidLock, when deciding
 1496 ** whether or not to call this function.
 1497 */
 1498 static SECStatus
 1499 secoid_InitDynOidData(void)
 1500 {
 1501     SECStatus   rv = SECSuccess;
 1502     NSSRWLock * lock;
 1503 
 1504     /* This function will create the lock if it doesn't exist,
 1505     ** and will return the address of the lock, whether it was 
 1506     ** previously created, or was created by the function.
 1507     */
 1508     lock = nssRWLock_AtomicCreate(&dynOidLock, 1, "dynamic OID data");
 1509     if (!lock) {
 1510             return SECFailure; /* Error code should already be set. */
 1511     }
 1512     PORT_Assert[1](lock == dynOidLock);
 1513     NSSRWLock_LockWrite(lock);
 1514     if (!dynOidPool) {
 1515             dynOidPool = PORT_NewArena(2048);
 1516         if (!dynOidPool) {
 1517             rv = SECFailure /* Error code should already be set. */;
 1518         }
 1519     }
 1520     NSSRWLock_UnlockWrite(lock);
 1521     return rv;
 1522 }

 1544  * Lookup a Dynamic OID. Dynamic OID's still change slowly, so it's
 1545  * cheaper to rehash the table when it changes than it is to do the loop
 1546  * each time. 
 1547  */

 1549 secoid_FindDynamic(const SECItem *key) 
 1550 {
 1551     SECOidData *ret = NULL;
 1552 
 1553     if (dynOidHash) {
 1554         NSSRWLock_LockRead(dynOidLock);
 1555         if (dynOidHash) { /* must check it again with lock held. */
 1556             ret = (SECOidData *)PL_HashTableLookup(dynOidHash, key);
 1557         }
 1558         NSSRWLock_UnlockRead(dynOidLock);
 1559     }
 1560     if (ret == NULL) {
 1561         PORT_SetError(SEC_ERROR_UNRECOGNIZED_OID);
 1562     }
 1563     return ret;

 1592 /*
 1593  * This routine is thread safe now.
 1594  */
 1595 SECOidTag
 1596 SECOID_AddEntry(const SECOidData * src)
 1597 {
 1598     SECOidData * dst;
 1599     SECOidData **table;
 1600     SECOidTag    ret         = SEC_OID_UNKNOWN;
 1601     SECStatus    rv;
 1602     int          tableEntries;
 1603     int          used;
 1604 
 1605     if (!src || !src->oid.data || !src->oid.len || \
 1606         !src->desc || !strlen(src->desc)) {
 1607         PORT_SetError(SEC_ERROR_INVALID_ARGS);
 1608         return ret;
 1609     }
 1610     if (src->supportedExtension != INVALID_CERT_EXTENSION     &&
 1611             src->supportedExtension != UNSUPPORTED_CERT_EXTENSION &&
 1612             src->supportedExtension != SUPPORTED_CERT_EXTENSION     ) {
 1613         PORT_SetError(SEC_ERROR_INVALID_ARGS);
 1614         return ret;
 1615     }
 1616 
 1617     if (!dynOidPool && secoid_InitDynOidData() != SECSuccess) {
 1618         /* Caller has set error code. */
 1619             return ret;
 1620     }
 1621 
 1622     NSSRWLock_LockWrite(dynOidLock);
 1623 
 1624     /* We've just acquired the write lock, and now we call FindOIDTag
 1625     ** which will acquire and release the read lock.  NSSRWLock has been
 1626     ** designed to allow this very case without deadlock.  This approach 
 1627     ** makes the test for the presence of the OID, and the subsequent 
 1628     ** addition of the OID to the table a single atomic write operation.
 1629     */
 1630     ret = SECOID_FindOIDTag(&src->oid);
 1631     if (ret != SEC_OID_UNKNOWN) {
 1632             /* we could return an error here, but I chose not to do that.
 1633         ** This way, if we add an OID to the shared library's built in
 1634         ** list of OIDs in some future release, and that OID is the same
 1635         ** as some OID that a program has been adding, the program will
 1636         ** not suddenly stop working.
 1637         */
 1638         goto done;
 1639     }
 1640 
 1641     table        = dynOidTable;
 1642     tableEntries = dynOidEntriesAllocated;
 1643     used         = dynOidEntriesUsed;
 1644 
 1645     if (used + 1 > tableEntries) {
 1646         SECOidData **newTable;
 1647         int          newTableEntries = tableEntries + 16;
 1648 
 1649         newTable = (SECOidData **)PORT_Realloc(table, 
 1650                                        newTableEntries * sizeof(SECOidData *));
 1651         if (newTable == NULL) {
 1652             goto done;
 1653         }
 1654         dynOidTable            = table        = newTable;
 1655         dynOidEntriesAllocated = tableEntries = newTableEntries;
 1656     }
 1657 
 1658     /* copy oid structure */
 1659     dst = PORT_ArenaNew(dynOidPool, SECOidData);
 1660     if (!dst) {
 1661             goto done;
 1662     }
 1663     rv  = SECITEM_CopyItem(dynOidPool, &dst->oid, &src->oid);
 1664     if (rv != SECSuccess) {
 1665         goto done;
 1666     }
 1667     dst->desc = PORT_ArenaStrdup(dynOidPool, src->desc);
 1668     if (!dst->desc) {
 1669         goto done;
 1670     }
 1671     dst->offset             = (SECOidTag)(used + SEC_OID_TOTAL);
 1672     dst->mechanism          = src->mechanism;
 1673     dst->supportedExtension = src->supportedExtension;
 1674 
 1675     rv = secoid_HashDynamicOiddata(dst);
 1676     if ( rv == SECSuccess ) {
 1677         table[used++] = dst;
 1678         dynOidEntriesUsed = used;
 1679         ret = dst->offset;
 1680     }
 1681 done:
 1682     NSSRWLock_UnlockWrite(dynOidLock);
 1683     return ret;
 1684 }

 1829 /*
 1830  * free up the oid tables.
 1831  */
 1832 SECStatus
 1833 SECOID_Shutdown(void)
 1834 {
 1835     if (oidhash) {
 1836         PL_HashTableDestroy(oidhash);
 1837         oidhash = NULL;
 1838     }
 1839     if (oidmechhash) {
 1840         PL_HashTableDestroy(oidmechhash);
 1841         oidmechhash = NULL;
 1842     }
 1843     /* Have to handle the case where the lock was created, but
 1844     ** the pool wasn't. 
 1845     ** I'm not going to attempt to create the lock, just to protect
 1846     ** the destruction of data the probably isn't inisialized anyway.
 1847     */
 1848     if (dynOidLock) {
 1849         NSSRWLock_LockWrite(dynOidLock);
 1850         if (dynOidHash) {
 1851             PL_HashTableDestroy(dynOidHash);
 1852             dynOidHash = NULL;
 1853         }
 1854         if (dynOidPool) {
 1855             PORT_FreeArena(dynOidPool, PR_FALSE);
 1856             dynOidPool = NULL;
 1857         }
 1858         if (dynOidTable) {
 1859             PORT_Free(dynOidTable);
 1860             dynOidTable = NULL;
 1861         }
 1862         dynOidEntriesAllocated = 0;
 1863         dynOidEntriesUsed = 0;
 1864 
 1865         NSSRWLock_UnlockWrite(dynOidLock);
 1866         NSSRWLock_Destroy(dynOidLock);
 1867         dynOidLock = NULL;
 1868     } else {
 1869             /* Since dynOidLock doesn't exist, then all the data it protects
 1870         ** should be uninitialized.  We'll check that (in DEBUG builds),
 1871         ** and then make sure it is so, in case NSS is reinitialized.
 1872         */
 1873         PORT_Assert[1](!dynOidHash && !dynOidPool && !dynOidTable && \
 1874                     !dynOidEntriesAllocated && !dynOidEntriesUsed);
 1875         dynOidHash = NULL;
 1876         dynOidPool = NULL;
 1877         dynOidTable = NULL;
 1878         dynOidEntriesAllocated = 0;
 1879         dynOidEntriesUsed = 0;
 1880     }
 1881     return SECSuccess;
