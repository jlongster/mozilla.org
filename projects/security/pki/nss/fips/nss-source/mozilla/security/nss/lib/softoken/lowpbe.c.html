   77 SECStatus
   78 nsspkcs5_HashBuf(const SECHashObject *hashObj, unsigned char *dest,
   79                                          unsigned char *src, int len)
   80 {
   81     void *ctx;
   82     unsigned int retLen;
   83 
   84     ctx = hashObj->create();
   85     if(ctx == NULL) {
   86         return SECFailure;
   87     }
   88     hashObj->begin(ctx);
   89     hashObj->update(ctx, src, len);
   90     hashObj->end(ctx, dest, &retLen, hashObj->length);
   91     hashObj->destroy(ctx, PR_TRUE);
   92     return SECSuccess;
   93 }

   97 static SECItem *
   98 nsspkcs5_PBKDF1(const SECHashObject *hashObj, SECItem *salt, SECItem *pwd, 
   99                                                 int iter, PRBool faulty3DES) 
  100 {
  101     SECItem *hash = NULL, *pre_hash = NULL;
  102     SECStatus rv = SECFailure;
  103 
  104     if((salt == NULL) || (pwd == NULL) || (iter < 0)) {
  105         return NULL;
  106     }
  107         
  108     hash = (SECItem *)PORT_ZAlloc(sizeof(SECItem));
  109     pre_hash = (SECItem *)PORT_ZAlloc(sizeof(SECItem));
  110 
  111     if((hash != NULL) && (pre_hash != NULL)) {
  112         int i, ph_len;
  113 
  114         ph_len = hashObj->length;
  115         if((salt->len + pwd->len) > hashObj->length) {
  116             ph_len = salt->len + pwd->len;
  117         }
  118 
  119         rv = SECFailure;
  120 
  121         /* allocate buffers */
  122         hash->len = hashObj->length;
  123         hash->data = (unsigned char *)PORT_ZAlloc(hash->len);
  124         pre_hash->data = (unsigned char *)PORT_ZAlloc(ph_len);
  125 
  126         /* in pbeSHA1TripleDESCBC there was an allocation error that made
  127          * it into the caller.  We do not want to propagate those errors
  128          * further, so we are doing it correctly, but reading the old method.
  129          */
  130         if (faulty3DES) {
  131             pre_hash->len = ph_len;
  132         } else {
  133             pre_hash->len = salt->len + pwd->len;
  134         }
  135 
  136         /* preform hash */
  137         if ((hash->data != NULL) && (pre_hash->data != NULL)) {
  138             rv = SECSuccess;
  139             /* check for 0 length password */
  140             if(pwd->len > 0) {
  141                 PORT_Memcpy(pre_hash->data, pwd->data, pwd->len);
  142             }
  143             if(salt->len > 0) {
  144                 PORT_Memcpy((pre_hash->data+pwd->len), salt->data, salt->len);
  145             }
  146             for(i = 0; ((i < iter) && (rv == SECSuccess)); i++) {
  147                 rv = nsspkcs5_HashBuf(hashObj, hash->data, 
  148                                         pre_hash->data, pre_hash->len);
  149                 if(rv != SECFailure) {
  150                     pre_hash->len = hashObj->length;
  151                     PORT_Memcpy(pre_hash->data, hash->data, hashObj->length);
  152                 }
  153             }
  154         }
  155     }
  156 
  157     if(pre_hash != NULL) {
  158         SECITEM_FreeItem(pre_hash, PR_TRUE);
  159     }
  160 
  161     if((rv != SECSuccess) && (hash != NULL)) {
  162         SECITEM_FreeItem(hash, PR_TRUE);
  163         hash = NULL;
  164     }
  165 
  166     return hash;
  167 }

  205      * data which are added based on iteration number 
  206      */
  207 
  208     if ((unsigned int)hash_size > pbe_param->salt.len) {
  209         state_len = hash_size;
  210     } else {
  211         state_len = pbe_param->salt.len;
  212     }
  213     state = (unsigned char *)PORT_ZAlloc(state_len);
  214     if(state == NULL) {
  215         rv = SECFailure;
  216         goto loser;
  217     }
  218     if(pbe_param->salt.len > 0) {
  219         PORT_Memcpy(state, pbe_param->salt.data, pbe_param->salt.len);
  220     }

  223     if (cx == NULL) {
  224         rv = SECFailure;
  225         goto loser;
  226     }
  227

  229

  231         HMAC_Begin(cx);
  232         HMAC_Update(cx, state, state_len);
  233         HMAC_Update(cx, pbe_param->salt.data, pbe_param->salt.len);

  236         if (rv != SECSuccess)
  237             goto loser;

  241         HMAC_Begin(cx);
  242         HMAC_Update(cx, state, state_len);
  243         rv = HMAC_Finish(cx, state, &state_len, state_len);

  257     }

  262 /* generate bits for the key and iv determination.  if enough bits
  263  * are not generated using PKCS 5, then we need to generate more bits
  264  * based on the extension proposed in PKCS 12
  265  */
  266 static SECItem *
  267 nsspkcs5_PBKDF1Extended(const SECHashObject *hashObj,
  268          NSSPKCS5PBEParameter *pbe_param, SECItem *pwitem, PRBool faulty3DES)
  269 {
  270     SECItem * hash                 = NULL;
  271     SECItem * newHash                 = NULL;
  272     int       bytes_needed;
  273     int       bytes_available;
  274     
  275     bytes_needed = pbe_param->ivLen + pbe_param->keyLen;
  276     bytes_available = hashObj->length;
  277     
  278     hash = nsspkcs5_PBKDF1(hashObj, &pbe_param->salt, pwitem, 
  279                                                 pbe_param->iter, faulty3DES);
  280 
  281     if(hash == NULL) {
  282         return NULL;
  283     }
  284 
  285     if(bytes_needed <= bytes_available) {
  286         return hash;
  287     } 
  288 
  289     newHash = nsspkcs5_PFXPBE(hashObj, pbe_param, hash, bytes_needed);
  290     if (hash != newHash)
  291         SECITEM_FreeItem(hash, PR_TRUE);
  292     return newHash;
  293 }

  347     last[salt->len+1] = (i >> 16) & 0xff;
  348     last[salt->len+2] = (i >> 8) & 0xff;
  349     last[salt->len+3] =    i  & 0xff;

  612             hash = nsspkcs5_PKCS12PBE(hashObj,pbe_param,pwitem,
  613                                                 pbeBitGenCipherIV,iv->len);
  614             if (hash == NULL) {
  615                 goto loser;
  616             }
  617             PORT_Memcpy(iv->data,hash->data,iv->len);
  618             SECITEM_ZfreeItem(hash,PR_TRUE);
  619             hash = NULL;
  620         }
  621         hash = nsspkcs5_PKCS12PBE(hashObj,pbe_param,pwitem,
  622                                                 pbe_param->keyID,key->len);
  623     default:
  624         break;
  625     }
  626 
  627     if (hash == NULL) {
  628         goto loser;
  629     }
  630 
  631     if (pbe_param->is2KeyDES) {
  632         PORT_Memcpy(key->data, hash->data, (key->len * 2) / 3);
  633         PORT_Memcpy(&(key->data[(key->len  * 2) / 3]), key->data,
  634                     key->len / 3);
  635     } else {
  636         PORT_Memcpy(key->data, hash->data, key->len);
  637     }
  638 
  639     SECITEM_ZfreeItem(hash, PR_TRUE);
  640     return key;
  641 
  642 loser:
  643     if (getIV && iv->data) {
  644         PORT_ZFree(iv->data,iv->len);
  645         iv->data = NULL;
  646     }
  647 
  648     SECITEM_ZfreeItem(key, PR_TRUE);
  649     return NULL;
  650 }

  652 static SECStatus
  653 nsspkcs5_FillInParam(SECOidTag algorithm, NSSPKCS5PBEParameter *pbe_param)
  654 {
  655     PRBool skipType = PR_FALSE;
  656 
  657     pbe_param->keyLen = 5;
  658     pbe_param->ivLen = 8;
  659     pbe_param->hashType = HASH_AlgSHA1;
  660     pbe_param->pbeType = NSSPKCS5_PBKDF1;
  661     pbe_param->encAlg = SEC_OID_RC2_CBC;
  662     pbe_param->is2KeyDES = PR_FALSE;
  663     switch(algorithm) {
  664     /* DES3 Algorithms */
  665     case SEC_OID_PKCS12_V2_PBE_WITH_SHA1_AND_2KEY_TRIPLE_DES_CBC:
  666         pbe_param->is2KeyDES = PR_TRUE;
  667         /* fall through */
  668     case SEC_OID_PKCS12_V2_PBE_WITH_SHA1_AND_3KEY_TRIPLE_DES_CBC:
  669         pbe_param->pbeType = NSSPKCS5_PKCS12_V2;
  670         /* fall through */
  671     case SEC_OID_PKCS12_PBE_WITH_SHA1_AND_TRIPLE_DES_CBC:
  672         pbe_param->keyLen = 24;
  673         pbe_param->encAlg = SEC_OID_DES_EDE3_CBC;
  674         break;
  675 
  676     /* DES Algorithms */
  677     case SEC_OID_PKCS5_PBE_WITH_MD2_AND_DES_CBC:
  678             pbe_param->hashType = HASH_AlgMD2;
  679         goto finish_des;
  680     case SEC_OID_PKCS5_PBE_WITH_MD5_AND_DES_CBC:
  681             pbe_param->hashType = HASH_AlgMD5;
  682         /* fall through */
  683     case SEC_OID_PKCS5_PBE_WITH_SHA1_AND_DES_CBC:
  684 finish_des:
  685         pbe_param->keyLen = 8;
  686         pbe_param->encAlg =  SEC_OID_DES_CBC;
  687         break;
  688 
  689     /* RC2 Algorithms */
  690     case SEC_OID_PKCS12_V2_PBE_WITH_SHA1_AND_128_BIT_RC2_CBC:
  691         pbe_param->keyLen = 16;
  692         /* fall through */
  693     case SEC_OID_PKCS12_V2_PBE_WITH_SHA1_AND_40_BIT_RC2_CBC:
  694         pbe_param->pbeType = NSSPKCS5_PKCS12_V2;
  695         break;
  696     case SEC_OID_PKCS12_PBE_WITH_SHA1_AND_128_BIT_RC2_CBC:
  697         pbe_param->keyLen = 16;
  698         /* fall through */
  699     case SEC_OID_PKCS12_PBE_WITH_SHA1_AND_40_BIT_RC2_CBC:
  700         break;
  701 
  702     /* RC4 algorithms */
  703     case SEC_OID_PKCS12_PBE_WITH_SHA1_AND_128_BIT_RC4:
  704         skipType = PR_TRUE;
  705         /* fall through */
  706     case SEC_OID_PKCS12_V2_PBE_WITH_SHA1_AND_128_BIT_RC4:
  707         pbe_param->keyLen = 16;
  708         /* fall through */
  709     case SEC_OID_PKCS12_V2_PBE_WITH_SHA1_AND_40_BIT_RC4:
  710         if (!skipType) {
  711                 pbe_param->pbeType = NSSPKCS5_PKCS12_V2;
  712         }
  713         /* fall through */
  714     case SEC_OID_PKCS12_PBE_WITH_SHA1_AND_40_BIT_RC4:
  715         pbe_param->ivLen = 0;
  716         pbe_param->encAlg =  SEC_OID_RC4;
  717         break;
  718     default:
  719         return SECFailure;
  720     }
  721 
  722     return SECSuccess;
  723 }

  725 /* decode the algid and generate a PKCS 5 parameter from it
  726  */
  727 NSSPKCS5PBEParameter *
  728 nsspkcs5_NewParam(SECOidTag alg, SECItem *salt, int iterator)
  729 {
  730     PRArenaPool *arena = NULL;
  731     NSSPKCS5PBEParameter *pbe_param = NULL;
  732     SECStatus rv = SECFailure;
  733 
  734     arena = PORT_NewArena(SEC_ASN1_DEFAULT_ARENA_SIZE);
  735     if (arena == NULL)
  736         return NULL;
  737 
  738     /* allocate memory for the parameter */
  739     pbe_param = (NSSPKCS5PBEParameter *)PORT_ArenaZAlloc(arena, 
  740         sizeof(NSSPKCS5PBEParameter));
  741 
  742     if (pbe_param == NULL) {
  743         goto loser;
  744     }
  745 
  746     pbe_param->poolp = arena;
  747 
  748     rv = nsspkcs5_FillInParam(alg, pbe_param);
  749     if (rv != SECSuccess) {
  750         goto loser;
  751     }
  752 
  753     pbe_param->iter = iterator;
  754     if (salt) {
  755         rv = SECITEM_CopyItem(arena,&pbe_param->salt,salt);
  756     }
  757 
  758     /* default key gen */
  759     pbe_param->keyID = pbeBitGenCipherKey;
  760 
  761 loser:
  762     if (rv != SECSuccess) {
  763         PORT_FreeArena(arena, PR_TRUE);
  764         pbe_param = NULL;
  765     }
  766 
  767     return pbe_param;
  768 }

  770 /* decode the algid and generate a PKCS 5 parameter from it
  771  */
  772 NSSPKCS5PBEParameter *
  773 nsspkcs5_AlgidToParam(SECAlgorithmID *algid)
  774 {
  775     NSSPKCS5PBEParameter *pbe_param = NULL;
  776     SECOidTag algorithm;
  777     SECStatus rv = SECFailure;
  778 
  779     if (algid == NULL) {
  780         return NULL;
  781     }
  782 
  783     algorithm = SECOID_GetAlgorithmTag(algid);
  784     if (algorithm == SEC_OID_UNKNOWN) {
  785         goto loser;
  786     }
  787 
  788     pbe_param = nsspkcs5_NewParam(algorithm, NULL, 1);
  789     if (pbe_param == NULL) {
  790         goto loser;
  791     }
  792 
  793     /* decode parameter */
  794     rv = SECFailure;
  795     switch (pbe_param->pbeType) {
  796     case NSSPKCS5_PBKDF1:
  797         rv = SEC_ASN1DecodeItem(pbe_param->poolp, pbe_param, 
  798             NSSPKCS5PBEParameterTemplate, &algid->parameters);
  799         break;
  800     case NSSPKCS5_PKCS12_V2:
  801         rv = SEC_ASN1DecodeItem(pbe_param->poolp, pbe_param, 
  802                 NSSPKCS5PKCS12V2PBEParameterTemplate, &algid->parameters);
  803         break;
  804     case NSSPKCS5_PBKDF2:
  805         break;
  806     }
  807 
  808 loser:
  809     if (rv == SECSuccess) {
  810             pbe_param->iter = DER_GetInteger(&pbe_param->iteration);
  811     } else {
  812         nsspkcs5_DestroyPBEParameter(pbe_param);
  813         pbe_param = NULL;
  814     }
  815 
  816     return pbe_param;
  817 }

  832 /* crypto routines */
  833 /* perform DES encryption and decryption.  these routines are called
  834  * by nsspkcs5_CipherData.  In the case of an error, NULL is returned.
  835  */
  836 static SECItem *
  837 sec_pkcs5_des(SECItem *key, SECItem *iv, SECItem *src, PRBool triple_des, 
  838                                                                 PRBool encrypt)
  839 {
  840     SECItem *dest;
  841     SECItem *dup_src;
  842     SECStatus rv = SECFailure;
  843     int pad;
  844 
  845     if((src == NULL) || (key == NULL) || (iv == NULL))
  846         return NULL;
  847 
  848     dup_src = SECITEM_DupItem(src);
  849     if(dup_src == NULL) {
  850         return NULL;
  851     }
  852 
  853     if(encrypt != PR_FALSE) {
  854         void *dummy;
  855 
  856         dummy = DES_PadBuffer(NULL, dup_src->data, 
  857             dup_src->len, &dup_src->len);
  858         if(dummy == NULL) {
  859             SECITEM_FreeItem(dup_src, PR_TRUE);
  860             return NULL;
  861         }
  862         dup_src->data = (unsigned char*)dummy;
  863     }
  864 
  865     dest = (SECItem *)PORT_ZAlloc(sizeof(SECItem));
  866     if(dest != NULL) {
  867         /* allocate with over flow */
  868         dest->data = (unsigned char *)PORT_ZAlloc(dup_src->len + 64);
  869         if(dest->data != NULL) {
  870             DESContext *ctxt;
  871             ctxt = DES_CreateContext(key->data, iv->data, 
  872                         (triple_des ? NSS_DES_EDE3_CBC : NSS_DES_CBC), 
  873                         encrypt);
  874         
  875             if(ctxt != NULL) {
  876                 rv = (encrypt ? DES_Encrypt : DES_Decrypt)(
  877                         ctxt, dest->data, &dest->len,
  878                         dup_src->len + 64, dup_src->data, dup_src->len);
  879 
  880                 /* remove padding -- assumes 64 bit blocks */
  881                 if((encrypt == PR_FALSE) && (rv == SECSuccess)) {
  882                     pad = dest->data[dest->len-1];
  883                     if((pad > 0) && (pad <= 8)) {
  884                         if(dest->data[dest->len-pad] != pad) {
  885                             rv = SECFailure;
  886                             PORT_SetError(SEC_ERROR_BAD_PASSWORD);
  887                         } else {
  888                             dest->len -= pad;
  889                         }
  890                     } else {
  891                         rv = SECFailure;
  892                         PORT_SetError(SEC_ERROR_BAD_PASSWORD);
  893                     }
  894                 }
  895                 DES_DestroyContext(ctxt, PR_TRUE);
  896             }
  897         }
  898     }
  899 
  900     if(rv == SECFailure) {
  901         if(dest != NULL) {
  902             SECITEM_FreeItem(dest, PR_TRUE);
  903         }
  904         dest = NULL;
  905     }
  906 
  907     if(dup_src != NULL) {
  908         SECITEM_FreeItem(dup_src, PR_TRUE);
  909     }
  910 
  911     return dest;
  912 }

  914 /* perform rc2 encryption/decryption if an error occurs, NULL is returned
  915  */
  916 static SECItem *
  917 sec_pkcs5_rc2(SECItem *key, SECItem *iv, SECItem *src, PRBool dummy, 
  918                                                                 PRBool encrypt)
  919 {
  920     SECItem *dest;
  921     SECItem *dup_src;
  922     SECStatus rv = SECFailure;
  923     int pad;
  924 
  925     if((src == NULL) || (key == NULL) || (iv == NULL)) {
  926         return NULL;
  927     }
  928 
  929     dup_src = SECITEM_DupItem(src);
  930     if(dup_src == NULL) {
  931         return NULL;
  932     }
  933 
  934     if(encrypt != PR_FALSE) {
  935         void *dummy;
  936 
  937         dummy = DES_PadBuffer(NULL, dup_src->data, 
  938                               dup_src->len, &dup_src->len);
  939         if(dummy == NULL) {
  940             SECITEM_FreeItem(dup_src, PR_TRUE);
  941             return NULL;
  942         }
  943         dup_src->data = (unsigned char*)dummy;
  944     }
  945 
  946     dest = (SECItem *)PORT_ZAlloc(sizeof(SECItem));
  947     if(dest != NULL) {
  948         dest->data = (unsigned char *)PORT_ZAlloc(dup_src->len + 64);
  949         if(dest->data != NULL) {
  950             RC2Context *ctxt;
  951 
  952             ctxt = RC2_CreateContext(key->data, key->len, iv->data,
  953                                                  NSS_RC2_CBC, key->len);
  954 
  955             if(ctxt != NULL) {
  956                 rv = (encrypt ? RC2_Encrypt: RC2_Decrypt)(
  957                         ctxt, dest->data, &dest->len,
  958                         dup_src->len + 64, dup_src->data, dup_src->len);
  959 
  960                 /* assumes 8 byte blocks  -- remove padding */        
  961                 if((rv == SECSuccess) && (encrypt != PR_TRUE)) {
  962                     pad = dest->data[dest->len-1];
  963                     if((pad > 0) && (pad <= 8)) {
  964                         if(dest->data[dest->len-pad] != pad) {
  965                             PORT_SetError(SEC_ERROR_BAD_PASSWORD);
  966                             rv = SECFailure;
  967                         } else {
  968                             dest->len -= pad;
  969                         }
  970                     } else {
  971                         PORT_SetError(SEC_ERROR_BAD_PASSWORD);
  972                         rv = SECFailure;
  973                     }
  974                 }
  975 
  976             }
  977         }
  978     }
  979 
  980     if((rv != SECSuccess) && (dest != NULL)) {
  981         SECITEM_FreeItem(dest, PR_TRUE);
  982         dest = NULL;
  983     }
  984 
  985     if(dup_src != NULL) {
  986         SECITEM_FreeItem(dup_src, PR_TRUE);
  987     }
  988 
  989     return dest;
  990 }

  992 /* perform rc4 encryption and decryption */
  993 static SECItem *
  994 sec_pkcs5_rc4(SECItem *key, SECItem *iv, SECItem *src, PRBool dummy_op,
  995                                                                  PRBool encrypt)
  996 {
  997     SECItem *dest;
  998     SECStatus rv = SECFailure;
  999 
 1000     if((src == NULL) || (key == NULL) || (iv == NULL)) {
 1001         return NULL;
 1002     }
 1003 
 1004     dest = (SECItem *)PORT_ZAlloc(sizeof(SECItem));
 1005     if(dest != NULL) {
 1006         dest->data = (unsigned char *)PORT_ZAlloc(sizeof(unsigned char) *
 1007             (src->len + 64));
 1008         if(dest->data != NULL) {
 1009             RC4Context *ctxt;
 1010 
 1011             ctxt = RC4_CreateContext(key->data, key->len);
 1012             if(ctxt) { 
 1013                 rv = (encrypt ? RC4_Encrypt : RC4_Decrypt)(
 1014                                 ctxt, dest->data, &dest->len,
 1015                                 src->len + 64, src->data, src->len);
 1016                 RC4_DestroyContext(ctxt, PR_TRUE);
 1017             }
 1018         }
 1019     }
 1020 
 1021     if((rv != SECSuccess) && (dest)) {
 1022         SECITEM_FreeItem(dest, PR_TRUE);
 1023         dest = NULL;
 1024     }
 1025 
 1026     return dest;

 1028 /* function pointer template for crypto functions */
 1029 typedef SECItem *(* pkcs5_crypto_func)(SECItem *key, SECItem *iv,
 1030                                          SECItem *src, PRBool op1, PRBool op2);
 1031 
 1032 /* performs the cipher operation on the src and returns the result.
 1033  * if an error occurs, NULL is returned. 
 1034  *
 1035  * a null length password is allowed.  this corresponds to encrypting
 1036  * the data with ust the salt.
 1037  */
 1038 /* change this to use PKCS 11? */
 1039 SECItem *
 1040 nsspkcs5_CipherData(NSSPKCS5PBEParameter *pbe_param, SECItem *pwitem, 
 1041                     SECItem *src, PRBool encrypt, PRBool *update)
 1042 {
 1043     SECItem *key = NULL, iv;
 1044     SECItem *dest = NULL;
 1045     PRBool tripleDES = PR_TRUE;
 1046     pkcs5_crypto_func cryptof;
 1047 
 1048     iv.data = NULL;
 1049 
 1050     if (update) { 
 1051         *update = PR_FALSE;
 1052     }
 1053 
 1054     if ((pwitem == NULL) || (src == NULL)) {
 1055         return NULL;
 1056     }
 1057 
 1058     /* get key, and iv */
 1059     key = nsspkcs5_ComputeKeyAndIV(pbe_param, pwitem, &iv, PR_FALSE);
 1060     if(key == NULL) {
 1061         return NULL;
 1062     }
 1063 
 1064     switch(pbe_param->encAlg) {
 1065     case SEC_OID_DES_EDE3_CBC:
 1066         cryptof = sec_pkcs5_des;
 1067         tripleDES = PR_TRUE;
 1068         break;
 1069     case SEC_OID_DES_CBC:
 1070         cryptof = sec_pkcs5_des;
 1071         tripleDES = PR_FALSE;
 1072         break;
 1073     case SEC_OID_RC2_CBC:
 1074         cryptof = sec_pkcs5_rc2;
 1075         break;
 1076     case SEC_OID_RC4:
 1077         cryptof = sec_pkcs5_rc4;
 1078         break;
 1079     default:
 1080         cryptof = NULL;
 1081         break;
 1082     }
 1083 
 1084     if (cryptof == NULL) {
 1085         goto loser;
 1086     }
 1087 
 1088     dest = (*cryptof)(key, &iv, src, tripleDES, encrypt);
 1089     /* 
 1090      * it's possible for some keys and keydb's to claim to
 1091      * be triple des when they're really des. In this case
 1092      * we simply try des. If des works we set the update flag
 1093      * so the key db knows it needs to update all it's entries.
 1094      *  The case can only happen on decrypted of a 
 1095      *  SEC_OID_DES_EDE3_CBD.
 1096      */
 1097     if ((dest == NULL) && (encrypt == PR_FALSE) && 
 1098                                 (pbe_param->encAlg == SEC_OID_DES_EDE3_CBC)) {
 1099         dest = (*cryptof)(key, &iv, src, PR_FALSE, encrypt);
 1100         if (update && (dest != NULL)) *update = PR_TRUE;
 1101     }
 1102 
 1103 loser:
 1104     if (key != NULL) {
 1105         SECITEM_ZfreeItem(key, PR_TRUE);
 1106     }
 1107     if (iv.data != NULL) {
 1108         SECITEM_ZfreeItem(&iv, PR_FALSE);
 1109     }
 1110 
 1111     return dest;
 1112 }

 1114 /* creates a algorithm ID containing the PBE algorithm and appropriate
 1115  * parameters.  the required parameter is the algorithm.  if salt is
 1116  * not specified, it is generated randomly.  if IV is specified, it overrides
 1117  * the PKCS 5 generation of the IV.  
 1118  *
 1119  * the returned SECAlgorithmID should be destroyed using 
 1120  * SECOID_DestroyAlgorithmID
 1121  */
 1122 SECAlgorithmID *
 1123 nsspkcs5_CreateAlgorithmID(PRArenaPool *arena, SECOidTag algorithm, 
 1124                                         NSSPKCS5PBEParameter *pbe_param)
 1125 {
 1126     SECAlgorithmID *algid, *ret_algid = NULL;
 1127     SECItem der_param;
 1128     SECStatus rv = SECFailure;
 1129     void *dummy = NULL;
 1130 
 1131     if (arena == NULL) {
 1132         return NULL;
 1133     }
 1134 
 1135     der_param.data = NULL;
 1136     der_param.len = 0;
 1137 
 1138     /* generate the algorithm id */
 1139     algid = (SECAlgorithmID *)PORT_ArenaZAlloc(arena, sizeof(SECAlgorithmID));
 1140     if (algid == NULL) {
 1141         goto loser;
 1142     }
 1143 
 1144     if (pbe_param->iteration.data == NULL) {
 1145         dummy = SEC_ASN1EncodeInteger(pbe_param->poolp,&pbe_param->iteration,
 1146                                                                 pbe_param->iter);
 1147         if (dummy == NULL) {
 1148             goto loser;
 1149         }
 1150     }
 1151     switch (pbe_param->pbeType) {
 1152     case NSSPKCS5_PBKDF1:
 1153         dummy = SEC_ASN1EncodeItem(arena, &der_param, pbe_param,
 1154                                         NSSPKCS5PBEParameterTemplate);
 1155         break;
 1156     case NSSPKCS5_PKCS12_V2:
 1157         dummy = SEC_ASN1EncodeItem(arena, &der_param, pbe_param,
 1158                                             NSSPKCS5PKCS12V2PBEParameterTemplate);
 1159         break;
 1160     default:
 1161         break;
 1162     }
 1163 
 1164     if (dummy == NULL) {
 1165         goto loser;
 1166     }
 1167         
 1168     rv = SECOID_SetAlgorithmID(arena, algid, algorithm, &der_param);
 1169     if (rv != SECSuccess) {
 1170         goto loser;
 1171     }
 1172 
 1173     ret_algid = (SECAlgorithmID *)PORT_ZAlloc(sizeof(SECAlgorithmID));
 1174     if (ret_algid == NULL) {
 1175         goto loser;
 1176     }
 1177 
 1178     rv = SECOID_CopyAlgorithmID(NULL, ret_algid, algid);
 1179     if (rv != SECSuccess) {
 1180         SECOID_DestroyAlgorithmID(ret_algid, PR_TRUE);
 1181         ret_algid = NULL;
 1182     }
 1183 
 1184 loser:        
 1185 
 1186     return ret_algid;
