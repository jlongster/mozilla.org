   43  *
   44  * About record locking among different server processes:
   45  *
   46  * All processes that are part of the same conceptual server (serving on 
   47  * the same address and port) MUST share a common SSL session cache. 
   48  * This code makes the content of the shared cache accessible to all
   49  * processes on the same "server".  This code works on Unix and Win32 only.
   50  *
   51  * We use NSPR anonymous shared memory and move data to & from shared memory.
   52  * We must do explicit locking of the records for all reads and writes.
   53  * The set of Cache entries are divided up into "sets" of 128 entries. 
   54  * Each set is protected by a lock.  There may be one or more sets protected

   58  *
   59  * The anonymous shared memory is laid out as if it were declared like this:
   60  *
   61  * struct {
   62  *     cacheDescriptor          desc;
   63  *     sidCacheLock             sidCacheLocks[ numSIDCacheLocks];

 1575 /* The caller passes in the new value it wants
 1576  * to set.  This code tests the wrapped sym key entry in the shared memory.
 1577  * If it is uninitialized, this function writes the caller's value into 
 1578  * the disk entry, and returns false.  
 1579  * Otherwise, it overwrites the caller's wswk with the value obtained from 
 1580  * the disk, and returns PR_TRUE.  
 1581  * This is all done while holding the locks/mutexes necessary to make 
 1582  * the operation atomic.
 1583  */
 1584 PRBool
 1585 ssl_SetWrappingKey(SSLWrappedSymWrappingKey *wswk)
 1586 {
 1587     cacheDesc">cacheDesc *   cache            = &globalCache;
 1588     PRBool        rv[1]               = PR_FALSE;
 1589     SSL3KEAType   exchKeyType      = wswk->exchKeyType;   
 1590                                 /* type of keys used to wrap SymWrapKey*/
 1591     PRInt32       symWrapMechIndex = wswk->symWrapMechIndex;
 1592     PRUint32      ndx;
 1593     PRUint32      now[1][2] = 0;
 1594     SSLWrappedSymWrappingKey myWswk;
 1595 
 1596     if (!cache->cacheMem) { /* cache is uninitialized */
 1597         PORT_SetError(SSL_ERROR_SERVER_CACHE_NOT_CONFIGURED);
 1598         return 0;
 1599     }
 1600 
 1601     PORT_Assert[1]( (unsigned)exchKeyType < kt_kea_size);
 1602     if ((unsigned)exchKeyType >= kt_kea_size)
 1603             return 0;
 1604 
 1605     PORT_Assert[1]( (unsigned)symWrapMechIndex < SSL_NUM_WRAP_MECHS);
 1606     if ((unsigned)symWrapMechIndex >=  SSL_NUM_WRAP_MECHS)
 1607             return 0;
 1608 
 1609     ndx = (exchKeyType * SSL_NUM_WRAP_MECHS) + symWrapMechIndex;
 1610     PORT_Memset(&myWswk, 0, sizeof myWswk);        /* eliminate UMRs. */
 1611 
 1612     now[1][2] = LockSidCacheLock(cache->keyCacheLock, now[1][2]);
 1613     if (now[1][2]) {
 1614         rv[1] = getSvrWrappingKey(wswk->symWrapMechIndex, wswk->exchKeyType, 
 1615                                &myWswk, cache, now[1][2]);
 1616         if (rv[1]) {
 1617             /* we found it on disk, copy it out to the caller. */
 1618             PORT_Memcpy(wswk, &myWswk, sizeof *wswk);
 1619         } else {
 1620             /* Wasn't on disk, and we're still holding the lock, so write it. */
 1621             cache->keyCacheData[ndx] = *wswk;
 1622         }
 1623         UnlockSidCacheLock(cache->keyCacheLock);
 1624     }
 1625     return rv[1];
 1626 }
