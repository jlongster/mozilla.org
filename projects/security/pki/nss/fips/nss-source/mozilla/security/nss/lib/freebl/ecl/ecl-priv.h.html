   68     { mp_word w; \
   69     w = ((mp_word)(cin)) + (a1) + (a2); \
   70     s = ACCUM(w); \
   71     cout = CARRYOUT(w); }
   72 
   73 #define MP_SUB_BORROW(a1, a2, s, bin, bout)   \
   74     { mp_word w; \
   75     w = ((mp_word)(a1)) - (a2) - (bin); \
   76     s = ACCUM(w); \
   77     bout = (w >> MP_DIGIT_BIT) & 1; }
   78 
   79 #else
   80 /* NOTE, 
   81  * cin and cout could be the same variable.
   82  * bin and bout could be the same variable.
   83  * a1 or a2 and s could be the same variable.
   84  * don't trash those outputs until their respective inputs have
   85  * been read. */
   86 #define MP_ADD_CARRY(a1, a2, s, cin, cout)   \
   87     { mp_digit tmp[1],sum; \
   88     tmp[1] = (a1); \
   89     sum = tmp[1] + (a2); \
   90     tmp[1] = (sum < tmp[1]);                     /* detect overflow */ \
   91     s = sum += (cin); \
   92     cout = tmp[1] + (sum < (cin)); }
   93 
   94 #define MP_SUB_BORROW(a1, a2, s, bin, bout)   \
   95     { mp_digit tmp[1]; \
   96     tmp[1] = (a1); \
   97     s = tmp[1] - (a2); \
   98     tmp[1] = (s > tmp[1]);                    /* detect borrow */ \
   99     if ((bin) && !s--) tmp[1]++;        \
  100     bout = tmp[1]; }
  101 #endif

  104 struct GFMethodStr;
  105 typedef struct GFMethodStr GFMethod;
  106 struct GFMethodStr {
  107         /* Indicates whether the structure was constructed from dynamic memory 
  108          * or statically created. */
  109         int constructed;
  110         /* Irreducible that defines the field. For prime fields, this is the
  111          * prime p. For binary polynomial fields, this is the bitstring
  112          * representation of the irreducible polynomial. */
  113         mp_int irr;
  114         /* For prime fields, the value irr_arr[0] is the number of bits in the 
  115          * field. For binary polynomial fields, the irreducible polynomial
  116          * f(t) is represented as an array of unsigned int[], where f(t) is
  117          * of the form: f(t) = t^p[0] + t^p[1] + ... + t^p[4] where m = p[0]
  118          * > p[1] > ... > p[4] = 0. */
  119         unsigned int irr_arr[5];
  120         /* Field arithmetic methods. All methods (except field_enc and
  121          * field_dec) are assumed to take field-encoded parameters and return
  122          * field-encoded values. All methods (except field_enc and field_dec)
  123          * are required to be implemented. */
  124         mp_err (*field_add) (const mp_int *a, const mp_int *b, mp_int *r,
  125                                                  const GFMethod *meth);
  126         mp_err (*field_neg) (const mp_int *a, mp_int *r, const GFMethod *meth);
  127         mp_err (*field_sub) (const mp_int *a, const mp_int *b, mp_int *r,
  128                                                  const GFMethod *meth);
  129         mp_err (*field_mod) (const mp_int *a, mp_int *r, const GFMethod *meth);
  130         mp_err (*field_mul) (const mp_int *a, const mp_int *b, mp_int *r,
  131                                                  const GFMethod *meth);
  132         mp_err (*field_sqr) (const mp_int *a, mp_int *r, const GFMethod *meth);
  133         mp_err (*field_div) (const mp_int *a, const mp_int *b, mp_int *r,
  134                                                  const GFMethod *meth);
  135         mp_err (*field_enc) (const mp_int *a, mp_int *r, const GFMethod *meth);
  136         mp_err (*field_dec) (const mp_int *a, mp_int *r, const GFMethod *meth);
  137         /* Extra storage for implementation-specific data.  Any memory
  138          * allocated to these extra fields will be cleared by extra_free. */
  139         void *extra1;
  140         void *extra2;
  141         void (*extra_free) (GFMethod *meth);
  142 };

  152 struct ECGroupStr {
  153         /* Indicates whether the structure was constructed from dynamic memory 
  154          * or statically created. */
  155         int constructed;
  156         /* Field definition and arithmetic. */
  157         GFMethod *meth;
  158         /* Textual representation of curve name, if any. */
  159         char *text;
  160         /* Curve parameters, field-encoded. */
  161         mp_int curvea, curveb;
  162         /* x and y coordinates of the base point, field-encoded. */
  163         mp_int genx, geny;
  164         /* Order and cofactor of the base point. */
  165         mp_int order;
  166         int cofactor;
  167         /* Point arithmetic methods. All methods are assumed to take
  168          * field-encoded parameters and return field-encoded values. All
  169          * methods (except base_point_mul and points_mul) are required to be
  170          * implemented. */
  171         mp_err (*point_add) (const mp_int *px, const mp_int *py,
  172                                                  const mp_int *qx, const mp_int *qy, mp_int *rx,
  173                                                  mp_int *ry, const ECGroup *group);
  174         mp_err (*point_sub) (const mp_int *px, const mp_int *py,
  175                                                  const mp_int *qx, const mp_int *qy, mp_int *rx,
  176                                                  mp_int *ry, const ECGroup *group);
  177         mp_err (*point_dbl) (const mp_int *px, const mp_int *py, mp_int *rx,
  178                                                  mp_int *ry, const ECGroup *group);
  179         mp_err (*point_mul) (const mp_int *n, const mp_int *px,
  180                                                  const mp_int *py, mp_int *rx, mp_int *ry,
  181                                                  const ECGroup *group);
  182         mp_err (*base_point_mul) (const mp_int *n, mp_int *rx, mp_int *ry,
  183                                                           const ECGroup *group);
  184         mp_err (*points_mul) (const mp_int *k1, const mp_int *k2,
  185                                                   const mp_int *px, const mp_int *py, mp_int *rx,
  186                                                   mp_int *ry, const ECGroup *group);
  187         mp_err (*validate_point) (const mp_int *px, const mp_int *py, const ECGroup *group);
  188         /* Extra storage for implementation-specific data.  Any memory
  189          * allocated to these extra fields will be cleared by extra_free. */
  190         void *extra1;
  191         void *extra2;
  192         void (*extra_free) (ECGroup *group);
  193 };
