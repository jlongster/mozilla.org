   55 typedef struct {
   56   unsigned int  sec;
   57   unsigned int  usec;
   58 } instant_t;
   59 
   60 instant_t[1][2] now(void)
   61 {
   62   struct timeval clk;
   63   instant_t[1][2]      res;
   64 
   65   res.sec = res.usec = 0;
   66 
   67   if(gettimeofday(&clk, NULL) != 0)
   68     return res;
   69 
   70   res.sec = clk.tv_sec;
   71   res.usec = clk.tv_usec;
   72 
   73   return res;
   74 }
   75 
   76 #define PRECISION 16
   77 
   78 int main(int argc, char *argv[])
   79 {
   80   int          ix, num, prec = PRECISION;
   81   mp_int       a, b, c, d;
   82   instant_t[1][2]    start, finish;
   83   time_t       seed;
   84   unsigned int d1, d2;
   85 
   86   seed = time(NULL);
   87 
   88   if(argc < 2) {
   89     fprintf(stderr, "Usage: %s <num-tests>\n", argv[0]);
   90     return 1;
   91   }
   92 
   93   if((num = atoi(argv[1])) < 0)
   94     num = -num;
   95 
   96   printf("Test 5a: Euclid vs. Binary, a GCD speed test\n\n"
   97          "Number of tests: %d\n"
   98          "Precision:       %d digits\n\n", num, prec);
   99 
  100   mp_init_size(&a, prec);
  101   mp_init_size(&b, prec);
  102   mp_init(&c);
  103   mp_init(&d);
  104 
  105   printf("Verifying accuracy ... \n");
  106   srand((unsigned int)seed);
  107   for(ix = 0; ix < num; ix++) {
  108     mpp_random_size(&a, prec);
  109     mpp_random_size(&b, prec);
  110 
  111     mp_gcd(&a, &b, &c);
  112     mp_bgcd(&a, &b, &d);
  113 
  114     if(mp_cmp(&c, &d) != 0) {
  115       printf("Error!  Results not accurate:\n");
  116       printf("a = "); mp_print(&a, stdout); fputc('\n', stdout);
  117       printf("b = "); mp_print(&b, stdout); fputc('\n', stdout);
  118       printf("c = "); mp_print(&c, stdout); fputc('\n', stdout);
  119       printf("d = "); mp_print(&d, stdout); fputc('\n', stdout);
  120 
  121       mp_clear(&a); mp_clear(&b); mp_clear(&c); mp_clear(&d);
  122       return 1;
  123     }
  124   }
  125   mp_clear(&d);
  126   printf("Accuracy confirmed for the %d test samples\n", num);
  127 
  128   printf("Testing Euclid ... \n");
  129   srand((unsigned int)seed);
  130   start = now[1][2]();
  131   for(ix = 0; ix < num; ix++) {
  132     mpp_random_size(&a, prec);
  133     mpp_random_size(&b, prec);
  134     mp_gcd(&a, &b, &c);
  135 
  136   }
  137   finish = now[1][2]();
  138 
  139   d1 = (finish.sec - start.sec) * 1000000;
  140   d1 -= start.usec; d1 += finish.usec;
  141 
  142   printf("Testing binary ... \n");
  143   srand((unsigned int)seed);
  144   start = now[1][2]();
  145   for(ix = 0; ix < num; ix++) {
  146     mpp_random_size(&a, prec);
  147     mpp_random_size(&b, prec);
  148     mp_bgcd(&a, &b, &c);
  149   }
  150   finish = now[1][2]();
  151 
  152   d2 = (finish.sec - start.sec) * 1000000;
  153   d2 -= start.usec; d2 += finish.usec;
  154 
  155   printf("Euclidean algorithm time: %u usec\n", d1);
  156   printf("Binary algorithm time:    %u usec\n", d2);
  157   printf("Improvement:              %.2f%%\n",
  158          (1.0 - ((double)d2 / (double)d1)) * 100.0);
  159 
  160   mp_clear(&c);
  161   mp_clear(&b);
  162   mp_clear(&a);
  163 
  164   return 0;
  165 }
