   69 NSS_IMPLEMENT NSSCryptoContext *
   70 nssCryptoContext_Create (
   71   NSSTrustDomain *td,
   72   NSSCallback *uhhOpt
   73 )
   74 {
   75     NSSArena *arena;
   76     NSSCryptoContext *rvCC;
   77     arena = NSSArena_Create();
   78     if (!arena) {
   79         return NULL;
   80     }
   81     rvCC = nss_ZNEW(arena, NSSCryptoContext);
   82     if (!rvCC) {
   83         return NULL;
   84     }
   85     rvCC->td = td;
   86     rvCC->arena = arena;
   87     rvCC->certStore = nssCertificateStore_Create(rvCC->arena);
   88     if (!rvCC->certStore) {
   89         nssArena_Destroy(arena);
   90         return NULL;
   91     }
   92 
   93     return rvCC;
   94 }

  100 {

  105         if (status == PR_FAILURE) {
  106             return status;
  107         }

  111     nssArena_Destroy(cc->arena);
  112     return status;
