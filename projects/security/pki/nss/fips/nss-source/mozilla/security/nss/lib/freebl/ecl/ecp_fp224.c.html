   47 /* Performs a single step of reduction, just on the uppermost float
   48  * (assumes already tidied), and then retidies. Note, this does not
   49  * guarantee that the result will be less than p. */
   50 void
   51 ecfp224_singleReduce(double *r, const EC_group_fp * group)
   52 {
   53         double q;
   54 
   55         ECFP_ASSERT(group->doubleBitSize == 24);
   56         ECFP_ASSERT(group->primeBitSize == 224);
   57         ECFP_ASSERT(group->numDoubles == 10);
   58 
   59         q = r[ECFP_NUMDOUBLES[1][2] - 1] - ecfp_beta_224;
   60         q += group->bitSize_alpha;
   61         q -= group->bitSize_alpha;
   62 
   63         r[ECFP_NUMDOUBLES[1][2] - 1] -= q;
   64         r[0] -= q * ecfp_twom224;
   65         r[4] += q * ecfp_twom128;
   66 
   67         ecfp_positiveTidy(r, group);
   68 }
   69 
   70 /* 
   71  * Performs imperfect reduction.  This might leave some negative terms,
   72  * and one more reduction might be required for the result to be between 0 
   73  * and p-1. x should be be an array of at least 20, and r at least 10 x
   74  * and r can be the same, but then the upper parts of r are not zeroed */
   75 void
   76 ecfp224_reduce(double *r, double *x, const EC_group_fp * group)
   77 {
   78 
   79         double x10, x11, x12, x13, x14, q;
   80 
   81         ECFP_ASSERT(group->doubleBitSize == 24);
   82         ECFP_ASSERT(group->primeBitSize == 224);
   83         ECFP_ASSERT(group->numDoubles == 10);
   84 
   85         /* Tidy just the upper bits of x.  Don't need to tidy the lower ones
   86          * yet. */
   87         ecfp_tidyUpper(x, group);
   88 
   89         x10 = x[10] + x[16] * ecfp_twom128;
   90         x11 = x[11] + x[17] * ecfp_twom128;
   91         x12 = x[12] + x[18] * ecfp_twom128;
   92         x13 = x[13] + x[19] * ecfp_twom128;
   93 
   94         /* Tidy up, or we won't have enough bits later to add it in */
   95         q = x10 + group->alpha[11];
   96         q -= group->alpha[11];
   97         x10 -= q;
   98         x11 = x11 + q;
   99 
  100         q = x11 + group->alpha[12];
  101         q -= group->alpha[12];
  102         x11 -= q;
  103         x12 = x12 + q;
  104 
  105         q = x12 + group->alpha[13];
  106         q -= group->alpha[13];
  107         x12 -= q;
  108         x13 = x13 + q;
  109 
  110         q = x13 + group->alpha[14];
  111         q -= group->alpha[14];
  112         x13 -= q;
  113         x14 = x[14] + q;
  114 
  115         r[9] = x[9] + x[15] * ecfp_twom128 - x[19] * ecfp_twom224;
  116         r[8] = x[8] + x14 * ecfp_twom128 - x[18] * ecfp_twom224;
  117         r[7] = x[7] + x13 * ecfp_twom128 - x[17] * ecfp_twom224;
  118         r[6] = x[6] + x12 * ecfp_twom128 - x[16] * ecfp_twom224;
  119         r[5] = x[5] + x11 * ecfp_twom128 - x[15] * ecfp_twom224;
  120         r[4] = x[4] + x10 * ecfp_twom128 - x14 * ecfp_twom224;
  121         r[3] = x[3] - x13 * ecfp_twom224;
  122         r[2] = x[2] - x12 * ecfp_twom224;
  123         r[1] = x[1] - x11 * ecfp_twom224;
  124         r[0] = x[0] - x10 * ecfp_twom224;
  125 
  126         /* 
  127          * Tidy up just r[ECFP_NUMDOUBLES-2] so that the number of reductions
  128          * is accurate plus or minus one.  (Rather than tidy all to make it
  129          * totally accurate) */
  130         q = r[ECFP_NUMDOUBLES[1][2] - 2] + group->alpha[ECFP_NUMDOUBLES[1][2] - 1];
  131         q -= group->alpha[ECFP_NUMDOUBLES[1][2] - 1];
  132         r[ECFP_NUMDOUBLES[1][2] - 2] -= q;
  133         r[ECFP_NUMDOUBLES[1][2] - 1] += q;
  134 
  135         /* Tidy up the excess bits on r[ECFP_NUMDOUBLES-1] using reduction */
  136         /* Use ecfp_beta so we get a positive res */
  137         q = r[ECFP_NUMDOUBLES[1][2] - 1] - ecfp_beta_224;
  138         q += group->bitSize_alpha;
  139         q -= group->bitSize_alpha;
  140 
  141         r[ECFP_NUMDOUBLES[1][2] - 1] -= q;
  142         r[0] -= q * ecfp_twom224;
  143         r[4] += q * ecfp_twom128;
  144 
  145         ecfp_tidyShort(r, group);
  146 }

  148 /* Sets group to use optimized calculations in this file */
  149 mp_err
  150 ec_group_set_nistp224_fp(ECGroup *group)
  151 {
  152 
  153         EC_group_fp *fpg;
  154 
  155         /* Allocate memory for floating point group data */
  156         fpg = (EC_group_fp *) malloc(sizeof(EC_group_fp));
  157         if (fpg == NULL) {
  158                 return MP_MEM;
  159         }
  160 
  161         fpg->numDoubles = ECFP_NUMDOUBLES[1][2];
  162         fpg->primeBitSize = ECFP_BSIZE[1][2];
  163         fpg->orderBitSize = 224;
  164         fpg->doubleBitSize = 24;
  165         fpg->numInts = (ECFP_BSIZE[1][2] + ECL_BITS - 1) / ECL_BITS;
  166         fpg->aIsM3 = 1;
  167         fpg->ecfp_singleReduce = &ecfp224_singleReduce;
  168         fpg->ecfp_reduce = &ecfp224_reduce;
  169         fpg->ecfp_tidy = &ecfp_tidy;
  170 
  171         fpg->pt_add_jac_aff = &ecfp224_pt_add_jac_aff;
  172         fpg->pt_add_jac = &ecfp224_pt_add_jac;
  173         fpg->pt_add_jm_chud = &ecfp224_pt_add_jm_chud;
  174         fpg->pt_add_chud = &ecfp224_pt_add_chud;
  175         fpg->pt_dbl_jac = &ecfp224_pt_dbl_jac;
  176         fpg->pt_dbl_jm = &ecfp224_pt_dbl_jm;
  177         fpg->pt_dbl_aff2chud = &ecfp224_pt_dbl_aff2chud;
  178         fpg->precompute_chud = &ecfp224_precompute_chud;
  179         fpg->precompute_jac = &ecfp224_precompute_jac;
  180 
  181         group->point_mul = &ec_GFp_point_mul_wNAF_fp;
  182         group->points_mul = &ec_pts_mul_basic;
  183         group->extra1 = fpg;
  184         group->extra_free = &ec_GFp_extra_free_fp;
  185 
  186         ec_set_fp_precision(fpg);
  187         fpg->bitSize_alpha = ECFP_TWO224 * fpg->alpha[0];
  188 
  189         return MP_OKAY;
  190 }
