   81 /*
   82  * See bugzilla bug 125359
   83  * Since NSS (via PKCS#11) wants to handle big integers as unsigned ints,
   84  * all of the templates above that en/decode into integers must be converted
   85  * from ASN.1's signed integer type.  This is done by marking either the
   86  * source or destination (encoding or decoding, respectively) type as
   87  * siUnsignedInteger.
   88  */

   90 static void
   91 prepare_low_rsa_pub_key_for_asn1(NSSLOWKEYPublicKey *pubk)
   92 {
   93     pubk->u.rsa.modulus.type = siUnsignedInteger;
   94     pubk->u.rsa.publicExponent.type = siUnsignedInteger;
   95 }

   97 static void
   98 prepare_low_dsa_pub_key_for_asn1(NSSLOWKEYPublicKey *pubk)
   99 {
  100     pubk->u.dsa.publicValue.type = siUnsignedInteger;
  101     pubk->u.dsa.params.prime.type = siUnsignedInteger;
  102     pubk->u.dsa.params.subPrime.type = siUnsignedInteger;
  103     pubk->u.dsa.params.base.type = siUnsignedInteger;
  104 }

  110     pubk->u.dh.base.type = siUnsignedInteger;
  111     pubk->u.dh.publicValue.type = siUnsignedInteger;

  134 /*
  135  * simple cert decoder to avoid the cost of asn1 engine
  136  */ 
  137 static unsigned char *
  138 nsslowcert_dataStart(unsigned char *buf, unsigned int length, 
  139                         unsigned int *data_length, PRBool includeTag,
  140                         unsigned char* rettag) {
  141     unsigned char tag;
  142     unsigned int used_length= 0;
  143 
  144     tag = buf[used_length++];
  145 
  146     if (rettag) {
  147         *rettag = tag;
  148     }
  149 
  150     /* blow out when we come to the end */
  151     if (tag == 0) {
  152         return NULL;
  153     }
  154 
  155     *data_length = buf[used_length++];
  156 
  157     if (*data_length&0x80) {
  158         int  len_count = *data_length & 0x7f;
  159 
  160         *data_length = 0;
  161 
  162         while (len_count-- > 0) {
  163             *data_length = (*data_length << 8) | buf[used_length++];
  164         } 
  165     }
  166 
  167     if (*data_length > (length-used_length) ) {
  168         *data_length = length-used_length;
  169         return NULL;
  170     }
  171     if (includeTag) *data_length += used_length;
  172 
  173     return (buf + (includeTag ? 0 : used_length));        
  174 }

  176 static void SetTimeType(SECItem* item, unsigned char tagtype)
  177 {
  178     switch (tagtype) {
  179         case SEC_ASN1_UTC_TIME:
  180             item->type = siUTCTime;
  181             break;
  182 
  183         case SEC_ASN1_GENERALIZED_TIME:
  184             item->type = siGeneralizedTime;
  185             break;
  186 
  187         default:
  188             PORT_Assert[1](0);
  189             break;
  190     }
  191 }

  193 static int
  194 nsslowcert_GetValidityFields(unsigned char *buf,int buf_length,
  195         SECItem *notBefore, SECItem *notAfter)
  196 {
  197     unsigned char tagtype;
  198     notBefore->data = nsslowcert_dataStart(buf,buf_length,
  199                                                 &notBefore->len,PR_FALSE, &tagtype);
  200     if (notBefore->data == NULL) return SECFailure;
  201     SetTimeType(notBefore, tagtype);
  202     buf_length -= (notBefore->data-buf) + notBefore->len;
  203     buf = notBefore->data + notBefore->len;
  204     notAfter->data = nsslowcert_dataStart(buf,buf_length,
  205                                                 &notAfter->len,PR_FALSE, &tagtype);
  206     if (notAfter->data == NULL) return SECFailure;
  207     SetTimeType(notAfter, tagtype);
  208     return SECSuccess;
  209 }

  211 static int
  212 nsslowcert_GetCertFields(unsigned char *cert,int cert_length,
  213         SECItem *issuer, SECItem *serial, SECItem *derSN, SECItem *subject,
  214         SECItem *valid, SECItem *subjkey)
  215 {
  216     unsigned char *buf;
  217     unsigned int buf_length;
  218     unsigned char *dummy;
  219     unsigned int dummylen;
  220 
  221     /* get past the signature wrap */
  222     buf = nsslowcert_dataStart(cert,cert_length,&buf_length,PR_FALSE, NULL);
  223     if (buf == NULL) return SECFailure;
  224     /* get into the raw cert data */
  225     buf = nsslowcert_dataStart(buf,buf_length,&buf_length,PR_FALSE, NULL);
  226     if (buf == NULL) return SECFailure;
  227     /* skip past any optional version number */
  228     if ((buf[0] & 0xa0) == 0xa0) {
  229         dummy = nsslowcert_dataStart(buf,buf_length,&dummylen,PR_FALSE, NULL);
  230         if (dummy == NULL) return SECFailure;
  231         buf_length -= (dummy-buf) + dummylen;
  232         buf = dummy + dummylen;
  233     }
  234     /* serial number */
  235     if (derSN) {
  236         derSN->data=nsslowcert_dataStart(buf,buf_length,&derSN->len,PR_TRUE, NULL);
  237     }
  238     serial->data = nsslowcert_dataStart(buf,buf_length,&serial->len,PR_FALSE, NULL);
  239     if (serial->data == NULL) return SECFailure;
  240     buf_length -= (serial->data-buf) + serial->len;
  241     buf = serial->data + serial->len;
  242     /* skip the OID */
  243     dummy = nsslowcert_dataStart(buf,buf_length,&dummylen,PR_FALSE, NULL);
  244     if (dummy == NULL) return SECFailure;
  245     buf_length -= (dummy-buf) + dummylen;
  246     buf = dummy + dummylen;
  247     /* issuer */
  248     issuer->data = nsslowcert_dataStart(buf,buf_length,&issuer->len,PR_TRUE, NULL);
  249     if (issuer->data == NULL) return SECFailure;
  250     buf_length -= (issuer->data-buf) + issuer->len;
  251     buf = issuer->data + issuer->len;
  252 
  253     /* only wanted issuer/SN */
  254     if (valid == NULL) {
  255         return SECSuccess;
  256     }
  257     /* validity */
  258     valid->data = nsslowcert_dataStart(buf,buf_length,&valid->len,PR_FALSE, NULL);
  259     if (valid->data == NULL) return SECFailure;
  260     buf_length -= (valid->data-buf) + valid->len;
  261     buf = valid->data + valid->len;
  262     /*subject */
  263     subject->data=nsslowcert_dataStart(buf,buf_length,&subject->len,PR_TRUE, NULL);
  264     if (subject->data == NULL) return SECFailure;
  265     buf_length -= (subject->data-buf) + subject->len;
  266     buf = subject->data + subject->len;
  267     /* subject  key info */
  268     subjkey->data=nsslowcert_dataStart(buf,buf_length,&subjkey->len,PR_TRUE, NULL);
  269     if (subjkey->data == NULL) return SECFailure;
  270     buf_length -= (subjkey->data-buf) + subjkey->len;
  271     buf = subjkey->data + subjkey->len;
  272     return SECSuccess;
  273 }

  275 SECStatus
  276 nsslowcert_GetCertTimes(NSSLOWCERTCertificate *c, PRTime *notBefore, PRTime *notAfter)
  277 {
  278     int rv;
  279     NSSLOWCERTValidity validity;
  280 
  281     rv = nsslowcert_GetValidityFields(c->validity.data,c->validity.len,
  282                                 &validity.notBefore,&validity.notAfter);
  283     if (rv != SECSuccess) {
  284         return rv;
  285     }
  286     
  287     /* convert DER not-before time */
  288     rv = DER_DecodeTimeChoice(notBefore, &validity.notBefore);
  289     if (rv) {
  290         return(SECFailure);
  291     }
  292     
  293     /* convert DER not-after time */
  294     rv = DER_DecodeTimeChoice(notAfter, &validity.notAfter);
  295     if (rv) {
  296         return(SECFailure);
  297     }
  298 
  299     return(SECSuccess);
  300 }

  302 /*
  303  * is certa newer than certb?  If one is expired, pick the other one.
  304  */
  305 PRBool
  306 nsslowcert_IsNewer(NSSLOWCERTCertificate *certa, NSSLOWCERTCertificate *certb)
  307 {
  308     PRTime notBeforeA, notAfterA, notBeforeB, notAfterB, now[1][2];
  309     SECStatus rv;
  310     PRBool newerbefore, newerafter;
  311     
  312     rv = nsslowcert_GetCertTimes(certa, &notBeforeA, &notAfterA);
  313     if ( rv != SECSuccess ) {
  314         return(PR_FALSE);
  315     }
  316     
  317     rv = nsslowcert_GetCertTimes(certb, &notBeforeB, &notAfterB);
  318     if ( rv != SECSuccess ) {
  319         return(PR_TRUE);
  320     }
  321 
  322     newerbefore = PR_FALSE;
  323     if ( LL_CMP(notBeforeA, >, notBeforeB) ) {
  324         newerbefore = PR_TRUE;
  325     }
  326 
  327     newerafter = PR_FALSE;
  328     if ( LL_CMP(notAfterA, >, notAfterB) ) {
  329         newerafter = PR_TRUE;
  330     }
  331     
  332     if ( newerbefore && newerafter ) {
  333         return(PR_TRUE);
  334     }
  335     
  336     if ( ( !newerbefore ) && ( !newerafter ) ) {
  337         return(PR_FALSE);
  338     }
  339 
  340     /* get current time */
  341     now[1][2] = PR_Now();
  342 
  343     if ( newerbefore ) {
  344         /* cert A was issued after cert B, but expires sooner */
  345         /* if A is expired, then pick B */
  346         if ( LL_CMP(notAfterA, <, now[1][2] ) ) {
  347             return(PR_FALSE);
  348         }
  349         return(PR_TRUE);
  350     } else {
  351         /* cert B was issued after cert A, but expires sooner */
  352         /* if B is expired, then pick A */
  353         if ( LL_CMP(notAfterB, <, now[1][2] ) ) {
  354             return(PR_TRUE);
  355         }
  356         return(PR_FALSE);
  357     }
  358 }

  362 static SECStatus
  363 nsslowcert_KeyFromIssuerAndSN(PRArenaPool *arena, 
  364                               SECItem *issuer, SECItem *sn, SECItem *key)
  365 {
  366     unsigned int len = sn->len + issuer->len;
  367 
  368     if (!arena) {
  369         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  370         goto loser;
  371     }
  372     key->data = (unsigned char*)PORT_ArenaAlloc(arena, len);
  373     if ( !key->data ) {
  374         goto loser;
  375     }
  376 
  377     key->len = len;
  378     /* copy the serialNumber */
  379     PORT_Memcpy(key->data, sn->data, sn->len);
  380 
  381     /* copy the issuer */
  382     PORT_Memcpy(&key->data[sn->len], issuer->data, issuer->len);
  383 
  384     return(SECSuccess);
  385 
  386 loser:
  387     return(SECFailure);
  388 }

  416 /*
  417  * take a DER certificate and decode it into a certificate structure
  418  */
  419 NSSLOWCERTCertificate *
  420 nsslowcert_DecodeDERCertificate(SECItem *derSignedCert, char *nickname)
  421 {
  422     NSSLOWCERTCertificate *cert;
  423     int rv;
  424 
  425     /* allocate the certificate structure */
  426     cert = nsslowcert_CreateCert();
  427     
  428     if ( !cert ) {
  429         goto loser;
  430     }
  431     
  432         /* point to passed in DER data */
  433     cert->derCert = *derSignedCert;
  434     cert->nickname = NULL;
  435     cert->certKey.data = NULL;
  436     cert->referenceCount = 1;
  437 
  438     /* decode the certificate info */
  439     rv = nsslowcert_GetCertFields(cert->derCert.data, cert->derCert.len,
  440         &cert->derIssuer, &cert->serialNumber, &cert->derSN, &cert->derSubject,
  441         &cert->validity, &cert->derSubjKeyInfo);
  442 
  443     /* cert->subjectKeyID;         x509v3 subject key identifier */
  444     cert->subjectKeyID.data = NULL;
  445     cert->subjectKeyID.len = 0;
  446     cert->dbEntry = NULL;
  447     cert ->trust = NULL;
  448     cert ->dbhandle = NULL;
  449 
  450     /* generate and save the database key for the cert */
  451     rv = nsslowcert_KeyFromIssuerAndSNStatic(cert->certKeySpace,
  452                 sizeof(cert->certKeySpace), &cert->derIssuer, 
  453                 &cert->serialNumber, &cert->certKey);
  454     if ( rv ) {
  455         goto loser;
  456     }
  457 
  458     /* set the nickname */
  459     if ( nickname == NULL ) {
  460         cert->nickname = NULL;
  461     } else {
  462         /* copy and install the nickname */
  463         cert->nickname = pkcs11_copyNickname(nickname,cert->nicknameSpace,
  464                                 sizeof(cert->nicknameSpace));
  465     }
  466 
  467 #ifdef FIXME
  468     /* initialize the subjectKeyID */
  469     rv = cert_GetKeyID(cert);
  470     if ( rv != SECSuccess ) {
  471         goto loser;
  472     }
  473 
  474     /* set the email address */
  475     cert->emailAddr = CERT_GetCertificateEmailAddress(cert);
  476     
  477 #endif
  478     
  479     cert->referenceCount = 1;
  480     
  481     return(cert);
  482     
  483 loser:
  484     if (cert) {
  485         nsslowcert_DestroyCertificate(cert);
  486     }
  487     
  488     return(0);
  489 }

  491 char *
  492 nsslowcert_FixupEmailAddr(char *emailAddr)
  493 {
  494     char *retaddr;
  495     char *str;
  496 
  497     if ( emailAddr == NULL ) {
  498         return(NULL);
  499     }
  500     
  501     /* copy the string */
  502     str = retaddr = PORT_Strdup(emailAddr);
  503     if ( str == NULL ) {
  504         return(NULL);
  505     }
  506     
  507     /* make it lower case */
  508     while ( *str ) {
  509         *str = tolower( *str );
  510         str++;
  511     }
  512     
  513     return(retaddr);
  514 }

  542 NSSLOWKEYPublicKey *
  543 nsslowcert_ExtractPublicKey(NSSLOWCERTCertificate *cert)
  544 {
  545     NSSLOWCERTSubjectPublicKeyInfo spki;
  546     NSSLOWKEYPublicKey *pubk;
  547     SECItem os;
  548     SECStatus rv;
  549     PRArenaPool *arena;
  550     SECOidTag tag;
  551     SECItem newDerSubjKeyInfo;
  552 
  553     arena = PORT_NewArena (DER_DEFAULT_CHUNKSIZE);
  554     if (arena == NULL)
  555         return NULL;
  556 
  557     pubk = (NSSLOWKEYPublicKey *) 
  558                 PORT_ArenaZAlloc(arena, sizeof(NSSLOWKEYPublicKey));
  559     if (pubk == NULL) {
  560         PORT_FreeArena (arena, PR_FALSE);
  561         return NULL;
  562     }
  563 
  564     pubk->arena = arena;
  565     PORT_Memset(&spki,0,sizeof(spki));
  566 
  567     /* copy the DER into the arena, since Quick DER returns data that points
  568        into the DER input, which may get freed by the caller */
  569     rv = SECITEM_CopyItem(arena, &newDerSubjKeyInfo, &cert->derSubjKeyInfo);
  570     if ( rv != SECSuccess ) {
  571         PORT_FreeArena (arena, PR_FALSE);
  572         return NULL;
  573     }
  574 
  575     /* we haven't bothered decoding the spki struct yet, do it now */
  576     rv = SEC_QuickDERDecodeItem(arena, &spki, 
  577                 nsslowcert_SubjectPublicKeyInfoTemplate, &newDerSubjKeyInfo);
  578     if (rv != SECSuccess) {
  579          PORT_FreeArena (arena, PR_FALSE);
  580          return NULL;
  581     }
  582 
  583     /* Convert bit string length from bits to bytes */
  584     os = spki.subjectPublicKey;
  585     DER_ConvertBitString (&os);
  586 
  587     tag = SECOID_GetAlgorithmTag(&spki.algorithm);
  588     switch ( tag ) {
  589       case SEC_OID_X500_RSA_ENCRYPTION:
  590       case SEC_OID_PKCS1_RSA_ENCRYPTION:
  591         pubk->keyType = NSSLOWKEYRSAKey;
  592         prepare_low_rsa_pub_key_for_asn1(pubk);
  593         rv = SEC_QuickDERDecodeItem(arena, pubk, 
  594                                 nsslowcert_RSAPublicKeyTemplate, &os);
  595         if (rv == SECSuccess)
  596             return pubk;
  597         break;
  598       case SEC_OID_ANSIX9_DSA_SIGNATURE:
  599         pubk->keyType = NSSLOWKEYDSAKey;
  600         prepare_low_dsa_pub_key_for_asn1(pubk);
  601         rv = SEC_QuickDERDecodeItem(arena, pubk,
  602                                  nsslowcert_DSAPublicKeyTemplate, &os);
  603         if (rv == SECSuccess) return pubk;
  604         break;
  605       case SEC_OID_X942_DIFFIE_HELMAN_KEY:
  606         pubk->keyType = NSSLOWKEYDHKey;
  607         prepare_low_dh_pub_key_for_asn1(pubk);
  608         rv = SEC_QuickDERDecodeItem(arena, pubk,
  609                                  nsslowcert_DHPublicKeyTemplate, &os);
  610         if (rv == SECSuccess) return pubk;
  611         break;
  612 #ifdef NSS_ENABLE_ECC
  613       case SEC_OID_ANSIX962_EC_PUBLIC_KEY:
  614         pubk->keyType = NSSLOWKEYECKey;
  615         /* Since PKCS#11 directly takes the DER encoding of EC params
  616          * and public value, we don't need any decoding here.
  617          */
  618         rv = SECITEM_CopyItem(arena, &pubk->u.ec.ecParams.DEREncoding, 
  619             &spki.algorithm.parameters);
  620         if ( rv != SECSuccess )
  621             break;        
  622 
  623         /* Fill out the rest of the ecParams structure 
  624          * based on the encoded params
  625          */
  626         if (EC_FillParams(arena, &pubk->u.ec.ecParams.DEREncoding,
  627             &pubk->u.ec.ecParams) != SECSuccess) 
  628             break;
  629
