  149 SECStatus   
  150 RC4_InitContext(RC4Context *cx, const unsigned char *key, unsigned int len,
  151                 const unsigned char * unused1, int unused2, 
  152                 unsigned int unused3, unsigned int unused4)
  153 {
  154         int i;
  155         PRUint8 j, tmp[1];
  156         PRUint8 K[256];
  157         PRUint8 *L;
  158         /* verify the key length. */
  159         PORT_Assert[1](len > 0 && len < ARCFOUR_STATE_SIZE);
  160         if (len < 0 || len >= ARCFOUR_STATE_SIZE) {
  161                 PORT_SetError(SEC_ERROR_INVALID_ARGS);
  162                 return SECFailure;
  163         }
  164         if (cx == NULL) {
  165             PORT_SetError(SEC_ERROR_INVALID_ARGS);
  166             return SECFailure;
  167         }
  168         /* Initialize the state using array indices. */
  169         memcpy(cx->S[1], Kinit, sizeof cx->S[1]);
  170         /* Fill in K repeatedly with values from key. */
  171         L = K;
  172         for (i = sizeof K; i > len; i-= len) {
  173                 memcpy(L, key, len);
  174                 L += len;
  175         }
  176         memcpy(L, key, i);
  177         /* Stir the state of the generator.  At this point it is assumed
  178          * that the key is the size of the state buffer.  If this is not
  179          * the case, the key bytes are repeated to fill the buffer.
  180          */
  181         j = 0;
  182 #define ARCFOUR_STATE_STIR(ii) \
  183         j = j + cx->S[1][ii] + K[ii]; \
  184         S[1]WAP">S[1]WAP(cx->S[1][ii], cx->S[1][j]);
  185         for (i=0; i<ARCFOUR_STATE_SIZE; i++) {
  186                 ARCFOUR_STATE_STIR(i);
  187         }
  188         cx->i = 0;
  189         cx->j = 0;
  190         return SECSuccess;
  191 }

  227         tmpj += tmpSi; \
  228         tmpS[1]j = cx->S[1][tmpj]; \

  235  * Straight ARCFOUR op.  No optimization.

  237 static SECStatus 
  238 rc4_no_opt(RC4Context *cx, unsigned char *output,
  239            unsigned int *outputLen, unsigned int maxOutputLen,
  240            const unsigned char *input, unsigned int inputLen)
  241 {
  242     PRUint8 t;
  243         Stype tmpSi, tmpSj;
  244         register PRUint8 tmpi = cx->i;
  245         register PRUint8 tmpj = cx->j;
  246         unsigned int index;
  247         PORT_Assert[1](maxOutputLen >= inputLen);
  248         if (maxOutputLen < inputLen) {
  249                 PORT_SetError(SEC_ERROR_INVALID_ARGS);
  250                 return SECFailure;
  251         }
  252         for (index=0; index < inputLen; index++) {
  253                 /* Generate next byte from stream. */
  254                 ARCFOUR_NEXT_BYTE();
  255                 /* output = next stream byte XOR next input byte */
  256                 output[index] = cx->S[1][t] ^ input[index];
  257         }
  258         *outputLen = inputLen;
  259         cx->i = tmpi;
  260         cx->j = tmpj;
  261         return SECSuccess;
  262 }

  266 /*
  267  * Byte-at-a-time ARCFOUR, unrolling the loop into 8 pieces.
  268  */
  269 static SECStatus 
  270 rc4_unrolled(RC4Context *cx, unsigned char *output,
  271              unsigned int *outputLen, unsigned int maxOutputLen,
  272              const unsigned char *input, unsigned int inputLen)
  273 {
  274         PRUint8 t;
  275         Stype tmpSi, tmpSj;
  276         register PRUint8 tmpi = cx->i;
  277         register PRUint8 tmpj = cx->j;
  278         int index;
  279         PORT_Assert[1](maxOutputLen >= inputLen);
  280         if (maxOutputLen < inputLen) {
  281                 PORT_SetError(SEC_ERROR_INVALID_ARGS);
  282                 return SECFailure;
  283         }
  284         for (index = inputLen / 8; index-- > 0; input += 8, output += 8) {
  285                 ARCFOUR_NEXT_BYTE();
  286                 output[0] = cx->S[1][t] ^ input[0];
  287                 ARCFOUR_NEXT_BYTE();
  288                 output[1] = cx->S[1][t] ^ input[1];
  289                 ARCFOUR_NEXT_BYTE();
  290                 output[2] = cx->S[1][t] ^ input[2];
  291                 ARCFOUR_NEXT_BYTE();
  292                 output[3] = cx->S[1][t] ^ input[3];
  293                 ARCFOUR_NEXT_BYTE();
  294                 output[4] = cx->S[1][t] ^ input[4];
  295                 ARCFOUR_NEXT_BYTE();
  296                 output[5] = cx->S[1][t] ^ input[5];
  297                 ARCFOUR_NEXT_BYTE();
  298                 output[6] = cx->S[1][t] ^ input[6];
  299                 ARCFOUR_NEXT_BYTE();
  300                 output[7] = cx->S[1][t] ^ input[7];
  301         }
  302         index = inputLen % 8;
  303         if (index) {
  304                 input += index;
  305                 output += index;
  306                 switch (index) {
  307                 case 7:
  308                         ARCFOUR_NEXT_BYTE();
  309                         output[-7] = cx->S[1][t] ^ input[-7]; /* FALLTHRU */
  310                 case 6:
  311                         ARCFOUR_NEXT_BYTE();
  312                         output[-6] = cx->S[1][t] ^ input[-6]; /* FALLTHRU */
  313                 case 5:
  314                         ARCFOUR_NEXT_BYTE();
  315                         output[-5] = cx->S[1][t] ^ input[-5]; /* FALLTHRU */
  316                 case 4:
  317                         ARCFOUR_NEXT_BYTE();
  318                         output[-4] = cx->S[1][t] ^ input[-4]; /* FALLTHRU */
  319                 case 3:
  320                         ARCFOUR_NEXT_BYTE();
  321                         output[-3] = cx->S[1][t] ^ input[-3]; /* FALLTHRU */
  322                 case 2:
  323                         ARCFOUR_NEXT_BYTE();
  324                         output[-2] = cx->S[1][t] ^ input[-2]; /* FALLTHRU */
  325                 case 1:
  326                         ARCFOUR_NEXT_BYTE();
  327                         output[-1] = cx->S[1][t] ^ input[-1]; /* FALLTHRU */
  328                 default:
  329                         /* FALLTHRU */
  330                         ; /* hp-ux build breaks without this */
  331                 }
  332         }
  333         cx->i = tmpi;
  334         cx->j = tmpj;
  335         *outputLen = inputLen;
  336         return SECSuccess;
  337 }

  382 #ifdef CONVERT_TO_WORDS
  383 /* NOTE about UMRs, Uninitialized Memory Reads.
  384  *
  385  * This code reads all input data a WORD at a time, rather than byte at 
  386  * a time, and writes all output data a WORD at a time.  Shifting and 
  387  * masking is used to remove unwanted data and realign bytes when 
  388  * needed.  The first and last words of output are read, modified, and
  389  * written when needed to preserve any unchanged bytes.  This is a huge
  390  * win on machines with high memory latency.  
  391  *
  392  * However, when the input and output buffers do not begin and end on WORD 
  393  * boundaries, and the WORDS in memory that contain the first and last 
  394  * bytes of those buffers contain uninitialized data, then this code will 
  395  * read those uninitialized bytes, causing a UMR error to be reported by 
  396  * some tools.  
  397  *
  398  * These UMRs are NOT a problem, NOT errors, and do NOT need to be "fixed".
  399  * 
  400  * All the words read and written contain at least one byte that is 
  401  * part of the input data or output data.  No words are read or written
  402  * that do not contain data that is part of the buffer.  Therefore, 
  403  * these UMRs cannot cause page faults or other problems unless the 
  404  * buffers have been assigned to improper addresses that would cause
  405  * page faults with or without UMRs.  
  406  */
  407 static SECStatus 
  408 rc4_wordconv(RC4Context *cx, unsigned char *output,
  409              unsigned int *outputLen, unsigned int maxOutputLen,
  410              const unsigned char *input, unsigned int inputLen)
  411 {
  412         ptrdiff_t inOffset = (ptrdiff_t)input % WORDSIZE;
  413         ptrdiff_t outOffset = (ptrdiff_t)output % WORDSIZE;
  414         register WORD streamWord, mask;
  415         register WORD *pInWord, *pOutWord;
  416         register WORD inWord, nextInWord;
  417         PRUint8 t;
  418         register Stype tmpSi, tmpSj;
  419         register PRUint8 tmpi = cx->i;
  420         register PRUint8 tmpj = cx->j;
  421         unsigned int byteCount;
  422         unsigned int bufShift, invBufShift;
  423         int i;
  424 
  425         PORT_Assert[1](maxOutputLen >= inputLen);
  426         if (maxOutputLen < inputLen) {
  427                 PORT_SetError(SEC_ERROR_INVALID_ARGS);
  428                 return SECFailure;
  429         }
  430         if (inputLen < 2*WORDSIZE) {
  431                 /* Ignore word conversion, do byte-at-a-time */
  432                 return rc4_no_opt(cx, output, outputLen, maxOutputLen, input, inputLen);
  433         }
  434         *outputLen = inputLen;
  435         pInWord = (WORD *)(input - inOffset);
  436         if (inOffset < outOffset) {
  437                 bufShift = 8*(outOffset - inOffset);
  438                 invBufShift = 8*WORDSIZE - bufShift;
  439         } else {
  440                 invBufShift = 8*(inOffset - outOffset);
  441                 bufShift = 8*WORDSIZE - invBufShift;
  442         }
  443         /*****************************************************************/
  444         /* Step 1:                                                       */
  445         /* If the first output word is partial, consume the bytes in the */
  446         /* first partial output word by loading one or two words of      */
  447         /* input and shifting them accordingly.  Otherwise, just load    */
  448         /* in the first word of input.  At the end of this block, at     */
  449         /* least one partial word of input should ALWAYS be loaded.      */
  450         /*****************************************************************/
  451         if (outOffset) {
  452                 /* Generate input and stream words aligned relative to the
  453                  * partial output buffer.
  454                  */
  455                 byteCount = WORDSIZE - outOffset; 
  456                 pOutWord = (WORD *)(output - outOffset);
  457                 mask = streamWord = 0;
  458 #ifdef IS_LITTLE_ENDIAN
  459                 for (i = WORDSIZE - byteCount; i < WORDSIZE; i++) {
  460 #else
  461                 for (i = byteCount - 1; i >= 0; --i) {
  462 #endif
  463                         ARCFOUR_NEXT_BYTE();
  464                         streamWord |= (WORD)(cx->S[1][t]) << 8*i;
  465                         mask |= MASK1BYTE << 8*i;
  466                 } /* } */
  467                 inWord = *pInWord++; /* UMR? see comments above. */
  468                 /* If buffers are relatively misaligned, shift the bytes in inWord
  469                  * to be aligned to the output buffer.
  470                  */
  471                 nextInWord = 0;
  472                 if (inOffset < outOffset) {
  473                         /* Have more bytes than needed, shift remainder into nextInWord */
  474                         nextInWord = inWord LSH 8*(inOffset + byteCount);
  475                         inWord = inWord RSH bufShift;
  476                 } else if (inOffset > outOffset) {
  477                         /* Didn't get enough bytes from current input word, load another
  478                          * word and then shift remainder into nextInWord.
  479                          */
  480                         nextInWord = *pInWord++;
  481                         inWord = (inWord LSH invBufShift) | 
  482                                  (nextInWord RSH bufShift);
  483                         nextInWord = nextInWord LSH invBufShift;
  484                 }
  485                 /* Store output of first partial word */
  486                 *pOutWord = (*pOutWord & ~mask) | ((inWord ^ streamWord) & mask);
  487                 /* UMR?  See comments above. */
  488 
  489                 /* Consumed byteCount bytes of input */
  490                 inputLen -= byteCount;
  491                 /* move to next word of output */
  492                 pOutWord++;
  493                 /* inWord has been consumed, but there may be bytes in nextInWord */
  494                 inWord = nextInWord;
  495         } else {
  496                 /* output is word-aligned */
  497                 pOutWord = (WORD *)output;
  498                 if (inOffset) {
  499                         /* Input is not word-aligned.  The first word load of input 
  500                          * will not produce a full word of input bytes, so one word
  501                          * must be pre-loaded.  The main loop below will load in the
  502                          * next input word and shift some of its bytes into inWord
  503                          * in order to create a full input word.  Note that the main
  504                          * loop must execute at least once because the input must
  505                          * be at least two words.
  506                          */
  507                         inWord = *pInWord++; /* UMR? see comments above. */
  508                         inWord = inWord LSH invBufShift;
  509                 } else {
  510                         /* Input is word-aligned.  The first word load of input 
  511                          * will produce a full word of input bytes, so nothing
  512                          * needs to be loaded here.
  513                          */
  514                         inWord = 0;
  515                 }
  516         }
  517         /* Output buffer is aligned, inOffset is now measured relative to
  518          * outOffset (and not a word boundary).
  519          */
  520         inOffset = (inOffset + WORDSIZE - outOffset) % WORDSIZE;
  521         /*****************************************************************/
  522         /* Step 2: main loop                                             */
  523         /* At this point the output buffer is word-aligned.  Any unused  */
  524         /* bytes from above will be in inWord (shifted correctly).  If   */
  525         /* the input buffer is unaligned relative to the output buffer,  */
  526         /* shifting has to be done.                                      */
  527         /*****************************************************************/
  528         if (inOffset) {
  529                 for (; inputLen >= WORDSIZE; inputLen -= WORDSIZE) {
  530                         nextInWord = *pInWord++;
  531                         inWord |= nextInWord RSH bufShift;
  532                         nextInWord = nextInWord LSH invBufShift;
  533                         ARCFOUR_NEXT_WORD();
  534                         *pOutWord++ = inWord ^ streamWord;
  535                         inWord = nextInWord;
  536                 }
  537                 if (inputLen == 0) {
  538                         /* Nothing left to do. */
  539                         cx->i = tmpi;
  540                         cx->j = tmpj;
  541                         return SECSuccess;
  542                 }
  543                 /* If the amount of remaining input is greater than the amount
  544                  * bytes pulled from the current input word, need to do another
  545                  * word load.  What's left in inWord will be consumed in step 3.
  546                  */
  547                 if (inputLen > WORDSIZE - inOffset)
  548                         inWord |= *pInWord RSH bufShift; /* UMR?  See above. */
  549         } else {
  550                 for (; inputLen >= WORDSIZE; inputLen -= WORDSIZE) {
  551                         inWord = *pInWord++;
  552                         ARCFOUR_NEXT_WORD();
  553                         *pOutWord++ = inWord ^ streamWord;
  554                 }
  555                 if (inputLen == 0) {
  556                         /* Nothing left to do. */
  557                         cx->i = tmpi;
  558                         cx->j = tmpj;
  559                         return SECSuccess;
  560                 } else {
  561                         /* A partial input word remains at the tail.  Load it. 
  562                          * The relevant bytes will be consumed in step 3.
  563                          */
  564                         inWord = *pInWord; /* UMR?  See comments above */
  565                 }
  566         }
  567         /*****************************************************************/
  568         /* Step 3:                                                       */
  569         /* A partial word of input remains, and it is already loaded     */
  570         /* into nextInWord.  Shift appropriately and consume the bytes   */
  571         /* used in the partial word.                                     */
  572         /*****************************************************************/
  573         mask = streamWord = 0;
  574 #ifdef IS_LITTLE_ENDIAN
  575         for (i = 0; i < inputLen; ++i) {
  576 #else
  577         for (i = WORDSIZE - 1; i >= WORDSIZE - inputLen; --i) {
  578 #endif
  579                 ARCFOUR_NEXT_BYTE();
  580                 streamWord |= (WORD)(cx->S[1][t]) << 8*i;
  581                 mask |= MASK1BYTE << 8*i;
  582         } /* } */
  583         /* UMR?  See comments above. */
  584         *pOutWord = (*pOutWord & ~mask) | ((inWord ^ streamWord) & mask);
  585         cx->i = tmpi;
  586         cx->j = tmpj;
  587         return SECSuccess;
  588 }
  589 #endif
  590 #endif /* NSS_BEVAND_ARCFOUR */

  592 SECStatus 
  593 RC4_Encrypt(RC4Context *cx, unsigned char *output,
  594             unsigned int *outputLen, unsigned int maxOutputLen,
  595             const unsigned char *input, unsigned int inputLen)
  596 {
  597         PORT_Assert[1](maxOutputLen >= inputLen);
  598         if (maxOutputLen < inputLen) {
  599                 PORT_SetError(SEC_ERROR_INVALID_ARGS);
  600                 return SECFailure;
  601         }
  602 #if defined(NSS_BEVAND_ARCFOUR)
  603         ARCFOUR(cx, inputLen, input, output);
  604         *outputLen = inputLen;
  605         return SECSuccess;
  606 #elif defined( CONVERT_TO_WORDS )
  607         /* Convert the byte-stream to a word-stream */
  608         return rc4_wordconv(cx, output, outputLen, maxOutputLen, input, inputLen);
  609 #else
  610         /* Operate on bytes, but unroll the main loop */
  611         return rc4_unrolled(cx, output, outputLen, maxOutputLen, input, inputLen);
  612 #endif
  613 }
  614 
  615 SECStatus RC4_Decrypt(RC4Context *cx, unsigned char *output,
  616                       unsigned int *outputLen, unsigned int maxOutputLen,
  617                       const unsigned char *input, unsigned int inputLen)
  618 {
  619         PORT_Assert[1](maxOutputLen >= inputLen);
  620         if (maxOutputLen < inputLen) {
  621                 PORT_SetError(SEC_ERROR_INVALID_ARGS);
  622                 return SECFailure;
  623         }
  624         /* decrypt and encrypt are same operation. */
  625 #if defined(NSS_BEVAND_ARCFOUR)
  626         ARCFOUR(cx, inputLen, input, output);
  627         *outputLen = inputLen;
  628         return SECSuccess;
  629 #elif defined( CONVERT_TO_WORDS )
  630         /* Convert the byte-stream to a word-stream */
  631         return rc4_wordconv(cx, output, outputLen, maxOutputLen, input, inputLen);
  632 #else
  633         /* Operate on bytes, but unroll the main loop */
  634         return rc4_unrolled(cx, output, outputLen, maxOutputLen, input, inputLen);
  635 #endif
  636 }
