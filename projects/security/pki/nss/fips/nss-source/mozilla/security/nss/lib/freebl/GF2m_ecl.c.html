   54 mp_err 
   55 GF2m_ec_pt_is_inf_aff(const mp_int *px, const mp_int *py)

   57 
   58     if ((mp_cmp_z(px) == 0) && (mp_cmp_z(py) == 0)) {
   59         return MP_YES;
   60     } else {
   61         return MP_NO;

   75 /* Computes R = P + Q based on IEEE P1363 A.10.2.
   76  * Elliptic curve points P, Q, and R can all be identical.
   77  * Uses affine coordinates.
   78  */
   79 mp_err 
   80 GF2m_ec_pt_add_aff(const mp_int *pp, const mp_int *a, const mp_int *px, 
   81     const mp_int *py, const mp_int *qx, const mp_int *qy, 
   82     mp_int *rx, mp_int *ry)
   83 {
   84     mp_err err = MP_OKAY;
   85     mp_int lambda, xtemp, ytemp;
   86     unsigned int *p;
   87     int p_size;
   88 
   89     p_size = mp_bpoly2arr(pp, p, 0) + 1;
   90     p = (unsigned int *) (malloc(sizeof(unsigned int) * p_size));
   91     if (p == NULL) goto cleanup;
   92     mp_bpoly2arr(pp, p, p_size);
   93     
   94     CHECK_MPI_OK( mp_init(&lambda) );
   95     CHECK_MPI_OK( mp_init(&xtemp) );
   96     CHECK_MPI_OK( mp_init(&ytemp) );
   97     /* if P = inf, then R = Q */
   98     if (GF2m_ec_pt_is_inf_aff(px, py) == 0) {
   99         CHECK_MPI_OK( mp_copy(qx, rx) );
  100         CHECK_MPI_OK( mp_copy(qy, ry) );
  101         err = MP_OKAY;
  102         goto cleanup;
  103     }
  104     /* if Q = inf, then R = P */
  105     if (GF2m_ec_pt_is_inf_aff(qx, qy) == 0) {
  106         CHECK_MPI_OK( mp_copy(px, rx) );
  107         CHECK_MPI_OK( mp_copy(py, ry) );
  108         err = MP_OKAY;
  109         goto cleanup;
  110     }
  111     /* if px != qx, then lambda = (py+qy) / (px+qx), 
  112      *                   xtemp = a + lambda^2 + lambda + px + qx
  113      */
  114     if (mp_cmp(px, qx) != 0) {
  115         CHECK_MPI_OK( mp_badd(py, qy, &ytemp) );
  116         CHECK_MPI_OK( mp_badd(px, qx, &xtemp) );
  117         CHECK_MPI_OK( mp_bdivmod(&ytemp, &xtemp, pp, p, &lambda) );
  118         CHECK_MPI_OK( mp_bsqrmod(&lambda, p, &xtemp) );
  119         CHECK_MPI_OK( mp_badd(&xtemp, &lambda, &xtemp) );
  120         CHECK_MPI_OK( mp_badd(&xtemp, a, &xtemp) );
  121         CHECK_MPI_OK( mp_badd(&xtemp, px, &xtemp) );
  122         CHECK_MPI_OK( mp_badd(&xtemp, qx, &xtemp) );
  123     } else {
  124         /* if py != qy or qx = 0, then R = inf */
  125         if (((mp_cmp(py, qy) != 0)) || (mp_cmp_z">mp_cmp_z(qx) == 0)) {
  126             mp_zero(rx);
  127             mp_zero(ry);
  128             err = MP_OKAY;
  129             goto cleanup;
  130         }
  131         /* lambda = qx + qy / qx  */
  132         CHECK_MPI_OK( mp_bdivmod(qy, qx, pp, p, &lambda) );
  133         CHECK_MPI_OK( mp_badd(&lambda, qx, &lambda) );
  134         /* xtemp = a + lambda^2 + lambda */
  135         CHECK_MPI_OK( mp_bsqrmod(&lambda, p, &xtemp) );
  136         CHECK_MPI_OK( mp_badd(&xtemp, &lambda, &xtemp) );
  137         CHECK_MPI_OK( mp_badd(&xtemp, a, &xtemp) );
  138     }
  139     /* ry = (qx + xtemp) * lambda + xtemp + qy */
  140     CHECK_MPI_OK( mp_badd(qx, &xtemp, &ytemp) );
  141     CHECK_MPI_OK( mp_bmulmod(&ytemp, &lambda, p, &ytemp) );
  142     CHECK_MPI_OK( mp_badd(&ytemp, &xtemp, &ytemp) );
  143     CHECK_MPI_OK( mp_badd(&ytemp, qy, ry) );
  144     /* rx = xtemp */
  145     CHECK_MPI_OK( mp_copy(&xtemp, rx) );
  146 
  147 cleanup:
  148     mp_clear(&lambda);
  149     mp_clear(&xtemp);
  150     mp_clear(&ytemp);
  151     free(p);
  152     return err;
  153 }

  297 /* Compute the x-coordinate x/z for the point 2*(x/z) in Montgomery projective 
  298  * coordinates.
  299  * Uses algorithm Mdouble in appendix of 
  300  *     Lopez, J. and Dahab, R.  "Fast multiplication on elliptic curves over 
  301  *     GF(2^m) without precomputation".
  302  * modified to not require precomputation of c=b^{2^{m-1}}.
  303  */
  304 static mp_err 
  305 gf2m_Mdouble(const mp_int *pp, const unsigned int p[], const mp_int *a, 
  306     const mp_int *b, mp_int *x, mp_int *z) 
  307 {
  308     mp_err err = MP_OKAY;
  309     mp_int t1;
  310 
  311     MP_DIGITS(&t1) = 0;
  312     CHECK_MPI_OK( mp_init(&t1) );
  313     
  314     CHECK_MPI_OK( mp_bsqrmod(x, p, x) );
  315     CHECK_MPI_OK( mp_bsqrmod(z, p, &t1) );
  316     CHECK_MPI_OK( mp_bmulmod(x, &t1, p, z) );
  317     CHECK_MPI_OK( mp_bsqrmod(x, p, x) );
  318     CHECK_MPI_OK( mp_bsqrmod(&t1, p, &t1) );
  319     CHECK_MPI_OK( mp_bmulmod(b, &t1, p, &t1) );
  320     CHECK_MPI_OK( mp_badd(x, &t1, x) );
  321 
  322 cleanup:
  323     mp_clear(&t1);
  324     return err;
  325 }

  327 /* Compute the x-coordinate x1/z1 for the point (x1/z1)+(x2/x2) in Montgomery

  329  * Uses algorithm Madd in appendix of 
  330  *     Lopex, J. and Dahab, R.  "Fast multiplication on elliptic curves over 
  331  *     GF(2^m) without precomputation".

  333 static mp_err 
  334 gf2m_Madd(const mp_int *pp, const unsigned int p[], const mp_int *a, 
  335     const mp_int *b, const mp_int *x, mp_int *x1, mp_int *z1, mp_int *x2,
  336     mp_int *z2)

  342     MP_DIGITS(&t2) = 0;
  343     CHECK_MPI_OK( mp_init(&t1) );
  344     CHECK_MPI_OK( mp_init(&t2) );
  345     
  346     CHECK_MPI_OK( mp_copy(x, &t1) );
  347     CHECK_MPI_OK( mp_bmulmod(x1, z2, p, x1) );
  348     CHECK_MPI_OK( mp_bmulmod(z1, x2, p, z1) );

  354     
  355 cleanup:
  356     mp_clear(&t1);
  357     mp_clear(&t2);

  361 /* Compute the x, y affine coordinates from the point (x1, z1) (x2, z2) 
  362  * using Montgomery point multiplication algorithm Mxy() in appendix of 
  363  *     Lopex, J. and Dahab, R.  "Fast multiplication on elliptic curves over 
  364  *     GF(2^m) without precomputation".
  365  * Returns:
  366  *     0 on error
  367  *     1 if return value should be the point at infinity
  368  *     2 otherwise
  369  */
  370 static int
  371 gf2m_Mxy(const mp_int *pp, const unsigned int p[], const mp_int *a, 
  372     const mp_int *b, const mp_int *x, const mp_int *y, mp_int *x1, mp_int *z1,

  378

  380     MP_DIGITS(&t4) = 0;
  381     MP_DIGITS(&t5) = 0;
  382     CHECK_MPI_OK( mp_init(&t3) );
  383     CHECK_MPI_OK( mp_init(&t4) );
  384     CHECK_MPI_OK( mp_init(&t5) );
  385

  390         goto cleanup;
  391     }
  392

  394         CHECK_MPI_OK( mp_copy(x, x2) );
  395         CHECK_MPI_OK( mp_badd(x, y, z2) );

  398     }

  407     CHECK_MPI_OK( mp_bmulmod(z2, x1, p, x1) );
  408     CHECK_MPI_OK( mp_badd(z2, x2, z2) );
  409 
  410     CHECK_MPI_OK( mp_bmulmod(z2, z1, p, z2) );
  411     CHECK_MPI_OK( mp_bsqrmod(x, p, &t4) );
  412     CHECK_MPI_OK( mp_badd(&t4, y, &t4) );
  413     CHECK_MPI_OK( mp_bmulmod(&t4, &t3, p, &t4) );
  414     CHECK_MPI_OK( mp_badd(&t4, z2, &t4) );
  415 
  416     CHECK_MPI_OK( mp_bmulmod(&t3, x, p, &t3) );
  417     CHECK_MPI_OK( mp_bdivmod(&t5, &t3, pp, p, &t3) );
  418     CHECK_MPI_OK( mp_bmulmod(&t3, &t4, p, &t4) );

  420     CHECK_MPI_OK( mp_badd(x2, x, z2) );
  421 
  422     CHECK_MPI_OK( mp_bmulmod(z2, &t4, p, z2) );
  423     CHECK_MPI_OK( mp_badd(z2, y, z2) );
  424 
  425     ret = 2;
  426     
  427 cleanup:
  428     mp_clear(&t3);
  429     mp_clear(&t4);
  430     mp_clear(&t5);

  432         return ret;
  433     } else {

  435     }
  436 }

  438 /* Computes R = nP based on algorithm 2P of
  439  *     Lopex, J. and Dahab, R.  "Fast multiplication on elliptic curves over 
  440  *     GF(2^m) without precomputation".
  441  * Elliptic curve points P and R can be identical.
  442  * Uses Montgomery projective coordinates.

  445 GF2m_ec_pt_mul_mont(const mp_int *pp, const mp_int *a, const mp_int *b,

  447     mp_int *rx, mp_int *ry)

  450     mp_int x1, x2, z1, z2;
  451     int i, j;
  452     mp_digit top_bit, mask;

  454     int p_size;
  455 
  456     MP_DIGITS(&x1) = 0;

  458     MP_DIGITS(&z1) = 0;
  459     MP_DIGITS(&z2) = 0;
  460     CHECK_MPI_OK( mp_init(&x1) );
  461     CHECK_MPI_OK( mp_init(&x2) );
  462     CHECK_MPI_OK( mp_init(&z1) );
  463     CHECK_MPI_OK( mp_init(&z2) );
  464 
  465     p_size = mp_bpoly2arr(pp, p, 0) + 1;

  483     CHECK_MPI_OK( mp_badd(&x2, b, &x2) );       /* x2 = px^4 + b */
  484     
  485     /* find top-most bit and go one past it */

  487     j = MP_DIGIT_BIT - 1;
  488     top_bit = 1;

  490     mask = top_bit;
  491     while (!(i/mpi.h.dep.html#MP_DIGITS">MP_DIGITS(n)[i] & mask)) {

  493         j--;
  494     }

  496     
  497     /* if top most bit was at word break, go to next word */
  498     if (!mask) {

  500         j = MP_DIGIT_BIT - 1;
  501         mask = top_bit;

  512             }
  513             mask >>= 1;
  514         }
  515         j = MP_DIGIT_BIT - 1;
  516         mask = top_bit;
  517     }
  518 
  519     /* convert out of "projective" coordinates */
  520     i = gf2m_Mxy[1](pp, p, a, b, px, py, &x1, &z1, &x2, &z2);
  521     if (i == 0) {
  522         err = MP_BADARG;
  523         goto cleanup;
  524     } else if (i == 1) {

  526     } else {
  527         CHECK_MPI_OK( mp_copy(&x2, rx) );
  528         CHECK_MPI_OK( mp_copy(&z2, ry) );
  529     }
