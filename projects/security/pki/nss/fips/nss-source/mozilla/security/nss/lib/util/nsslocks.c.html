   50 /* Given the address of a (global) pointer to a PZLock, 
   51  * atomicly create the lock and initialize the (global) pointer, 
   52  * if it is not already created/initialized.
   53  */
   54 
   55 SECStatus 
   56 __nss_InitLock(   PZLock    **ppLock, nssILockType ltype )
   57 {
   58     static PRInt32  initializers;
   59 
   60     PORT_Assert[1]( ppLock != NULL);
   61 
   62     /* atomically initialize the lock */
   63     while (!*ppLock) {
   64         PRInt32 myAttempt = PR_AtomicIncrement(&initializers);
   65         if (myAttempt == 1) {
   66             if (!*ppLock) {
   67                 *ppLock = PZ_NewLock(ltype);
   68             }
   69             (void) PR_AtomicDecrement(&initializers);
   70             break;
   71         }
   72         PR_Sleep(PR_INTERVAL_NO_WAIT);          /* PR_Yield() */
   73         (void) PR_AtomicDecrement(&initializers);
   74     }
   75 
   76     return (*ppLock != NULL) ? SECSuccess : SECFailure;
   77 }
   78 
   79 SECStatus 
   80 nss_InitLock(   PZLock    **ppLock, nssILockType ltype )
   81 {
   82     return __nss_InitLock(ppLock, ltype);
   83 }
