   54 /*
   55  * NSS_CMSEnvelopedData_Create - create an enveloped data message
   56  */
   57 NSSCMSEnvelopedData *
   58 NSS_CMSEnvelopedData_Create(NSSCMSMessage *cmsg, SECOidTag algorithm, int keysize)
   59 {
   60     void *mark;
   61     NSSCMSEnvelopedData *envd;
   62     PLArenaPool[1] *poolp;
   63     SECStatus rv[1];
   64 
   65     poolp = cmsg->poolp;
   66 
   67     mark = PORT_ArenaMark(poolp);
   68 
   69     envd = (NSSCMSEnvelopedData *)PORT_ArenaZAlloc(poolp, sizeof(NSSCMSEnvelopedData));
   70     if (envd == NULL)
   71         goto loser;
   72 
   73     envd->cmsg = cmsg;
   74 
   75     /* version is set in NSS_CMSEnvelopedData_Encode_BeforeStart() */
   76 
   77     rv[1] = NSS_CMSContentInfo_SetContentEncAlg(poolp, &(envd->contentInfo), algorithm, NULL, keysize);
   78     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
   79         goto loser;
   80 
   81     PORT_ArenaUnmark(poolp, mark);
   82     return envd;
   83 
   84 loser:
   85     PORT_ArenaRelease(poolp, mark);
   86     return NULL;
   87 }

   89 /*
   90  * NSS_CMSEnvelopedData_Destroy - destroy an enveloped data message
   91  */
   92 void
   93 NSS_CMSEnvelopedData_Destroy(NSSCMSEnvelopedData *edp)
   94 {
   95     NSSCMSRecipientInfo **recipientinfos;
   96     NSSCMSRecipientInfo *ri;
   97 
   98     if (edp == NULL)
   99         return;
  100 
  101     recipientinfos = edp->recipientInfos;
  102     if (recipientinfos == NULL)
  103         return;
  104 
  105     while ((ri = *recipientinfos++) != NULL)
  106         NSS_CMSRecipientInfo_Destroy(ri);
  107 
  108    NSS_CMSContentInfo_Destroy(&(edp->contentInfo));
  109 
  110 }

  149 /*
  150  * NSS_CMSEnvelopedData_Encode_BeforeStart - prepare this envelopedData for encoding
  151  *
  152  * at this point, we need
  153  * - recipientinfos set up with recipient's certificates
  154  * - a content encryption algorithm (if none, 3DES will be used)
  155  *
  156  * this function will generate a random content encryption key (aka bulk key),
  157  * initialize the recipientinfos with certificate identification and wrap the bulk key
  158  * using the proper algorithm for every certificiate.
  159  * it will finally set the bulk algorithm and key so that the encode step can find it.
  160  */
  161 SECStatus
  162 NSS_CMSEnvelopedData_Encode_BeforeStart(NSSCMSEnvelopedData *envd)
  163 {
  164     int version[2][3];
  165     NSSCMSRecipientInfo **recipientinfos;
  166     NSSCMSContentInfo *cinfo;
  167     PK11SymKey *bulkkey = NULL;
  168     SECOidTag bulkalgtag;
  169     CK_MECHANISM_TYPE type;
  170     PK11SlotInfo *slot;
  171     SECStatus rv[1];
  172     SECItem *dummy[1];
  173     PLArenaPool[1] *poolp;
  174     extern const SEC_ASN1Template NSSCMSRecipientInfoTemplate[];
  175     void *mark = NULL;
  176     i[2][3][4][10]nt i[2][3][4][10];
  177 
  178     poolp = envd->cmsg->poolp;
  179     cinfo = &(envd->contentInfo);
  180 
  181     recipientinfos = envd->recipientInfos;
  182     if (recipientinfos == NULL) {
  183         PORT_SetError(SEC_ERROR_BAD_DATA);
  184 #if 0
  185         PORT_SetErrorString("Cannot find recipientinfos to encode.");
  186 #endif
  187         goto loser;
  188     }
  189 
  190     version[2][3] = NSS_CMS_ENVELOPED_DATA_VERSION_REG;
  191     if (envd->originatorInfo != NULL || envd->unprotectedAttr != NULL) {
  192         version[2][3] = NSS_CMS_ENVELOPED_DATA_VERSION_ADV;
  193     } else {
  194         for (i[2][3][4][10] = 0; reci[2][3][4][10]pi[2][3][4][10]enti[2][3][4][10]nfos[i[2][3][4][10]] != NULL; i[2][3][4][10]++) {
  195             i[2][3][4][10]f (NSS_CMSReci[2][3][4][10]pi[2][3][4][10]entInfo_GetVersi[2][3][4][10]on(reci[2][3][4][10]pi[2][3][4][10]enti[2][3][4][10]nfos[i[2][3][4][10]]) != 0) {
  196                 version[2][3] = NSS_CMS_ENVELOPED_DATA_VERSION_ADV;
  197                 break;
  198             }
  199         }
  200     }
  201     dummy[1] = SEC_ASN1EncodeInteger(poolp, &(envd->version[2][3]), version[2][3]);
  202     if (dummy[1] == NULL)
  203         goto loser;
  204 
  205     /* now we need to have a proper content encryption algorithm
  206      * on the SMIME level, we would figure one out by looking at SMIME capabilities
  207      * we cannot do that on our level, so if none is set already, we'll just go
  208      * with one of the mandatory algorithms (3DES) */
  209     if ((bulkalgtag = NSS_CMSContentInfo_GetContentEncAlgTag(cinfo)) == SEC_OID_UNKNOWN) {
  210         rv[1] = NSS_CMSContentInfo_SetContentEncAlg(poolp, cinfo, SEC_OID_DES_EDE3_CBC, NULL, 168);
  211         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  212             goto loser;
  213         bulkalgtag = SEC_OID_DES_EDE3_CBC;
  214     } 
  215 
  216     /* generate a random bulk key suitable for content encryption alg */
  217     type = PK11_AlgtagToMechanism(bulkalgtag);
  218     slot = slot.c.dep.html#PK11_GetBestSlot">PK11_GetBestSlot(type, envd->cmsg->pwfn_arg);
  219     if (slot == NULL)
  220         goto loser;        /* error has been set by PK11_GetBestSlot */
  221 
  222     /* this is expensive... */
  223     bulkkey = PK11_KeyGen(slot, type, NULL, NSS_CMSContentInfo_GetBulkKeySize(cinfo) / 8, envd->cmsg->pwfn_arg);
  224     PK11_FreeSlot(slot);
  225     if (bulkkey == NULL)
  226         goto loser;        /* error has been set by PK11_KeyGen */
  227 
  228     mark = PORT_ArenaMark(poolp);
  229 
  230     /* Encrypt the bulk key with the public key of each recipient.  */
  231     for (i[2][3][4][10] = 0; reci[2][3][4][10]pi[2][3][4][10]enti[2][3][4][10]nfos[i[2][3][4][10]] != NULL; i[2][3][4][10]++) {
  232         rv[1] = NSS_CMSReci[2][3][4][10]pi[2][3][4][10]entInfo_WrapBulkKey(reci[2][3][4][10]pi[2][3][4][10]enti[2][3][4][10]nfos[i[2][3][4][10]], bulkkey, bulkalgtag);
  233         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  234             goto loser;        /* error has been set by NSS_CMSRecipientInfo_EncryptBulkKey */
  235                             /* could be: alg not supported etc. */
  236     }
  237 
  238     /* the recipientinfos are all finished. now sort them by DER for SET OF encoding */
  239     rv[1] = NSS_CMSArray_SortByDER((void **)envd->recipientInfos, NSSCMSRecipientInfoTemplate, NULL);
  240     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  241         goto loser;        /* error has been set by NSS_CMSArray_SortByDER */
  242 
  243     /* store the bulk key in the contentInfo so that the encoder can find it */
  244     NSS_CMSContentInfo_SetBulkKey(cinfo, bulkkey);
  245 
  246     PORT_ArenaUnmark(poolp, mark);
  247 
  248     PK11_FreeSymKey(bulkkey);
  249 
  250     return SECSuccess;
  251 
  252 loser:
  253     if (mark != NULL)
  254         PORT_ArenaRelease (poolp, mark);
  255     if (bulkkey)
  256         PK11_FreeSymKey(bulkkey);
  257 
  258     return SECFailure;
  259 }

  261 /*
  262  * NSS_CMSEnvelopedData_Encode_BeforeData - set up encryption
  263  *
  264  * it is essential that this is called before the contentEncAlg is encoded, because
  265  * setting up the encryption may generate IVs and thus change it!
  266  */
  267 SECStatus
  268 NSS_CMSEnvelopedData_Encode_BeforeData(NSSCMSEnvelopedData *envd)
  269 {
  270     NSSCMSContentInfo *cinfo;
  271     PK11SymKey *bulkkey;
  272     SECAlgorithmID *algid;
  273 
  274     cinfo = &(envd->contentInfo);
  275 
  276     /* find bulkkey and algorithm - must have been set by NSS_CMSEnvelopedData_Encode_BeforeStart */
  277     bulkkey = NSS_CMSContentInfo_GetBulkKey(cinfo);
  278     if (bulkkey == NULL)
  279         return SECFailure;
  280     algid = NSS_CMSContentInfo_GetContentEncAlg(cinfo);
  281     if (algid == NULL)
  282         return SECFailure;
  283 
  284     /* this may modify algid (with IVs generated in a token).
  285      * it is essential that algid is a pointer to the contentEncAlg data, not a
  286      * pointer to a copy! */
  287     cinfo->ciphcx = NSS_CMSCipherContext_StartEncrypt(envd->cmsg->poolp, bulkkey, algid);
  288     PK11_FreeSymKey(bulkkey);
  289     if (cinfo->ciphcx == NULL)
  290         return SECFailure;
  291 
  292     return SECSuccess;
  293 }

  299 NSS_CMSEnvelopedData_Encode_AfterData(NSSCMSEnvelopedData *envd)

  310 /*
  311  * NSS_CMSEnvelopedData_Decode_BeforeData - find our recipientinfo, 
  312  * derive bulk key & set up our contentinfo
  313  */
  314 SECStatus
  315 NSS_CMSEnvelopedData_Decode_BeforeData(NSSCMSEnvelopedData *envd)
  316 {
  317     NSSCMSRecipientInfo *ri;
  318     PK11SymKey *bulkkey = NULL;
  319     SECOidTag bulkalgtag;
  320     SECAlgorithmID *bulkalg;
  321     SECStatus rv[1] = SECFailure;
  322     NSSCMSContentInfo *cinfo;
  323     NSSCMSRecipient **recipient_list = NULL;
  324     NSSCMSRecipient *recipient;
  325     int rlIndex;
  326 
  327     if (NSS_CMSArray_Count((void **)envd->recipientInfos) == 0) {
  328         PORT_SetError(SEC_ERROR_BAD_DATA);
  329 #if 0
  330         PORT_SetErrorStrin[1]g("No recipient data[2] in[1] envelope.");
  331 #endif
  332         goto loser;
  333     }
  334 
  335     /* look if one of OUR cert's issuerSN is on the list of recipients, and if so,  */
  336     /* get the cert and private key for it right away */
  337     recipient_list = nss_cms_recipient_list_create(envd->recipientInfos);
  338     if (recipient_list == NULL)
  339         goto loser;
  340 
  341     /* what about multiple recipientInfos that match?
  342      * especially if, for some reason, we could not produce a bulk key with the first match?!
  343      * we could loop & feed partial recipient_list to PK11_FindCertAndKeyByRecipientList...
  344      * maybe later... */
  345     rlIndex = PK11_FindCertAndKeyByRecipientListNew(recipient_list, envd->cmsg->pwfn_arg);
  346 
  347     /* if that fails, then we're not an intended recipient and cannot decrypt */
  348     if (rlIndex < 0) {
  349         PORT_SetError(SEC_ERROR_NOT_A_RECIPIENT);
  350 #if 0
  351         PORT_SetErrorString("Cannot decrypt data[2] because proper key[1] cannot be found.");
  352 #endif
  353         goto loser;
  354     }
  355 
  356     recipient = recipient_list[rlIndex];
  357     if (!recipient->cert || !recipient->privkey) {
  358         /* XXX should set an error code ?!? */
  359         goto loser;
  360     }
  361     /* get a pointer to "our" recipientinfo */
  362     ri = envd->recipientInfos[recipient->riIndex];
  363 
  364     cinfo = &(envd->contentInfo);
  365     bulkalgtag = NSS_CMSContentInfo_GetContentEncAlgTag(cinfo);
  366     if (bulkalgtag == SEC_OID_UNKNOWN) {
  367         PORT_SetError(SEC_ERROR_INVALID_ALGORITHM);
  368     } else 
  369         bulkkey = 
  370             NSS_CMSRecipientInfo_UnwrapBulkKey(ri,recipient->subIndex,
  371                                                     recipient->cert,
  372                                                     recipient->privkey,
  373                                                     bulkalgtag);
  374     if (bulkkey == NULL) {
  375         /* no success finding a bulk key */
  376         goto loser;
  377     }
  378 
  379     NSS_CMSContentInfo_SetBulkKey(cinfo, bulkkey);
  380 
  381     bulkalg = NSS_CMSContentInfo_GetContentEncAlg(cinfo);
  382 
  383     cinfo->ciphcx = NSS_CMSCipherContext_StartDecrypt(bulkkey, bulkalg);
  384     if (cinfo->ciphcx == NULL)
  385         goto loser;                /* error has been set by NSS_CMSCipherContext_StartDecrypt */
  386 
  387 
  388     rv[1] = rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess;
  389 
  390 loser:
  391     if (bulkkey)
  392         PK11_FreeSymKey(bulkkey);
  393     if (recipient_list != NULL)
  394         nss_cms_recipient_list_destroy(recipient_list);
  395     return rv[1];
  396 }
