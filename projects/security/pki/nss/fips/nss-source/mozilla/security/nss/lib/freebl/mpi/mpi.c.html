  161 /* {{{ mp_init_copy(mp, from) */
  162 
  163 /*
  164   mp_init_copy(mp, from)
  165 
  166   Initialize mp as an exact copy of from.  Returns MP_OKAY if
  167   successful, MP_MEM if memory could not be allocated for the new
  168   structure.
  169  */
  170 
  171 mp_err mp_init_copy(mp_int *mp, const mp_int *from)
  172 {
  173   ARGCHK(mp != NULL && from != NULL, MP_BADARG);
  174 
  175   if(mp == from)
  176     return MP_OKAY;
  177 
  178   if((DIGITS(mp) = s_mp_alloc[1](ALLOC(from), sizeof(mp_digit))) == NULL)
  179     return MP_MEM;
  180 
  181   s_mp_copy[1](DIGITS(from), DIGITS(mp), USED(from));
  182   USED(mp) = USED(from);
  183   ALLOC(mp) = ALLOC(from);
  184   SIGN(mp) = SIGN(from);
  185 
  186   return MP_OKAY;
  187 
  188 } /* end mp_init_copy() */

  192 /* {{{ mp_copy(from, to) */
  193 
  194 /*
  195   mp_copy(from, to)
  196 
  197   Copies the mp_int 'from' to the mp_int 'to'.  It is presumed that
  198   'to' has already been initialized (if not, use mp_init_copy()
  199   instead). If 'from' and 'to' are identical, nothing happens.
  200  */
  201 
  202 mp_err mp_copy(const mp_int *from, mp_int *to)
  203 {
  204   ARGCHK(from != NULL && to != NULL, MP_BADARG);
  205 
  206   if(from == to)
  207     return MP_OKAY;
  208 
  209   ++mp_copies[1];
  210   { /* copy */
  211     mp_digit   *tmp[1];
  212 
  213     /*
  214       If the allocated buffer in 'to' already has enough space to hold
  215       all the used digits of 'from', we'll re-use it to avoid hitting
  216       the memory allocater more than necessary; otherwise, we'd have
  217       to grow anyway, so we just allocate a hunk and make the copy as
  218       usual
  219      */
  220     if(ALLOC(to) >= USED(from)) {
  221       s_mp_setz[1](DIGITS(to) + USED(from), ALLOC(to) - USED(from));
  222       s_mp_copy[1](DIGITS(from), DIGITS(to), USED(from));
  223       
  224     } else {
  225       if((tmp[1] = s_mp_alloc[1](ALLOC(from), sizeof(mp_digit))) == NULL)
  226         return MP_MEM;
  227 
  228       s_mp_copy[1](DIGITS(from), tmp[1], USED(from));
  229 
  230       if(DIGITS(to) != NULL) {
  231 #if MP_CRYPTO
  232         s_mp_setz[1](DIGITS(to), ALLOC(to));
  233 #endif
  234         s_mp_free[1](DIGITS(to));
  235       }
  236 
  237       DIGITS(to) = tmp[1];
  238       ALLOC(to) = ALLOC(from);
  239     }
  240 
  241     /* Copy the precision and sign from the original */
  242     USED(to) = USED(from);
  243     SIGN(to) = SIGN(from);
  244   } /* end copy */
  245 
  246   return MP_OKAY;
  247 
  248 } /* end mp_copy() */

  342 /* {{{ mp_set_int(mp, z) */
  343 
  344 mp_err mp_set_int(mp_int *mp, long z)
  345 {
  346   int            ix;
  347   unsigned long  v = labs(z);
  348   mp_err         res;
  349 
  350   ARGCHK(mp != NULL, MP_BADARG);
  351 
  352   mp_zero(mp);
  353   if(z == 0)
  354     return MP_OKAY;  /* shortcut for zero */
  355 
  356   if (sizeof v <= sizeof(mp_digit)) {
  357     DIGIT(mp,0) = v;
  358   } else {
  359     for (ix = sizeof(long) - 1; ix >= 0; ix--) {
  360       if ((res = s_mp_mul_d(mp, (UCHAR_MAX + 1))) != MP_OKAY)
  361         return res;
  362 
  363       res = s_mp_add_d(mp, (mp_digit)((v >> (ix * CHAR_BIT)) & UCHAR_MAX));
  364       if (res != MP_OKAY)
  365         return res;
  366     }
  367   }
  368   if(z < 0)
  369     SIGN(mp) = NEG;
  370 
  371   return MP_OKAY;
  372 
  373 } /* end mp_set_int() */

  410 /* {{{ mp_add_d(a, d, b) */
  411 
  412 /*
  413   mp_add_d(a, d, b)
  414 
  415   Compute the sum b = a + d, for a single digit d.  Respects the sign of
  416   its primary addend (single digits are unsigned anyway).
  417  */
  418 
  419 mp_err mp_add_d(const mp_int *a, mp_digit d, mp_int *b)
  420 {
  421   mp_int   tmp[1];
  422   mp_err   res;
  423 
  424   ARGCHK(a != NULL && b != NULL, MP_BADARG);
  425 
  426   if((res = mp_init_copy(&tmp[1], a)) != MP_OKAY)
  427     return res;
  428 
  429   if(SIGN(&tmp[1]) == ZPOS) {
  430     if((res = s_mp_add_d(&tmp[1], d)) != MP_OKAY)
  431       goto CLEANUP;
  432   } else if(s_mp_cmp_d(&tmp[1], d) >= 0) {
  433     if((res = s_mp_sub_d(&tmp[1], d)) != MP_OKAY)
  434       goto CLEANUP;
  435   } else {
  436     mp_neg(&tmp[1], &tmp[1]);
  437 
  438     DIGIT(&tmp[1], 0) = d - DIGIT(&tmp[1], 0);
  439   }
  440 
  441   if(s_mp_cmp_d(&tmp[1], 0) == 0)
  442     SIGN(&tmp[1]) = ZPOS;
  443 
  444   s_mp_exch(&tmp[1], b);
  445 
  446 CLEANUP:
  447   mp_clear(&tmp[1]);
  448   return res;
  449 
  450 } /* end mp_add_d() */
  451 
  452 /* }}} */

  547 /* {{{ mp_div_d(a, d, q, r) */
  548 
  549 /*
  550   mp_div_d(a, d, q, r)
  551 
  552   Compute the quotient q = a / d and remainder r = a mod d, for a
  553   single digit d.  Respects the sign of its divisor (single digits are
  554   unsigned anyway).
  555  */
  556 
  557 mp_err mp_div_d(const mp_int *a, mp_digit d, mp_int *q, mp_digit *r)
  558 {
  559   mp_err   res;
  560   mp_int   qp;
  561   mp_digit rem;
  562   int      pow;
  563 
  564   ARGCHK(a != NULL, MP_BADARG);
  565 
  566   if(d == 0)
  567     return MP_RANGE;
  568 
  569   /* Shortcut for powers of two ... */
  570   if((pow = s_mp_ispow2d(d)) >= 0) {
  571     mp_digit  mask;
  572 
  573     mask = ((mp_digit)1 << pow) - 1;
  574     rem = DIGIT(a, 0) & mask;
  575 
  576     if(q) {
  577       mp_copy(a, q);
  578       s_mp_div_2d(q, pow);
  579     }
  580 
  581     if(r)
  582       *r = rem;
  583 
  584     return MP_OKAY;
  585   }
  586 
  587   if((res = mp_init_copy(&qp, a)) != MP_OKAY)
  588     return res;
  589 
  590   res = s_mp_div_d(&qp, d, &rem);
  591 
  592   if(s_mp_cmp_d(&qp, 0) == 0)
  593     SIGN(q) = ZPOS;
  594 
  595   if(r)
  596     *r = rem;
  597 
  598   if(q)
  599     s_mp_exch(&qp, q);
  600 
  601   mp_clear(&qp);
  602   return res;
  603 
  604 } /* end mp_div_d() */

  633 /* {{{ mp_expt_d(a, d, b) */
  634 
  635 mp_err mp_expt_d(const mp_int *a, mp_digit d, mp_int *c)
  636 {
  637   mp_int   s, x;
  638   mp_err   res;
  639 
  640   ARGCHK(a != NULL && c != NULL, MP_BADARG);
  641 
  642   if((res = mp_init(&s)) != MP_OKAY)
  643     return res;
  644   if((res = mp_init_copy(&x, a)) != MP_OKAY)
  645     goto X;
  646 
  647   DIGIT(&s, 0) = 1;
  648 
  649   while(d != 0) {
  650     if(d & 1) {
  651       if((res = s_mp_mul(&s, &x)) != MP_OKAY)
  652         goto CLEANUP;
  653     }
  654 
  655     d /= 2;
  656 
  657     if((res = s_mp_sqr[1](&x)) != MP_OKAY)
  658       goto CLEANUP;
  659   }
  660 
  661   s_mp_exch(&s, c);
  662 
  663 CLEANUP:
  664   mp_clear(&x);
  665 X:
  666   mp_clear(&s);
  667 
  668   return res;
  669 
  670 } /* end mp_expt_d() */

  806 /* {{{ mp_mul(a, b, c) */
  807 
  808 /*
  809   mp_mul(a, b, c)
  810 
  811   Compute c = a * b.  All parameters may be identical.
  812  */
  813 mp_err   mp_mul(const mp_int *a, const mp_int *b, mp_int * c)
  814 {
  815   mp_digit *pb;
  816   mp_int   tmp[1];
  817   mp_err   res;
  818   mp_size  ib;
  819   mp_size  useda, usedb;
  820 
  821   ARGCHK(a != NULL && b != NULL && c != NULL, MP_BADARG);
  822 
  823   if (a == c) {
  824     if ((res = mp_init_copy(&tmp[1], a)) != MP_OKAY)
  825       return res;
  826     if (a == b) 
  827       b = &tmp[1];
  828     a = &tmp[1];
  829   } else if (b == c) {
  830     if ((res = mp_init_copy(&tmp[1], b)) != MP_OKAY)
  831       return res;
  832     b = &tmp[1];
  833   } else {
  834     MP_DIGITS(&tmp[1]) = 0;
  835   }
  836 
  837   if (MP_USED(a) < MP_USED(b)) {
  838     const mp_int *xch = b;        /* switch a and b, to do fewer outer loops */
  839     b = a;
  840     a = xch;
  841   }
  842 
  843   MP_USED(c) = 1; MP_DIGIT(c, 0) = 0;
  844   if((res = s_mp_pad(c, USED(a) + USED(b))) != MP_OKAY)
  845     goto CLEANUP;
  846 
  847 #ifdef NSS_USE_COMBA
  848   if ((MP_USED(a) == MP_USED(b)) && IS_POWER_OF_2(MP_USED(b))) {
  849       if (MP_USED(a) == 4) {
  850           s_mp_mul_comba_4(a, b, c);
  851           goto CLEANUP;
  852       }
  853       if (MP_USED(a) == 8) {
  854           s_mp_mul_comba_8(a, b, c);
  855           goto CLEANUP;
  856       }
  857       if (MP_USED(a) == 16) {
  858           s_mp_mul_comba_16(a, b, c);
  859           goto CLEANUP;
  860       }
  861       if (MP_USED(a) == 32) {
  862           s_mp_mul_comba_32(a, b, c);
  863           goto CLEANUP;
  864       } 
  865   }
  866 #endif
  867 
  868   pb = MP_DIGITS(b);
  869   s_mpv_mul_d[1][2][3][4][5](MP_DIGITS(a), MP_USED(a), *pb++, MP_DIGITS(c));
  870 
  871   /* Outer loop:  Digits of b */
  872   useda = MP_USED(a);
  873   usedb = MP_USED(b);
  874   for (ib = 1; ib < usedb; ib++) {
  875     mp_digit b_i    = *pb++;
  876 
  877     /* Inner product:  Digits of a */
  878     if (b_i)
  879       s_mpv_mul_d_add[1][2][3][4][5](MP_DIGITS(a), useda, b_i, MP_DIGITS(c) + ib);
  880     else
  881       MP_DIGIT(c, ib + useda) = b_i;
  882   }
  883 
  884   s_mp_clamp[1](c);
  885 
  886   if(SIGN(a) == SIGN(b) || s_mp_cmp_d(c, 0) == MP_EQ)
  887     SIGN(c) = ZPOS;
  888   else
  889     SIGN(c) = NEG;
  890 
  891 CLEANUP:
  892   mp_clear(&tmp[1]);
  893   return res;
  894 } /* end mp_mul() */

  910 mp_err   mp_sqr(const mp_int *a, mp_int *sqr)

  949           s_mp_sqr_comba_16(a, sqr);

  952       if (MP_USED(a) == 32) {
  953           s_mp_sqr_comba_32(a, sqr);
  954           goto CLEANUP;

  972   } else {
  973     MP_DIGIT(sqr, 1) = 0;
  974   }
  975 
  976   /* now add the squares of the digits of a to sqr. */
  977   s_mpv_sqr_add_prop[1][2](MP_DIGITS(a), MP_USED(a), MP_DIGITS(sqr));
  978 
  979   SIGN(sqr) = ZPOS;
  980   s_mp_clamp[1](sqr);
  981 
  982 CLEANUP:
  983   mp_clear(&tmp[1]);
  984   return res;
  985 
  986 } /* end mp_sqr() */

  991 /* {{{ mp_div(a, b, q, r) */
  992 
  993 /*
  994   mp_div(a, b, q, r)
  995 
  996   Compute q = a / b and r = a mod b.  Input parameters may be re-used
  997   as output parameters.  If q or r is NULL, that portion of the
  998   computation will be discarded (although it will still be computed)
  999  */
 1000 mp_err mp_div(const mp_int *a, const mp_int *b, mp_int *q, mp_int *r)
 1001 {
 1002   mp_err   res;
 1003   mp_int   *pQ, *pR;
 1004   mp_int   qtmp, rtmp, btmp;
 1005   int      cmp;
 1006   mp_sign  signA = MP_SIGN(a);
 1007   mp_sign  signB = MP_SIGN(b);
 1008 
 1009   ARGCHK(a != NULL && b != NULL, MP_BADARG);
 1010 
 1011   if(mp_cmp_z(b) == MP_EQ)
 1012     return MP_RANGE;
 1013 
 1014   DIGITS(&qtmp) = 0;
 1015   DIGITS(&rtmp) = 0;
 1016   DIGITS(&btmp) = 0;
 1017 
 1018   /* Set up some temporaries... */
 1019   if (!r || r == a || r == b) {
 1020     MP_CHECKOK( mp_init_copy(&rtmp, a) );
 1021     pR = &rtmp;
 1022   } else {
 1023     MP_CHECKOK( mp_copy(a, r) );
 1024     pR = r;
 1025   }
 1026 
 1027   if (!q || q == a || q == b) {
 1028     MP_CHECKOK( mp_init_size(&qtmp, MP_USED(a)) );
 1029     pQ = &qtmp;
 1030   } else {
 1031     MP_CHECKOK( s_mp_pad(q, MP_USED(a)) );
 1032     pQ = q;
 1033     mp_zero(pQ);
 1034   }
 1035 
 1036   /*
 1037     If |a| <= |b|, we can compute the solution without division;
 1038     otherwise, we actually do the work required.
 1039    */
 1040   if ((cmp = s_mp_cmp(a, b)) <= 0) {
 1041     if (cmp) {
 1042       /* r was set to a above. */
 1043       mp_zero(pQ);
 1044     } else {
 1045       mp_set(pQ, 1);
 1046       mp_zero(pR);
 1047     }
 1048   } else {
 1049     MP_CHECKOK( mp_init_copy(&btmp, b) );
 1050     MP_CHECKOK( s_mp_div(pR, &btmp, pQ) );
 1051   }
 1052 
 1053   /* Compute the signs for the output  */
 1054   MP_SIGN(pR) = signA;   /* Sr = Sa              */
 1055   /* Sq = ZPOS if Sa == Sb */ /* Sq = NEG if Sa != Sb */
 1056   MP_SIGN(pQ) = (signA == signB) ? ZPOS : NEG;
 1057 
 1058   if(s_mp_cmp_d(pQ, 0) == MP_EQ)
 1059     SIGN(pQ) = ZPOS;
 1060   if(s_mp_cmp_d(pR, 0) == MP_EQ)
 1061     SIGN(pR) = ZPOS;
 1062 
 1063   /* Copy output, if it is needed      */
 1064   if(q && q != pQ) 
 1065     s_mp_exch(pQ, q);
 1066 
 1067   if(r && r != pR) 
 1068     s_mp_exch(pR, r);
 1069 
 1070 CLEANUP:
 1071   mp_clear(&btmp);
 1072   mp_clear(&rtmp);
 1073   mp_clear(&qtmp);
 1074 
 1075   return res;
 1076 
 1077 } /* end mp_div() */

 1113   mp_expt(a, b, c)
 1114 
 1115   Compute c = a ** b, that is, raise a to the b power.  Uses a
 1116   standard iterative square-and-multiply technique.
 1117  */
 1118 
 1119 mp_err mp_expt(mp_int *a, mp_int *b, mp_int *c)
 1120 {
 1121   mp_int   s, x;
 1122   mp_err   res;
 1123   mp_digit d;
 1124   int      dig, bit;
 1125 
 1126   ARGCHK(a != NULL && b != NULL && c != NULL, MP_BADARG);
 1127 
 1128   if(mp_cmp_z(b) < 0)
 1129     return MP_RANGE;
 1130 
 1131   if((res = mp_init(&s)) != MP_OKAY)
 1132     return res;
 1133 
 1134   mp_set(&s, 1);
 1135 
 1136   if((res = mp_init_copy(&x, a)) != MP_OKAY)
 1137     goto X;
 1138 
 1139   /* Loop over low-order digits in ascending order */
 1140   for(dig = 0; dig < (USED(b) - 1); dig++) {
 1141     d = DIGIT(b, dig);
 1142 
 1143     /* Loop over bits of each non-maximal digit */
 1144     for(bit = 0; bit < DIGIT_BIT; bit++) {
 1145       if(d & 1) {
 1146         if((res = s_mp_mul(&s, &x)) != MP_OKAY) 
 1147           goto CLEANUP;
 1148       }
 1149 
 1150       d >>= 1;
 1151       
 1152       if((res = s_mp_sqr[1](&x)) != MP_OKAY)
 1153         goto CLEANUP;
 1154     }
 1155   }
 1156 
 1157   /* Consider now the last digit... */
 1158   d = DIGIT(b, dig);
 1159 
 1160   while(d) {
 1161     if(d & 1) {
 1162       if((res = s_mp_mul(&s, &x)) != MP_OKAY)
 1163         goto CLEANUP;
 1164     }
 1165 
 1166     d >>= 1;
 1167 
 1168     if((res = s_mp_sqr[1](&x)) != MP_OKAY)
 1169       goto CLEANUP;
 1170   }
 1171   
 1172   if(mp_iseven(b))
 1173     SIGN(&s) = SIGN(a);
 1174 
 1175   res = mp_copy(&s, c);
 1176 
 1177 CLEANUP:
 1178   mp_clear(&x);
 1179 X:
 1180   mp_clear(&s);
 1181 
 1182   return res;
 1183 
 1184 } /* end mp_expt() */

 1203 
 1204 /*
 1205   mp_mod(a, m, c)
 1206 
 1207   Compute c = a (mod m).  Result will always be 0 <= c < m.
 1208  */
 1209 
 1210 mp_err mp_mod(const mp_int *a, const mp_int *m, mp_int *c)
 1211 {
 1212   mp_err  res;
 1213   int     mag;
 1214 
 1215   ARGCHK(a != NULL && m != NULL && c != NULL, MP_BADARG);
 1216 
 1217   if(SIGN(m) == NEG)
 1218     return MP_RANGE;
 1219 
 1220   /*
 1221      If |a| > m, we need to divide to get the remainder and take the
 1222      absolute value.  
 1223 
 1224      If |a| < m, we don't need to do any division, just copy and adjust
 1225      the sign (if a is negative).
 1226 
 1227      If |a| == m, we can simply set the result to zero.
 1228 
 1229      This order is intended to minimize the average path length of the
 1230      comparison chain on common workloads -- the most frequent cases are
 1231      that |a| != m, so we do those first.
 1232    */
 1233   if((mag = s_mp_cmp(a, m)) > 0) {
 1234     if((res = mp_div(a, m, NULL, c)) != MP_OKAY)
 1235       return res;
 1236     
 1237     if(SIGN(c) == NEG) {
 1238       if((res = mp_add(c, m, c)) != MP_OKAY)
 1239         return res;
 1240     }
 1241 
 1242   } else if(mag < 0) {
 1243     if((res = mp_copy(a, c)) != MP_OKAY)
 1244       return res;
 1245 
 1246     if(mp_cmp_z(a) < 0) {
 1247       if((res = mp_add(c, m, c)) != MP_OKAY)
 1248         return res;
 1249 
 1250     }
 1251     
 1252   } else {
 1253     mp_zero(c);
 1254 
 1255   }
 1256 
 1257   return MP_OKAY;
 1258 
 1259 } /* end mp_mod() */

 1297 /* {{{ mp_sqrt(a, b) */
 1298 
 1299 /*
 1300   mp_sqrt(a, b)
 1301 
 1302   Compute the integer square root of a, and store the result in b.
 1303   Uses an integer-arithmetic version of Newton's iterative linear
 1304   approximation technique to determine this value; the result has the
 1305   following two properties:
 1306 
 1307      b^2 <= a
 1308      (b+1)^2 >= a
 1309 
 1310   It is a range error to pass a negative value.
 1311  */
 1312 mp_err mp_sqrt(const mp_int *a, mp_int *b)
 1313 {
 1314   mp_int   x, t;
 1315   mp_err   res;
 1316   mp_size  used;
 1317 
 1318   ARGCHK(a != NULL && b != NULL, MP_BADARG);
 1319 
 1320   /* Cannot take square root of a negative value */
 1321   if(SIGN(a) == NEG)
 1322     return MP_RANGE;
 1323 
 1324   /* Special cases for zero and one, trivial     */
 1325   if(mp_cmp_d(a, 1) <= 0)
 1326     return mp_copy(a, b);
 1327     
 1328   /* Initialize the temporaries we'll use below  */
 1329   if((res = mp_init_size(&t, USED(a))) != MP_OKAY)
 1330     return res;
 1331 
 1332   /* Compute an initial guess for the iteration as a itself */
 1333   if((res = mp_init_copy(&x, a)) != MP_OKAY)
 1334     goto X;
 1335 
 1336   used = MP_USED(&x);
 1337   if (used > 1) {
 1338     s_mp_rshd(&x, used / 2);
 1339   }
 1340 
 1341   for(;;) {
 1342     /* t = (x * x) - a */
 1343     mp_copy(&x, &t);      /* can't fail, t is big enough for original x */
 1344     if((res = mp_sqr[1](&t, &t)) != MP_OKAY ||
 1345        (res = mp_sub(&t, a, &t)) != MP_OKAY)
 1346       goto CLEANUP;
 1347 
 1348     /* t = t / 2x       */
 1349     s_mp_mul_2(&x);
 1350     if((res = mp_div(&t, &x, &t, NULL)) != MP_OKAY)
 1351       goto CLEANUP;
 1352     s_mp_div_2(&x);
 1353 
 1354     /* Terminate the loop, if the quotient is zero */
 1355     if(mp_cmp_z(&t) == MP_EQ)
 1356       break;
 1357 
 1358     /* x = x - t       */
 1359     if((res = mp_sub(&x, &t, &x)) != MP_OKAY)
 1360       goto CLEANUP;
 1361 
 1362   }
 1363 
 1364   /* Copy result to output parameter */
 1365   mp_sub_d(&x, 1, &x);
 1366   s_mp_exch(&x, b);
 1367 
 1368  CLEANUP:
 1369   mp_clear(&x);
 1370  X:
 1371   mp_clear(&t); 
 1372 
 1373   return res;
 1374 
 1375 } /* end mp_sqrt() */

 1922 mp_err mp_xgcd(const mp_int *a, const mp_int *b, mp_int *g, mp_int *x, mp_int *y)
 1923 {

 1925   mp_int  *clean[9];

 1927   int      last = -1;
 1928 
 1929   if(mp_cmp_z(b) == 0)
 1930     return MP_RANGE;
 1931 
 1932   /* Initialize all these variables we need */
 1933   MP_CHECKOK( mp_init(&u) );
 1934   clean[++last] = &u;
 1935   MP_CHECKOK( mp_init(&v) );
 1936   clean[++last] = &v;
 1937   MP_CHECKOK( mp_init(&gx) );
 1938   clean[++last] = &gx;
 1939   MP_CHECKOK( mp_init(&A) );
 1940   clean[++last] = &A;
 1941   MP_CHECKOK( mp_init(&B[1][2]) );
 1942   clean[++last] = &B[1][2];
 1943   MP_CHECKOK( mp_init(&C) );
 1944   clean[++last] = &C;
 1945   MP_CHECKOK( mp_init(&D) );
 1946   clean[++last] = &D;
 1947   MP_CHECKOK( mp_init_copy(&xc, a) );
 1948   clean[++last] = &xc;
 1949   mp_abs(&xc, &xc);
 1950   MP_CHECKOK( mp_init_copy(&yc, b) );
 1951   clean[++last] = &yc;
 1952   mp_abs(&yc, &yc);
 1953 
 1954   mp_set(&gx, 1);
 1955 
 1956   /* Divide by two until at least one of them is odd */
 1957   while(mp_iseven(&xc) && mp_iseven(&yc)) {
 1958     mp_size nx = mp_trailing_zeros(&xc);
 1959     mp_size ny = mp_trailing_zeros(&yc);
 1960     mp_size n  = MP_MIN(nx, ny);
 1961     s_mp_div_2d(&xc,n);
 1962     s_mp_div_2d(&yc,n);
 1963     MP_CHECKOK( s_mp_mul_2d(&gx,n) );
 1964   }
 1965 
 1966   mp_copy(&xc, &u);
 1967   mp_copy(&yc, &v);
 1968   mp_set(&A, 1); mp_set(&D, 1);
 1969 
 1970   /* Loop through binary GCD algorithm */
 1971   do {
 1972     while(mp_iseven(&u)) {
 1973       s_mp_div_2(&u);
 1974

 1976         s_mp_div_2(&A); s_mp_div_2(&B[1][2]);
 1977       } else {
 1978         MP_CHECKOK( mp_add(&A, &yc, &A) );
 1979         s_mp_div_2(&A);
 1980         MP_CHECKOK( mp_sub(&B[1][2], &xc, &B[1][2]) );
 1981         s_mp_div_2(&B[1][2]);
 1982       }
 1983     }

 1989         s_mp_div_2(&C); s_mp_div_2(&D);
 1990       } else {

 1992         s_mp_div_2(&C);
 1993         MP_CHECKOK( mp_sub(&D, &xc, &D) );
 1994         s_mp_div_2(&D);
 1995       }
 1996     }
 1997 
 1998     if(mp_cmp(&u, &v) >= 0) {
 1999       MP_CHECKOK( mp_sub(&u, &v, &u) );
 2000       MP_CHECKOK( mp_sub(&A, &C, &A) );
 2001       MP_CHECKOK( mp_sub(&B[1][2], &D, &B[1][2]) );
 2002     } else {
 2003       MP_CHECKOK( mp_sub(&v, &u, &v) );
 2004       MP_CHECKOK( mp_sub(&C, &A, &C) );
 2005       MP_CHECKOK( mp_sub(&D, &B[1][2], &D) );
 2006     }
 2007   } while (mp_cmp_z(&u) != 0);
 2008 
 2009   /* copy results to output */
 2010   if(x)
 2011     MP_CHECKOK( mp_copy(&C, x) );
 2012 
 2013   if(y)
 2014     MP_CHECKOK( mp_copy(&D, y) );
 2015       
 2016   if(g)
 2017     MP_CHECKOK( mp_mul(&gx, &v, g) );
 2018 
 2019  CLEANUP:
 2020   while(last >= 0)
 2021     mp_clear(clean[last--]);
 2022 
 2023   return res;
 2024 
 2025 } /* end mp_xgcd() */

 2030 {

 2036     return n;

 2039     n += MP_DIGIT_BIT;

 2042 #if !defined(MP_USE_UINT_DIGIT)

 2044     d >>= 32;

 2046   }

 2049     d >>= 16;

 2053     d >>= 8;
 2054     n  += 8;

 2057     d >>= 4;
 2058     n  += 4;

 2061     d >>= 2;
 2062     n  += 2;

 2065     d >>= 1;

 4354   mu = b^2k / m, where b = RADIX and k = #digits(m).  This should be

 4356   value of m are required (such as in modular exponentiation).  This
 4357   can nearly halve the time required to do modular exponentiation,

 4361   Cryptography_ by Menezes, Oorschot and VanStone, Ch. 14,

 4371     return res;
 4372

 4374   s_mp_mul(&q, mu);            /* q2 = q1 * mu      */
