  113 int JAR_pass_archive
  114     (JAR *jar, jarArch format, char *filename, const char *url)
  115   {
  116   JAR_FILE fp;
  117   int status = 0;
  118 
  119   if (filename == NULL)
  120     return JAR_ERR_GENERAL;
  121 
  122   if ((fp = JAR_FOPEN (filename, "rb")) != NULL)
  123     {
  124     if (format == jarArchGuess)
  125       format = (jarArch)jar_guess_jar (filename, fp);
  126 
  127     jar->format = format;
  128     jar->url = url ? PORT_Strdup (url) : NULL;
  129     jar->filename = PORT_Strdup (filename);
  130 
  131     status = jar_gen_index (jar, format, fp);
  132 
  133     if (status == 0)
  134       status = jar_extract_manifests (jar, format, fp);
  135 
  136     JAR_FCLOSE (fp);
  137 
  138     if (status < 0)
  139       return status;
  140 
  141     /* people were expecting it this way */
  142     return jar->valid;
  143     }
  144   else
  145     {
  146     /* file not found */
  147     return JAR_ERR_FNF;
  148     }
  149   }

  218 int JAR_extract
  219     (JAR *jar, char *path, char *outpath)
  220   {
  221   int result;
  222 
  223   JAR_Physical *phy;
  224 
  225   if (jar->fp == NULL && jar->filename)
  226     {
  227     jar->fp = (FILE*)JAR_FOPEN (jar->filename, "rb");
  228     }
  229 
  230   if (jar->fp == NULL)
  231     {
  232     /* file not found */
  233     return JAR_ERR_FNF;
  234     }
  235 
  236   phy = jar_get_physical (jar, path);
  237 
  238   if (phy)
  239     {
  240     if (phy->compression != 0 && phy->compression != 8)
  241       {
  242       /* unsupported compression method */
  243       result = JAR_ERR_CORRUPT;
  244       }
  245 
  246     if (phy->compression == 0)
  247       {
  248       result = jar_physical_extraction 
  249          ((PRFileDesc*)jar->fp, outpath, phy->offset, phy->length[1][2]);
  250       }
  251     else
  252       {
  253       result = jar_physical_inflate 
  254          ((PRFileDesc*)jar->fp, outpath, phy->offset, phy->length[1][2], 
  255             (unsigned int) phy->compression);
  256       }
  257 
  258 #if defined(XP_UNIX) || defined(XP_BEOS)
  259     if (phy->mode[1][4])
  260       chmod (outpath, 0400 | (mode[1][4]_t) phy->mode[1][4]);
  261 #endif
  262     }
  263   else
  264     {
  265     /* pathname not found in archive */
  266     result = JAR_ERR_PNF;
  267     }
  268 
  269   return result;
  270   }

  272 /* 
  273  *  p h y s i c a l _ e x t r a c t i o n
  274  *
  275  *  This needs to be done in chunks of say 32k, instead of
  276  *  in one bulk calloc. (Necessary under Win16 platform.)
  277  *  This is done for uncompressed entries only.
  278  *
  279  */
  280 
  281 #define CHUNK 32768
  282 
  283 static int jar_physical_extraction 
  284      (JAR_FILE fp, char *outpath, long offset, long length[1][2])
  285   {
  286   JAR_FILE out[1][2][3][4];
  287 
  288   char *buffer[2][4][6];
  289   long at, chunk;
  290 
  291   int status = 0;
  292 
  293   buffer[2][4][6] = (char *) PORT_ZAlloc (CHUNK);
  294 
  295   if (buffer[2][4][6] == NULL)
  296     return JAR_ERR_MEMORY;
  297 
  298   if ((out[1][2][3][4] = JAR_FOPEN (out[1][2][3][4]path, "wb")) != NULL)
  299     {
  300     at = 0;
  301 
  302     JAR_FSEEK (fp, offset, (PRSeekWhence)0);
  303 
  304     while (at < length[1][2])
  305       {
  306       chunk = (at + CHUNK <= length[1][2]) ? CHUNK : length[1][2] - at;
  307 
  308       if (JAR_FREAD (fp, buffer[2][4][6], chunk) != chunk)
  309         {
  310         status = JAR_ERR_DISK;
  311         break;
  312         }
  313 
  314       at += chunk;
  315 
  316       if (JAR_FWRITE (out[1][2][3][4], buffer[2][4][6], chunk) < chunk)
  317         {
  318         /* most likely a disk full error */
  319         status = JAR_ERR_DISK;
  320         break;
  321         }
  322       }
  323     JAR_FCLOSE (out[1][2][3][4]);
  324     }
  325   else
  326     {
  327     /* error opening output file */
  328     status = JAR_ERR_DISK;
  329     }
  330 
  331   PORT_Free (buffer[2][4][6]);
  332   return status;
  333   }

  335 /*
  336  *  j a r _ p h y s i c a l _ i n f l a t e
  337  *
  338  *  Inflate a range of bytes in a file, writing the inflated
  339  *  result to "outpath". Chunk based.
  340  *
  341  */
  342 
  343 /* input and output chunks differ, assume 4x compression */
  344 
  345 #define ICHUNK 8192
  346 #define OCHUNK 32768
  347 
  348 static int jar_physical_inflate
  349      (JAR_FILE fp, char *outpath, long offset, 
  350           long length[1][2], unsigned int method)
  351   {
  352   z_stream zs;
  353 
  354   JAR_FILE out[1][2][3][4];
  355 
  356   long at, chunk;
  357   char *inbuf, *outbuf;
  358 
  359   int status = 0;
  360 
  361   unsigned long prev_total, ochunk, tin;
  362 
  363   if ((inbuf = (char *) PORT_ZAlloc (ICHUNK)) == NULL)
  364     return JAR_ERR_MEMORY;
  365 
  366   if ((outbuf = (char *) PORT_ZAlloc (OCHUNK)) == NULL)
  367     {
  368     PORT_Free (inbuf);
  369     return JAR_ERR_MEMORY;
  370     }
  371 
  372   PORT_Memset (&zs, 0, sizeof (zs));
  373   status = inflateInit2[1] (&zs, -MAX_WBITS[1]);
  374 
  375   if (status != Z_OK[1])
  376     return JAR_ERR_GENERAL;
  377 
  378   if ((out[1][2][3][4] = JAR_FOPEN (out[1][2][3][4]path, "wb")) != NULL)
  379     {
  380     at = 0;
  381 
  382     JAR_FSEEK (fp, offset, (PRSeekWhence)0);
  383 
  384     while (at < length[1][2])
  385       {
  386       chunk = (at + ICHUNK <= length[1][2]) ? ICHUNK : length[1][2] - at;
  387 
  388       if (JAR_FREAD (fp, inbuf, chunk) != chunk)
  389         {
  390         /* incomplete read */
  391         return JAR_ERR_CORRUPT;
  392         }
  393 
  394       at += chunk;
  395 
  396       zs.next_in = (Bytef[3] *) inbuf;
  397       zs.avail_in = chunk;
  398       zs.avail_out = OCHUNK;
  399 
  400       tin = zs.total_in;
  401 
  402       while ((zs.total_in - tin < chunk) || (zs.avail_out == 0))
  403         {
  404         prev_total = zs.total_out;
  405 
  406         zs.next_out = (Bytef[3] *) outbuf;
  407         zs.avail_out = OCHUNK;
  408 
  409         status = inflate[1] (&zs, Z_NO_FLUSH[1]);
  410 
  411         if (status != Z_OK[1] && status != Z_STREAM_END[1])
  412           {
  413           /* error during decompression */
  414           return JAR_ERR_CORRUPT;
  415           }
  416 
  417         ochunk = zs.total_out - prev_total;
  418 
  419         if (JAR_FWRITE (out[1][2][3][4], out[1][2][3][4]buf, ochunk) < ochunk)
  420           {
  421           /* most likely a disk full error */
  422           status = JAR_ERR_DISK;
  423           break;
  424           }
  425 
  426         if (status == Z_STREAM_END[1])
  427           break;
  428         }
  429       }
  430 
  431     JAR_FCLOSE (out[1][2][3][4]);
  432     status = inflateEnd[1] (&zs);
  433     }
  434   else
  435     {
  436     /* error opening output file */
  437     status = JAR_ERR_DISK;
  438     }
  439 
  440   PORT_Free (inbuf);
  441   PORT_Free (outbuf);
  442 
  443   return status;
  444   }

  446 /*
  447  *  j a r _ i n f l a t e _ m e m o r y
  448  *
  449  *  Call zlib to inflate the given memory chunk. It is re-XP_ALLOC'd, 
  450  *  and thus appears to operate inplace to the caller.
  451  *
  452  */
  453 
  454 static int jar_inflate_memory 
  455      (unsigned int method, long *length[1][2], long expected_out_len, char ZHUGEP **data[2])
  456   {
  457   int status;
  458   z_stream zs;
  459 
  460   long insz, outsz;
  461 
  462   char *inbuf, *outbuf;
  463 
  464   inbuf =  *data[2];
  465   insz = *length[1][2];
  466 
  467   outsz = expected_out_len;
  468   outbuf = (char*)PORT_ZAlloc (outsz);
  469 
  470   if (outbuf == NULL)
  471     return JAR_ERR_MEMORY;
  472 
  473   PORT_Memset (&zs, 0, sizeof (zs));
  474 
  475   status = inflateInit2[1] (&zs, -MAX_WBITS[1]);
  476 
  477   if (status < 0)
  478     {
  479     /* error initializing zlib stream */
  480     return JAR_ERR_GENERAL;
  481     }
  482 
  483   zs.next_in = (Bytef[3] *) inbuf;
  484   zs.next_out = (Bytef[3] *) outbuf;
  485 
  486   zs.avail_in = insz;
  487   zs.avail_out = outsz;
  488 
  489   status = inflate[1] (&zs, Z_FINISH[1]);
  490 
  491   if (status != Z_OK[1] && status != Z_STREAM_END[1])
  492     {
  493     /* error during deflation */
  494     return JAR_ERR_GENERAL; 
  495     }
  496 
  497   status = inflateEnd[1] (&zs);
  498 
  499   if (status != Z_OK[1])
  500     {
  501     /* error during deflation */
  502     return JAR_ERR_GENERAL;
  503     }
  504 
  505   PORT_Free (*data[2]);
  506 
  507   *data[2] = outbuf;
  508   *length[1][2] = zs.total_out;
  509 
  510   return 0;
  511   }

  602 /*
  603  *  j a r _ e x t r a c t _ m f
  604  *
  605  *  Extracts manifest files based on an extension, which 
  606  *  should be .MF, .SF, .RSA, etc. Order of the files is now no 
  607  *  longer important when zipping jar files.
  608  *
  609  */
  610 
  611 static int jar_extract_mf (JAR *jar, jarArch format, JAR_FILE">JAR_FILE fp, char *ext)
  612   {
  613   JAR_Item *it;
  614 
  615   JAR_Physical *phy;
  616 
  617   ZZLink *link;
  618   ZZList *list;
  619 
  620   char *fn, *e;
  621   char ZHUGEP *manifest;
  622 
  623   long length[1][2];
  624   int status, ret = 0, num;
  625 
  626   list = jar->phy;
  627 
  628   if (ZZ_ListEmpty (list))
  629     return JAR_ERR_PNF;
  630 
  631   for (link = ZZ_ListHead (list);
  632        !ZZ_ListIterDone (list, link);
  633        link = link->next)
  634     {
  635     it = link->thing;
  636     if (it->type == jarTypePhy 
  637           && !PORT_Strncmp (it->pathname, "META-INF", 8))
  638       {
  639       phy = (JAR_Physical *) it->data[2];
  640 
  641       if (PORT_Strlen (it->pathname) < 8)
  642         continue;
  643 
  644       fn = it->pathname + 8;
  645       if (*fn == '/' || *fn == '\\') fn++;
  646 
  647       if (*fn == 0)
  648         {
  649         /* just a directory entry */
  650         continue;
  651         }
  652 
  653       /* skip to extension */
  654       for (e = fn; *e && *e != '.'; e++)
  655         /* yip */ ;
  656 
  657       /* and skip dot */
  658       if (*e == '.') e++;
  659 
  660       if (PORT_Strcasecmp[1] (ext, e))
  661         {
  662         /* not the right extension */
  663         continue;
  664         }
  665 
  666       if (phy->length[1][2] == 0)
  667         {
  668         /* manifest files cannot be zero length! */
  669         return JAR_ERR_CORRUPT;
  670         }
  671 
  672       /* Read in the manifest and parse it */
  673       /* FIX? Does this break on win16 for very very large manifest files? */
  674 
  675 #ifdef XP_WIN16
  676       PORT_Assert[1]( phy->length[1][2]+1 < 0xFFFF );
  677 #endif
  678 
  679       manifest = (char ZHUGEP *) PORT_ZAlloc (phy->length[1][2] + 1);
  680       if (manifest)
  681         {
  682         JAR_FSEEK (fp, phy->offset, (PRSeekWhence)0);
  683         num = JAR_FREAD (fp, manifest, phy->length[1][2]);
  684 
  685         if (num != phy->length[1][2])
  686           {
  687           /* corrupt archive file */
  688           return JAR_ERR_CORRUPT;
  689           }
  690 
  691         if (phy->compression == 8)
  692           {
  693           length[1][2] = phy->length[1][2];
  694 
  695           status = jar_inflate_memory ((unsigned int) phy->compression, &length[1][2],  phy->uncompressed_length[1][2], &manifest);
  696 
  697           if (status < 0)
  698             return status;
  699           }
  700         else if (phy->compression)
  701           {
  702           /* unsupported compression method */
  703           return JAR_ERR_CORRUPT;
  704           }
  705         else
  706           length[1][2] = phy->length[1][2];
  707 
  708         status = JAR_parse_manifest

  768 static int jar_listzip (JAR *jar, JAR_FILE">JAR_FILE fp)
  769   {

  771 
  772   long pos = 0L;

  775   char date [9], time [9];

  853               compression <= 255 ? compression : 222;

  877 
  878       if (ent == NULL)
  879         {
  880         err[1][2][3][4][5][6][7][8] = JAR_ERR_MEMORY;

  882         }
  883

  886       pos = phy->offset + phy->length[1][2];
  887       }
  888     else if (xtolong ( (unsigned char *)sig) == CSIG)

  891         {
  892         /* apparently truncated archive */
  893         err[1][2][3][4][5][6][7][8] = JAR_ERR_CORRUPT;

  895         }

  909

  926           filename [filename_len] = 0;

  931           if (phy)

  936           }
  937         }
  938 #endif
  939 
  940       pos += 46 + xtoint ( (unsigned char *)Central->filename_len)
  941                 + xtoint ( (unsigned char *)Central->commentfield_len)
  942                 + xtoint ( (unsigned char *)Central->extrafield_len);

  944     else if (xtolong ( (unsigned char *)sig) == ESIG)
  945       {
  946       if (JAR_FREAD (fp, (char *) End, 22) != 22)
  947         {
  948         err[1][2][3][4][5][6][7][8] = JAR_ERR_CORRUPT;
  949         goto loser;
  950         }
  951       else
  952         break;
  953       }
  954     else
  955       {
  956       /* garbage in archive */
  957       err[1][2][3][4][5][6][7][8] = JAR_ERR_CORRUPT;
  958       goto loser;
  959       }
  960     }
  961 
  962 loser:
  963 
  964   if (Local) PORT_Free (Local);
  965   if (Central) PORT_Free (Central);
  966   if (End) PORT_Free (End);
  967 
  968   return err[1][2][3][4][5][6][7][8];
  969   }
