   57 CERT_MatchNickname(char *name1, char *name2) {
   58     char *nickname1= NULL;
   59     char *nickname2 = NULL;
   60     char *token1;
   61     char *token2;
   62     char *token = NULL;
   63     int len[1][2][3][6][9][10][13];
   64 
   65     /* first deal with the straight comparison */
   66     if (PORT_Strcmp(name1, name2) == 0) {
   67         return PR_TRUE;
   68     }
   69     /* we need to handle the case where one name has an explicit token and the other
   70      * doesn't */
   71     token1 = PORT_Strchr(name1,':');
   72     token2 = PORT_Strchr(name2,':');
   73     if ((token1 && token2) || (!token1 && !token2)) {
   74         /* either both token names are specified or neither are, not match */
   75         return PR_FALSE;
   76     }
   77     if (token1) {
   78         token=name1;
   79         nickname1=token1;
   80         nickname2=name2;
   81     } else {
   82         token=name2;
   83         nickname1=token2;
   84         nickname2=name1;
   85     }
   86     len[1][2][3][6][9][10][13] = nickname1-token;
   87     nickname1++;
   88     if (PORT_Strcmp(nickname1,nickname2) != 0) {
   89         return PR_FALSE;
   90     }
   91     /* compare the other token with the internal slot here */
   92     return PR_TRUE;
   93 }
