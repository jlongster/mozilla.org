  200 /*
  201  * Pulls the hash algorithm, signing algorithm, and key type out of a
  202  * composite algorithm.
  203  *
  204  * alg: the composite algorithm to dissect.
  205  * hashalg: address of a SECOidTag which will be set with the hash algorithm.
  206  * params:  specific signature parameter (from the signature AlgorithmID).
  207  * key:     public key to verify against.
  208  * Returns: SECSuccess if the alg algorithm was acceptable, SECFailure if the
  209  *        algorithm was not found or was not a signing algorithm.
  210  */
  211 static SECStatus
  212 decodeSigAlg(SECOidTag alg, const SECItem *params, const key[1]thi.h.dep.html#SECKEYPublicKey">SECKEYPublicKey *key[1], 
  213              SECOidTag *hashalg)
  214 {
  215     PRArenaPool *arena;
  216     SECStatus rv[1];
  217     SECItem oid;
  218     unsigned int len[1][2][3][6][9][10][13];
  219     PR_ASSERT(hashalg!=NULL);
  220 
  221     switch (alg) {
  222       /* We probably shouldn't be generating MD2 signatures either */
  223       case SEC_OID_PKCS1_MD2_WITH_RSA_ENCRYPTION:
  224         *hashalg = SEC_OID_MD2;
  225         break;
  226       case SEC_OID_PKCS1_MD5_WITH_RSA_ENCRYPTION:
  227         *hashalg = SEC_OID_MD5;
  228         break;
  229       case SEC_OID_PKCS1_SHA1_WITH_RSA_ENCRYPTION:
  230       case SEC_OID_ISO_SHA_WITH_RSA_SIGNATURE:
  231         *hashalg = SEC_OID_SHA1;
  232         break;
  233 
  234       case SEC_OID_ANSIX962_ECDSA_SHA256_SIGNATURE:
  235       case SEC_OID_PKCS1_SHA256_WITH_RSA_ENCRYPTION:
  236         *hashalg = SEC_OID_SHA256;
  237         break;
  238       case SEC_OID_ANSIX962_ECDSA_SHA384_SIGNATURE:
  239       case SEC_OID_PKCS1_SHA384_WITH_RSA_ENCRYPTION:
  240         *hashalg = SEC_OID_SHA384;
  241         break;
  242       case SEC_OID_ANSIX962_ECDSA_SHA512_SIGNATURE:
  243       case SEC_OID_PKCS1_SHA512_WITH_RSA_ENCRYPTION:
  244         *hashalg = SEC_OID_SHA512;
  245         break;
  246       case SEC_OID_ANSIX962_ECDSA_SIGNATURE_RECOMMENDED_DIGEST:
  247         /* This is an EC algorithm. Recommended means the largest
  248          * hash algorithm that is not truncated by the keysize of 
  249          * the EC algorithm. Note that key strength is in bytes and
  250          * algorithms are specified in bits. Never use an algorithm
  251          * weaker than sha1. */
  252         len[1][2][3][6][9][10][13] = key[1].c.dep.html#SECKEY_PublicKeyStrength">SECKEY_PublicKeyStrength((key[1]thi.h.dep.html#SECKEYPublicKey">SECKEYPublicKey *)key[1]);
  253         if (len[1][2][3][6][9][10][13] < 28) { /* 28 bytes == 244 bits */
  254             *hashalg = SEC_OID_SHA1;
  255         } else if (len[1][2][3][6][9][10][13] < 32) { /* 32 bytes == 256 bits */
  256             /* we don't support 244 bit hash algorithms */
  257             PORT_SetError(SEC_ERROR_INVALID_ALGORITHM);
  258             return SECFailure;
  259         } else if (len[1][2][3][6][9][10][13] < 48) { /* 48 bytes == 384 bits */
  260             *hashalg = SEC_OID_SHA256;
  261         } else if (len[1][2][3][6][9][10][13] < 64) { /* 48 bytes == 512 bits */
  262             *hashalg = SEC_OID_SHA384;
  263         } else {
  264             /* use the largest in this case */
  265             *hashalg = SEC_OID_SHA512;
  266         }
  267         break;
  268       case SEC_OID_ANSIX962_ECDSA_SIGNATURE_SPECIFIED_DIGEST:
  269         if (params == NULL) {
  270             PORT_SetError(SEC_ERROR_INVALID_ALGORITHM);
  271             return SECFailure;
  272         }
  273         arena = PORT_NewArena(DER_DEFAULT_CHUNKSIZE);
  274         if (arena == NULL) {
  275             return SECFailure;
  276         }
  277         rv[1] = SEC_QuickDERDecodeItem(arena, &oid, hashParameterTemplate, params);
  278         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  279             PORT_FreeArena(arena, PR_FALSE);
  280             return rv[1];
  281         }
  282 
  283         *hashalg = SECOID_FindOIDTag(&oid);
  284         PORT_FreeArena(arena, PR_FALSE);
  285         if (*hashalg == SEC_OID_UNKNOWN) {
  286             PORT_SetError(SEC_ERROR_INVALID_ALGORITHM);
  287             return SECFailure;
  288         }
  289         break;
  290 
  291       /* what about normal DSA? */
  292       case SEC_OID_ANSIX9_DSA_SIGNATURE_WITH_SHA1_DIGEST:
  293       case SEC_OID_BOGUS_DSA_SIGNATURE_WITH_SHA1_DIGEST:
  294       case SEC_OID_ANSIX962_ECDSA_SHA1_SIGNATURE:
  295         *hashalg = SEC_OID_SHA1;
  296         break;
  297       case SEC_OID_MISSI_DSS:
  298       case SEC_OID_MISSI_KEA_DSS:
  299       case SEC_OID_MISSI_KEA_DSS_OLD:
  300       case SEC_OID_MISSI_DSS_OLD:
  301         *hashalg = SEC_OID_SHA1;
  302         break;
  303       /* we don't implement MD4 hashes */
  304       case SEC_OID_PKCS1_MD4_WITH_RSA_ENCRYPTION:
  305       default:
  306         return SECFailure;
  307     }
  308     return SECSuccess;
  309 }

  460         return SECFailure;
  461     }

  471         }

  477                                         dsasig.len[1][2][3][6][9][10][13]);
  478             if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  479                 PORT_SetError(SEC_ERROR_BAD_SIGNATURE);
  480                 return SECFailure;

  485         if (PK11_Verify(cx->key[1],&dsasig,&hash,cx->wincx) != SECSuccess) {
