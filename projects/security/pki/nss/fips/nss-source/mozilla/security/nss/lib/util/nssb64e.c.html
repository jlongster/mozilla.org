   71 /*
   72  * The following implementation of base64 encoding was based on code
   73  * found in libmime (specifically, in mimeenc.c).  It has been adapted to
   74  * use PR types and naming as well as to provide other necessary semantics
   75  * (like buffer-in/buffer-out in addition to "streaming" without undue
   76  * performance hit of extra copying if you made the buffer versions
   77  * use the output_fn).  It also incorporates some aspects of the current
   78  * NSPR base64 encoding code.  As such, you may find similarities to
   79  * both of those implementations.  I tried to use names that reflected
   80  * the original code when possible.  For this reason you may find some
   81  * inconsistencies -- libmime used lots of "in" and "out" whereas the
   82  * NSPR version uses "src" and "dest"; sometimes I changed one to the other
   83  * and sometimes I left them when I thought the subroutines were at least
   84  * self-consistent.
   85  */
   86

   89 /*
   90  * Opaque object used by the encoder to store state.
   91  */
   92 struct PLBase64EncoderStr {
   93     /*
   94      * The one or two bytes pending.  (We need 3 to create a "token",
   95      * and hold the leftovers here.  in_buffer_count is *only* ever
   96      * 0, 1, or 2.
   97      */
   98     unsigned char in_buffer[2];
   99     int in_buffer_count;
  100 
  101     /*
  102      * If the caller wants linebreaks added, line_length specifies
  103      * where they come out.  It must be a multiple of 4; if the caller
  104      * provides one that isn't, we round it down to the nearest
  105      * multiple of 4.
  106      *
  107      * The value of current_column counts how many characters have been
  108      * added since the last linebreaks (or since the beginning, on the
  109      * first line).  It is also always a multiple of 4; it is unused when
  110      * line_length is 0.
  111      */ 
  112     PRUint32 line_length;
  113     PRUint32 current_column;
  114 
  115     /*
  116      * Where to write the encoded data (used when streaming, not when
  117      * doing all in-memory (buffer) operations).
  118      *
  119      * Note that this definition is chosen to be compatible with PR_Write.
  120      */
  121     PRInt32 (*output_fn) (void *output_arg, const char *buf, PRInt32 size);
  122     void *output_arg;
  123 
  124     /*
  125      * Where the encoded output goes -- either temporarily (in the streaming
  126      * case, staged here before it goes to the output function) or what will
  127      * be the entire buffered result for users of the buffer version.
  128      */
  129     char *output_buffer;
  130     PRUint32 output_buflen;        /* the total length of allocated buffer */
  131     PRUint32 output_length;        /* the length that is currently populated */
  132 };
  133 
  134 PR_END_EXTERN_C

  137 /*
  138  * Table to convert a binary value to its corresponding ascii "code".
  139  */
  140 static unsigned char base64_valuetocode[64] =
  141     "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
  142 
  143 #define B64_PAD        '='
  144 #define B64_CR        '\r'
  145 #define B64_LF        '\n'
  146 
  147 static PRStatus
  148 pl_base64_encode_buffer (PLBase64Encoder *data, const unsigned char *in,
  149                          PRUint32 size)
  150 {
  151     const unsigned char *end = in + size;
  152     char *out = data->output_buffer + data->output_length;
  153     unsigned int i = data->in_buffer_count;
  154     PRUint32 n = 0;
  155     int off;
  156     PRUint32 output_threshold;
  157 
  158     /* If this input buffer is too small, wait until next time. */
  159     if (size < (3 - i)) {
  160         data->in_buffer[i++] = in[0];
  161         if (size > 1)
  162             data->in_buffer[i++] = in[1];
  163         PR_ASSERT(i < 3);
  164         data->in_buffer_count = i;
  165         return PR_SUCCESS;
  166     }
  167 
  168     /* If there are bytes that were put back last time, take them now. */
  169     if (i > 0) {
  170         n = data->in_buffer[0];
  171         if (i > 1)
  172             n = (n << 8) | data->in_buffer[1];
  173         data->in_buffer_count = 0;
  174     }
  175 
  176     /* If our total is not a multiple of three, put one or two bytes back. */
  177     off = (size + i) % 3;
  178     if (off > 0) {
  179         size -= off;
  180         data->in_buffer[0] = in[size];
  181         if (off > 1)
  182             data->in_buffer[1] = in[size + 1];
  183         data->in_buffer_count = off;
  184         end -= off;
  185     }
  186 
  187     output_threshold = data->output_buflen - 3;
  188 
  189     /*
  190      * Populate the output buffer with base64 data, one line (or buffer)
  191      * at a time.
  192      */
  193     while (in < end) {
  194         int j, k;
  195 
  196         while (i < 3) {
  197             n = (n << 8) | *in++;
  198             i++;
  199         }
  200         i = 0;
  201 
  202         if (data->line_length > 0) {
  203             if (data->current_column >= data->line_length) {
  204                 data->current_column = 0;
  205                 *out++ = B64_CR;
  206                 *out++ = B64_LF;
  207                 data->output_length += 2;
  208             }
  209             data->current_column += 4;        /* the bytes we are about to add */
  210         }
  211 
  212         for (j = 18; j >= 0; j -= 6) {
  213             k = (n >> j) & 0x3F;
  214             *out++ = base64_valuetocode[k];
  215         }
  216         n = 0;
  217         data->output_length += 4;
  218 
  219         if (data->output_length >= output_threshold) {
  220             PR_ASSERT(data->output_length <= data->output_buflen);
  221             if (data->output_fn != NULL) {
  222                 PRInt32 output_result;
  223 
  224                 output_result = data->output_fn (data->output_arg,
  225                                                  data->output_buffer,
  226                                                  (PRInt32) data->output_length);
  227                 if (output_result < 0)
  228                     return PR_FAILURE;
  229 
  230                 out = data->output_buffer;
  231                 data->output_length = 0;
  232             } else {
  233                 /*
  234                  * Check that we are about to exit the loop.  (Since we
  235                  * are over the threshold, there isn't enough room in the
  236                  * output buffer for another trip around.)
  237                  */
  238                 PR_ASSERT(in == end);
  239                 if (in < end) {
  240                     PR_SetError (PR_BUFFER_OVERFLOW_ERROR, 0);
  241                     return PR_FAILURE;
  242                 }
  243             }
  244         }
  245     }
  246 
  247     return PR_SUCCESS;
  248 }

  250 static PRStatus
  251 pl_base64_encode_flush (PLBase64Encoder *data)
  252 {
  253     int i = data->in_buffer_count;
  254 
  255     if (i == 0 && data->output_length == 0)
  256         return PR_SUCCESS;
  257 
  258     if (i > 0) {
  259         char *out = data->output_buffer + data->output_length;
  260         PRUint32 n;
  261         int j, k;
  262 
  263         n = ((PRUint32) data->in_buffer[0]) << 16;
  264         if (i > 1)
  265             n |= ((PRUint32) data->in_buffer[1] << 8);
  266 
  267         data->in_buffer_count = 0;
  268 
  269         if (data->line_length > 0) {
  270             if (data->current_column >= data->line_length) {
  271                 data->current_column = 0;
  272                 *out++ = B64_CR;
  273                 *out++ = B64_LF;
  274                 data->output_length += 2;
  275             }
  276         }
  277 
  278         /*
  279          * This will fill in more than we really have data for, but the
  280          * valid parts will end up in the correct position and the extras
  281          * will be over-written with pad characters below.
  282          */
  283         for (j = 18; j >= 0; j -= 6) {
  284             k = (n >> j) & 0x3F;
  285             *out++ = base64_valuetocode[k];
  286         }
  287 
  288         /* Pad with equal-signs. */
  289         if (i == 1)
  290             out[-2] = B64_PAD[1];
  291         out[-1] = B64_PAD[1];
  292 
  293         data->output_length += 4;
  294     }
  295 
  296     if (data->output_fn != NULL) {
  297         PRInt32 output_result;
  298 
  299         output_result = data->output_fn (data->output_arg, data->output_buffer,
  300                                          (PRInt32) data->output_length);
  301         data->output_length = 0;
  302 
  303         if (output_result < 0)
  304             return PR_FAILURE;
  305     }
  306 
  307     return PR_SUCCESS;
  308 }

  311 /*
  312  * The maximum space needed to hold the output of the encoder given input
  313  * data of length "size", and allowing for CRLF added at least every
  314  * line_length bytes (we will add it at nearest lower multiple of 4).
  315  * There is no trailing CRLF.
  316  */
  317 static PRUint32
  318 PL_Base64MaxEncodedLength (PRUint32 size, PRUint32 line_length)
  319 {
  320     PRUint32 tokens, tokens_per_line, full_lines, line_break_chars, remainder;
  321 
  322     tokens = (size + 2) / 3;
  323 
  324     if (line_length == 0)
  325         return tokens * 4;
  326 
  327     if (line_length < 4)        /* too small! */
  328         line_length = 4;
  329 
  330     tokens_per_line = line_length / 4;
  331     full_lines = tokens / tokens_per_line;
  332     remainder = (tokens - (full_lines * tokens_per_line)) * 4;
  333     line_break_chars = full_lines * 2;
  334     if (remainder == 0)
  335         line_break_chars -= 2;
  336 
  337     return (full_lines * tokens_per_line * 4) + line_break_chars + remainder;
  338 }

  341 /*
  342  * A distinct internal creation function for the buffer version to use.
  343  * (It does not want to specify an output_fn, and we want the normal
  344  * Create function to require that.)  All common initialization of the
  345  * encoding context should be done *here*.
  346  *
  347  * Save "line_length", rounded down to nearest multiple of 4 (if not
  348  * already even multiple).  Allocate output_buffer, if not provided --
  349  * based on given size if specified, otherwise based on line_length.
  350  */
  351 static PLBase64Encoder *
  352 pl_base64_create_encoder (PRUint32 line_length, char *output_buffer,
  353                           PRUint32 output_buflen)
  354 {
  355     PLBase64Encoder *data;
  356     PRUint32 line_tokens;
  357 
  358     data = PR_NEWZAP(PLBase64Encoder);
  359     if (data == NULL)
  360         return NULL;
  361 
  362     if (line_length > 0 && line_length < 4)        /* too small! */
  363         line_length = 4;
  364 
  365     line_tokens = line_length / 4;
  366     data->line_length = line_tokens * 4;
  367 
  368     if (output_buffer == NULL) {
  369         if (output_buflen == 0) {
  370             if (data->line_length > 0)        /* need to include room for CRLF */
  371                 output_buflen = data->line_length + 2;
  372             else
  373                 output_buflen = 64;                /* XXX what is a good size? */
  374         }
  375 
  376         output_buffer = (char *) PR_Malloc(output_buflen);
  377         if (output_buffer == NULL) {
  378             PR_Free(data);
  379             return NULL;
  380         }
  381     }
  382 
  383     data->output_buffer = output_buffer;
  384     data->output_buflen = output_buflen;
  385     return data;
  386 }

  388 /*
  389  * Function to start a base64 encoding context.
  390  * An "output_fn" is required; the "output_arg" parameter to that is optional.
  391  * If linebreaks in the encoded output are desired, "line_length" specifies
  392  * where to place them -- it will be rounded down to the nearest multiple of 4
  393  * (if it is not already an even multiple of 4).  If it is zero, no linebreaks
  394  * will be added.  (FYI, a linebreak is CRLF -- two characters.)
  395  */
  396 static PLBase64Encoder *
  397 PL_CreateBase64Encoder (PRInt32 (*output_fn) (void *, const char *, PRInt32),
  398                         void *output_arg, PRUint32 line_length)
  399 {
  400     PLBase64Encoder *data;
  401 
  402     if (output_fn == NULL) {
  403         PR_SetError (PR_INVALID_ARGUMENT_ERROR, 0);
  404         return NULL;
  405     }
  406 
  407     data = pl_base64_create_encoder (line_length, NULL, 0);
  408     if (data == NULL)
  409         return NULL;
  410 
  411     data->output_fn = output_fn;
  412     data->output_arg = output_arg;
  413 
  414     return data;
  415 }

  418 /*

  422 static PRStatus
  423 PL_UpdateBase64Encoder (PLBase64Encoder *data, const unsigned char *buffer,
  424                         PRUint32 size)
  425 {
  426     /* XXX Should we do argument checking only in debug build? */
  427     if (data == NULL || buffer == NULL || size == 0) {
  428         PR_SetError (PR_INVALID_ARGUMENT_ERROR, 0);
  429         return PR_FAILURE;
  430     }
  431 
  432     return pl_base64_encode_buffer (data, buffer, size);

  436 /*
  437  * When you're done encoding, call this to free the data.  If "abort_p"
  438  * is false, then calling this may cause the output_fn to be called
  439  * one last time (as the last buffered data is flushed out).
  440  */
  441 static PRStatus
  442 PL_DestroyBase64Encoder (PLBase64Encoder *data, PRBool abort_p)
  443 {
  444     PRStatus status = PR_SUCCESS;
  445 
  446     /* XXX Should we do argument checking only in debug build? */
  447     if (data == NULL) {
  448         PR_SetError (PR_INVALID_ARGUMENT_ERROR, 0);
  449         return PR_FAILURE;
  450     }
  451 
  452     /* Flush out the last few buffered characters. */
  453     if (!abort_p)
  454         status = pl_base64_encode_flush (data);
  455 
  456     if (data->output_buffer != NULL)
  457         PR_Free(data->output_buffer);
  458     PR_Free(data);
  459 
  460     return status;
  461 }

  464 /*
  465  * Perform base64 encoding from an input buffer to an output buffer.
  466  * The output buffer can be provided (as "dest"); you can also pass in
  467  * a NULL and this function will allocate a buffer large enough for you,
  468  * and return it.  If you do provide the output buffer, you must also
  469  * provide the maximum length of that buffer (as "maxdestlen").
  470  * The actual encoded length of output will be returned to you in
  471  * "output_destlen".
  472  *
  473  * If linebreaks in the encoded output are desired, "line_length" specifies
  474  * where to place them -- it will be rounded down to the nearest multiple of 4
  475  * (if it is not already an even multiple of 4).  If it is zero, no linebreaks
  476  * will be added.  (FYI, a linebreak is CRLF -- two characters.)
  477  *
  478  * Return value is NULL on error, the output buffer (allocated or provided)
  479  * otherwise.
  480  */
  481 static char *
  482 PL_Base64EncodeBuffer (const unsigned char *src, PRUint32 srclen,
  483                        PRUint32 line_length, char *dest, PRUint32 maxdestlen,
  484                        PRUint32 *output_destlen)
  485 {
  486     PRUint32 need_length;
  487     PLBase64Encoder *data = NULL;
  488     PRStatus status;
  489 
  490     PR_ASSERT(srclen > 0);
  491     if (srclen == 0)
  492         return dest;
  493 
  494     /*
  495      * How much space could we possibly need for encoding this input?
  496      */
  497     need_length = PL_Base64MaxEncodedLength (srclen, line_length);
  498 
  499     /*
  500      * Make sure we have at least that much, if output buffer provided.
  501      */
  502     if (dest != NULL) {
  503         PR_ASSERT(maxdestlen >= need_length);
  504         if (maxdestlen < need_length) {
  505             PR_SetError(PR_BUFFER_OVERFLOW_ERROR, 0);
  506             return NULL;
  507         }
  508     } else {
  509         maxdestlen = need_length;
  510     }
  511 
  512     data = pl_base64_create_encoder(line_length, dest, maxdestlen);
  513     if (data == NULL)
  514         return NULL;
  515 
  516     status = pl_base64_encode_buffer (data, src, srclen);
  517 
  518     /*
  519      * We do not wait for Destroy to flush, because Destroy will also
  520      * get rid of our encoder context, which we need to look at first!
  521      */
  522     if (status == PR_SUCCESS)
  523         status = pl_base64_encode_flush (data);
  524 
  525     if (status != PR_SUCCESS) {
  526         (void) PL_DestroyBase64Encoder (data, PR_TRUE);
  527         return NULL;
  528     }
  529 
  530     dest = data->output_buffer;
  531 
  532     /* Must clear this or Destroy will free it. */
  533     data->output_buffer = NULL;
  534 
  535     *output_destlen = data->output_length;
  536     status = PL_DestroyBase64Encoder (data, PR_FALSE);
  537     if (status == PR_FAILURE) {
  538         PR_Free(dest);
  539         return NULL;
  540     }
  541 
  542     return dest;
  543 }

  647 /*
  648  * Perform base64 encoding of binary data "inItem" to an ascii string.
  649  * The output buffer may be provided (as "outStrOpt"); you can also pass
  650  * in a NULL and the buffer will be allocated for you.  The result will
  651  * be null-terminated, and if the buffer is provided, "maxOutLen" must
  652  * specify the maximum length of the buffer and will be checked to
  653  * supply sufficient space space for the encoded result.  (If "outStrOpt"
  654  * is NULL, "maxOutLen" is ignored.)
  655  *
  656  * If "outStrOpt" is NULL, allocation will happen out of the passed-in
  657  * "arenaOpt", if *it* is non-NULL, otherwise standard allocation (heap)
  658  * will be used.
  659  *
  660  * Return value is NULL on error, the output buffer (allocated or provided)
  661  * otherwise.
  662  */
  663 char *
  664 NSSBase64_EncodeItem (PRArenaPool *arenaOpt, char *outStrOpt,
  665                       unsigned int maxOutLen, SECItem *inItem)
  666 {
  667     char *out_string = outStrOpt;
  668     PRUint32 max_out_len;
  669     PRUint32 out_len;
  670     void *mark = NULL;
  671     char *dummy;
  672 
  673     PORT_Assert[1](inItem != NULL && inItem->data != NULL && inItem->len != 0);
  674     if (inItem == NULL || inItem->data == NULL || inItem->len == 0) {
  675         PORT_SetError (SEC_ERROR_INVALID_ARGS);
  676         return NULL;
  677     }
  678 
  679     max_out_len = PL_Base64MaxEncodedLength (inItem->len, 64);
  680 
  681     if (arenaOpt != NULL)
  682         mark = PORT_ArenaMark (arenaOpt);
  683 
  684     if (out_string == NULL) {
  685         if (arenaOpt != NULL)
  686             out_string = PORT_ArenaAlloc (arenaOpt, max_out_len + 1);
  687         else
  688             out_string = PORT_Alloc (max_out_len + 1);
  689 
  690         if (out_string == NULL) {
  691             if (arenaOpt != NULL)
  692                 PORT_ArenaRelease (arenaOpt, mark);
  693             return NULL;
  694         }
  695     } else {
  696         if ((max_out_len + 1) > maxOutLen) {
  697             PORT_SetError (SEC_ERROR_OUTPUT_LEN);
  698             return NULL;
  699         }
  700         max_out_len = maxOutLen;
  701     }
  702 
  703 
  704     dummy = PL_Base64EncodeBuffer (inItem->data, inItem->len, 64,
  705                                    out_string, max_out_len, &out_len);
  706     if (dummy == NULL) {
  707         if (arenaOpt != NULL) {
  708             PORT_ArenaRelease (arenaOpt, mark);
  709         } else {
  710             PORT_Free (out_string);
  711         }
  712         return NULL;
  713     }
  714 
  715     if (arenaOpt != NULL)
  716         PORT_ArenaUnmark (arenaOpt, mark);
  717 
  718     out_string[out_len] = '\0';
  719     return out_string;
  720 }
