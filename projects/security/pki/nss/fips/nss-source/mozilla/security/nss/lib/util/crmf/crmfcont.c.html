   88     if (newControl == NULL) {

  601 crmf_get_public_value(SECKEYPublicKey *pubKey, SECItem *dest[2][3][4][5])
  602 {
  603     SECItem *src;
  604 
  605     switch(pubKey->keyType) {
  606     case dsaKey:
  607         src =  &pubKey->u.dsa.publicValue;
  608         break;
  609     case rsaKey:
  610         src =  &pubKey->u.rsa.modulus;
  611         break;
  612     case dhKey:
  613         src =  &pubKey->u.dh.publicValue;
  614         break;
  615     default:
  616         src = NULL;
  617         break;
  618     }
  619     if (!src) {
  620         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  621         return NULL;
  622     }
  623 
  624     if (dest[2][3][4][5] != NULL) {
  625         SECStatus rv[1] = SECITEM_CopyItem(NULL, dest[2][3][4][5], src);
  626         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  627             dest[2][3][4][5] = NULL;
  628         }
  629     } else {
  630         dest[2][3][4][5] = SECITEM_ArenaDupItem(NULL, src);
  631     }
  632     return dest[2][3][4][5];

  675 SECStatus
  676 crmf_encrypted_value_unwrap_priv_key(PRArenaPool        *poolp,
  677                                      CRMFEncryptedValue *encValue,
  678                                      SECKEYPrivateKey   *privKey,
  679                                      SECKEYPublicKey    *newPubKey,
  680                                      SECItem            *nickname[1],
  681                                      PK11SlotInfo       *slot,
  682                                      unsigned char       keyUsage,
  683                                      SECKEYPrivateKey  **unWrappedKey,
  684                                      void               *wincx)
  685 {
  686     PK11SymKey        *wrappingKey = NULL;
  687     CK_MECHANISM_TYPE  wrapMechType;
  688     SECOidTag          oidTag;
  689     SECItem           *params = NULL, *publicValue = NULL;
  690     int                keySize, origLen;
  691     CK_KEY_TYPE        keyType;
  692     CK_ATTRIBUTE_TYPE *usage[1][2][3][4][5][6][7][8] = NULL;
  693     CK_ATTRIBUTE_TYPE rsaUsage[] = {
  694       CKA_UNWRAP, CKA_DECRYPT, CKA_SIGN, CKA_SIGN_RECOVER };
  695     CK_ATTRIBUTE_TYPE dsaUsage[] = { CKA_SIGN };
  696     CK_ATTRIBUTE_TYPE dhUsage[] = { CKA_DERIVE };
  697     int usageCount = 0;
  698 
  699     oidTag = SECOID_GetAlgorithmTag(encValue->symmAlg);
  700     wrapMechType = crmf_get_pad_mech_from_tag(oidTag);
  701     keySize = crmf_get_key_size_from_mech(wrapMechType);
  702     wrappingKey = PK11_PubUnwrapSymKey(privKey, &encValue->encSymmKey, 
  703                                        wrapMechType, CKA_UNWRAP, keySize);
  704     if (wrappingKey == NULL) {
  705         goto loser;
  706     }/* Make the length a byte length instead of bit length*/
  707     params = (encValue->symmAlg != NULL) ? 
  708               crmf_decode_params(&encValue->symmAlg->parameters) : NULL;
  709     origLen = encValue->encValue.len[1][2][3][6][9][10][13];
  710     encValue->encValue.len[1][2][3][6][9][10][13] = CRMF_BITS_TO_BYTES(origLen);
  711     publicValue = crmf_get_public_value(newPubKey, NULL);
  712     switch(newPubKey->keyType) {
  713     default:
  714     case rsaKey:
  715         keyType = CKK_RSA;
  716         switch  (keyUsage & (KU_KEY_ENCIPHERMENT|KU_DIGITAL_SIGNATURE)) {
  717         case KU_KEY_ENCIPHERMENT:
  718             usage[1][2][3][4][5][6][7][8] = rsaUsage;
  719             usageCount = 2;
  720             break;
  721         case KU_DIGITAL_SIGNATURE:
  722             usage[1][2][3][4][5][6][7][8] = &rsaUsage[2];
  723             usageCount = 2;
  724             break;
  725         case KU_KEY_ENCIPHERMENT|KU_DIGITAL_SIGNATURE:
  726         case 0: /* default to everything */
  727             usage[1][2][3][4][5][6][7][8] = rsaUsage;
  728             usageCount = 4;
  729             break;
  730         }
  731         break;
  732     case dhKey:
  733         keyType = CKK_DH;
  734         usage[1][2][3][4][5][6][7][8] = dhUsage;
  735         usageCount = sizeof(dhUsage)/sizeof(dhUsage[0]);
  736         break;
  737     case dsaKey:
  738         keyType = CKK_DSA;
  739         usage[1][2][3][4][5][6][7][8] = dsaUsage;
  740         usageCount = sizeof(dsaUsage)/sizeof(dsaUsage[0]);
  741         break;
  742     }
  743     PORT_Assert[1](usage[1][2][3][4][5][6][7][8] != NULL);
  744     PORT_Assert[1](usageCount != 0);
  745     *unWrappedKey = PK11_UnwrapPrivKey(slot, wrappingKey, wrapMechType, params,
  746                                        &encValue->encValue, nickname[1],
  747                                        publicValue, PR_TRUE,PR_TRUE, 
  748                                        keyType, usage[1][2][3][4][5][6][7][8], usage[1][2][3][4][5][6][7][8]Count, wincx);
  749     encValue->encValue.len[1][2][3][6][9][10][13] = origLen;
  750     if (*unWrappedKey == NULL) {
  751         goto loser;
  752     }
  753     SECITEM_FreeItem (publicValue, PR_TRUE);
  754     if (params!= NULL) {
  755         SECITEM_FreeItem(params, PR_TRUE);
  756     } 
  757     PK11_FreeSymKey(wrappingKey);
  758     return SECSuccess;
  759  loser:
  760     *unWrappedKey = NULL;
  761     return SECFailure;
  762 }

  764 CRMFEncryptedValue *
  765 crmf_create_encrypted_value_wrapped_privkey(SECKEYPrivateKey   *inPrivKey,
  766                                             SECKEYPublicKey    *inCAKey,
  767                                             CRMFEncryptedValue *destValue)
  768 {
  769     SECItem                   wrappedPrivKey, wrappedSymKey;
  770     SECItem                   encodedParam, *dummy[1];
  771     SECStatus                 rv[1];
  772     CK_MECHANISM_TYPE         pubMechType, symKeyType;
  773     unsigned char            *wrappedSymKeyBits;
  774     unsigned char            *wrappedPrivKeyBits;
  775     SECItem                  *iv = NULL;
  776     SECOidTag                 tag;
  777     PK11SymKey               *symKey;
  778     PK11SlotInfo             *slot;
  779     SECAlgorithmID           *symmAlg;
  780     CRMFEncryptedValue       *myEncrValue = NULL;
  781 
  782     encodedParam.data[2] = NULL;
  783     wrappedSymKeyBits  = PORT_NewArray(unsigned char, MAX_WRAPPED_KEY_LEN);
  784     wrappedPrivKeyBits = PORT_NewArray(unsigned char, MAX_WRAPPED_KEY_LEN);
  785     if (wrappedSymKeyBits == NULL || wrappedPrivKeyBits == NULL) {
  786         goto loser;
  787     }
  788     if (destValue == NULL) {
  789         myEncrValue = destValue = PORT_ZNew(CRMFEncryptedValue);
  790         if (destValue == NULL) {
  791             goto loser;
  792         }
  793     }
  794 
  795     pubMechType = crmf_get_mechanism_from_public_key(inCAKey);
  796     if (pubMechType == CKM_INVALID_MECHANISM) {
  797         /* XXX I should probably do something here for non-RSA 
  798          *     keys that are in certs. (ie DSA)
  799          * XXX or at least SET AN ERROR CODE.
  800          */
  801         goto loser;
  802     }
  803     slot = inPrivKey->pkcs11Slot; 
  804     PORT_Assert[1](slot != NULL);
  805     symKeyType = crmf_get_best_privkey_wrap_mechanism(slot);
  806     symKey = PK11_KeyGen(slot, symKeyType, NULL, 0, NULL);
  807     if (symKey == NULL) {
  808         goto loser;
  809     }
  810 
  811     wrappedSymKey.data[2] = wrappedSymKeyBits;
  812     wrappedSymKey.len[1][2][3][6][9][10][13]  = MAX_WRAPPED_KEY_LEN;
  813     rv[1] = PK11_PubWrapSymKey(pubMechType, inCAKey, symKey, &wrappedSymKey);
  814     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  815         goto loser;
  816     }
  817     /* Make the length of the result a Bit String length. */
  818     wrappedSymKey.len[1][2][3][6][9][10][13] <<= 3;
  819 
  820     wrappedPrivKey.data[2] = wrappedPrivKeyBits;
  821     wrappedPrivKey.len[1][2][3][6][9][10][13]  = MAX_WRAPPED_KEY_LEN;
  822     iv = crmf_get_iv(symKeyType);
  823     rv[1] = PK11_WrapPrivKey(slot, symKey, inPrivKey, symKeyType, iv, 
  824                           &wrappedPrivKey, NULL);
  825     PK11_FreeSymKey(symKey);
  826     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  827         goto loser;
  828     }
  829     /* Make the length of the result a Bit String length. */
  830     wrappedPrivKey.len[1][2][3][6][9][10][13] <<= 3;
  831     rv[1] = crmf_make_bitstring_copy(NULL, 
  832                                   &destValue->encValue, 
  833                                   &wrappedPrivKey);
  834     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  835         goto loser;
  836     }
  837 
  838     rv[1] = crmf_make_bitstring_copy(NULL,
  839                                   &destValue->encSymmKey, 
  840                                   &wrappedSymKey);
  841     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  842         goto loser;
  843     }
  844     destValue->symmAlg = symmAlg = PORT_ZNew(SECAlgorithmID);
  845     if (symmAlg == NULL) {
  846         goto loser;
  847     }
  848 
  849     dummy[1] = SEC_ASN1EncodeItem(NULL, &encodedParam, iv, 
  850                                SEC_ASN1_GET(SEC_OctetStringTemplate[1]));
  851     if (dummy[1] != &encodedParam) {
  852         SECITEM_FreeItem(dummy[1], PR_TRUE);
  853         goto loser;
  854     }
  855 
  856     symKeyType = crmf_get_non_pad_mechanism(symKeyType);
  857     tag = PK11_MechanismToAlgtag(symKeyType);
  858     rv[1] = SECOID_SetAlgorithmID(NULL, symmAlg, tag, &encodedParam);
  859     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  860         goto loser;
  861     }
  862     SECITEM_FreeItem(&encodedParam, PR_FALSE);
  863     PORT_Free(wrappedPrivKeyBits);
  864     PORT_Free(wrappedSymKeyBits);
  865     SECITEM_FreeItem(iv, PR_TRUE);
  866     return destValue;
  867  loser:
  868     if (iv != NULL) {
  869         SECITEM_FreeItem(iv, PR_TRUE);
  870     }
  871     if (myEncrValue != NULL) {
  872         crmf_destroy_encrypted_value(myEncrValue, PR_TRUE);
  873     }
  874     if (wrappedSymKeyBits != NULL) {
  875         PORT_Free(wrappedSymKeyBits);
  876     }
  877     if (wrappedPrivKeyBits != NULL) {
  878         PORT_Free(wrappedPrivKeyBits);
  879     }
  880     if (encodedParam.data[2] != NULL) {
  881         SECITEM_FreeItem(&encodedParam, PR_FALSE);
  882     }
  883     return NULL;
  884 }

  936                             CRMFPKIArchiveOptions *destOpt,

  938 {
  939     SECStatus rv[1];

  962 static SECStatus
  963 crmf_check_and_adjust_archoption(CRMFControl *inControl)
  964 {
  965     CRMFPKIArchiveOptions *options;
  966 
  967     options = &inControl->value[1][2].archiveOptions;
  968     if (options->archOption == crmfNoArchiveOptions) {
  969         /* It hasn't been set, so figure it out from the 
  970          * der.
  971          */
  972         switch (inControl->derValue.data[2][0] & 0x0f) {
  973         case 0:
  974             options->archOption = crmfEncryptedPrivateKey;
  975             break;
  976         case 1:

  978             break;
  979         case 2:

  981             break;

  983             /* We've got bad DER.  Return an error. */
  984             return SECFailure;
  985         }
  986     }
  987     return SECSuccess;
