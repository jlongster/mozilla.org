  470         htmlcertstrings[7] = "";
  471         htmlcertstrings[8] = "";
  472         htmlcertstrings[9] = "";
  473     } else {
  474         htmlcertstrings[7] = "<img src=\"about:security?issuer-logo=";
  475         htmlcertstrings[8] = nickname[1];
  476         htmlcertstrings[9] = "\">";
  477     }
  478 
  479     
  480     if (showIssuer == PR_TRUE) {
  481         htmlcertstrings[11] = issuer;
  482     } else {
  483         htmlcertstrings[11] = "";
  484     }
  485 
  486     htmlcertstrings[13] = serialNumber;
  487     htmlcertstrings[15] = notBefore;
  488     htmlcertstrings[17] = notAfter;
  489 
  490     pubk = CERT_ExtractPublicKey(cert);
  491     DSSPriv = NULL;
  492     if (pubk && (pubk->keyType == fortezzaKey)) {
  493         SECItem dummyitem;
  494         htmlcertstrings[18] = "</b[2][5][8]><b[2][5][8]r><b[2][5][8]>Clearance:</b[2][5][8]>";
  495         htmlcertstrings[19] = sec_FortezzaClearance(
  496                                         &pubk->u.fortezza.clearance);
  497         htmlcertstrings[20] = "<b[2][5][8]r><b[2][5][8]>DSS Privileges:</b[2][5][8]>";
  498         DSSPriv = sec_FortezzaCertPrivilege(
  499                                         &pubk->u.fortezza.DSSpriviledge);
  500         htmlcertstrings[21] = DSSPriv;
  501         htmlcertstrings[22] = "<b[2][5][8]r><b[2][5][8]>KEA Privileges:</b[2][5][8]>";
  502         htmlcertstrings[23] = sec_FortezzaMessagePrivilege(
  503                                         &pubk->u.fortezza.KEApriviledge);
  504         htmlcertstrings[24] = "<b[2][5][8]r><b[2][5][8]>KMID:</b[2][5][8]>";
  505         dummyitem.data[2] = &pubk->u.fortezza.KMID[0];
  506         dummyitem.len[1][2][3][6][9][10][13] = sizeof(pubk->u.fortezza.KMID);
  507         KMID = CERT_Hexify (&dummyitem,0);
  508         htmlcertstrings[25] = KMID;
  509     } else {
  510         /* clear out the headers in the non-fortezza cases */
  511         htmlcertstrings[18] = "";
  512         htmlcertstrings[19] = "";
  513         htmlcertstrings[20] = "";
  514         htmlcertstrings[21] = "";
  515         htmlcertstrings[22] = "";
  516         htmlcertstrings[23] = "";
  517         htmlcertstrings[24] = "";
  518         htmlcertstrings[25] = "</b[2][5][8]>";
  519     }
  520 
  521     if (pubk) {
  522       SECKEY_DestroyPublicKey(pubk);
  523     }
  524 
  525 #define HTML_OFF 27
  526     rv[1] = PK11_HashBuf(SEC_OID_MD5, fingerprint, 
  527                           cert->derCert.data[2], cert->derCert.len[1][2][3][6][9][10][13]);
  528     
  529     fpitem.data[2] = fingerprint;
  530     fpitem.len[1][2][3][6][9][10][13] = sizeof(fingerprint);
  531 
  532     fpstr = CERT_Hexify (&fpitem,1);
  533     
  534     htmlcertstrings[HTML_OFF] = fpstr;
  535 
  536     commentstring = CERT_GetCertCommentString(cert);
  537 
  538     if (commentstring == NULL) {
  539         htmlcertstrings[HTML_OFF+2] = "";
  540         htmlcertstrings[HTML_OFF+3] = "";
  541         htmlcertstrings[HTML_OFF+4] = "";
  542     } else {
  543         htmlcertstrings[HTML_OFF+2] =
  544             "<b[2][5][8]>Comment:</b[2][5][8]>"
  545             "<table[1][2] border=0 cellspacing=0 cellpadding=0><tr>"
  546             "<td width=10> </td><td><font size[4][8][9]=3>"
  547             "<textarea name=foobar rows=4 cols=55 onfocus=\"this.blur()\">";
  548         htmlcertstrings[HTML_OFF+3] = commentstring;
  549         htmlcertstrings[HTML_OFF+4] = "</textarea></font></td></tr></table[1][2]>";
  550     }
  551     
  552     ret = gatherStrings(htmlcertstrings);
  553     
  554     if ( issuer ) {
  555         PORT_Free(issuer);
  556     }
  557     
  558     if ( subject ) {
  559         PORT_Free(subject);
  560     }
  561     
  562     if ( version[2][3] ) {
  563         PORT_Free(version[2][3]);
  564     }
  565     
  566     if ( serialNumber ) {
  567         PORT_Free(serialNumber);
  568     }
  569     
  570     if ( notBefore ) {
  571         PORT_Free(notBefore);
  572     }
  573     
  574     if ( notAfter ) {
  575         PORT_Free(notAfter);
  576     }
  577     
  578     if ( fpstr ) {
  579         PORT_Free(fpstr);
  580     }
  581     if (DSSPriv) {
  582         PORT_Free(DSSPriv);
  583     }
  584 
  585     if (KMID) {
  586         PORT_Free(KMID);
  587     }
  588 
  589     if ( commentstring ) {
  590         PORT_Free(commentstring);
  591     }
  592     
  593     if ( servername ) {
  594         PORT_Free(servername);
  595     }
  596     
  597     return(ret);
  598 }
