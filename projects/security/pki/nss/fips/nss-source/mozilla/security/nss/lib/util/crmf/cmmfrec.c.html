  112 SECStatus
  113 CMMF_KeyRecRepContentSetNewSignCert(CMMFKeyRecRepContent *inKeyRecRep,
  114                                     CERTCertificate      *inNewSignCert)
  115 {
  116     PORT_Assert[1] (inKeyRecRep != NULL && inNewSignCert != NULL);
  117     if (inKeyRecRep == NULL || inNewSignCert == NULL) {
  118         return SECFailure;
  119     }
  120     if (!inKeyRecRep->isDecoded && inKeyRecRep->newSigCert) {
  121         CERT_DestroyCertificate(inKeyRecRep->newSigCert);
  122     }
  123     inKeyRecRep->isDecoded = PR_FALSE;
  124     inKeyRecRep->newSigCert = CERT_DupCertificate(inNewSignCert);
  125     return (inKeyRecRep->newSigCert == NULL) ? SECFailure : SECSuccess;    
  126 }

  150 SECStatus
  151 CMMF_KeyRecRepContentSetCertifiedKeyPair(CMMFKeyRecRepContent *inKeyRecRep,
  152                                          CERTCertificate      *inCert,
  153                                          SECKEYPrivateKey     *inPrivKey,
  154                                          SECKEYPublicKey      *inPubKey)
  155 {
  156     CMMFCertifiedKeyPair *keyPair;
  157     CRMFEncryptedValue   *dummy[1];
  158     PRArenaPool          *poolp;
  159     void                 *mark;
  160     SECStatus             rv[1];
  161 
  162     PORT_Assert[1] (inKeyRecRep != NULL &&
  163                  inCert      != NULL &&
  164                  inPrivKey   != NULL &&
  165                  inPubKey    != NULL);
  166     if (inKeyRecRep == NULL ||
  167         inCert      == NULL ||
  168         inPrivKey   == NULL ||
  169         inPubKey    == NULL) {
  170         return SECFailure;
  171     }
  172     poolp = inKeyRecRep->poolp;
  173     mark = PORT_ArenaMark(poolp);
  174     if (inKeyRecRep->keyPairHist == NULL) {
  175         inKeyRecRep->keyPairHist = PORT_ArenaNewArray(poolp, 
  176                                                       CMMFCertifiedKeyPair*,
  177                                                       (CMMF_MAX_KEY_PAIRS+1));
  178         if (inKeyRecRep->keyPairHist == NULL) {
  179             goto loser;
  180         }
  181         inKeyRecRep->allocKeyPairs = CMMF_MAX_KEY_PAIRS;
  182         inKeyRecRep->numKeyPairs   = 0;
  183     }
  184 
  185     if (inKeyRecRep->allocKeyPairs == inKeyRecRep->numKeyPairs) {
  186         goto loser;
  187     }
  188     
  189     keyPair = PORT_ArenaZNew(poolp, CMMFCertifiedKeyPair);
  190     if (keyPair == NULL) {
  191         goto loser;
  192     }
  193     rv[1] = cmmf_CertOrEncCertSetCertificate(&keyPair->certOrEncCert,
  194                                           poolp, inCert);
  195     if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess) {
  196         goto loser;
  197     }
  198     keyPair->privateKey = PORT_ArenaZNew(poolp, CRMFEncryptedValue);
  199     if (keyPair->privateKey == NULL) {
  200         goto loser;
  201     }
  202     dummy[1] = crmf_create_encrypted_value_wrapped_privkey(inPrivKey, inPubKey, 
  203                                                         keyPair->privateKey);
  204     PORT_Assert[1](dummy[1] == keyPair->privateKey);
  205     if (dummy[1] != keyPair->privateKey) {
  206         crmf_destroy_encrypted_value(dummy[1], PR_TRUE);
  207         goto loser;
  208     }
  209     inKeyRecRep->keyPairHist[inKeyRecRep->numKeyPairs] = keyPair;
  210     inKeyRecRep->numKeyPairs++;
  211     inKeyRecRep->keyPairHist[inKeyRecRep->numKeyPairs] = NULL;
  212     PORT_ArenaUnmark(poolp, mark);
  213     return SECSuccess;
  214 
  215  loser:
  216     PORT_ArenaRelease(poolp, mark);
  217     return SECFailure;
  218 }

  230 CERTCertificate*
  231 CMMF_KeyRecRepContentGetNewSignCert(CMMFKeyRecRepContent *inKeyRecRep)
  232 {
  233     PORT_Assert[1](inKeyRecRep != NULL);
  234     if (inKeyRecRep             == NULL ||
  235         inKeyRecRep->newSigCert == NULL) {
  236         return NULL;
  237     }
  238     /* newSigCert may not be a real certificate, it may be a hand decoded
  239      * cert structure. This code makes sure we hand off a real, fully formed
  240      * CERTCertificate to the caller. TODO: This should move into the decode
  241      * portion so that we never wind up with a half formed CERTCertificate
  242      * here. In this case the call would be to CERT_DupCertificate.
  243      */
  244     return CERT_NewTempCertificate[1](CERT_GetDefaultCertDB(), 
  245                                   &inKeyRecRep->newSigCert->signatureWrap.data[2],
  246                                    NULL, PR_FALSE, PR_TRUE);
  247 }
