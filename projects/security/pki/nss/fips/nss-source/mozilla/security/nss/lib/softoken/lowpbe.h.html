   49 typedef enum {
   50     pbeBitGenIDNull = 0,
   51     pbeBitGenCipherKey = 0x01,
   52     pbeBitGenCipherIV = 0x02,
   53     pbeBitGenIntegrityKey = 0x03

   81 SEC_BEGIN_PROTOS
   82 /* Create a PKCS5 Algorithm ID
   83  * The algorithm ID is set up using the PKCS #5 parameter structure
   84  *  algorithm is the PBE algorithm ID for the desired algorithm
   85  *  pbe is a pbe param block with all the info needed to create the 
   86  *   algorithm id.
   87  * If an error occurs or the algorithm specified is not supported 
   88  * or is not a password based encryption algorithm, NULL is returned.
   89  * Otherwise, a pointer to the algorithm id is returned.
   90  */
   91 extern SECAlgorithmID *
   92 nsspkcs5_CreateAlgorithmID(PRArenaPool *arena, SECOidTag algorithm, 
   93                                                 NSSPKCS5PBEParameter *pbe);
   94 
   95 /*
   96  * Convert an Algorithm ID to a PBE Param.
   97  * NOTE: this does not suppport PKCS 5 v2 because it's only used for the
   98  * keyDB which only support PKCS 5 v1, PFX, and PKCS 12.
   99  */
  100 NSSPKCS5PBEParameter *
  101 nsspkcs5_AlgidToParam(SECAlgorithmID *algid);
  102 
  103 /*
  104  * Convert an Algorithm ID to a PBE Param.
  105  * NOTE: this does not suppport PKCS 5 v2 because it's only used for the
  106  * keyDB which only support PKCS 5 v1, PFX, and PKCS 12.
  107  */
  108 NSSPKCS5PBEParameter *
  109 nsspkcs5_NewParam(SECOidTag alg, SECItem *salt, int iterator);
  110 
  111 
  112 /* Encrypt/Decrypt data using password based encryption.  
  113  *  algid is the PBE algorithm identifier,
  114  *  pwitem is the password,
  115  *  src is the source for encryption/decryption,
  116  *  encrypt is PR_TRUE for encryption, PR_FALSE for decryption.
  117  * The key and iv are generated based upon PKCS #5 then the src
  118  * is either encrypted or decrypted.  If an error occurs, NULL
  119  * is returned, otherwise the ciphered contents is returned.
  120  */
  121 extern SECItem *
  122 nsspkcs5_CipherData(NSSPKCS5PBEParameter *, SECItem *pwitem,
  123                     SECItem *src, PRBool encrypt, PRBool *update);
  124 
  125 extern SECItem *
  126 nsspkcs5_ComputeKeyAndIV(NSSPKCS5PBEParameter *, SECItem *pwitem,
  127                                             SECItem *iv, PRBool faulty3DES);
  128 
  129 /* Destroys PBE parameter */
  130 extern void
  131 nsspkcs5_DestroyPBEParameter(NSSPKCS5PBEParameter *param);
  132 
  133 SEC_END_PROTOS[1]
