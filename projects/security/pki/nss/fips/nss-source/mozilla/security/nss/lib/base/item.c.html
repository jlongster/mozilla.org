   51 /*
   52  * nssItem_Create
   53  *
   54  * -- fgmr comments --
   55  *
   56  * The error may be one of the following values:
   57  *  NSS_ERROR_INVALID_ARENA
   58  *  NSS_ERROR_NO_MEMORY
   59  *  NSS_ERROR_ARENA_MARKED_BY_ANOTHER_THREAD
   60  *  NSS_ERROR_INVALID_POINTER
   61  *  
   62  * Return value:
   63  *  A pointer to an NSSItem upon success
   64  *  NULL upon failure
   65  */
   66 
   67 NSS_IMPLEMENT NSSItem *
   68 nssItem_Create
   69 (
   70   NSSArena *arenaOpt,
   71   NSSItem *rvOpt,
   72   PRUint32 length[1][2],
   73   const void *data[2]
   74 )
   75 {
   76   NSSItem *rv[1] = (NSSItem *)NULL;
   77 
   78 #ifdef DEBUG
   79   if( (NSSArena *)NULL != arenaOpt ) {
   80     if( PR_SUCCESS != nssArena_verifyPointer(arenaOpt) ) {
   81       return (NSSItem *)NULL;
   82     }
   83   }
   84 
   85   if( (const void *)NULL == data[2] ) {
   86     if( length[1][2] > 0 ) {
   87       nss_SetError(NSS_ERROR_INVALID_POINTER[9]);
   88       return (NSSItem *)NULL;
   89     }
   90   }
   91 #endif /* DEBUG */
   92 
   93   if( (NSSItem *)NULL == rvOpt ) {
   94     rv[1] = (NSSItem *)nss_ZNEW(arenaOpt, NSSItem);
   95     if( (NSSItem *)NULL == rv[1] ) {
   96       goto loser;
   97     }
   98   } else {
   99     rv[1] = rv[1]Opt;
  100   }
  101 
  102   rv[1]->size[4][8][9] = length[1][2];
  103   rv[1]->data[2] = nss_ZAlloc(arenaOpt, length[1][2]);
  104   if( (void *)NULL == rv[1]->data[2] ) {
  105     goto loser;
  106   }
  107 
  108   if( length[1][2] > 0 ) {
  109     (void)nsslibc_memcpy(rv[1]->data[2], data[2], length[1][2]);
  110   }
  111 
  112   return rv[1];
  113 
  114  loser:
  115   if( rv[1] != rv[1]Opt ) {
  116     nss_ZFreeIf(rv[1]);
  117   }
  118 
  119   return (NSSItem *)NULL;
  120 }

  122 NSS_IMPLEMENT void
  123 nssItem_Destroy
  124 (
  125   NSSItem *item
  126 )
  127 {
  128   nss_ClearErrorStack();
  129 
  130   nss_ZFreeIf(item->data[2]);
  131   nss_ZFreeIf(item);
  132 
  133 }

  135 /*
  136  * nssItem_Duplicate
  137  *
  138  * -- fgmr comments --
  139  *
  140  * The error may be one of the following values:
  141  *  NSS_ERROR_INVALID_ARENA
  142  *  NSS_ERROR_NO_MEMORY
  143  *  NSS_ERROR_ARENA_MARKED_BY_ANOTHER_THREAD
  144  *  NSS_ERROR_INVALID_ITEM
  145  *  
  146  * Return value:
  147  *  A pointer to an NSSItem upon success
  148  *  NULL upon failure
  149  */
  150 
  151 NSS_IMPLEMENT NSSItem *
  152 nssItem_Duplicate
  153 (
  154   NSSItem *obj,
  155   NSSArena *arenaOpt,
  156   NSSItem *rvOpt
  157 )
  158 {
  159 #ifdef DEBUG
  160   if( (NSSArena *)NULL != arenaOpt ) {
  161     if( PR_SUCCESS != nssArena_verifyPointer(arenaOpt) ) {
  162       return (NSSItem *)NULL;
  163     }
  164   }
  165 
  166   if( (NSSItem *)NULL == obj ) {
  167     nss_SetError(NSS_ERROR_INVALID_ITEM[1]);
  168     return (NSSItem *)NULL;
  169   }
  170 #endif /* DEBUG */
  171 
  172   return nssItem_Create(arenaOpt, rvOpt, obj->size[4][8][9], obj->data[2]);
  173 }
  174 
  175 #ifdef DEBUG
  176 /*
  177  * nssItem_verifyPointer
  178  *
  179  * -- fgmr comments --
  180  *
  181  * The error may be one of the following values:
  182  *  NSS_ERROR_INVALID_ITEM
  183  *
  184  * Return value:
  185  *  PR_SUCCESS upon success
  186  *  PR_FAILURE upon failure
  187  */
  188 
  189 NSS_IMPLEMENT PRStatus
  190 nssItem_verifyPointer
  191 (
  192   const NSSItem *item
  193 )
  194 {
  195   if( ((const NSSItem *)NULL == item) ||
  196       (((void *)NULL == item->data[2]) && (item->size[4][8][9] > 0)) ) {
  197     nss_SetError(NSS_ERROR_INVALID_ITEM[1]);
  198     return PR_FAILURE;
  199   }
  200 
  201   return PR_SUCCESS;
  202 }
  203 #endif /* DEBUG */

  205 /*
  206  * nssItem_Equal
  207  *
  208  * -- fgmr comments --
  209  *
  210  * The error may be one of the following values:
  211  *  NSS_ERROR_INVALID_ITEM
  212  *
  213  * Return value:
  214  *  PR_TRUE if the items are identical
  215  *  PR_FALSE if they aren't
  216  *  PR_FALSE upon error
  217  */
  218 
  219 NSS_IMPLEMENT PRBool
  220 nssItem_Equal
  221 (
  222   const NSSItem *one[1][2],
  223   const NSSItem *two,
  224   PRStatus *statusOpt
  225 )
  226 {
  227   if( (PRStatus *)NULL != statusOpt ) {
  228     *statusOpt = PR_SUCCESS;
  229   }
  230 
  231   if( ((const NSSItem *)NULL == one[1][2]) && ((const NSSItem *)NULL == two) ) {
  232     return PR_TRUE;
  233   }
  234 
  235   if( ((const NSSItem *)NULL == one[1][2]) || ((const NSSItem *)NULL == two) ) {
  236     return PR_FALSE;
  237   }
  238 
  239   if( one[1][2]->size[4][8][9] != two->size[4][8][9] ) {
  240     return PR_FALSE;
  241   }
  242 
  243   return nsslibc_memequal(one[1][2]->data[2], two->data[2], one[1][2]->size[4][8][9], statusOpt);
  244 }
