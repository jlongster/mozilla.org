  284 /*
  285  * NSS_CMSCipherContext_DecryptLength - find the output length of the next call to decrypt.
  286  *
  287  * cc - the cipher context
  288  * input_len - number of bytes used as input
  289  * final - true if this is the final chunk of data
  290  *
  291  * Result can be used to perform memory allocations.  Note that the amount
  292  * is exactly accurate only when not doing a block cipher or when final
  293  * is false, otherwise it is an upper bound on the amount because until
  294  * we see the data we do not know how many padding bytes there are
  295  * (always between 1 and bsize).
  296  *
  297  * Note that this can return zero, which does not mean that the decrypt
  298  * operation can be skipped!  (It simply means that there are not enough
  299  * bytes to make up an entire block; the bytes will be reserved until
  300  * there are enough to encrypt/decrypt at least one block.)  However,
  301  * if zero is returned it *does* mean that no output buffer need be
  302  * passed in to the subsequent decrypt operation, as no output bytes
  303  * will be stored.
  304  */
  305 unsigned int
  306 NSS_CMSCipherContext_DecryptLength(NSSCMSCipherContext *cc, unsigned int input_len, PRBool final)
  307 {
  308     int blocks, block_size;
  309 
  310     PORT_Assert[1] (! cc->encrypt);
  311 
  312     block_size = cc->block_size;
  313 
  314     /*
  315      * If this is not a block cipher, then we always have the same
  316      * number of output bytes as we had input bytes.
  317      */
  318     if (block_size == 0)
  319         return input_len;
  320 
  321     /*
  322      * On the final call, we will always use up all of the pending
  323      * bytes plus all of the input bytes, *but*, there will be padding
  324      * at the end and we cannot predict how many bytes of padding we
  325      * will end up removing.  The amount given here is actually known
  326      * to be at least 1 byte too long (because we know we will have
  327      * at least 1 byte of padding), but seemed clearer/better to me.
  328      */
  329     if (final)
  330         return cc->pending_count + input_len;
  331 
  332     /*
  333      * Okay, this amount is exactly what we will output on the
  334      * next cipher operation.  We will always hang onto the last
  335      * 1 - block_size bytes for non-final operations.  That is,
  336      * we will do as many complete blocks as we can *except* the
  337      * last block (complete or partial).  (This is because until
  338      * we know we are at the end, we cannot know when to interpret
  339      * and removing the padding byte(s), which are guaranteed to
  340      * be there.)
  341      */
  342     blocks = (cc->pending_count + input_len - 1) / block_size;
  343     return blocks * block_size;
  344 }

  346 /*
  347  * NSS_CMSCipherContext_EncryptLength - find the output length of the next call to encrypt.
  348  *
  349  * cc - the cipher context
  350  * input_len - number of bytes used as input
  351  * final - true if this is the final chunk of data
  352  *
  353  * Result can be used to perform memory allocations.
  354  *
  355  * Note that this can return zero, which does not mean that the encrypt
  356  * operation can be skipped!  (It simply means that there are not enough
  357  * bytes to make up an entire block; the bytes will be reserved until
  358  * there are enough to encrypt/decrypt at least one block.)  However,
  359  * if zero is returned it *does* mean that no output buffer need be
  360  * passed in to the subsequent encrypt operation, as no output bytes
  361  * will be stored.
  362  */
  363 unsigned int
  364 NSS_CMSCipherContext_EncryptLength(NSSCMSCipherContext *cc, unsigned int input_len, PRBool final)
  365 {
  366     int blocks, block_size;
  367     int pad_size;
  368 
  369     PORT_Assert[1] (cc->encrypt);
  370 
  371     block_size = cc->block_size;
  372     pad_size = cc->pad_size;
  373 
  374     /*
  375      * If this is not a block cipher, then we always have the same
  376      * number of output bytes as we had input bytes.
  377      */
  378     if (block_size == 0)
  379         return input_len;
  380 
  381     /*
  382      * On the final call, we only send out what we need for
  383      * remaining bytes plus the padding.  (There is always padding,
  384      * so even if we have an exact number of blocks as input, we
  385      * will add another full block that is just padding.)
  386      */
  387     if (final) {
  388         if (pad_size == 0) {
  389                 return cc->pending_count + input_len;
  390         } else {
  391                 blocks = (cc->pending_count + input_len) / pad_size;
  392             blocks++;
  393             return blocks*pad_size;
  394         }
  395     }
  396 
  397     /*
  398      * Now, count the number of complete blocks of data we have.
  399      */
  400     blocks = (cc->pending_count + input_len) / block_size;
  401 
  402 
  403     return blocks * block_size;
  404 }

  407 /*
  408  * NSS_CMSCipherContext_Decrypt - do the decryption
  409  *
  410  * cc - the cipher context
  411  * output - buffer for decrypted result bytes
  412  * output_len_p - number of bytes in output
  413  * max_output_len - upper bound on bytes to put into output
  414  * input - pointer to input bytes
  415  * input_len - number of input bytes
  416  * final - true if this is the final chunk of data
  417  *
  418  * Decrypts a given length of input buffer (starting at "input" and
  419  * containing "input_len" bytes), placing the decrypted bytes in
  420  * "output" and storing the output length in "*output_len_p".
  421  * "cc" is the return value from NSS_CMSCipher_StartDecrypt.
  422  * When "final" is true, this is the last of the data to be decrypted.
  423  *
  424  * This is much more complicated than it sounds when the cipher is
  425  * a block-type, meaning that the decryption function will only
  426  * operate on whole blocks.  But our caller is operating stream-wise,
  427  * and can pass in any number of bytes.  So we need to keep track
  428  * of block boundaries.  We save excess bytes between calls in "cc".
  429  * We also need to determine which bytes are padding, and remove
  430  * them from the output.  We can only do this step when we know we
  431  * have the final block of data.  PKCS #7 specifies that the padding
  432  * used for a block cipher is a string of bytes, each of whose value is
  433  * the same as the length of the padding, and that all data is padded.
  434  * (Even data that starts out with an exact multiple of blocks gets
  435  * added to it another block, all of which is padding.)
  436  */ 
  437 SECStatus
  438 NSS_CMSCipherContext_Decrypt(NSSCMSCipherContext *cc, unsigned char *output[1][2][3][4][5][6][7][8],
  439                   unsigned int *output_len_p, unsigned int max_output_len,
  440                   const unsigned char *input[1], unsigned int input[1]_len,
  441                   PRBool final)
  442 {
  443     int blocks, bsize, pcount, padsize;
  444     unsigned int max_needed, ifraglen, ofraglen, output_len;
  445     unsigned char *pbuf;
  446     SECStatus rv[1];
  447 
  448     PORT_Assert[1] (! cc->encrypt);
  449 
  450     /*
  451      * Check that we have enough room for the output.  Our caller should
  452      * already handle this; failure is really an internal error (i.e. bug).
  453      */
  454     max_needed = NSS_CMSCipherContext_DecryptLength(cc, input_len, final);
  455     PORT_Assert[1] (max_output_len >= max_needed);
  456     if (max_output_len < max_needed) {
  457         /* PORT_SetError (XXX); */
  458         return SECFailure;
  459     }
  460 
  461     /*
  462      * hardware encryption does not like small decryption sizes here, so we
  463      * allow both blocking and padding.
  464      */
  465     bsize = cc->block_size;
  466     padsize = cc->pad_size;
  467 
  468     /*
  469      * When no blocking or padding work to do, we can simply call the
  470      * cipher function and we are done.
  471      */
  472     if (bsize == 0) {
  473         return (* cc->doit) (cc->cx, output[1][2][3][4][5][6][7][8], output[1][2][3][4][5][6][7][8]_len_p, max_output[1][2][3][4][5][6][7][8]_len,
  474                               input[1], input[1]_len);
  475     }
  476 
  477     pcount = cc->pending_count;
  478     pbuf = cc->pending_buf;
  479 
  480     output_len = 0;
  481 
  482     if (pcount) {
  483         /*
  484          * Try to fill in an entire block, starting with the bytes
  485          * we already have saved away.
  486          */
  487         while (input_len && pcount < bsize) {
  488             pbuf[pcount++] = *input[1]++;
  489             input_len--;
  490         }
  491         /*
  492          * If we have at most a whole block and this is not our last call,
  493          * then we are done for now.  (We do not try to decrypt a lone
  494          * single block because we cannot interpret the padding bytes
  495          * until we know we are handling the very last block of all input.)
  496          */
  497         if (input_len == 0 && !final) {
  498             cc->pending_count = pcount;
  499             if (output_len_p)
  500                 *output_len_p = 0;
  501             return SECSuccess;
  502         }
  503         /*
  504          * Given the logic above, we expect to have a full block by now.
  505          * If we do not, there is something wrong, either with our own
  506          * logic or with (length of) the data given to us.
  507          */
  508         if ((padsize != 0) && (pcount % padsize) != 0) {
  509             PORT_Assert[1] (final);        
  510             PORT_SetError (SEC_ERROR_BAD_DATA);
  511             return SECFailure;
  512         }
  513         /*
  514          * Decrypt the block.
  515          */
  516         rv[1] = (*cc->doit)(cc->cx, output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8], &ofraglen, max_output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8]_len,
  517                             pbuf, pcount);
  518         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  519             return rv[1];
  520 
  521         /*
  522          * For now anyway, all of our ciphers have the same number of
  523          * bytes of output as they do input.  If this ever becomes untrue,
  524          * then NSS_CMSCipherContext_DecryptLength needs to be made smarter!
  525          */
  526         PORT_Assert[1](ofraglen == pcount);
  527 
  528         /*
  529          * Account for the bytes now in output.
  530          */
  531         max_output_len -= ofraglen;
  532         output_len += ofraglen;
  533         output[1][2][3][4][5][6][7][8] += ofraglen;
  534     }
  535 
  536     /*
  537      * If this is our last call, we expect to have an exact number of
  538      * blocks left to be decrypted; we will decrypt them all.
  539      * 
  540      * If not our last call, we always save between 1 and bsize bytes
  541      * until next time.  (We must do this because we cannot be sure
  542      * that none of the decrypted bytes are padding bytes until we
  543      * have at least another whole block of data.  You cannot tell by
  544      * looking -- the data could be anything -- you can only tell by
  545      * context, knowing you are looking at the last block.)  We could
  546      * decrypt a whole block now but it is easier if we just treat it
  547      * the same way we treat partial block bytes.
  548      */
  549     if (final) {
  550         if (padsize) {
  551             blocks = input_len / padsize;
  552             ifraglen = blocks * padsize;
  553         } else ifraglen = input_len;
  554         PORT_Assert[1] (ifraglen == input_len);
  555 
  556         if (ifraglen != input_len) {
  557             PORT_SetError(SEC_ERROR_BAD_DATA);
  558             return SECFailure;
  559         }
  560     } else {
  561         blocks = (input_len - 1) / bsize;
  562         ifraglen = blocks * bsize;
  563         PORT_Assert[1] (ifraglen < input_len);
  564 
  565         pcount = input_len - ifraglen;
  566         PORT_Memcpy (pbuf, input[1] + ifraglen, pcount);
  567         cc->pending_count = pcount;
  568     }
  569 
  570     if (ifraglen) {
  571         rv[1] = (* cc->doit)(cc->cx, output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8], &ofraglen, max_output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8]_len,
  572                             input[1], ifraglen);
  573         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  574             return rv[1];
  575 
  576         /*
  577          * For now anyway, all of our ciphers have the same number of
  578          * bytes of output as they do input.  If this ever becomes untrue,
  579          * then sec_PKCS7DecryptLength needs to be made smarter!
  580          */
  581         PORT_Assert[1] (ifraglen == ofraglen);
  582         if (ifraglen != ofraglen) {
  583             PORT_SetError(SEC_ERROR_BAD_DATA);
  584             return SECFailure;
  585         }
  586 
  587         output_len += ofraglen;
  588     } else {
  589         ofraglen = 0;
  590     }
  591 
  592     /*
  593      * If we just did our very last block, "remove" the padding by
  594      * adjusting the output length.
  595      */
  596     if (final && (padsize != 0)) {
  597         unsigned int padlen = *(output[1][2][3][4][5][6][7][8] + ofraglen - 1);
  598 
  599         if (padlen == 0 || padlen > padsize) {
  600             PORT_SetError(SEC_ERROR_BAD_DATA);
  601             return SECFailure;
  602         }
  603         output_len -= padlen;
  604     }
  605 
  606     PORT_Assert[1] (output_len_p != NULL || output_len == 0);
  607     if (output_len_p != NULL)
  608         *output_len_p = output_len;
  609 
  610     return SECSuccess;
  611 }

  613 /*
  614  * NSS_CMSCipherContext_Encrypt - do the encryption
  615  *
  616  * cc - the cipher context
  617  * output - buffer for decrypted result bytes
  618  * output_len_p - number of bytes in output
  619  * max_output_len - upper bound on bytes to put into output
  620  * input - pointer to input bytes
  621  * input_len - number of input bytes
  622  * final - true if this is the final chunk of data
  623  *
  624  * Encrypts a given length of input buffer (starting at "input" and
  625  * containing "input_len" bytes), placing the encrypted bytes in
  626  * "output" and storing the output length in "*output_len_p".
  627  * "cc" is the return value from NSS_CMSCipher_StartEncrypt.
  628  * When "final" is true, this is the last of the data to be encrypted.
  629  *
  630  * This is much more complicated than it sounds when the cipher is
  631  * a block-type, meaning that the encryption function will only
  632  * operate on whole blocks.  But our caller is operating stream-wise,
  633  * and can pass in any number of bytes.  So we need to keep track
  634  * of block boundaries.  We save excess bytes between calls in "cc".
  635  * We also need to add padding bytes at the end.  PKCS #7 specifies
  636  * that the padding used for a block cipher is a string of bytes,
  637  * each of whose value is the same as the length of the padding,
  638  * and that all data is padded.  (Even data that starts out with
  639  * an exact multiple of blocks gets added to it another block,
  640  * all of which is padding.)
  641  *
  642  * XXX I would kind of like to combine this with the function above
  643  * which does decryption, since they have a lot in common.  But the
  644  * tricky parts about padding and filling blocks would be much
  645  * harder to read that way, so I left them separate.  At least for
  646  * now until it is clear that they are right.
  647  */ 
  648 SECStatus
  649 NSS_CMSCipherContext_Encrypt(NSSCMSCipherContext *cc, unsigned char *output[1][2][3][4][5][6][7][8],
  650                   unsigned int *output_len_p, unsigned int max_output_len,
  651                   const unsigned char *input[1], unsigned int input[1]_len,
  652                   PRBool final)
  653 {
  654     int blocks, bsize, padlen, pcount, padsize;
  655     unsigned int max_needed, ifraglen, ofraglen, output_len;
  656     unsigned char *pbuf;
  657     SECStatus rv[1];
  658 
  659     PORT_Assert[1] (cc->encrypt);
  660 
  661     /*
  662      * Check that we have enough room for the output.  Our caller should
  663      * already handle this; failure is really an internal error (i.e. bug).
  664      */
  665     max_needed = NSS_CMSCipherContext_EncryptLength (cc, input_len, final);
  666     PORT_Assert[1] (max_output_len >= max_needed);
  667     if (max_output_len < max_needed) {
  668         /* PORT_SetError (XXX); */
  669         return SECFailure;
  670     }
  671 
  672     bsize = cc->block_size;
  673     padsize = cc->pad_size;
  674 
  675     /*
  676      * When no blocking and padding work to do, we can simply call the
  677      * cipher function and we are done.
  678      */
  679     if (bsize == 0) {
  680         return (*cc->doit)(cc->cx, output[1][2][3][4][5][6][7][8], output[1][2][3][4][5][6][7][8]_len_p, max_output[1][2][3][4][5][6][7][8]_len,
  681                               input[1], input[1]_len);
  682     }
  683 
  684     pcount = cc->pending_count;
  685     pbuf = cc->pending_buf;
  686 
  687     output_len = 0;
  688 
  689     if (pcount) {
  690         /*
  691          * Try to fill in an entire block, starting with the bytes
  692          * we already have saved away.
  693          */
  694         while (input_len && pcount < bsize) {
  695             pbuf[pcount++] = *input[1]++;
  696             input_len--;
  697         }
  698         /*
  699          * If we do not have a full block and we know we will be
  700          * called again, then we are done for now.
  701          */
  702         if (pcount < bsize && !final) {
  703             cc->pending_count = pcount;
  704             if (output_len_p != NULL)
  705                 *output_len_p = 0;
  706             return SECSuccess;
  707         }
  708         /*
  709          * If we have a whole block available, encrypt it.
  710          */
  711         if ((padsize == 0) || (pcount % padsize) == 0) {
  712             rv[1] = (* cc->doit) (cc->cx, output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8], &ofraglen, max_output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8]_len,
  713                                 pbuf, pcount);
  714             if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  715                 return rv[1];
  716 
  717             /*
  718              * For now anyway, all of our ciphers have the same number of
  719              * bytes of output as they do input.  If this ever becomes untrue,
  720              * then sec_PKCS7EncryptLength needs to be made smarter!
  721              */
  722             PORT_Assert[1] (ofraglen == pcount);
  723 
  724             /*
  725              * Account for the bytes now in output.
  726              */
  727             max_output_len -= ofraglen;
  728             output_len += ofraglen;
  729             output[1][2][3][4][5][6][7][8] += ofraglen;
  730 
  731             pcount = 0;
  732         }
  733     }
  734 
  735     if (input_len) {
  736         PORT_Assert[1] (pcount == 0);
  737 
  738         blocks = input_len / bsize;
  739         ifraglen = blocks * bsize;
  740 
  741         if (ifraglen) {
  742             rv[1] = (* cc->doit) (cc->cx, output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8], &ofraglen, max_output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8]_len,
  743                                 input[1], ifraglen);
  744             if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  745                 return rv[1];
  746 
  747             /*
  748              * For now anyway, all of our ciphers have the same number of
  749              * bytes of output as they do input.  If this ever becomes untrue,
  750              * then sec_PKCS7EncryptLength needs to be made smarter!
  751              */
  752             PORT_Assert[1] (ifraglen == ofraglen);
  753 
  754             max_output_len -= ofraglen;
  755             output_len += ofraglen;
  756             output[1][2][3][4][5][6][7][8] += ofraglen;
  757         }
  758 
  759         pcount = input_len - ifraglen;
  760         PORT_Assert[1] (pcount < bsize);
  761         if (pcount)
  762             PORT_Memcpy (pbuf, input[1] + ifraglen, pcount);
  763     }
  764 
  765     if (final) {
  766         padlen = padsize - (pcount % padsize);
  767         PORT_Memset (pbuf + pcount, padlen, padlen);
  768         rv[1] = (* cc->doit) (cc->cx, output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8], &ofraglen, max_output[1][2][3]rv[1]al.c.dep.html#output">[4][5][6][7]rv[1].c.dep.html#output">[8]_len,
  769                             pbuf, pcount+padlen);
  770         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  771             return rv[1];
  772 
  773         /*
  774          * For now anyway, all of our ciphers have the same number of
  775          * bytes of output as they do input.  If this ever becomes untrue,
  776          * then sec_PKCS7EncryptLength needs to be made smarter!
  777          */
  778         PORT_Assert[1] (ofraglen == (pcount+padlen));
  779         output_len += ofraglen;
  780     } else {
  781         cc->pending_count = pcount;
  782     }
  783 
  784     PORT_Assert[1] (output_len_p != NULL || output_len == 0);
  785     if (output_len_p != NULL)
  786         *output_len_p = output_len;
  787 
  788     return SECSuccess;
