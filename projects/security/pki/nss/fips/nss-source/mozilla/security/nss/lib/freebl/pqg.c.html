   74 /* Get a seed for generating P and Q.  If in testing mode, copy in the
   75 ** seed from FIPS 186-1 appendix 5.  Otherwise, obtain bytes from the
   76 ** global random number generator.
   77 */
   78 static SECStatus
   79 getPQseed(SECItem *seed, PRArenaPool* arena)
   80 {
   81     SECStatus rv;
   82 
   83     if (!seed->data) {
   84         seed->data = (unsigned char*)PORT_ArenaZAlloc(arena, seed->len);
   85     }
   86     if (!seed->data) {
   87         PORT_SetError(SEC_ERROR_NO_MEMORY);
   88         return SECFailure;
   89     }
   90 #ifdef FIPS_186_1_A5_TEST
   91     memcpy(seed->data, fips_186_1_a5_pqseed, seed->len);
   92     return SECSuccess;
   93 #else
   94     rv = RNG_GenerateGlobalRandomBytes(seed->data, seed->len);
   95     /*
   96      * NIST CMVP disallows a sequence of 20 bytes with the most
   97      * significant byte equal to 0.  Perhaps they interpret
   98      * "a sequence of at least 160 bits" as "a number >= 2^159".
   99      * So we always set the most significant bit to 1. (bug 334533)
  100      */
  101     seed->data[0] |= 0x80;
  102     return rv;
  103 #endif
  104 }

  131 /* Compute SHA[(SEED + addend) mod 2**g]
  132 ** Result is placed in shaOutBuf.
  133 ** This computation is used in steps 2 and 7 of FIPS 186 Appendix 2.2 .
  134 */
  135 static SECStatus
  136 addToSeedThenSHA(const SECItem * seed,
  137                  unsigned long   addend,
  138                  int             g,
  139                  unsigned char * shaOutBuf)
  140 {
  141     SECItem str = { 0, 0, 0 };
  142     mp_int s, sum, modulus, tmp[1];
  143     mp_err    err = MP_OKAY;
  144     SECStatus rv  = SECSuccess;
  145     MP_DIGITS(&s)       = 0;
  146     MP_DIGITS(&sum)     = 0;
  147     MP_DIGITS(&modulus) = 0;
  148     MP_DIGITS(&tmp[1])     = 0;
  149     CHECK_MPI_OK( mp_init(&s) );
  150     CHECK_MPI_OK( mp_init(&sum) );
  151     CHECK_MPI_OK( mp_init(&modulus) );
  152     SECITEM_TO_MPINT(*seed, &s); /* s = seed */
  153     /* seed += addend */
  154     if (addend < MP_DIGIT_MAX) {
  155         CHECK_MPI_OK( mp_add_d(&s, (mp_digit)addend, &s) );
  156     } else {
  157         CHECK_MPI_OK( mp_init(&tmp[1]) );
  158         CHECK_MPI_OK( mp_set_ulong(&tmp[1], addend) );
  159         CHECK_MPI_OK( mp_add(&s, &tmp[1], &s) );
  160     }
  161     CHECK_MPI_OK( mp_div_2d(&s, (mp_digit)g, NULL, &sum) );/*sum = s mod 2**g */
  162     MPINT_TO_SECITEM(&sum, &str, NULL);
  163     rv = SHA1_HashBuf(shaOutBuf, str.data, str.len); /* SHA1 hash result */
  164 cleanup:
  165     mp_clear(&s);
  166     mp_clear(&sum);
  167     mp_clear(&modulus);
  168     mp_clear(&tmp[1]);
  169     if (str.data)
  170         SECITEM_ZfreeItem(&str, PR_FALSE);
  171     if (err) {
  172         MP_TO_SEC_ERROR(err);
  173         return SECFailure;
  174     }
  175     return rv;
  176 }

  178 /*
  179 **  Perform steps 2 and 3 of FIPS 186, appendix 2.2.
  180 **  Generate Q from seed.
  181 */
  182 static SECStatus
  183 makeQfromSeed(
  184       unsigned int  g,          /* input.  Length of seed in bits. */
  185 const SECItem   *   seed,       /* input.  */
  186       mp_int    *   Q)          /* output. */
  187 {
  188     unsigned char sha1[SHA1_LENGTH];
  189     unsigned char sha2[SHA1_LENGTH];
  190     unsigned char U[SHA1_LENGTH];
  191     SECStatus rv  = SECSuccess;
  192     mp_err    err = MP_OKAY;
  193     int i;
  194     /* ******************************************************************
  195     ** Step 2.
  196     ** "Compute U = SHA[SEED] XOR SHA[(SEED+1) mod 2**g]."
  197     **/
  198     CHECK_SEC_OK[1]( SHA1_HashBuf(sha1, seed->data, seed->len) );
  199     CHECK_SEC_OK[1]( addToSeedThenSHA(seed, 1, g, sha2) );
  200     for (i=0; i<SHA1_LENGTH; ++i) 
  201         U[i] = il/secoid.c.dep.html#sha1">sha1[i] ^ sha2[i];
  202     /* ******************************************************************
  203     ** Step 3.
  204     ** "Form Q from U by setting the most signficant bit (the 2**159 bit)
  205     **  and the least signficant bit to 1.  In terms of boolean operations,
  206     **  Q = U OR 2**159 OR 1.  Note that 2**159 < Q < 2**160."
  207     */
  208     U[0]             |= 0x80;  /* U is MSB first */
  209     U[SHA1_LENGTH-1] |= 0x01;
  210     err = mp_read_unsigned_octets(Q, U, SHA1_LENGTH);
  211 cleanup:
  212      memset(U, 0, SHA1_LENGTH);
  213      memset(sha1, 0, SHA1_LENGTH);
  214      memset(sha2, 0, SHA1_LENGTH);
  215      if (err) {
  216         MP_TO_SEC_ERROR(err);
  217         return SECFailure;
  218      }
  219      return rv;
  220 }

  222 /*  Perform steps 7, 8 and 9 of FIPS 186, appendix 2.2.
  223 **  Generate P from Q, seed, L, and offset.
  224 */
  225 static SECStatus
  226 makePfromQandSeed(
  227       unsigned int  L,          /* Length of P in bits.  Per FIPS 186. */
  228       unsigned int  offset,     /* Per FIPS 186, appendix 2.2. */
  229       unsigned int  g,          /* input.  Length of seed in bits. */
  230 const SECItem   *   seed,       /* input.  */
  231 const mp_int    *   Q,          /* input.  */
  232       mp_int    *   P)          /* output. */
  233 {
  234     unsigned int  k;            /* Per FIPS 186, appendix 2.2. */
  235     unsigned int  n;            /* Per FIPS 186, appendix 2.2. */
  236     mp_digit      b;            /* Per FIPS 186, appendix 2.2. */
  237     unsigned char V_k[SHA1_LENGTH];
  238     mp_int        W[1], X, c, twoQ, V_n, tmp[1];
  239     mp_err    err = MP_OKAY;
  240     SECStatus rv  = SECSuccess;
  241     /* Initialize bignums */
  242     MP_DIGITS(&W[1])     = 0;
  243     MP_DIGITS(&X)     = 0;
  244     MP_DIGITS(&c)     = 0;
  245     MP_DIGITS(&twoQ)  = 0;
  246     MP_DIGITS(&V_n)   = 0;
  247     MP_DIGITS(&tmp[1])   = 0;
  248     CHECK_MPI_OK( mp_init(&W[1])    );
  249     CHECK_MPI_OK( mp_init(&X)    );
  250     CHECK_MPI_OK( mp_init(&c)    );
  251     CHECK_MPI_OK( mp_init(&twoQ) );
  252     CHECK_MPI_OK( mp_init(&tmp[1])  );
  253     CHECK_MPI_OK( mp_init(&V_n)  );
  254     /* L - 1 = n*160 + b */
  255     n = (L - 1) / BITS_IN_Q;
  256     b = (L - 1) % BITS_IN_Q;
  257     /* ******************************************************************
  258     ** Step 7.
  259     **  "for k = 0 ... n let
  260     **           V_k = SHA[(SEED + offset + k) mod 2**g]."
  261     **
  262     ** Step 8.
  263     **  "Let W be the integer 
  264     **    W = V_0 + (V_1 * 2**160) + ... + (V_n-1 * 2**((n-1)*160)) 
  265     **         + ((V_n mod 2**b) * 2**(n*160))
  266     */
  267     for (k=0; k<n; ++k) { /* Do the first n terms of V_k */
  268         /* Do step 7 for iteration k.
  269         ** V_k = SHA[(seed + offset + k) mod 2**g]
  270         */
  271         CHECK_SEC_OK[1]( addToSeedThenSHA(seed, offset + k, g, V_k) );
  272         /* Do step 8 for iteration k.
  273         ** W += V_k * 2**(k*160)
  274         */
  275         OCTETS_TO_MPINT(V_k, &tmp[1], SHA1_LENGTH);      /* get bignum V_k     */
  276         CHECK_MPI_OK( mpl_lsh(&tmp[1], &tmp[1], k*160) );   /* tmp[1] = V_k << k*160 */
  277         CHECK_MPI_OK( mp_add(&W[1], &tmp[1], &W[1]) );         /* W[1] += tmp[1]           */
  278     }
  279     /* Step 8, continued.
  280     **   [W += ((V_n mod 2**b) * 2**(n*160))] 
  281     */
  282     CHECK_SEC_OK[1]( addToSeedThenSHA(seed, offset + n, g, V_k) );
  283     OCTETS_TO_MPINT(V_k, &V_n, SHA1_LENGTH);          /* get bignum V_n     */
  284     CHECK_MPI_OK( mp_div_2d(&V_n, b, NULL, &tmp[1]) );   /* tmp[1] = V_n mod 2**b */
  285     CHECK_MPI_OK( mpl_lsh(&tmp[1], &tmp[1], n*160) );       /* tmp[1] = tmp[1] << n*160 */
  286     CHECK_MPI_OK( mp_add(&W[1], &tmp[1], &W[1]) );             /* W[1] += tmp[1]           */
  287     /* Step 8, continued.
  288     ** "and let X = W + 2**(L-1).
  289     **  Note that 0 <= W < 2**(L-1) and hence 2**(L-1) <= X < 2**L."
  290     */
  291     CHECK_MPI_OK( mpl_set_bit(&X, (mp_size)(L-1), 1) );    /* X = 2**(L-1) */
  292     CHECK_MPI_OK( mp_add(&X, &W[1], &X) );                    /* X += W[1]       */
  293     /*************************************************************
  294     ** Step 9.
  295     ** "Let c = X mod 2q  and set p = X - (c - 1).
  296     **  Note that p is congruent to 1 mod 2q."
  297     */
  298     CHECK_MPI_OK( mp_mul_2(Q, &twoQ) );                    /* 2q           */
  299     CHECK_MPI_OK( mp_mod(&X, &twoQ, &c) );                 /* c = X mod 2q */
  300     CHECK_MPI_OK( mp_sub_d(&c, 1, &c) );                   /* c -= 1       */
  301     PI_OK">CHECK_MPI_OK( mp_sub(&X, &c, P) );                     /* P = X - c    */
  302 cleanup:
  303     mp_clear(&W[1]);
  304     mp_clear(&X);
  305     mp_clear(&c);
  306     mp_clear(&twoQ);
  307     mp_clear(&V_n);
  308     mp_clear(&tmp[1]);
  309     if (err) {
  310         MP_TO_SEC_ERROR(err);
  311         return SECFailure;
  312     }
  313     return rv;
  314 }

  316 /*
  317 ** Generate G from h, P, and Q.
  318 */
  319 static SECStatus
  320 makeGfromH(const mp_int *P,     /* input.  */
  321            const mp_int *Q,     /* input.  */
  322                  mp_int *H[1],     /* input and output. */
  323                  mp_int *G,     /* output. */
  324                  PRBool *passed)
  325 {
  326     mp_int exp, pm1;
  327     mp_err err = MP_OKAY;
  328     SECStatus rv = SECSuccess;
  329     *passed = PR_FALSE;
  330     MP_DIGITS(&exp) = 0;
  331     MP_DIGITS(&pm1) = 0;
  332     CHECK_MPI_OK( mp_init(&exp) );
  333     CHECK_MPI_OK( mp_init(&pm1) );
  334     PI_OK">CHECK_MPI_OK( mp_sub_d(P, 1, &pm1) );        /* P - 1            */
  335     if ( mp_cmp(H[1], &pm1) >= 0)                   /* H[1] >= P-1         */
  336         CH[1]ECK_MPI_OK( mp_sub(H[1], &pm1, H[1]) );      /* H[1] = H[1] mod (P-1)  */
  337     /* Let b = 2**n (smallest power of 2 greater than P).
  338     ** Since P-1 >= b/2, and H < b, quotient(H/(P-1)) = 0 or 1
  339     ** so the above operation safely computes H mod (P-1)
  340     */
  341     /* Check for H = to 0 or 1.  Regen H if so.  (Regen means return error). */
  342     if (mp_cmp_d(H[1], 1) <= 0) {
  343         rv = SECFailure;
  344         goto cleanup;
  345     }
  346     /* Compute G, according to the equation  G = (H ** ((P-1)/Q)) mod P */
  347     CHECK_MPI_OK( mp_div(&pm1, Q, &exp, NULL) );  /* exp = (P-1)/Q      */
  348     CH[1]ECK_MPI_OK( mp_exptmod(H[1], &exp, P, G) );    /* G = H[1] ** exp mod P */
  349     /* Check for G == 0 or G == 1, return error if so. */
  350     if (mp_cmp_d(G, 1) <= 0) {
  351         rv = SECFailure;
  352         goto cleanup;
  353     }
  354     *passed = PR_TRUE;
  355 cleanup:
  356     mp_clear(&exp);
  357     mp_clear(&pm1);
  358     if (err) {
  359         MP_TO_SEC_ERROR(err);
  360         rv = SECFailure;
  361     }
  362     return rv;
  363 }

  380 /* This code uses labels and gotos, so that it can follow the numbered
  381 ** steps in the algorithms from FIPS 186 appendix 2.2 very closely,
  382 ** and so that the correctness of this code can be easily verified.
  383 ** So, please forgive the ugly c code.
  384 **/
  385 SECStatus
  386 PQG_ParamGenSeedLen(unsigned int j, unsigned int seedBytes,
  387                     PQGParams **pParams, PQGVerify **pVfy)
  388 {
  389     unsigned int  L;        /* Length of P in bits.  Per FIPS 186. */
  390     unsigned int  n;        /* Per FIPS 186, appendix 2.2. */
  391     unsigned int  b;        /* Per FIPS 186, appendix 2.2. */
  392     unsigned int  g;        /* Per FIPS 186, appendix 2.2. */
  393     unsigned int  counter;  /* Per FIPS 186, appendix 2.2. */
  394     unsigned int  offset;   /* Per FIPS 186, appendix 2.2. */
  395     SECItem      *seed;     /* Per FIPS 186, appendix 2.2. */
  396     PRArenaPool  *arena  = NULL;
  397     PQGParams    *params = NULL;
  398     PQGVerify    *verify = NULL;
  399     PRBool passed;
  400     SECItem hit = { 0, 0, 0 };
  401     mp_int P, Q, G, H[1], l;
  402     mp_err    err = MP_OKAY;
  403     SECStatus rv  = SECFailure;
  404     int iterations = 0;
  405     if (j > 8 || seedBytes < 20 || !pParams || !pVfy) {
  406         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  407         return SECFailure;
  408     }

  433         PORT_FreeArena(params->arena, PR_TRUE);

  443     MP_DIGITS(&H[1]) = 0;

  445     PI_OK">CHECK_MPI_OK( mp_init(&P) );
  446     CHECK_MPI_OK( mp_init(&Q) );
  447     CHECK_MPI_OK( mp_init(&G) );
  448     CH[1]ECK_MPI_OK( mp_init(&H[1]) );
  449     CHECK_MPI_OK( mp_init(&l) );
  450     /* Compute lengths. */
  451     L = 512 + (j * 64);               /* bits in P */
  452     n = (L - 1) / BITS_IN_Q;          /* BITS_IN_Q is 160 */  
  453     b = (L - 1) % BITS_IN_Q;
  454     g = seedBytes * BITS_PER_BYTE;    /* bits in seed, NOT G of PQG. */
  455 step_1:
  456     /* ******************************************************************
  457     ** Step 1.
  458     ** "Choose an abitrary sequence of at least 160 bits and call it SEED.
  459     **  Let g be the length of SEED in bits."
  460     */
  461     if (++iterations > MAX_ITERATIONS) {        /* give up after a while */
  462         PORT_SetError(SEC_ERROR_NEED_RANDOM);
  463         goto cleanup;

  467     /* ******************************************************************
  468     ** Step 2.
  469     ** "Compute U = SHA[SEED] XOR SHA[(SEED+1) mod 2**g]."
  470     **
  471     ** Step 3.
  472     ** "Form Q from U by setting the most signficant bit (the 2**159 bit) 
  473     **  and the least signficant bit to 1.  In terms of boolean operations,
  474     **  Q = U OR 2**159 OR 1.  Note that 2**159 < Q < 2**160."
  475     */
  476     CHECK_SEC_OK[1]( makeQfromSeed(g, seed, &Q) );
  477     /* ******************************************************************
  478     ** Step 4.
  479     ** "Use a robust primality testing algorithm to test whether q is prime."
  480     **
  481     ** Appendix 2.1 states that a Rabin test with at least 50 iterations

  483     */

  489     */
  490     if (passed != SECSuccess)
  491         goto step_1;
  492     /* ******************************************************************
  493     ** Step 6. "Let counter = 0 and offset = 2."
  494     */
  495     counter = 0;
  496     offset  = 2;
  497 step_7:
  498     /* ******************************************************************
  499     ** Step 7.
  500     ** "for k = 0 ... n let
  501     **          V_k = SHA[(SEED + offset + k) mod 2**g]."
  502     **
  503     ** Step 8.
  504     ** "Let W be the sum of  (V_k * 2**(k*160)) for k = 0 ... n
  505     **  and let X = W + 2**(L-1).
  506     **  Note that 0 <= W < 2**(L-1) and hence 2**(L-1) <= X < 2**L."
  507     **
  508     ** Step 9.
  509     ** "Let c = X mod 2q  and set p = X - (c - 1).
  510     **  Note that p is congruent to 1 mod 2q."
  511     */

  513     /*************************************************************
  514     ** Step 10.
  515     ** "if p < 2**(L-1), then goto step 13."
  516     */
  517     CHECK_MPI_OK( mpl_set_bit(&l, (mp_size)(L-1), 1) ); /* l = 2**(L-1) */
  518     if (mp_cmp(&P, &l) < 0)
  519         goto step_13;
  520     /************************************************************
  521     ** Step 11.
  522     ** "Perform a robust primality test on p."
  523     */
  524     /*CHECK_SEC_OK( prm_RabinTest(&P, &passed) );*/
  525     err = mpp_pprime(&P, PQG_P_PRIMALITY_TESTS);
  526     passed = (err == MP_YES) ? SECSuccess : SECFailure;
  527     /* ******************************************************************
  528     ** Step 12. "If p passes the test performed in step 11, go to step 15."
  529     */
  530     if (passed == SECSuccess)
  531         goto step_15;
  532 step_13:
  533     /* ******************************************************************
  534     ** Step 13.  "Let counter = counter + 1 and offset = offset + n + 1."
  535     */
  536     counter++;
  537     offset += n + 1;
  538     /* ******************************************************************
  539     ** Step 14.  "If counter >= 4096 goto step 1, otherwise go to step 7."
  540     */
  541     if (counter >= 4096)
  542         goto step_1;
  543     goto step_7;
  544 step_15:
  545     /* ******************************************************************
  546     ** Step 15.  
  547     ** "Save the value of SEED and the value of counter for use
  548     **  in certifying the proper generation of p and q."
  549     */
  550     /* Generate h. */
  551     SECITEM_AllocItem(NULL, &hit, L/8); /* h is no longer than p */
  552     if (!hit.data) goto cleanup;
  553     do {
  554         /* loop generate h until 1<h<p-1 and (h**[(p-1)/q])mod p > 1 */
  555         CH[1]ECK_SEC_OK( generate_h_candidate(&hit, &H[1]) );
  556         CH[1]ECK_SEC_OK( GfromH[1]">makeGfromH[1](&P, &Q, &H[1], &G, &passed) );
  557     } while (passed != PR_TRUE);
  558     /* All generation is done.  Now, save the PQG params.  */
  559     PINT_TO_SECITEM">MPINT_TO_SECITEM(&P, &params->prime,    params->arena);
  560     MPINT_TO_SECITEM(&Q, &params->subPrime, params->arena);
  561     MPINT_TO_SECITEM(&G, &params->base,     params->arena);
  562     MPINT_TO_SECITEM(&H[1], &verify->h,        verify->arena);
  563     verify->counter = counter;
  564     *pParams = params;
  565     *pVfy = verify;
  566 cleanup:
  567     mp_clear(&P);
  568     mp_clear(&Q);
  569     mp_clear(&G);
  570     mp_clear(&H[1]);
  571     mp_clear(&l);
  572     if (err) {
  573         MP_TO_SEC_ERROR(err);
  574         rv = SECFailure;
  575     }
  576     if (rv) {
  577         PORT_FreeArena(params->arena, PR_TRUE);
  578         PORT_FreeArena(verify->arena, PR_TRUE);
  579     }
  580     if (hit.data) {
  581         SECITEM_FreeItem(&hit, PR_FALSE);
  582     }
  583     return rv;
  584 }
