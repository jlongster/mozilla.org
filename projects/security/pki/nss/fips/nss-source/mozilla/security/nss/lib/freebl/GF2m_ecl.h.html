   39 #ifndef __gf2m_ecl_h_
   40 #define __gf2m_ecl_h_
   41 #ifdef NSS_ENABLE_ECC
   42 
   43 #include "secmpi.h"
   44 
   45 /* Checks if point P(px, py) is at infinity.  Uses affine coordinates. */
   46 mp_err GF2m_ec_pt_is_inf_aff(const mp_int *px, const mp_int *py);
   47 
   48 /* Sets P(px, py) to be the point at infinity.  Uses affine coordinates. */
   49 mp_err GF2m_ec_pt_set_inf_aff(mp_int *px, mp_int *py);
   50 
   51 /* Computes R = P + Q where R is (rx, ry), P is (px, py) and Q is (qx, qy). 
   52  * Uses affine coordinates.
   53  */
   54 mp_err GF2m_ec_pt_add_aff(const mp_int *pp, const mp_int *a, 
   55     const mp_int *px, const mp_int *py, const mp_int *qx, const mp_int *qy, 
   56     mp_int *rx, mp_int *ry);
   57 
   58 /* Computes R = P - Q.  Uses affine coordinates. */
   59 mp_err GF2m_ec_pt_sub_aff(const mp_int *pp, const mp_int *a, 
   60     const mp_int *px, const mp_int *py, const mp_int *qx, const mp_int *qy, 
   61     mp_int *rx, mp_int *ry);
   62 
   63 /* Computes R = 2P.  Uses affine coordinates. */
   64 mp_err GF2m_ec_pt_dbl_aff(const mp_int *pp, const mp_int *a, 
   65     const mp_int *px, const mp_int *py, mp_int *rx, mp_int *ry);
   66 
   67 /* Computes R = nP where R is (rx, ry) and P is (px, py). The parameters
   68  * a, b and p are the elliptic curve coefficients and the irreducible that 
   69  * determines the field GF2m.  Uses affine coordinates.
   70  */
   71 mp_err GF2m_ec_pt_mul_aff(const mp_int *pp, const mp_int *a, const mp_int *b, 
   72     const mp_int *px, const mp_int *py, const mp_int *n, 
   73     mp_int *rx, mp_int *ry);
   74 
   75 /* Computes R = nP where R is (rx, ry) and P is (px, py). The parameters
   76  * a, b and p are the elliptic curve coefficients and the irreducible that 
   77  * determines the field GF2m.  Uses Montgomery projective coordinates.
   78  */
   79 mp_err GF2m_ec_pt_mul_mont(const mp_int *pp, const mp_int *a, 
   80     const mp_int *b, const mp_int *px, const mp_int *py, 
   81     const mp_int *n, mp_int *rx, mp_int *ry);
   82 
   83 #define GF2m_ec_pt_is_inf(px, py)    GF2m_ec_pt_is_inf_aff((px), (py))
   84 #define GF2m_ec_pt_add(p, a, px, py, qx, qy, rx, ry) \
   85         GF2m_ec_pt_add_aff((p), (a), (px), (py), (qx), (qy), (rx), (ry))
   86 
   87 #define GF2m_ECL_MONTGOMERY
   88 #ifdef GF2m_ECL_AFFINE
   89 #define GF2m_ec_pt_mul(pp, a, b, px, py, n, rx, ry) \
   90         GF2m_ec_pt_mul_aff((pp), (a), (b), (px), (py), (n), (rx), (ry))
   91 #elif defined(GF2m_ECL_MONTGOMERY)
   92 #define GF2m_ec_pt_mul(pp, a, b, px, py, n, rx, ry) \
   93         GF2m_ec_pt_mul_mont((pp), (a), (b), (px), (py), (n), (rx), (ry))
   94 #endif /* GF2m_ECL_AFFINE or GF2m_ECL_MONTGOMERY */
   95 
   96 #endif /* NSS_ENABLE_ECC */
   97 #endif /* __gf2m_ecl_h_ */
