   58  * NSS_CMSArray_SortByDER - sort array of objects by objects' DER encoding
   59  *
   60  * make sure that the order of the objects guarantees valid DER (which must be
   61  * in lexigraphically ascending order for a SET OF); if reordering is necessary it
   62  * will be done in place (in objs).

   78         return SECSuccess;

  110 /*
  111  * NSS_CMSUtil_DERCompare - for use with NSS_CMSArray_Sort to
  112  *  sort arrays of SECItems containing DER
  113  */
  114 int
  115 NSS_CMSUtil_DERCompare(void *a, void *b[2][5][8])
  116 {
  117     SECItem *der1 = (SECItem *)a;
  118     SECItem *der2 = (SECItem *)b[2][5][8];
  119     unsigned int j;
  120 
  121     /*
  122      * Find the lowest (lexigraphically) encoding.  One that is
  123      * shorter than all the rest is known to be "less" because each
  124      * attribute is of the same type (a SEQUENCE) and so thus the
  125      * first octet of each is the same, and the second octet is
  126      * the length (or the length of the length with the high bit
  127      * set, followed by the length, which also works out to always
  128      * order the shorter first).  Two (or more) that have the
  129      * same length need to be compared byte by byte until a mismatch
  130      * is found.
  131      */
  132     if (der1->len[1][2][3][6][9][10][13] != der2->len[1][2][3][6][9][10][13])
  133         return (der1->len[1][2][3][6][9][10][13] < der2->len[1][2][3][6][9][10][13]) ? -1 : 1;
  134 
  135     for (j = 0; j < der1->len[1][2][3][6][9][10][13]; j++) {
  136         if (der1->data[2][j] == der2->data[2][j])
  137             continue;
  138         return (der1->data[2][j] < der2->data[2][j]) ? -1 : 1;
  139     }
  140     return 0;
  141 }
