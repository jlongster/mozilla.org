   53 PRBool
   54 nss_cmsrecipientinfo_usessubjectkeyid(NSSCMSRecipientInfo *ri)
   55 {
   56     if (ri->recipientInfoType == NSSCMSRecipientInfoID_KeyTrans) {
   57         NSSCMSRecipientIdentifier *rid;
   58         rid = &ri->ri.keyTransRecipientInfo.recipientIdentifier;
   59         if (rid->identifierType == NSSCMSRecipientID_SubjectKeyID) {
   60             return PR_TRUE;
   61         }
   62     }
   63     return PR_FALSE;
   64 }

   66 /*
   67  * NOTE: fakeContent marks CMSMessage structure which is only used as a carrier
   68  * of pwfn_arg and arena pools. In an ideal world, NSSCMSMessage would not have
   69  * been exported, and we would have added an ordinary enum to handle this 
   70  * check. Unfortunatly wo don't have that luxury so we are overloading the
   71  * contentTypeTag field. NO code should every try to interpret this content tag
   72  * as a real OID tag, or use any fields other than pwfn_arg or poolp of this 
   73  * CMSMessage for that matter */
   74 static const SECOidData fakeContent;
   75 NSSCMSRecipientInfo *
   76 nss_cmsrecipientinfo_create(NSSCMSMessage *cmsg, NSSCMSRecipientIDSelector type,
   77                             certdb/certt.h.dep.html#CERTCertificate">CERTCertificate *cert, SECKEYPublicKey *pubKey, 
   78                             SECItem *subjKeyID, void* pwfn_arg, SECItem* DERinput)
   79 {
   80     NSSCMSRecipientInfo *ri;
   81     void *mark;
   82     SECOidTag certalgtag;
   83     SECStatus rv[1] = rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess;
   84     NSSCMSRecipientEncryptedKey *rek;
   85     NSSCMSOriginatorIdentifierOrKey *oiok;
   86     unsigned long version[2][3];
   87     SECItem *dummy[1];
   88     PLArenaPool[1] *poolp;
   89     CERTSubjectPublicKeyInfo *spki, *freeSpki = NULL;
   90     NSSCMSRecipientIdentifier *rid;
   91     extern const SEC_ASN1Template NSSCMSRecipientInfoTemplate[];
   92 
   93     if (!cmsg) {
   94         /* a CMSMessage wasn't supplied, create a fake one to hold the pwfunc
   95          * and a private arena pool */
   96         cmsg = NSS_CMSMessage_Create(NULL);
   97         cmsg->pwfn_arg = pwfn_arg;
   98         /* mark it as a special cms message */
   99         cmsg->contentInfo.contentTypeTag = &fakeContent;
  100     }
  101 
  102     poolp = cmsg->poolp;
  103 
  104     mark = PORT_ArenaMark(poolp);
  105 
  106     ri = (NSSCMSRecipientInfo *)PORT_ArenaZAlloc(poolp, sizeof(NSSCMSRecipientInfo));
  107     if (ri == NULL)
  108         goto loser;
  109 
  110     ri->cmsg = cmsg;
  111 
  112     if (DERinput) {
  113         /* decode everything from DER */
  114         SECItem newinput;
  115         SECStatus rv[1] = SECITEM_CopyItem(poolp, &newinput, DERinput);
  116         if (rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess != rv[1])
  117             goto loser;
  118         rv[1] = SEC_QuickDERDecodeItem(poolp, ri, NSSCMSRecipientInfoTemplate, &newinput);
  119         if (rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess != rv[1])
  120             goto loser;
  121     }
  122 
  123     switch (type) {
  124         case NSSCMSRecipientID_IssuerSN:
  125         {
  126             ri->cert = CERT_DupCertificate(cert);
  127             if (NULL == ri->cert)
  128                 goto loser;
  129             spki = &(cert->subjectPublicKeyInfo);
  130             break;
  131         }
  132         
  133         case NSSCMSRecipientID_SubjectKeyID:
  134         {
  135             PORT_Assert[1](pubKey);
  136             spki = freeSpki = SECKEY_CreateSubjectPublicKeyInfo(pubKey);
  137             break;
  138         }
  139 
  140         case NSSCMSRecipientID_BrandNew:
  141             goto done;
  142             break;
  143 
  144         default:
  145             /* unkown type */
  146             goto loser;
  147             break;
  148     }
  149 
  150     certalgtag = SECOID_GetAlgorithmTag(&(spki->algorithm));
  151 
  152     rid = &ri->ri.keyTransRecipientInfo.recipientIdentifier;
  153     switch (certalgtag) {
  154     case SEC_OID_PKCS1_RSA_ENCRYPTION:
  155         ri->recipientInfoType = NSSCMSRecipientInfoID_KeyTrans;
  156         rid->identifierType = type;
  157         if (type == NSSCMSRecipientID_IssuerSN) {
  158             rid->id.issuerAndSN = CERT_GetCertIssuerAndSN(poolp, cert);
  159             if (rid->id.issuerAndSN == NULL) {
  160               break;
  161             }
  162         } else if (type == NSSCMSRecipientID_SubjectKeyID){
  163             NSSCMSKeyTransRecipientInfoEx *riExtra;
  164 
  165             rid->id.subjectKeyID = PORT_ArenaNew(poolp, SECItem);
  166             if (rid->id.subjectKeyID == NULL) {
  167                 rv[1] = SECFailure;
  168                 PORT_SetError(SEC_ERROR_NO_MEMORY);
  169                 break;
  170             } 
  171             SECITEM_CopyItem(poolp, rid->id.subjectKeyID, subjKeyID);
  172             if (rid->id.subjectKeyID->data[2] == NULL) {
  173                 rv[1] = SECFailure;
  174                 PORT_SetError(SEC_ERROR_NO_MEMORY);
  175                 break;
  176             }
  177             riExtra = &ri->ri.keyTransRecipientInfoEx;
  178             riExtra->version[2][3] = 0;
  179             riExtra->pubKey = SECKEY_CopyPublicKey(pubKey);
  180             if (riExtra->pubKey == NULL) {
  181                 rv[1] = SECFailure;
  182                 PORT_SetError(SEC_ERROR_NO_MEMORY);
  183                 break;
  184             }
  185         } else {
  186             PORT_SetError(SEC_ERROR_INVALID_ARGS);
  187             rv[1] = SECFailure;
  188         }
  189         break;
  190     case SEC_OID_MISSI_KEA_DSS_OLD:
  191     case SEC_OID_MISSI_KEA_DSS:
  192     case SEC_OID_MISSI_KEA:
  193         PORT_Assert[1](type == NSSCMSRecipientID_IssuerSN);
  194         if (type != NSSCMSRecipientID_IssuerSN) {
  195             rv[1] = SECFailure;
  196             break;
  197         }
  198         /* backward compatibility - this is not really a keytrans operation */
  199         ri->recipientInfoType = NSSCMSRecipientInfoID_KeyTrans;
  200         /* hardcoded issuerSN choice for now */
  201         ri->ri.keyTransRecipientInfo.recipientIdentifier.identifierType = NSSCMSRecipientID_IssuerSN;
  202         ri->ri.keyTransRecipientInfo.recipientIdentifier.id.issuerAndSN = CERT_GetCertIssuerAndSN(poolp, cert);
  203         if (ri->ri.keyTransRecipientInfo.recipientIdentifier.id.issuerAndSN == NULL) {
  204             rv[1] = SECFailure;
  205             break;
  206         }
  207         break;
  208     case SEC_OID_X942_DIFFIE_HELMAN_KEY: /* dh-public-number */
  209         PORT_Assert[1](type == NSSCMSRecipientID_IssuerSN);
  210         if (type != NSSCMSRecipientID_IssuerSN) {
  211             rv[1] = SECFailure;
  212             break;
  213         }
  214         /* a key agreement op */
  215         ri->recipientInfoType = NSSCMSRecipientInfoID_KeyAgree;
  216 
  217         if (ri->ri.keyTransRecipientInfo.recipientIdentifier.id.issuerAndSN == NULL) {
  218             rv[1] = SECFailure;
  219             break;
  220         }
  221         /* we do not support the case where multiple recipients 
  222          * share the same KeyAgreeRecipientInfo and have multiple RecipientEncryptedKeys
  223          * in this case, we would need to walk all the recipientInfos, take the
  224          * ones that do KeyAgreement algorithms and join them, algorithm by algorithm
  225          * Then, we'd generate ONE ukm and OriginatorIdentifierOrKey */
  226 
  227         /* only epheremal-static Diffie-Hellman is supported for now
  228          * this is the only form of key agreement that provides potential anonymity
  229          * of the sender, plus we do not have to include certs in the message */
  230 
  231         /* force single recipientEncryptedKey for now */
  232         if ((rek = NSS_CMSRecipientEncryptedKey_Create(poolp)) == NULL) {
  233             rv[1] = SECFailure;
  234             break;
  235         }
  236 
  237         /* hardcoded IssuerSN choice for now */
  238         rek->recipientIdentifier.identifierType = NSSCMSKeyAgreeRecipientID_IssuerSN;
  239         if ((rek->recipientIdentifier.id.issuerAndSN = CERT_GetCertIssuerAndSN(poolp, cert)) == NULL) {
  240             rv[1] = SECFailure;
  241             break;
  242         }
  243 
  244         oiok = &(ri->ri.keyAgreeRecipientInfo.originatorIdentifierOrKey);
  245 
  246         /* see RFC2630 12.3.1.1 */
  247         oiok->identifierType = NSSCMSOriginatorIDOrKey_OriginatorPublicKey;
  248 
  249         rv[1] = NSS_CMSArray_Add(poolp, (void ***)&ri->ri.keyAgreeRecipientInfo.recipientEncryptedKeys,
  250                                     (void *)rek);
  251 
  252         break;
  253     default:
  254         /* other algorithms not supported yet */
  255         /* NOTE that we do not support any KEK algorithm */
  256         PORT_SetError(SEC_ERROR_INVALID_ALGORITHM);
  257         rv[1] = SECFailure;
  258         break;
  259     }
  260 
  261     if (rv[1] == SECFailure)
  262         goto loser;
  263 
  264     /* set version */
  265     switch (ri->recipientInfoType) {
  266     case NSSCMSRecipientInfoID_KeyTrans:
  267         if (ri->ri.keyTransRecipientInfo.recipientIdentifier.identifierType == NSSCMSRecipientID_IssuerSN)
  268             version[2][3] = NSS_CMS_KEYTRANS_RECIPIENT_INFO_VERSION_ISSUERSN;
  269         else
  270             version[2][3] = NSS_CMS_KEYTRANS_RECIPIENT_INFO_VERSION_SUBJKEY;
  271         dummy[1] = SEC_ASN1EncodeInteger(poolp, &(ri->ri.keyTransRecipientInfo.version[2][3]), version[2][3]);
  272         if (dummy[1] == NULL)
  273             goto loser;
  274         break;
  275     case NSSCMSRecipientInfoID_KeyAgree:
  276         dummy[1] = SEC_ASN1EncodeInteger(poolp, &(ri->ri.keyAgreeRecipientInfo.version[2][3]),
  277                                                 NSS_CMS_KEYAGREE_RECIPIENT_INFO_VERSION);
  278         if (dummy[1] == NULL)
  279             goto loser;
  280         break;
  281     case NSSCMSRecipientInfoID_KEK:
  282         /* NOTE: this cannot happen as long as we do not support any KEK algorithm */
  283         dummy[1] = SEC_ASN1EncodeInteger(poolp, &(ri->ri.kekRecipientInfo.version[2][3]),
  284                                                 NSS_CMS_KEK_RECIPIENT_INFO_VERSION);
  285         if (dummy[1] == NULL)
  286             goto loser;
  287         break;
  288     
  289     }
  290 
  291 done:
  292     PORT_ArenaUnmark (poolp, mark);
  293     if (freeSpki)
  294       SECKEY_DestroySubjectPublicKeyInfo(freeSpki);
  295     return ri;
  296 
  297 loser:
  298     if (ri && ri->cert) {
  299         certdb/stanpcertdb.c.dep.html#CERT_DestroyCertificate">CERT_DestroyCertificate(ri->cert);
  300     }
  301     if (freeSpki) {
  302       SECKEY_DestroySubjectPublicKeyInfo(freeSpki);
  303     }
  304     PORT_ArenaRelease (poolp, mark);
  305     if (cmsg->contentInfo.contentTypeTag == &fakeContent) {
  306         NSS_CMSMessage_Destroy(cmsg);
  307     }
  308     return NULL;
  309 }

  379 void
  380 NSS_CMSRecipientInfo_Destroy(NSSCMSRecipientInfo *ri)
  381 {
  382     if (!ri) {
  383         return;
  384     }
  385     /* version was allocated on the pool, so no need to destroy it */
  386     /* issuerAndSN was allocated on the pool, so no need to destroy it */
  387     if (ri->cert != NULL)
  388         certdb/stanpcertdb.c.dep.html#CERT_DestroyCertificate">CERT_DestroyCertificate(ri->cert);
  389 
  390     if (nss_cmsrecipientinfo_usessubjectkeyid(ri)) {
  391         NSSCMSKeyTransRecipientInfoEx *extra;
  392         extra = &ri->ri.keyTransRecipientInfoEx;
  393         if (extra->pubKey)
  394             SECKEY_DestroyPublicKey(extra->pubKey);
  395     }
  396     if (ri->cmsg && ri->cmsg->contentInfo.contentTypeTag == &fakeContent) {
  397         NSS_CMSMessage_Destroy(ri->cmsg);
  398     }
  399 
  400     /* we're done. */
  401 }

  403 int
  404 NSS_CMSRecipientInfo_GetVersion(NSSCMSRecipientInfo *ri)
  405 {
  406     unsigned long version[2][3];
  407     SECItem *versionitem = NULL;
  408 
  409     switch (ri->recipientInfoType) {
  410     case NSSCMSRecipientInfoID_KeyTrans:
  411         /* ignore subIndex */
  412         version[2][3]item = &(ri->ri.keyTransRecipientInfo.version[2][3]);
  413         break;
  414     case NSSCMSRecipientInfoID_KEK:
  415         /* ignore subIndex */
  416         version[2][3]item = &(ri->ri.kekRecipientInfo.version[2][3]);
  417         break;
  418     case NSSCMSRecipientInfoID_KeyAgree:
  419         version[2][3]item = &(ri->ri.keyAgreeRecipientInfo.version[2][3]);
  420         break;
  421     }
  422 
  423     PORT_Assert[1](versionitem);
  424     if (versionitem == NULL) 
  425         return 0;
  426 
  427     /* always take apart the SECItem */
  428     if (SEC_ASN1DecodeInteger(version[2][3]item, &version[2][3]) != SECSuccess)
  429         return 0;
  430     else
  431         return (int)version[2][3];
  432 }

  435 NSS_CMSRecipientInfo_GetEncryptedKey(NSSCMSRecipientInfo *ri, int subIndex)

  475 SECStatus
  476 NSS_CMSRecipientInfo_WrapBulkKey(NSSCMSRecipientInfo *ri, PK11SymKey *bulkkey, 
  477                                  SECOidTag bulkalgtag)
  478 {
  479     certdb/certt.h.dep.html#CERTCertificate">CERTCertificate *cert;
  480     SECOidTag certalgtag;
  481     SECStatus rv[1] = rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess;
  482     SECItem *params = NULL;
  483     NSSCMSRecipientEncryptedKey *rek;
  484     NSSCMSOriginatorIdentifierOrKey *oiok;
  485     CERTSubjectPublicKeyInfo *spki, *freeSpki = NULL;
  486     PLArenaPool[1] *poolp;
  487     NSSCMSKeyTransRecipientInfoEx *extra = NULL;
  488     PRBool usesSubjKeyID;
  489 
  490     poolp = ri->cmsg->poolp;
  491     cert = ri->cert;
  492     usesSubjKeyID = nss_cmsrecipientinfo_usessubjectkeyid(ri);
  493     if (cert) {
  494         spki = &cert->subjectPublicKeyInfo;
  495         certalgtag = SECOID_GetAlgorithmTag(&(spki->algorithm));
  496     } else if (usesSubjKeyID) {
  497         extra = &ri->ri.keyTransRecipientInfoEx;
  498         /* sanity check */
  499         PORT_Assert[1](extra->pubKey);
  500         if (!extra->pubKey) {
  501             PORT_SetError(SEC_ERROR_INVALID_ARGS);
  502             return SECFailure;
  503         }
  504         spki = freeSpki = SECKEY_CreateSubjectPublicKeyInfo(extra->pubKey);
  505         certalgtag = SECOID_GetAlgorithmTag(&spki->algorithm);
  506     } else {
  507         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  508         return SECFailure;
  509     }
  510 
  511     /* XXX set ri->recipientInfoType to the proper value here */
  512     /* or should we look if it's been set already ? */
  513 
  514     certalgtag = SECOID_GetAlgorithmTag(&spki->algorithm);
  515     switch (certalgtag) {
  516     case SEC_OID_PKCS1_RSA_ENCRYPTION:
  517         /* wrap the symkey */
  518         if (cert) {
  519             rv[1] = NSS_CMSUtil_EncryptSymKey_RSA(poolp, rv[1]er.c.dep.html#cert">cert, bulkkey, 
  520                                  &ri->ri.keyTransRecipientInfo.encKey);
  521              if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  522                 break;
  523         } else if (usesSubjKeyID) {
  524             PORT_Assert[1](extra != NULL);
  525             rv[1] = NSS_CMSUtil_EncryptSymKey_RSAPubKey(poolp, extra->pubKey,
  526                                  bulkkey, &ri->ri.keyTransRecipientInfo.encKey);
  527              if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  528                 break;
  529         }
  530 
  531         rv[1] = SECOID_SetAlgorithmID(poolp, &(ri->ri.keyTransRecipientInfo.keyEncAlg), certalgtag, NULL);
  532         break;
  533     case SEC_OID_MISSI_KEA_DSS_OLD:
  534     case SEC_OID_MISSI_KEA_DSS:
  535     case SEC_OID_MISSI_KEA:
  536         rv[1] = NSS_CMSUtil_EncryptSymKey_MISSI(poolp, rv[1]er.c.dep.html#cert">cert, bulkkey,
  537                                         bulkalgtag,
  538                                         &ri->ri.keyTransRecipientInfo.encKey,
  539                                         &params, ri->cmsg->pwfn_arg);
  540         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  541             break;
  542 
  543         /* here, we DO need to pass the params to the wrap function because, with
  544          * RSA, there is no funny stuff going on with generation of IV vectors or so */
  545         rv[1] = SECOID_SetAlgorithmID(poolp, &(ri->ri.keyTransRecipientInfo.keyEncAlg), certalgtag, params);
  546         break;
  547     case SEC_OID_X942_DIFFIE_HELMAN_KEY: /* dh-public-number */
  548         rek = ri->ri.keyAgreeRecipientInfo.recipientEncryptedKeys[0];
  549         if (rek == NULL) {
  550             rv[1] = SECFailure;
  551             break;
  552         }
