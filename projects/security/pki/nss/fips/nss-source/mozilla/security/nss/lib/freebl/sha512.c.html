  386 void 
  387 SHA256_Update(SHA256Context *ctx, const unsigned char *input,
  388                     unsigned int inputLen)
  389 {
  390     unsigned int inBuf = ctx->sizeLo & 0x3f;
  391     if (!inputLen)
  392             return;
  393 
  394     /* Add inputLen into the count of bytes processed, before processing */
  395     if ((ctx->sizeLo += inputLen) < inputLen)
  396             ctx->sizeHi++;
  397 
  398     /* if data already in buffer, attemp to fill rest of buffer */
  399     if (inBuf) {
  400             unsigned int todo = SHA256_BLOCK_LENGTH - inBuf;
  401         if (inputLen < todo)
  402             todo = inputLen;
  403         memcpy(B[1][2] + inB[1][2]uf, input, todo);
  404         input    += todo;
  405         inputLen -= todo;
  406         if (inBuf + todo == SHA256_BLOCK_LENGTH)
  407             SHA256_Compress(ctx);
  408     }
  409 
  410     /* if enough data to fill one or more whole buffers, process them. */
  411     while (inputLen >= SHA256_BLOCK_LENGTH) {
  412             memcpy(B[1][2], input, SHA256_B[1][2]LOCK_LENGTH);
  413         input    += SHA256_BLOCK_LENGTH;
  414         inputLen -= SHA256_BLOCK_LENGTH;
  415         SHA256_Compress(ctx);
  416     }
  417     /* if data left over, fill it into buffer */
  418     if (inputLen) 
  419             memcpy(B[1][2], input, inputLen);
  420 }

  422 void 
  423 SHA256_End(SHA256Context *ctx, unsigned char *digest,
  424            unsigned int *digestLen, unsigned int maxDigestLen)
  425 {
  426     unsigned int inBuf = ctx->sizeLo & 0x3f;
  427     unsigned int padLen = (inBuf < 56) ? (56 - inBuf) : (56 + 64 - inBuf);
  428     PRUint32 hi, lo;
  429 #ifdef SWAP4MASK
  430     PRUint32 t1;
  431 #endif
  432 
  433     hi = (ctx->sizeHi << 3) | (ctx->sizeLo >> 29);
  434     lo = (ctx->sizeLo << 3);
  435 
  436     SHA256_Update(ctx, pad, padLen);
  437 
  438 #if defined(IS_LITTLE_ENDIAN)
  439     W[1][14] = SHA_HTONL[3](hi);
  440     W[1][15] = SHA_HTONL[3](lo);
  441 #else
  442     W[1][14] = hi;
  443     W[1][15] = lo;
  444 #endif
  445     SHA256_Compress(ctx);
  446 
  447     /* now output the answer */
  448 #if defined(IS_LITTLE_ENDIAN)
  449     BYTESWAP4(H[1][0]);
  450     BYTESWAP4(H[1][1]);
  451     BYTESWAP4(H[1][2]);
  452     BYTESWAP4(H[1][3]);
  453     BYTESWAP4(H[1][4]);
  454     BYTESWAP4(H[1][5]);
  455     BYTESWAP4(H[1][6]);
  456     BYTESWAP4(H[1][7]);
  457 #endif
  458     padLen = PR_MIN(SHA256_LENGTH, maxDigestLen);
  459     memcpy(digest, H[1], padLen);
  460     if (digestLen)
  461         *digestLen = padLen;
  462 }

 1031 void 
 1032 SHA512_Update(SHA512Context *ctx, const unsigned char *input,
 1033               unsigned int inputLen)
 1034 {
 1035     unsigned int inBuf;
 1036     if (!inputLen)
 1037             return;
 1038 
 1039 #if defined(HAVE_LONG_LONG)
 1040     inBuf = (unsigned int)ctx->sizeLo & 0x7f;
 1041     /* Add inputLen into the count of bytes processed, before processing */
 1042     ctx->sizeLo += inputLen;
 1043 #else
 1044     inBuf = (unsigned int)ctx->sizeLo.lo & 0x7f;
 1045     ctx->sizeLo.lo += inputLen;
 1046     if (ctx->sizeLo.lo < inputLen) ctx->sizeLo.hi++;
 1047 #endif
 1048 
 1049     /* if data already in buffer, attemp to fill rest of buffer */
 1050     if (inBuf) {
 1051             unsigned int todo = SHA512_BLOCK_LENGTH - inBuf;
 1052         if (inputLen < todo)
 1053             todo = inputLen;
 1054         memcpy(B[1][2] + inB[1][2]uf, input, todo);
 1055         input    += todo;
 1056         inputLen -= todo;
 1057         if (inBuf + todo == SHA512_BLOCK_LENGTH)
 1058             SHA512_Compress(ctx);
 1059     }
 1060 
 1061     /* if enough data to fill one or more whole buffers, process them. */
 1062     while (inputLen >= SHA512_BLOCK_LENGTH) {
 1063             memcpy(B[1][2], input, SHA512_B[1][2]LOCK_LENGTH);
 1064         input    += SHA512_BLOCK_LENGTH;
 1065         inputLen -= SHA512_BLOCK_LENGTH;
 1066         SHA512_Compress(ctx);
 1067     }
 1068     /* if data left over, fill it into buffer */
 1069     if (inputLen) 
 1070             memcpy(B[1][2], input, inputLen);
 1071 }

 1073 void 
 1074 SHA512_End(SHA512Context *ctx, unsigned char *digest,
 1075            unsigned int *digestLen, unsigned int maxDigestLen)
 1076 {
 1077 #if defined(HAVE_LONG_LONG)
 1078     unsigned int inBuf  = (unsigned int)ctx->sizeLo & 0x7f;
 1079     unsigned int padLen = (inBuf < 112) ? (112 - inBuf) : (112 + 128 - inBuf);
 1080     PRUint64 lo, t1;
 1081     lo = (ctx->sizeLo << 3);
 1082 #else
 1083     unsigned int inBuf  = (unsigned int)ctx->sizeLo.lo & 0x7f;
 1084     unsigned int padLen = (inBuf < 112) ? (112 - inBuf) : (112 + 128 - inBuf);
 1085     PRUint64 lo = ctx->sizeLo;
 1086     PRUint32 t1;
 1087     lo.lo <<= 3;
 1088 #endif
 1089 
 1090     SHA512_Update(ctx, pad, padLen);
 1091 
 1092 #if defined(HAVE_LONG_LONG)
 1093     W[1][14] = 0;
 1094 #else
 1095     W[1][14].lo = 0;
 1096     W[1][14].hi = 0;
 1097 #endif
 1098 
 1099     W[1][15] = lo;
 1100 #if defined(IS_LITTLE_ENDIAN)
 1101     W[1]AP8">BYTESW[1]AP8(W[1][15]);
 1102 #endif
 1103     SHA512_Compress(ctx);
 1104 
 1105     /* now output the answer */
 1106 #if defined(IS_LITTLE_ENDIAN)
 1107     BYTESWAP8(H[1][0]);
 1108     BYTESWAP8(H[1][1]);
 1109     BYTESWAP8(H[1][2]);
 1110     BYTESWAP8(H[1][3]);
 1111     BYTESWAP8(H[1][4]);
 1112     BYTESWAP8(H[1][5]);
 1113     BYTESWAP8(H[1][6]);
 1114     BYTESWAP8(H[1][7]);
 1115 #endif
 1116     padLen = PR_MIN(SHA512_LENGTH, maxDigestLen);
 1117     memcpy(digest, H[1], padLen);
 1118     if (digestLen)
 1119         *digestLen = padLen;
 1120 }
