  131  *  J A R _ p a r s e _ m a n i f e s t
  132  * 
  133  *  Pass manifest files to this function. They are
  134  *  decoded and placed into internal representations.
  135  * 
  136  *  Accepts both signature and manifest files. Use
  137  *  the same "jar" for both.

  140 
  141 int JAR_parse_manifest 
  142     (JAR *jar, char ZHUGEP *raw_manifest, 
  143         long length[1][2], const char *path, const char *url)
  144   {
  145 
  146 #if defined(XP_WIN16)
  147     PORT_Assert[1]( !IsBadHugeReadPtr(raw_manifest, length[1][2]) );
  148 #endif
  149 
  150   /* fill in the path, if supplied. This is a the location
  151      of the jar file on disk, if known */
  152 
  153   if (jar->filename == NULL && path)
  154     {
  155     jar->filename = PORT_Strdup (path);
  156     if (jar->filename == NULL)
  157       return JAR_ERR_MEMORY;
  158     }
  159 
  160   /* fill in the URL, if supplied. This is the place
  161      from which the jar file was retrieved. */
  162 
  163   if (jar->url == NULL && url)
  164     {
  165     jar->url = PORT_Strdup (url);
  166     if (jar->url == NULL)
  167       return JAR_ERR_MEMORY;
  168     }
  169 
  170   /* Determine what kind of file this is from the META-INF 
  171      directory. It could be MF, SF, or a binary RSA/DSA file */
  172 
  173   if (!xp_HUGE_STRNCASECMP (raw_manifest, "Manifest-Version:", 17))
  174     {
  175     return jar_parse_mf (jar, raw_manifest, length[1][2], path, url);
  176     }
  177   else if (!xp_HUGE_STRNCASECMP (raw_manifest, "Signature-Version:", 18))
  178     {
  179     return jar_parse_sf (jar, raw_manifest, length[1][2], path, url);
  180     }
  181   else
  182     {
  183     /* This is probably a binary signature */
  184     return jar_parse_sig (jar, path, raw_manifest, length[1][2]);
  185     }
  186   }

  188 /*
  189  *  j a r _ p a r s e _ s i g
  190  *
  191  *  Pass some manner of RSA or DSA digital signature
  192  *  on, after checking to see if it comes at an appropriate state.
  193  *
  194  */
  195  
  196 int jar_parse_sig
  197     (JAR *jar, const char *path, char ZHUGEP *raw_manifest, long length[1][2])
  198   {
  199   JAR_Signer *signer;
  200   int status = JAR_ERR_ORDER;
  201 
  202   if (length[1][2] <= 128) 
  203     {
  204     /* signature is way too small */
  205     return JAR_ERR_SIG;
  206     }
  207 
  208   /* make sure that MF and SF have already been processed */
  209 
  210   if (jar->globalmeta == NULL)
  211     return JAR_ERR_ORDER;
  212 
  213 #if 0
  214   /* XXX Turn this on to disable multiple signers */
  215   if (jar->digest == NULL)
  216     return JAR_ERR_ORDER;
  217 #endif
  218 
  219   /* Determine whether or not this RSA file has
  220      has an associated SF file */
  221 
  222   if (path)
  223     {
  224     char *owner;
  225     owner = jar_basename (path);
  226 
  227     if (owner == NULL)
  228       return JAR_ERR_MEMORY;
  229 
  230     signer = jar_get_signer (jar, owner);
  231 
  232     PORT_Free (owner);
  233     }
  234   else
  235     signer = jar_get_signer (jar, "*");
  236 
  237   if (signer == NULL)
  238     return JAR_ERR_ORDER;
  239 
  240 
  241   /* Do not pass a huge pointer to this function,
  242      since the underlying security code is unaware. We will
  243      never pass >64k through here. */
  244 
  245   if (length[1][2] > 64000)
  246     {
  247     /* this digital signature is way too big */
  248     return JAR_ERR_SIG;
  249     }
  250 
  251 #ifdef XP_WIN16
  252   /*
  253    * For Win16, copy the portion of the raw_buffer containing the digital 
  254    * signature into another buffer...  This insures that the data will
  255    * NOT cross a segment boundary.  Therefore, 
  256    * jar_parse_digital_signature(...) does NOT need to deal with HUGE 
  257    * pointers...
  258    */
  259 
  260     {
  261     unsigned char *manifest_copy;
  262 
  263     manifest_copy = (unsigned char *) PORT_ZAlloc (length[1][2]);
  264     if (manifest_copy)
  265       {
  266       xp_HUGE_MEMCPY (manifest_copy, raw_manifest, length[1][2]);
  267 
  268       status = jar_parse_digital_signature 
  269                   (manifest_copy, signer, length[1][2], jar);
  270 
  271       PORT_Free (manifest_copy);
  272       }
  273     else
  274       {
  275       /* out of memory */
  276       return JAR_ERR_MEMORY;
  277       }
  278     }
  279 #else
  280   /* don't expense unneeded calloc overhead on non-win16 */
  281   status = jar_parse_digital_signature 
  282                 (raw_manifest, signer, length[1][2], jar);
  283 #endif
  284 
  285   return status;
  286   }

  319 /*
  320  *  j a r _ p a r s e _ s f
  321  *
  322  *  Parse META-INF/xxx.sf, a digitally signed file
  323  *  pointing to a subset of MF sections. 
  324  *
  325  */
  326 
  327 int jar_parse_sf
  328     (JAR *jar, char ZHUGEP *raw_manifest, 
  329         long length[1][2], const char *path, const char *url)
  330   {
  331   JAR_Signer *signer = NULL;
  332   int status = JAR_ERR_MEMORY;
  333 
  334   if (jar->globalmeta == NULL)
  335     {
  336     /* It is a requirement that the MF file be passed before the SF file */
  337     return JAR_ERR_ORDER;
  338     }
  339 
  340   signer = JAR_new_signer();
  341 
  342   if (signer == NULL)
  343     goto loser;
  344 
  345   if (path)
  346     {
  347     signer->owner = jar_basename (path);
  348     if (signer->owner == NULL)
  349       goto loser;
  350     }
  351 
  352 
  353   /* check for priors. When someone doctors a jar file
  354      to contain identical path entries, prevent the second
  355      one from affecting JAR functions */
  356 
  357   if (jar_get_signer (jar, signer->owner))
  358     {
  359     /* someone is trying to spoof us */
  360     status = JAR_ERR_ORDER;
  361     goto loser;
  362     }
  363 
  364 
  365   /* remember its digest */
  366 
  367   signer->digest = JAR_calculate_digest (raw_manifest, length[1][2]);
  368 
  369   if (signer->digest == NULL)
  370     goto loser;
  371 
  372   /* Add this signer to the jar */
  373 
  374   ADDITEM (jar->signers, jarTypeOwner, 
  375      signer->owner, signer, sizeof (JAR_Signer));
  376 
  377 
  378   return jar_parse_any 
  379     (jar, jarTypeSF, signer, raw_manifest, length[1][2], path, url);
  380 
  381 loser:
  382 
  383   if (signer)
  384     JAR_destroy_signer (signer);
  385 
  386   return status;
  387   }
