  217 /* Affine EC Point. This is the basic representation (x, y) of an elliptic 
  218  * curve point. */
  219 typedef struct {
  220         double x[ECFP_MAXDOUBLES];
  221         double y[ECFP_MAXDOUBLES];
  222 } ecfp_aff_pt;
  223 
  224 /* Jacobian EC Point. This coordinate system uses X = x/z^2, Y = y/z^3,
  225  * which enables calculations with fewer inversions than affine
  226  * coordinates. */
  227 typedef struct {
  228         double x[ECFP_MAXDOUBLES];
  229         double y[ECFP_MAXDOUBLES];
  230         double z[ECFP_MAXDOUBLES];
  231 } ecfp_jac_pt;
  232 
  233 /* Chudnovsky Jacobian EC Point. This coordinate system is the same as
  234  * Jacobian, except it keeps z^2, z^3 for faster additions. */
  235 typedef struct {
  236         double x[ECFP_MAXDOUBLES];
  237         double y[ECFP_MAXDOUBLES];
  238         double z[ECFP_MAXDOUBLES];
  239         double z2[ECFP_MAXDOUBLES];
  240         double z3[ECFP_MAXDOUBLES];
  241 } ecfp_chud_pt;
  242 
  243 /* Modified Jacobian EC Point. This coordinate system is the same as
  244  * Jacobian, except it keeps a*z^4 for faster doublings. */
  245 typedef struct {
  246         double x[ECFP_MAXDOUBLES];
  247         double y[ECFP_MAXDOUBLES];
  248         double z[ECFP_MAXDOUBLES];
  249         double az4[ECFP_MAXDOUBLES];
  250 } ecfp_jm_pt;

  254 typedef struct EC_group_fp_str">EC_group_fp_str EC_group_fp;
  255 struct EC_group_fp_str {
  256         int fpPrecision;                        /* Set to number of bits in mantissa, 53
  257                                                                  * or 64 */
  258         int numDoubles;
  259         int primeBitSize;
  260         int orderBitSize;
  261         int doubleBitSize;
  262         int numInts;
  263         int aIsM3;                                        /* True if curvea == -3 (mod p), then we
  264                                                                  * can optimize doubling */
  265         double curvea[ECFP_MAXDOUBLES];
  266         /* Used to truncate a double to the number of bits in the curve */
  267         double bitSize_alpha;
  268         /* Pointer to either ecfp_alpha_53 or ecfp_alpha_64 */
  269         const double *alpha;
  270 
  271         void (*ecfp_singleReduce) (double *r, const EC_group_fp * group);
  272         void (*ecfp_reduce) (double *r, double *x, const EC_group_fp * group);
  273         /* Performs a "tidy" operation, which performs carrying, moving excess 
  274          * bits from one double to the next double, so that the precision of
  275          * the doubles is reduced to the regular precision ECFP_DSIZE. This
  276          * might result in some float digits being negative. */
  277         void (*ecfp_tidy) (double *t, const double *alpha,
  278                                            const EC_group_fp * group);
  279         /* Perform a point addition using coordinate system Jacobian + Affine
  280          * -> Jacobian. Input and output should be multi-precision floating
  281          * point integers. */
  282         void (*pt_add_jac_aff) (const ecfp_jac_pt * p, const ecfp_aff_pt * q,
  283                                                         ecfp_jac_pt * r, const EC_group_fp * group);
  284         /* Perform a point doubling in Jacobian coordinates. Input and output
  285          * should be multi-precision floating point integers. */
  286         void (*pt_dbl_jac) (const ecfp_jac_pt * dp, ecfp_jac_pt * dr,
  287                                                 const EC_group_fp * group);
  288         /* Perform a point addition using Jacobian coordinate system. Input
  289          * and output should be multi-precision floating point integers. */
  290         void (*pt_add_jac) (const ecfp_jac_pt * p, const ecfp_jac_pt * q,
  291                                                 ecfp_jac_pt * r, const EC_group_fp * group);
  292         /* Perform a point doubling in Modified Jacobian coordinates. Input
  293          * and output should be multi-precision floating point integers. */
  294         void (*pt_dbl_jm) (const ecfp_jm_pt * p, ecfp_jm_pt * r,
  295                                            const EC_group_fp * group);
  296         /* Perform a point doubling using coordinates Affine -> Chudnovsky
  297          * Jacobian. Input and output should be multi-precision floating point 
  298          * integers. */
  299         void (*pt_dbl_aff2chud) (const ecfp_aff_pt * p, ecfp_chud_pt * r,
  300                                                          const EC_group_fp * group);
  301         /* Perform a point addition using coordinates: Modified Jacobian +
  302          * Chudnovsky Jacobian -> Modified Jacobian. Input and output should
  303          * be multi-precision floating point integers. */
  304         void (*pt_add_jm_chud) (ecfp_jm_pt * p, ecfp_chud_pt * q,
  305                                                         ecfp_jm_pt * r, const EC_group_fp * group);
  306         /* Perform a point addition using Chudnovsky Jacobian coordinates.
  307          * Input and output should be multi-precision floating point integers. 
  308          */
  309         void (*pt_add_chud) (const ecfp_chud_pt * p, const ecfp_chud_pt * q,
  310                                                  ecfp_chud_pt * r, const EC_group_fp * group);
  311         /* Expects out to be an array of size 16 of Chudnovsky Jacobian
  312          * points. Fills in Chudnovsky Jacobian form (x, y, z, z^2, z^3), for
  313          * -15P, -13P, -11P, -9P, -7P, -5P, -3P, -P, P, 3P, 5P, 7P, 9P, 11P,

  316                                                          const EC_group_fp * group);
  317         /* Expects out to be an array of size 16 of Jacobian points. Fills in
  318          * Chudnovsky Jacobian form (x, y, z), for O, P, 2P, ... 15P */
  319         void (*precompute_jac) (ecfp_jac_pt * out, const ecfp_aff_pt * p,
  320                                                         const EC_group_fp * group);
  321 
  322 };

  359 /* Computes R = nP where R is (rx, ry) and P is the base point. The
  360  * parameters a, b and p are the elliptic curve coefficients and the prime 
  361  * that determines the field GFp.  Elliptic curve points P and R can be
  362  * identical.  Uses mixed Jacobian-affine coordinates (Jacobian
  363  * coordinates for doubles and affine coordinates for additions; based on
  364  * recommendation from Brown et al.). Uses window NAF method (algorithm
  365  * 11) for scalar-point multiplication from Brown, Hankerson, Lopez,
  366  * Menezes. Software Implementation of the NIST Elliptic Curves Over Prime
  367  * Fields. */
  368 mp_err ec_GFp_point_mul_wNAF_fp(const mp_int *n, const mp_int *px,
  369                                                                 const mp_int *py, mp_int *rx, mp_int *ry,
  370                                                                 const ECGroup *ecgroup);

  372 /* Uses mixed Jacobian-affine coordinates to perform a point
  373  * multiplication: R = n * P, n scalar. Uses mixed Jacobian-affine
  374  * coordinates (Jacobian coordinates for doubles and affine coordinates
  375  * for additions; based on recommendation from Brown et al.). Not very
  376  * time efficient but quite space efficient, no precomputation needed.
  377  * group contains the elliptic curve coefficients and the prime that
  378  * determines the field GFp.  Elliptic curve points P and R can be
  379  * identical. Performs calculations in floating point number format, since 
  380  * this is faster than the integer operations on the ULTRASPARC III.
  381  * Uses left-to-right binary method (double & add) (algorithm 9) for
  382  * scalar-point multiplication from Brown, Hankerson, Lopez, Menezes.
  383  * Software Implementation of the NIST Elliptic Curves Over Prime Fields. */
  384 mp_err
