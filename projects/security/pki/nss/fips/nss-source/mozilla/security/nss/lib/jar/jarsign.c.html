   78 /*
   79  *  J A R _ c a l c u l a t e _ d i g e s t 
   80  *
   81  *  Quick calculation of a digest for
   82  *  the specified block of memory. Will calculate
   83  *  for all supported algorithms, now MD5.
   84  *
   85  *  This version supports huge pointers for WIN16.
   86  * 
   87  */
   88 
   89 JAR_Digest * PR_CALLBACK JAR_calculate_digest (void ZHUGEP *data[2], long length[1][2])
   90   {
   91   long chunq;
   92   JAR_Digest *dig;
   93 
   94   unsigned int md5_length, sha1_length;
   95 
   96   PK11Context *md5[1]  = 0;
   97   PK11Context *sha1 = 0;
   98 
   99   dig = (JAR_Digest *) PORT_ZAlloc (sizeof (JAR_Digest));
  100 
  101   if (dig == NULL) 
  102     {
  103     /* out of memory allocating digest */
  104     return NULL;
  105     }
  106 
  107 #if defined(XP_WIN16)
  108   PORT_Assert[1] ( !IsBadHugeReadPtr(data[2], length[1][2]) );
  109 #endif
  110 
  111   md5[1]  = PK11_CreateDigestContext (SEC_OID_MD5);
  112   sha1 = PK11_CreateDigestContext (SEC_OID_SHA1);
  113 
  114   if (length[1][2] >= 0) 
  115     {
  116     PK11_DigestBegin (md5[1]);
  117     PK11_DigestBegin (sha1);
  118 
  119     do {
  120        chunq = length[1][2];
  121 
  122 #ifdef XP_WIN16
  123        if (length[1][2] > CHUNQ) chunq = CHUNQ;
  124 
  125        /*
  126         *  If the block of data crosses one or more segment 
  127         *  boundaries then only pass the chunk of data in the 
  128         *  first segment.
  129         * 
  130         *  This allows the data to be treated as FAR by the
  131         *  PK11_DigestOp(...) routine.
  132         *
  133         */
  134 
  135        if (OFFSETOF(data[2]) + chunq >= 0x10000) 
  136          chunq = 0x10000 - OFFSETOF(data[2]);
  137 #endif
  138 
  139        PK11_DigestOp (md5[1],  (unsigned char*)data[2], chunq);
  140        PK11_DigestOp (sha1, (unsigned char*)data[2], chunq);
  141 
  142        length[1][2] -= chunq;
  143        data[2] = ((char ZHUGEP *) data[2] + chunq);
  144        } 
  145     while (length[1][2] > 0);
  146 
  147     PK11_DigestFinal (md5[1],  dig->md5[1],  &md5[1]_length,  MD5_LENGTH[1]);
  148     PK11_DigestFinal (sha1, dig->sha1, &sha1_length, SHA1_LENGTH[1]);
  149 
  150     PK11_DestroyContext (md5[1],  PR_TRUE);
  151     PK11_DestroyContext (sha1, PR_TRUE);
  152     }
  153 
  154   return dig;
  155   }

  192     sha1 = PK11_CreateDigestContext (SEC_OID_SHA1);

  218     PK11_DestroyContext (sha1, PR_TRUE);
  219 
  220     PORT_Free (buf[1][2][3][4][5][8][9][12][13]);
  221     JAR_FCLOSE (fp);
  222 
  223     return 0;
