  107 SECStatus
  108 nsslowcert_InitLocks(void)
  109 {
  110     if (freeListLock == NULL) {
  111         nss_InitLock(&freeListLock, nssILockRefLock);
  112         if (freeListLock == NULL) {
  113             return SECFailure;
  114         }
  115     }
  116     if (certRefCountLock == NULL) {
  117         nss_InitLock(&certRefCountLock, nssILockRefLock);
  118         if (certRefCountLock == NULL) {
  119             return SECFailure;
  120         }
  121     }
  122     if (certTrustLock == NULL ) {
  123         nss_InitLock(&certTrustLock, nssILockCertDB);
  124         if (certTrustLock == NULL) {
  125             return SECFailure;
  126         }
  127     }
  128     
  129     return SECSuccess;
  130 }

  431 /*
  432  * destroy a database entry
  433  */
  434 static void
  435 DestroyDBEntry(certDBEntry *entry)
  436 {
  437     PRArenaPool *arena = entry->common.arena;
  438 
  439     /* must be one of our certDBEntry from the free list */
  440     if (arena == NULL) {
  441         certDBEntryCert *certEntry;
  442         if ( entry->common.type != certDBEntryTypeCert) {
  443             return;
  444         }
  445         certEntry = (certDBEntryCert *)entry;
  446 
  447         pkcs11_freeStaticData(certEntry->derCert.data, certEntry->derCertSpace);
  448         pkcs11_freeNickname(certEntry->nickname, certEntry->nicknameSpace);
  449 
  450         nsslowcert_LockFreeList();
  451         if (entryListCount > MAX_ENTRY_LIST_COUNT) {
  452             PORT_Free(certEntry);
  453         } else {
  454             entryListCount++;
  455             PORT_Memset(certEntry, 0, sizeof( *certEntry));
  456             certEntry->next = entryListHead;
  457             entryListHead = certEntry;
  458         }
  459         nsslowcert_UnlockFreeList();
  460         return;
  461     }
  462 
  463 
  464     /* Zero out the entry struct, so that any further attempts to use it
  465      * will cause an exception (e.g. null pointer reference). */
  466     PORT_Memset(&entry->common, 0, sizeof entry->common);
  467     PORT_FreeArena(arena, PR_FALSE);
  468 
  469     return;
  470 }

  475 static SECStatus
  476 DeleteDBEntry(NSSLOWCERTCertDBHandle *handle, certDBEntryType type, SECItem *dbkey)
  477 {
  478     DBT key;
  479     int ret;
  480 
  481     /* init the database key */
  482     key.data = dbkey->data;
  483     key.size = dbkey->len;
  484     
  485     dbkey->data[0] = (unsigned char)type;
  486 
  487     /* delete entry from database */
  488     ret = certdb_Del(handle->permCertDB, &key, 0 );
  489     if ( ret != 0 ) {
  490         PORT_SetError(SEC_ERROR_BAD_DATABASE);
  491         goto loser;
  492     }
  493 
  494     ret = certdb_Sync(handle->permCertDB, 0);
  495     if ( ret ) {
  496         PORT_SetError(SEC_ERROR_BAD_DATABASE);
  497         goto loser;
  498     }
  499 
  500     return(SECSuccess);
  501     
  502 loser:
  503     return(SECFailure);
  504 }

  506 static SECStatus
  507 ReadDBEntry(NSSLOWCERTCertDBHandle *handle, certDBEntryCommon *entry,
  508             SECItem *dbkey, SECItem *dbentry, PRArenaPool *arena)
  509 {
  510     DBT data, key;
  511     int ret;
  512     unsigned char *buf;
  513     
  514     /* init the database key */
  515     key.data = dbkey->data;
  516     key.size = dbkey->len;
  517     
  518     dbkey->data[0] = (unsigned char)entry->type;
  519 
  520     /* read entry from database */
  521     ret = certdb_Get(handle->permCertDB, &key, &data, 0 );
  522     if ( ret != 0 ) {
  523         PORT_SetError(SEC_ERROR_BAD_DATABASE);
  524         goto loser;
  525     }
  526     
  527     /* validate the entry */
  528     if ( data.size < SEC_DB_ENTRY_HEADER_LEN ) {
  529         PORT_SetError(SEC_ERROR_BAD_DATABASE);
  530         goto loser;
  531     }
  532     buf = (unsigned char *)data.data;
  533     /* version 7 has the same schema, we may be using a v7 db if we openned
  534      * the databases readonly. */
  535     if (!((buf[0] == (unsigned char)CERT_DB_FILE_VERSION) 
  536                 || (buf[0] == (unsigned char) CERT_DB_V7_FILE_VERSION))) {
  537         PORT_SetError(SEC_ERROR_BAD_DATABASE);
  538         goto loser;
  539     }
  540     if ( buf[1] != (unsigned char)entry->type ) {
  541         PORT_SetError(SEC_ERROR_BAD_DATABASE);
  542         goto loser;
  543     }
  544 
  545     /* copy out header information */
  546     entry->version = (unsigned int)buf[0];
  547     entry->type = (certDBEntryType)buf[1];
  548     entry->flags = (unsigned int)buf[2];
  549     
  550     /* format body of entry for return to caller */
  551     dbentry->len = data.size - SEC_DB_ENTRY_HEADER_LEN;
  552     if ( dbentry->len ) {
  553         if (arena) {
  554             dbentry->data = (unsigned char *)
  555                                 PORT_ArenaAlloc(arena, dbentry->len);
  556             if ( dbentry->data == NULL ) {
  557                 PORT_SetError(SEC_ERROR_NO_MEMORY);
  558                 goto loser;
  559             }
  560     
  561             PORT_Memcpy(dbentry->data, &buf[SEC_DB_ENTRY_HEADER_LEN],
  562                   dbentry->len);
  563         } else {
  564             dbentry->data = &buf[SEC_DB_ENTRY_HEADER_LEN];
  565         }
  566     } else {
  567         dbentry->data = NULL;
  568     }
  569     
  570     return(SECSuccess);
  571 
  572 loser:
  573     return(SECFailure);
  574 }

  579 static SECStatus
  580 WriteDBEntry(NSSLOWCERTCertDBHandle *handle, certDBEntryCommon *entry,
  581              SECItem *dbkey, SECItem *dbentry)
  582 {
  583     int ret;
  584     DBT data, key;
  585     unsigned char *buf;
  586     
  587     data.data = dbentry->data;
  588     data.size = dbentry->len;
  589     
  590     buf = (unsigned char*)data.data;
  591     
  592     buf[0] = (unsigned char)entry->version;
  593     buf[1] = (unsigned char)entry->type;
  594     buf[2] = (unsigned char)entry->flags;
  595     
  596     key.data = dbkey->data;
  597     key.size = dbkey->len;
  598     
  599     dbkey->data[0] = (unsigned char)entry->type;

  602     ret = certdb_Put(handle->permCertDB, &key, &data, 0);
  603 
  604     if ( ret != 0 ) {
  605         goto loser;
  606     }
  607 
  608     ret = certdb_Sync( handle->permCertDB, 0 );
  609     
  610     if ( ret ) {
  611         goto loser;
  612     }
  613 
  614     return(SECSuccess);
  615 
  616 loser:
  617     return(SECFailure);
  618 }

  620 /*
  621  * encode a database cert record
  622  */
  623 static SECStatus
  624 EncodeDBCertEntry(certDBEntryCert *entry, PRArenaPool *arena, SECItem *dbitem)
  625 {
  626     unsigned int nnlen;
  627     unsigned char *buf;
  628     char *nn;
  629     char zbuf = 0;
  630     
  631     if ( entry->nickname ) {
  632         nn = entry->nickname;
  633     } else {
  634         nn = &zbuf;
  635     }
  636     nnlen = PORT_Strlen(nn) + 1;
  637     
  638     /* allocate space for encoded database record, including space
  639      * for low level header
  640      */
  641     dbitem->len = entry->derCert.len + nnlen + DB_CERT_ENTRY_HEADER_LEN +
  642         SEC_DB_ENTRY_HEADER_LEN;
  643     
  644     dbitem->data = (unsigned char *)PORT_ArenaAlloc(arena, dbitem->len);
  645     if ( dbitem->data == NULL) {
  646         PORT_SetError(SEC_ERROR_NO_MEMORY);
  647         goto loser;
  648     }
  649     
  650     /* fill in database record */
  651     buf = &dbitem->data[SEC_DB_ENTRY_HEADER_LEN];
  652     
  653     buf[0] = ( entry->trust.sslFlags >> 8 ) & 0xff;
  654     buf[1] = entry->trust.sslFlags & 0xff;
  655     buf[2] = ( entry->trust.emailFlags >> 8 ) & 0xff;
  656     buf[3] = entry->trust.emailFlags & 0xff;
  657     buf[4] = ( entry->trust.objectSigningFlags >> 8 ) & 0xff;
  658     buf[5] = entry->trust.objectSigningFlags & 0xff;
  659     buf[6] = ( entry->derCert.len >> 8 ) & 0xff;
  660     buf[7] = entry->derCert.len & 0xff;
  661     buf[8] = ( nnlen >> 8 ) & 0xff;
  662     buf[9] = nnlen & 0xff;
  663     
  664     PORT_Memcpy(&buf[DB_CERT_ENTRY_HEADER_LEN], entry->derCert.data,
  665               entry->derCert.len);
  666 
  667     PORT_Memcpy(&buf[DB_CERT_ENTRY_HEADER_LEN + entry->derCert.len],
  668               nn, nnlen);
  669 
  670     return(SECSuccess);
  671 
  672 loser:
  673     return(SECFailure);
  674 }

  735 static SECStatus
  736 DecodeDBCertEntry(certDBEntryCert *entry, SECItem *dbentry)
  737 {
  738     unsigned int nnlen;
  739     unsigned int headerlen;
  740     int lenoff;
  741 
  742     /* allow updates of old versions of the database */
  743     switch ( entry->common.version ) {
  744       case 5:
  745         headerlen = DB_CERT_V5_ENTRY_HEADER_LEN;
  746         lenoff = 3;
  747         break;
  748       case 6:
  749         /* should not get here */
  750         PORT_Assert[1](0);
  751         headerlen = DB_CERT_V6_ENTRY_HEADER_LEN;
  752         lenoff = 3;
  753         break;
  754       case 7:
  755       case 8:
  756         headerlen = DB_CERT_ENTRY_HEADER_LEN;
  757         lenoff = 6;
  758         break;
  759       default:
  760         /* better not get here */
  761         PORT_Assert[1](0);
  762         headerlen = DB_CERT_V5_ENTRY_HEADER_LEN;
  763         lenoff = 3;
  764         break;
  765     }
  766     
  767     /* is record long enough for header? */
  768     if ( dbentry->len < headerlen ) {
  769         PORT_SetError(SEC_ERROR_BAD_DATABASE);
  770         goto loser;
  771     }
  772     
  773     /* is database entry correct length? */
  774     entry->derCert.len = ( ( dbentry->data[lenoff] << 8 ) |
  775                           dbentry->data[lenoff+1] );
  776     nnlen = ( ( dbentry->data[lenoff+2] << 8 ) | dbentry->data[lenoff+3] );
  777     if ( ( entry->derCert.len + nnlen + headerlen )
  778         != dbentry->len) {
  779         PORT_SetError(SEC_ERROR_BAD_DATABASE);
  780         goto loser;
  781     }
  782     
  783     /* copy the dercert */
  784 
  785     entry->derCert.data = pkcs11_copyStaticData(&dbentry->data[headerlen],
  786         entry->derCert.len,entry->derCertSpace,sizeof(entry->derCertSpace));
  787     if ( entry->derCert.data == NULL ) {
  788         PORT_SetError(SEC_ERROR_NO_MEMORY);
  789         goto loser;
  790     }
  791 
  792     /* copy the nickname */
  793     if ( nnlen > 1 ) {
  794         entry->nickname = (char *)pkcs11_copyStaticData(
  795                         &dbentry->data[headerlen+entry->derCert.len], nnlen,
  796                         (unsigned char *)entry->nicknameSpace, 
  797                         sizeof(entry->nicknameSpace));
  798         if ( entry->nickname == NULL ) {
  799             PORT_SetError(SEC_ERROR_NO_MEMORY);
  800             goto loser;
  801         }
  802     } else {
  803         entry->nickname = NULL;
  804     }
  805     
  806     if ( entry->common.version < 7 ) {
  807         /* allow updates of v5 db */
  808         entry->trust.sslFlags = dbentry->data[0];
  809         entry->trust.emailFlags = dbentry->data[1];
  810         entry->trust.objectSigningFlags = dbentry->data[2];
  811     } else {
  812         entry->trust.sslFlags = ( dbentry->data[0] << 8 ) | dbentry->data[1];
  813         entry->trust.emailFlags = ( dbentry->data[2] << 8 ) | dbentry->data[3];
  814         entry->trust.objectSigningFlags =
  815             ( dbentry->data[4] << 8 ) | dbentry->data[5];
  816     }
  817     
  818     return(SECSuccess);
  819 loser:
  820     return(SECFailure);
  821 }

  889 /*
  890  * Decode a version 4 DBCert from the byte stream database format
  891  * and construct a current database entry struct
  892  */
  893 static certDBEntryCert *
  894 DecodeV4DBCertEntry(unsigned char *buf, int len)
  895 {
  896     certDBEntryCert *entry;

  898     int nnlen;
  899     PRArenaPool *arena;
  900     
  901     /* make sure length is at least long enough for the header */
  902     if ( len < DBCERT_V4_HEADER_LEN ) {
  903         PORT_SetError(SEC_ERROR_BAD_DATABASE);
  904         return(0);
  905     }
  906 
  907     /* get other lengths */
  908     certlen = buf[3] << 8 | buf[4];
  909     nnlen = buf[5] << 8 | buf[6];
  910     
  911     /* make sure DB entry is the right size */
  912     if ( ( certlen + nnlen + DBCERT_V4_HEADER_LEN ) != len ) {
  913         PORT_SetError(SEC_ERROR_BAD_DATABASE);
  914         return(0);
  915     }
  916 
  917     /* allocate arena */
  918     arena = PORT_NewArena( DER_DEFAULT_CHUNKSIZE );
  919 
  920     if ( !arena ) {
  921         PORT_SetError(SEC_ERROR_NO_MEMORY);
  922         return(0);
  923     }
  924         
  925     /* allocate structure and members */
  926     entry = (certDBEntryCert *)  PORT_ArenaAlloc(arena, sizeof(certDBEntryCert));
  927 
  928     if ( !entry ) {
  929         goto loser;
  930     }
  931 
  932     entry->common.arena = arena;
  933     entry->common.version = CERT_DB_FILE_VERSION;
  934     entry->common.type = certDBEntryTypeCert;
  935     entry->common.flags = 0;
  936     entry->trust.sslFlags = buf[0];
  937     entry->trust.emailFlags = buf[1];
  938     entry->trust.objectSigningFlags = buf[2];

  940     entry->derCert.data = (unsigned char *)PORT_ArenaAlloc(arena, certlen);
  941     if ( !entry->derCert.data ) {
  942         goto loser;
  943     }
  944     entry->derCert.len = certlen;
  945     PORT_Memcpy(entry->derCert.data, &buf[DBCERT_V4_HEADER_LEN], certlen);
  946 
  947     if ( nnlen ) {
  948         entry->nickname = (char *) PORT_ArenaAlloc(arena, nnlen);
  949         if ( !entry->nickname ) {
  950             goto loser;
  951         }

  953         
  954         if (PORT_Strcmp(entry->nickname, "Server-Cert") == 0) {
  955             entry->trust.sslFlags |= CERTDB_USER;
  956         }
  957     } else {
  958         entry->nickname = 0;
  959     }
  960 
  961     return(entry);
  962     
  963 loser:
  964     PORT_FreeArena(arena, PR_FALSE);
  965     PORT_SetError(SEC_ERROR_NO_MEMORY);
  966     return(0);
  967 }

 3492 UpdateV7DB(NSSLOWCERTCertDBHandle *handle, DB *updatedb)

 3504

 3528         case certDBEntryTypeContentVersion:
 3529         case certDBEntryTypeNickname:

 3533             break;
 3534

 3536             /* decode Entry */

 3590     }

 3599             continue;

 3602         dbKey.len = key.size - SEC_DB_KEY_HEADER_LEN;
 3603         if ((dbEntry.len <= 0) || (dbKey.len <= 0)) {
 3604             continue;
 3605         }
 3606         smimeEntry.common.version = (unsigned int)dataBuf[0];
 3607         smimeEntry.common.type = entryType;
 3608         smimeEntry.common.flags = (unsigned int)dataBuf[2];
 3609         smimeEntry.common.arena = PORT_NewArena(DER_DEFAULT_CHUNKSIZE);

 3624      * the keys and certs */

 4801 {
 4802     SECItem certKey;
 4803     SECItem *sn = &issuerAndSN->serialNumber;
 4804     SECItem *issuer = &issuerAndSN->derIssuer;
 4805     NSSLOWCERTCertificate *cert;
 4806     int data_left = sn->len-1;
 4807     int data_len = sn->len;
 4808     int index = 0;
 4809 
 4810     /* automatically detect DER encoded serial numbers and remove the der
 4811      * encoding since the database expects unencoded data. 
 4812      * if it's DER encoded, there must be at least 3 bytes, tag, len, data */
 4813     if ((sn->len >= 3) && (sn->data[0] == 0x2)) {
 4814         /* remove the der encoding of the serial number before generating the
 4815          * key.. */
 4816         data_left = sn->len-2;
 4817         data_len = sn->data[1];
 4818         index = 2;
 4819 
 4820         /* extended length ? (not very likely for a serial number) */
 4821         if (data_len & 0x80) {
 4822             int len_count = data_len & 0x7f;
 4823 
 4824             data_len = 0;
 4825             data_left -= len_count;
 4826             if (data_left > 0) {
 4827                 while (len_count --) {
 4828                     data_len = (data_len << 8) | sn->data[index++];
 4829                 }
 4830             } 
 4831         }
 4832         /* XXX leaving any leading zeros on the serial number for backwards
 4833          * compatibility
 4834          */
 4835         /* not a valid der, must be just an unlucky serial number value */
 4836         if (data_len != data_left) {
 4837             data_len = sn->len;
 4838             index = 0;
 4839         }
 4840     }

 4878 /*
 4879  * Generate a key from an issuerAndSerialNumber, and find the
 4880  * associated cert in the database.
 4881  */
 4882 NSSLOWCERTTrust *
 4883 nsslowcert_FindTrustByIssuerAndSN(NSSLOWCERTCertDBHandle *handle, 
 4884                                         NSSLOWCERTIssuerAndSN *issuerAndSN)
 4885 {
 4886     SECItem certKey;
 4887     SECItem *sn = &issuerAndSN->serialNumber;
 4888     SECItem *issuer = &issuerAndSN->derIssuer;
 4889     NSSLOWCERTTrust *trust;
 4890     unsigned char keyBuf[512];
 4891     int data_left = sn->len-1;
 4892     int data_len = sn->len;
 4893     int index = 0;
 4894     int len;
 4895 
 4896     /* automatically detect DER encoded serial numbers and remove the der
 4897      * encoding since the database expects unencoded data. 
 4898      * if it's DER encoded, there must be at least 3 bytes, tag, len, data */
 4899     if ((sn->len >= 3) && (sn->data[0] == 0x2)) {
 4900         /* remove the der encoding of the serial number before generating the
 4901          * key.. */
 4902         data_left = sn->len-2;
 4903         data_len = sn->data[1];
 4904         index = 2;
 4905 
 4906         /* extended length ? (not very likely for a serial number) */
 4907         if (data_len & 0x80) {
 4908             int len_count = data_len & 0x7f;
 4909 
 4910             data_len = 0;
 4911             data_left -= len_count;
 4912             if (data_left > 0) {
 4913                 while (len_count --) {
 4914                     data_len = (data_len << 8) | sn->data[index++];
 4915                 }
 4916             } 
 4917         }
 4918         /* XXX leaving any leading zeros on the serial number for backwards
 4919          * compatibility
 4920          */
 4921         /* not a valid der, must be just an unlucky serial number value */
 4922         if (data_len != data_left) {
 4923             data_len = sn->len;
 4924             index = 0;
 4925         }
 4926     }
 4927 
 4928     certKey.type = 0;
 4929     certKey.len = data_len + issuer->len;
 4930     len = sn->len + issuer->len;
 4931     if (len > sizeof (keyBuf)) {
 4932         certKey.data = (unsigned char*)PORT_Alloc(len);
 4933     } else {
 4934         certKey.data = keyBuf;
 4935     }
 4936     
 4937     if ( certKey.data == NULL ) {
 4938         return(0);
 4939     }
 4940 
 4941     /* first try the serial number as hand-decoded above*/
 4942     /* copy the serialNumber */
 4943     PORT_Memcpy(certKey.data, &sn->data[index], data_len);
 4944 
 4945     /* copy the issuer */
 4946     PORT_Memcpy( &certKey.data[data_len],issuer->data,issuer->len);
 4947 
 4948     trust = nsslowcert_FindTrustByKey(handle, &certKey);
 4949     if (trust) {
 4950         pkcs11_freeStaticData(certKey.data, keyBuf);
 4951         return (trust);
 4952     }
 4953 
 4954     if (index == 0) {
 4955         pkcs11_freeStaticData(certKey.data, keyBuf);
 4956         return NULL;
 4957     }
 4958 
 4959     /* didn't find it, try by der encoded serial number */
 4960     /* copy the serialNumber */
 4961     PORT_Memcpy(certKey.data, sn->data, sn->len);
 4962 
 4963     /* copy the issuer */
 4964     PORT_Memcpy( &certKey.data[sn->len], issuer->data, issuer->len);
 4965     certKey.len = sn->len + issuer->len;
 4966 
 4967     trust = nsslowcert_FindTrustByKey(handle, &certKey);
 4968     
 4969     pkcs11_freeStaticData(certKey.data, keyBuf);
 4970     
 4971     return(trust);
 4972 }

 5005 static void
 5006 DestroyCertificate(NSSLOWCERTCertificate *cert, PRBool lockdb)
 5007 {
 5008     int refCount;
 5009     NSSLOWCERTCertDBHandle *handle;
 5010     
 5011     if ( cert ) {
 5012 
 5013         handle = cert->dbhandle;
 5014 
 5015         /*
 5016          * handle may be NULL, for example if the cert was created with
 5017          * nsslowcert_DecodeDERCertificate.
 5018          */
 5019         if ( lockdb && handle ) {
 5020             nsslowcert_LockDB(handle);
 5021             /* keep a reference until we unlock, so handle won't disappear
 5022              * before we are through with it */
 5023             nsslowcert_reference(handle);
 5024         }
 5025 
 5026         nsslowcert_LockCertRefCount(cert);
 5027         PORT_Assert[1](cert->referenceCount > 0);
 5028         refCount = --cert->referenceCount;
 5029         nsslowcert_UnlockCertRefCount(cert);
 5030 
 5031         if ( ( refCount == 0 ) ) {
 5032             certDBEntryCert *entry  = cert->dbEntry;
 5033 
 5034             if ( entry ) {
 5035                 DestroyDBEntry((certDBEntry *)entry);
 5036             }
 5037 
 5038             pkcs11_freeNickname(cert->nickname,cert->nicknameSpace);
 5039             pkcs11_freeStaticData(cert->certKey.data,cert->certKeySpace);
 5040             cert->certKey.data = NULL;
 5041             cert->nickname = NULL;
 5042 
 5043             /* zero cert before freeing. Any stale references to this cert
 5044              * after this point will probably cause an exception.  */
 5045             PORT_Memset(cert, 0, sizeof *cert);
 5046 
 5047             /* use reflock to protect the free list */
 5048             nsslowcert_LockFreeList();
 5049             if (certListCount > MAX_CERT_LIST_COUNT) {
 5050                 PORT_Free(cert);
 5051             } else {
 5052                 certListCount++;
 5053                 cert->next = certListHead;
 5054                 certListHead = cert;
 5055             }
 5056             nsslowcert_UnlockFreeList();
 5057             if (handle) {
 5058                 sftk_freeCertDB(handle);
 5059             }
 5060                 
 5061             cert = NULL;
 5062         }
 5063         if ( lockdb && handle ) {
 5064             nsslowcert_UnlockDB(handle);
 5065             sftk_freeCertDB(handle);
 5066         }
 5067     }
 5068 
 5069     return;
 5070 }

 5376 certDBEntry *
 5377 nsslowcert_DecodeAnyDBEntry(SECItem *dbData, SECItem *dbKey, 
 5378                  certDBEntryType entryType, void *pdata)
 5379 {
 5380     PLArenaPool *arena = NULL;
 5381     certDBEntry *entry;
 5382     SECStatus rv;
 5383     SECItem dbEntry;
 5384 
 5385 
 5386     if ((dbData->len < SEC_DB_ENTRY_HEADER_LEN) || (dbKey->len == 0)) {
 5387         PORT_SetError(SEC_ERROR_INVALID_ARGS);
 5388         goto loser;
 5389     }
 5390     dbEntry.data = &dbData->data[SEC_DB_ENTRY_HEADER_LEN];
 5391     dbEntry.len  = dbData->len - SEC_DB_ENTRY_HEADER_LEN;
 5392 
 5393     arena = PORT_NewArena(DER_DEFAULT_CHUNKSIZE);
 5394     if (arena == NULL) {
 5395         goto loser;
 5396     }
 5397     entry = PORT_ArenaZNew(arena, certDBEntry);
 5398     if (!entry)
 5399             goto loser;
 5400 
 5401     entry->common.version = (unsigned int)dbData->data[0];
 5402     entry->common.flags   = (unsigned int)dbData->data[2];
 5403     entry->common.type    = entryType;
 5404     entry->common.arena   = arena;
 5405 
 5406     switch (entryType) {
 5407     case certDBEntryTypeContentVersion: /* This type appears to be unused */
 5408     case certDBEntryTypeVersion:        /* This type has only the common hdr */
 5409         rv = SECSuccess;
 5410             break;
 5411 
 5412     case certDBEntryTypeSubject:
 5413         rv = DecodeDBSubjectEntry(&entry->subject, &dbEntry, dbKey);
 5414             break;
 5415 
 5416     case certDBEntryTypeNickname:
 5417         rv = DecodeDBNicknameEntry(&entry->nickname, &dbEntry,
 5418                                    (char *)dbKey->data);
 5419             break;
 5420 
 5421     /* smime profiles need entries created after the certs have
 5422      * been imported, loop over them in a second run */
 5423     case certDBEntryTypeSMimeProfile:
 5424         rv = DecodeDBSMimeEntry(&entry->smime, &dbEntry, (char *)dbKey->data);
 5425         break;
 5426 
 5427     case certDBEntryTypeCert:
 5428         rv = DecodeDBCertEntry(&entry->cert, &dbEntry);
 5429         break;
 5430 
 5431     case certDBEntryTypeKeyRevocation:
 5432     case certDBEntryTypeRevocation:
 5433         rv = DecodeDBCrlEntry(&entry->revocation, &dbEntry);
 5434         break;
 5435 
 5436     default:
 5437         PORT_SetError(SEC_ERROR_INVALID_ARGS);
 5438         rv = SECFailure;
 5439     }
 5440 
 5441     if (rv == SECSuccess)
 5442         return entry;
 5443 
 5444 loser:
 5445     if (arena)
 5446         PORT_FreeArena(arena, PR_FALSE);
 5447     return NULL;
 5448 }
