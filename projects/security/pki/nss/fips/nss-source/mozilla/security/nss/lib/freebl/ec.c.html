   51 /* 
   52  * Returns true if pointP is the point at infinity, false otherwise
   53  */
   54 PRBool
   55 ec_point_at_infinity(SECItem *pointP)
   56 {
   57     unsigned int i;
   58 
   59     for (i = 1; i < pointP->len; i++) {
   60         if (pointP->data[i] != 0x00) return PR_FALSE;
   61     }
   62 
   63     return PR_TRUE;
   64 }

   66 /* 
   67  * Computes scalar point multiplication pointQ = k1 * G + k2 * pointP for
   68  * the curve whose parameters are encoded in params with base point G.
   69  */
   70 SECStatus 
   71 ec_points_mul(const ECParams *params, const mp_int *k1, const mp_int *k2,
   72              const SECItem *pointP, SECItem *pointQ)
   73 {
   74     mp_int Px, Py, Qx, Qy;
   75     mp_int Gx, Gy, order, irreducible, a, b;
   76 #if 0 /* currently don't support non-named curves */
   77     unsigned int irr_arr[5];
   78 #endif
   79     ECGroup *group = NULL;
   80     SECStatus rv = SECFailure;
   81     mp_err err = MP_OKAY;
   82     int len;
   83 
   84 #if EC_DEBUG
   85     int i;
   86     char mpstr[256];
   87 
   88     printf("ec_points_mul: params [len=%d]:", params->DEREncoding.len);
   89     for (i = 0; i < params->DEREncoding.len; i++) 
   90             printf("%02x:", params->DEREncoding.data[i]);
   91     printf("\n");
   92 
   93         if (k1 != NULL) {
   94                 mp_tohex(k1, mpstr);
   95                 printf("ec_points_mul: scalar k1: %s\n", mpstr);
   96                 mp_todecimal(k1, mpstr);
   97                 printf("ec_points_mul: scalar k1: %s (dec)\n", mpstr);
   98         }
   99 
  100         if (k2 != NULL) {
  101                 mp_tohex(k2, mpstr);
  102                 printf("ec_points_mul: scalar k2: %s\n", mpstr);
  103                 mp_todecimal(k2, mpstr);
  104                 printf("ec_points_mul: scalar k2: %s (dec)\n", mpstr);
  105         }
  106 
  107         if (pointP != NULL) {
  108                 printf("ec_points_mul: pointP [len=%d]:", pointP->len);
  109                 for (i = 0; i < pointP->len; i++) 
  110                         printf("%02x:", pointP->data[i]);
  111                 printf("\n");
  112         }
  113 #endif
  114 
  115         /* NOTE: We only support uncompressed points for now */
  116         len = (params->fieldID.size + 7) >> 3;
  117         if (pointP != NULL) {
  118                 if ((pointP->data[0] != EC_POINT_FORM_UNCOMPRESSED) ||
  119                         (pointP->len != (2 * len + 1))) {
  120                         return SECFailure;
  121                 };
  122         }
  123 
  124         MP_DIGITS(&Px) = 0;
  125         MP_DIGITS(&Py) = 0;
  126         MP_DIGITS(&Qx) = 0;
  127         MP_DIGITS(&Qy) = 0;
  128         MP_DIGITS(&Gx) = 0;
  129         MP_DIGITS(&Gy) = 0;
  130         MP_DIGITS(&order) = 0;
  131         MP_DIGITS(&irreducible) = 0;
  132         MP_DIGITS(&a) = 0;
  133         MP_DIGITS(&b) = 0;
  134         CHECK_MPI_OK( mp_init(&Px) );
  135         CHECK_MPI_OK( mp_init(&Py) );
  136         CHECK_MPI_OK( mp_init(&Qx) );
  137         CHECK_MPI_OK( mp_init(&Qy) );
  138         CHECK_MPI_OK( mp_init(&Gx) );
  139         CHECK_MPI_OK( mp_init(&Gy) );
  140         CHECK_MPI_OK( mp_init(&order) );
  141         CHECK_MPI_OK( mp_init(&irreducible) );
  142         CHECK_MPI_OK( mp_init(&a) );
  143         CHECK_MPI_OK( mp_init(&b) );
  144 
  145         if ((k2 != NULL) && (pointP != NULL)) {
  146                 /* Initialize Px and Py */
  147                 CHECK_MPI_OK( mp_read_unsigned_octets(&Px, pointP->data + 1, (mp_size) len) );
  148                 CHECK_MPI_OK( mp_read_unsigned_octets(&Py, pointP->data + 1 + len, (mp_size) len) );
  149         }
  150 
  151         /* construct from named params, if possible */
  152         if (params->name != ECCurve_noName) {
  153                 group = ECGroup_fromName(params->name);
  154         }
  155 
  156 #if 0 /* currently don't support non-named curves */
  157         if (group == NULL) {
  158                 /* Set up mp_ints containing the curve coefficients */
  159                 CHECK_MPI_OK( mp_read_unsigned_octets(&Gx, params->base.data + 1, 
  160                                                                                   (mp_size) len) );
  161                 CHECK_MPI_OK( mp_read_unsigned_octets(&Gy, params->base.data + 1 + len, 
  162                                                                                   (mp_size) len) );
  163                 SECITEM_TO_MPINT( params->order, &order );
  164                 SECITEM_TO_MPINT( params->curve.a, &a );
  165                 SECITEM_TO_MPINT( params->curve.b, &b );
  166                 if (params->fieldID.type == ec_field_GFp) {
  167                         SECITEM_TO_MPINT( params->fieldID.u.prime, &irreducible );
  168                         group = ECGroup_consGFp(&irreducible, &a, &b, &Gx, &Gy, &order, params->cofactor);
  169                 } else {
  170                         SECITEM_TO_MPINT( params->fieldID.u.poly, &irreducible );
  171                         irr_arr[0] = params->fieldID.size;
  172                         irr_arr[1] = params->fieldID.k1;
  173                         irr_arr[2] = params->fieldID.k2;
  174                         irr_arr[3] = params->fieldID.k3;
  175                         irr_arr[4] = 0;
  176                         group = ECGroup_consGF2m(&irreducible, irr_arr, &a, &b, &Gx, &Gy, &order, params->cofactor);
  177                 }
  178         }
  179 #endif
  180         if (group == NULL)
  181                 goto cleanup;
  182 
  183         if ((k2 != NULL) && (pointP != NULL)) {
  184                 CHECK_MPI_OK( ECPoints_mul(group, k1, k2, &Px, &Py, &Qx, &Qy) );
  185         } else {
  186                 CHECK_MPI_OK( ECPoints_mul(group, k1, NULL, NULL, NULL, &Qx, &Qy) );
  187     }
  188 
  189     /* Construct the SECItem representation of point Q */
  190     pointQ->data[0] = EC_POINT_FORM_UNCOMPRESSED;
  191     CHECK_MPI_OK( mp_to_fixlen_octets(&Qx, pointQ->data + 1,
  192                                       (mp_size) len) );
  193     CHECK_MPI_OK( mp_to_fixlen_octets(&Qy, pointQ->data + 1 + len,
  194                                       (mp_size) len) );
  195 
  196     rv = SECSuccess;
  197 
  198 #if EC_DEBUG
  199     printf("ec_points_mul: pointQ [len=%d]:", pointQ->len);
  200     for (i = 0; i < pointQ->len; i++) 
  201             printf("%02x:", pointQ->data[i]);
  202     printf("\n");
  203 #endif
  204 
  205 cleanup:
  206     ECGroup_free(group);
  207     mp_clear(&Px);
  208     mp_clear(&Py);
  209     mp_clear(&Qx);
  210     mp_clear(&Qy);
  211     mp_clear(&Gx);
  212     mp_clear(&Gy);
  213     mp_clear(&order);
  214     mp_clear(&irreducible);
  215     mp_clear(&a);
  216     mp_clear(&b);
  217     if (err) {
  218         MP_TO_SEC_ERROR(err);
  219         rv = SECFailure;
  220     }
  221 
  222     return rv;
  223 }
  224 #endif /* NSS_ENABLE_ECC */

  226 /* Generates a new EC key pair. The private key is a supplied
  227  * value and the public key is the result of performing a scalar 
  228  * point multiplication of that value with the curve's base point.
  229  */
  230 SECStatus 
  231 ec_NewKey(ECParams *ecParams, ECPrivateKey **privKey, 
  232     const unsigned char *privKeyBytes, int privKeyLen)
  233 {
  234     SECStatus rv = SECFailure;
  235 #ifdef NSS_ENABLE_ECC
  236     PRArenaPool *arena;
  237     ECPrivateKey *key;
  238     mp_int k;
  239     mp_err err = MP_OKAY;
  240     int len;
  241 
  242 #if EC_DEBUG
  243     printf("ec_NewKey called\n");
  244 #endif
  245 
  246     if (!ecParams || !privKey || !privKeyBytes || (privKeyLen < 0)) {
  247         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  248         return SECFailure;
  249     }
  250 
  251     /* Initialize an arena for the EC key. */
  252     if (!(arena = PORT_NewArena(NSS_FREEBL_DEFAULT_CHUNKSIZE)))
  253         return SECFailure;
  254 
  255     key = (ECPrivateKey *)PORT_ArenaZAlloc(arena, sizeof(ECPrivateKey));
  256     if (!key) {
  257         PORT_FreeArena(arena, PR_TRUE);
  258         return SECFailure;
  259     }
  260 
  261     /* Set the version number (SEC 1 section C.4 says it should be 1) */
  262     SECITEM_AllocItem(arena, &key->version, 1);
  263     key->version.data[0] = 1;
  264 
  265     /* Copy all of the fields from the ECParams argument to the
  266      * ECParams structure within the private key.
  267      */
  268     key->ecParams.arena = arena;
  269     key->ecParams.type = ecParams->type;
  270     key->ecParams.fieldID.size = ecParams->fieldID.size;
  271     key->ecParams.fieldID.type = ecParams->fieldID.type;
  272     if (ecParams->fieldID.type == ec_field_GFp) {
  273         CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &key->ecParams.fieldID.u.prime,
  274             &ecParams->fieldID.u.prime));
  275     } else {
  276         CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &key->ecParams.fieldID.u.poly,
  277             &ecParams->fieldID.u.poly));
  278     }
  279     key->ecParams.fieldID.k1 = ecParams->fieldID.k1;
  280     key->ecParams.fieldID.k2 = ecParams->fieldID.k2;
  281     key->ecParams.fieldID.k3 = ecParams->fieldID.k3;
  282     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &key->ecParams.curve.a,
  283         &ecParams->curve.a));
  284     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &key->ecParams.curve.b,
  285         &ecParams->curve.b));
  286     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &key->ecParams.curve.seed,
  287         &ecParams->curve.seed));
  288     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &key->ecParams.base,
  289         &ecParams->base));
  290     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &key->ecParams.order,
  291         &ecParams->order));
  292     key->ecParams.cofactor = ecParams->cofactor;
  293     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &key->ecParams.DEREncoding,
  294         &ecParams->DEREncoding));
  295     key->ecParams.name = ecParams->name;
  296     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &key->ecParams.curveOID,
  297         &ecParams->curveOID));
  298 
  299     len = (ecParams->fieldID.size + 7) >> 3;
  300     SECITEM_AllocItem(arena, &key->publicValue, 2*len + 1);
  301     len = ecParams->order.len;
  302     SECITEM_AllocItem(arena, &key->privateValue, len);
  303 
  304     /* Copy private key */
  305     if (privKeyLen >= len) {
  306         memcpy(key->privateValue.data, privKeyBytes, len);
  307     } else {
  308         memset(key->privateValue.data, 0, (len - privKeyLen));
  309         memcpy(key->privateValue.data + (len - privKeyLen), privKeyBytes, privKeyLen);
  310     }
  311 
  312     /* Compute corresponding public key */
  313     MP_DIGITS(&k) = 0;
  314     CHECK_MPI_OK( mp_init(&k) );
  315     CHECK_MPI_OK( mp_read_unsigned_octets(&k, key->privateValue.data, 
  316         (mp_size) len) );
  317 
  318     rv = ec_points_mul(ecParams, &k, NULL, NULL, &(key->publicValue));
  319     if (rv != SECSuccess) goto cleanup;
  320     *privKey = key;
  321 
  322 cleanup:
  323     mp_clear(&k);
  324     if (rv)
  325         PORT_FreeArena(arena, PR_TRUE);
  326 
  327 #if EC_DEBUG
  328     printf("ec_NewKey returning %s\n", 
  329         (rv == SECSuccess) ? "success" : "failure");
  330 #endif
  331 #else
  332     PORT_SetError(SEC_ERROR_UNSUPPORTED_KEYALG);
  333 #endif /* NSS_ENABLE_ECC */
  334 
  335     return rv;
  336 
  337 }

  357 /* Generate a random private key using the algorithm A.4.1 of ANSI X9.62,
  358  * modified a la FIPS 186-2 Change Notice 1 to eliminate the bias in the
  359  * random number generator.
  360  *
  361  * Parameters
  362  * - order: a buffer that holds the curve's group order
  363  * - len: the length in octets of the order buffer
  364  *
  365  * Return Value
  366  * Returns a buffer of len octets that holds the private key. The caller
  367  * is responsible for freeing the buffer with PORT_ZFree.
  368  */
  369 static unsigned char *
  370 ec_GenerateRandomPrivateKey(const unsigned char *order, int len)
  371 {
  372     SECStatus rv = SECSuccess;
  373     mp_err err;
  374     unsigned char *privKeyBytes = NULL;
  375     mp_int privKeyVal, order_1, one;
  376 
  377     MP_DIGITS(&privKeyVal) = 0;
  378     MP_DIGITS(&order_1) = 0;
  379     MP_DIGITS(&one) = 0;
  380     CHECK_MPI_OK( mp_init(&privKeyVal) );
  381     CHECK_MPI_OK( mp_init(&order_1) );
  382     CHECK_MPI_OK( mp_init(&one) );
  383 
  384     /* Generates 2*len random bytes using the global random bit generator
  385      * (which implements Algorithm 1 of FIPS 186-2 Change Notice 1) then
  386      * reduces modulo the group order.
  387      */
  388     if ((privKeyBytes = PORT_Alloc(2*len)) == NULL) goto cleanup;
  389     CHECK_SEC_OK[1]( RNG_GenerateGlobalRandomBytes(privKeyBytes, 2*len) );
  390     CHECK_MPI_OK( mp_read_unsigned_octets(&privKeyVal, privKeyBytes, 2*len) );
  391     CHECK_MPI_OK( mp_read_unsigned_octets(&order_1, order, len) );
  392     CHECK_MPI_OK( mp_set_int(&one, 1) );
  393     CHECK_MPI_OK( mp_sub(&order_1, &one, &order_1) );
  394     CHECK_MPI_OK( mp_mod(&privKeyVal, &order_1, &privKeyVal) );
  395     CHECK_MPI_OK( mp_add(&privKeyVal, &one, &privKeyVal) );
  396     CHECK_MPI_OK( mp_to_fixlen_octets(&privKeyVal, privKeyBytes, len) );
  397     memset(privKeyBytes+len, 0, len);
  398 cleanup:
  399     mp_clear(&privKeyVal);
  400     mp_clear(&order_1);
  401     mp_clear(&one);
  402     if (err < MP_OKAY) {
  403         MP_TO_SEC_ERROR(err);
  404         rv = SECFailure;
  405     }
  406     if (rv != SECSuccess && privKeyBytes) {
  407         PORT_Free(privKeyBytes);
  408         privKeyBytes = NULL;
  409     }
  410     return privKeyBytes;
  411 }

  413 /* Generates a new EC key pair. The private key is a random value and
  414  * the public key is the result of performing a scalar point multiplication
  415  * of that value with the curve's base point.
  416  */
  417 SECStatus 
  418 EC_NewKey(ECParams *ecParams, ECPrivateKey **privKey)
  419 {
  420     SECStatus rv = SECFailure;
  421 #ifdef NSS_ENABLE_ECC
  422     int len;
  423     unsigned char *privKeyBytes = NULL;
  424 
  425     if (!ecParams) {
  426         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  427         return SECFailure;
  428     }
  429 
  430     len = ecParams->order.len;
  431     privKeyBytes = ec_GenerateRandomPrivateKey(ecParams->order.data, len);
  432     if (privKeyBytes == NULL) goto cleanup;
  433     /* generate public key */
  434     CHECK_SEC_OK[1]( ec_NewKey(ecParams, privKey, privKeyBytes, len) );
  435 
  436 cleanup:
  437     if (privKeyBytes) {
  438         PORT_ZFree(privKeyBytes, len);
  439     }
  440 #if EC_DEBUG
  441     printf("EC_NewKey returning %s\n", 
  442         (rv == SECSuccess) ? "success" : "failure");
  443 #endif
  444 #else
  445     PORT_SetError(SEC_ERROR_UNSUPPORTED_KEYALG);
  446 #endif /* NSS_ENABLE_ECC */
  447     
  448     return rv;
  449 }

  451 /* Validates an EC public key as described in Section 5.2.2 of
  452  * X9.62. The ECDH primitive when used without the cofactor does
  453  * not address small subgroup attacks, which may occur when the
  454  * public key is not valid. These attacks can be prevented by 
  455  * validating the public key before using ECDH.
  456  */
  457 SECStatus 
  458 EC_ValidatePublicKey(ECParams *ecParams, SECItem *publicValue)
  459 {
  460 #ifdef NSS_ENABLE_ECC
  461     mp_int Px, Py;
  462     ECGroup *group = NULL;
  463     SECStatus rv = SECFailure;
  464     mp_err err = MP_OKAY;
  465     int len;
  466 
  467     if (!ecParams || !publicValue) {
  468         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  469         return SECFailure;
  470     }
  471         
  472     /* NOTE: We only support uncompressed points for now */
  473     len = (ecParams->fieldID.size + 7) >> 3;
  474     if (publicValue->data[0] != EC_POINT_FORM_UNCOMPRESSED) {
  475         PORT_SetError(SEC_ERROR_UNSUPPORTED_EC_POINT_FORM);
  476         return SECFailure;
  477     } else if (publicValue->len != (2 * len + 1)) {
  478         PORT_SetError(SEC_ERROR_BAD_KEY);
  479         return SECFailure;
  480     }
  481 
  482     MP_DIGITS(&Px) = 0;
  483     MP_DIGITS(&Py) = 0;
  484     CHECK_MPI_OK( mp_init(&Px) );
  485     CHECK_MPI_OK( mp_init(&Py) );
  486 
  487     /* Initialize Px and Py */
  488     CHECK_MPI_OK( mp_read_unsigned_octets(&Px, publicValue->data + 1, (mp_size) len) );
  489     CHECK_MPI_OK( mp_read_unsigned_octets(&Py, publicValue->data + 1 + len, (mp_size) len) );
  490 
  491     /* construct from named params */
  492     group = ECGroup_fromName(ecParams->name);
  493     if (group == NULL) {
  494         /*
  495          * ECGroup_fromName fails if ecParams->name is not a valid
  496          * ECCurveName value, or if we run out of memory, or perhaps
  497          * for other reasons.  Unfortunately if ecParams->name is a
  498          * valid ECCurveName value, we don't know what the right error
  499          * code should be because ECGroup_fromName doesn't return an
  500          * error code to the caller.  Set err to MP_UNDEF because
  501          * that's what ECGroup_fromName uses internally.
  502          */
  503         if ((ecParams->name <= ECCurve_noName) ||
  504             (ecParams->name >= ECCurve_pastLastCurve)) {
  505             err = MP_BADARG;
  506         } else {
  507             err = MP_UNDEF;
  508         }
  509         goto cleanup;
  510     }
  511 
  512     /* validate public point */
  513     if ((err = ECPoint_validate(group, &Px, &Py)) < MP_YES) {
  514         if (err == MP_NO) {
  515             PORT_SetError(SEC_ERROR_BAD_KEY);
  516             rv = SECFailure;
  517             err = MP_OKAY;  /* don't change the error code */
  518         }
  519         goto cleanup;
  520     }
  521 
  522     rv = SECSuccess;
  523 
  524 cleanup:
  525     ECGroup_free(group);
  526     mp_clear(&Px);
  527     mp_clear(&Py);
  528     if (err) {
  529         MP_TO_SEC_ERROR(err);
  530         rv = SECFailure;
  531     }
  532     return rv;
  533 #else
  534     PORT_SetError(SEC_ERROR_UNSUPPORTED_KEYALG);
  535     return SECFailure;
  536 #endif /* NSS_ENABLE_ECC */
  537 }

  539 /* 
  540 ** Performs an ECDH key derivation by computing the scalar point
  541 ** multiplication of privateValue and publicValue (with or without the
  542 ** cofactor) and returns the x-coordinate of the resulting elliptic
  543 ** curve point in derived secret.  If successful, derivedSecret->data
  544 ** is set to the address of the newly allocated buffer containing the
  545 ** derived secret, and derivedSecret->len is the size of the secret
  546 ** produced. It is the caller's responsibility to free the allocated
  547 ** buffer containing the derived secret.
  548 */
  549 SECStatus 
  550 ECDH_Derive(SECItem  *publicValue, 
  551             ECParams *ecParams,
  552             SECItem  *privateValue,
  553             PRBool    withCofactor,
  554             SECItem  *derivedSecret)
  555 {
  556     SECStatus rv = SECFailure;
  557 #ifdef NSS_ENABLE_ECC
  558     unsigned int len = 0;
  559     SECItem pointQ = {siBuffer, NULL, 0};
  560     mp_int k; /* to hold the private value */
  561     mp_int cofactor;
  562     mp_err err = MP_OKAY;
  563 #if EC_DEBUG
  564     int i;
  565 #endif
  566 
  567     if (!publicValue || !ecParams || !privateValue || 
  568         !derivedSecret) {
  569         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  570         return SECFailure;
  571     }
  572 
  573     memset(derivedSecret, 0, sizeof *derivedSecret);
  574     len = (ecParams->fieldID.size + 7) >> 3;  
  575     pointQ.len = 2*len + 1;
  576     if ((pointQ.data = PORT_Alloc(2*len + 1)) == NULL) goto cleanup;
  577 
  578     MP_DIGITS(&k) = 0;
  579     CHECK_MPI_OK( mp_init(&k) );
  580     CHECK_MPI_OK( mp_read_unsigned_octets(&k, privateValue->data, 
  581                                           (mp_size) privateValue->len) );
  582 
  583     if (withCofactor && (ecParams->cofactor != 1)) {
  584             /* multiply k with the cofactor */
  585             MP_DIGITS(&cofactor) = 0;
  586             CHECK_MPI_OK( mp_init(&cofactor) );
  587             mp_set(&cofactor, ecParams->cofactor);
  588             CHECK_MPI_OK( mp_mul(&k, &cofactor, &k) );
  589     }
  590 
  591     /* Multiply our private key and peer's public point */
  592     if ((ec_points_mul(ecParams, NULL, &k, publicValue, &pointQ) != SECSuccess) ||
  593         ec_point_at_infinity(&pointQ))
  594         goto cleanup;
  595 
  596     /* Allocate memory for the derived secret and copy
  597      * the x co-ordinate of pointQ into it.
  598      */
  599     SECITEM_AllocItem(NULL, derivedSecret, len);
  600     memcpy(derivedSecret->data, pointQ.data + 1, len);
  601 
  602     rv = SECSuccess;
  603 
  604 #if EC_DEBUG
  605     printf("derived_secret:\n");
  606     for (i = 0; i < derivedSecret->len; i++) 
  607         printf("%02x:", derivedSecret->data[i]);
  608     printf("\n");
  609 #endif
  610 
  611 cleanup:
  612     mp_clear(&k);
  613 
  614     if (pointQ.data) {
  615         PORT_ZFree(pointQ.data, 2*len + 1);
  616     }
  617 #else
  618     PORT_SetError(SEC_ERROR_UNSUPPORTED_KEYALG);
  619 #endif /* NSS_ENABLE_ECC */
  620 
  621     return rv;
  622 }

  624 /* Computes the ECDSA signature (a concatenation of two values r and s)
  625  * on the digest using the given key and the random value kb (used in
  626  * computing s).
  627  */
  628 SECStatus 
  629 ECDSA_SignDigestWithSeed(ECPrivateKey *key, SECItem *signature, 
  630     const SECItem *digest, const unsigned char *kb, const int kblen)
  631 {
  632     SECStatus rv = SECFailure;
  633 #ifdef NSS_ENABLE_ECC
  634     mp_int x1;
  635     mp_int d, k;     /* private key, random integer */
  636     mp_int r, s;     /* tuple (r, s) is the signature */
  637     mp_int n;
  638     mp_err err = MP_OKAY;
  639     ECParams *ecParams = NULL;
  640     SECItem kGpoint = { siBuffer, NULL, 0};
  641     int flen = 0;    /* length in bytes of the field size */
  642     unsigned olen;   /* length in bytes of the base point order */
  643 
  644 #if EC_DEBUG
  645     char mpstr[256];
  646 #endif
  647 
  648     /* Initialize MPI integers. */
  649     /* must happen before the first potential call to cleanup */
  650     MP_DIGITS(&x1) = 0;
  651     MP_DIGITS(&d) = 0;
  652     MP_DIGITS(&k) = 0;
  653     MP_DIGITS(&r) = 0;
  654     MP_DIGITS(&s) = 0;
  655     MP_DIGITS(&n) = 0;
  656 
  657     /* Check args */
  658     if (!key || !signature || !digest || !kb || (kblen < 0)) {
  659         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  660         goto cleanup;
  661     }
  662 
  663     ecParams = &(key->ecParams);
  664     flen = (ecParams->fieldID.size + 7) >> 3;
  665     olen = ecParams->order.len;  
  666     if (signature->data == NULL) {
  667         /* a call to get the signature length only */
  668         goto finish;
  669     }
  670     if (signature->len < 2*olen) {
  671         PORT_SetError(SEC_ERROR_OUTPUT_LEN);
  672         goto cleanup;
  673     }
  674 
  675 
  676     CHECK_MPI_OK( mp_init(&x1) );
  677     CHECK_MPI_OK( mp_init(&d) );
  678     CHECK_MPI_OK( mp_init(&k) );
  679     CHECK_MPI_OK( mp_init(&r) );
  680     CHECK_MPI_OK( mp_init(&s) );
  681     CHECK_MPI_OK( mp_init(&n) );
  682 
  683     SECITEM_TO_MPINT( ecParams->order, &n );
  684     SECITEM_TO_MPINT( key->privateValue, &d );
  685     CHECK_MPI_OK( mp_read_unsigned_octets(&k, kb, kblen) );
  686     /* Make sure k is in the interval [1, n-1] */
  687     if ((mp_cmp_z(&k) <= 0) || (mp_cmp(&k, &n) >= 0)) {
  688 #if EC_DEBUG
  689         printf("k is outside [1, n-1]\n");
  690         mp_tohex(&k, mpstr);
  691         printf("k : %s \n", mpstr);
  692         mp_tohex(&n, mpstr);
  693         printf("n : %s \n", mpstr);
  694 #endif
  695         PORT_SetError(SEC_ERROR_NEED_RANDOM);
  696         goto cleanup;
  697     }
  698 
  699     /* 
  700     ** ANSI X9.62, Section 5.3.2, Step 2
  701     **
  702     ** Compute kG
  703     */
  704     kGpoint.len = 2*flen + 1;
  705     kGpoint.data = PORT_Alloc(2*flen + 1);
  706     if ((kGpoint.data == NULL) ||
  707         (ec_points_mul(ecParams, &k, NULL, NULL, &kGpoint)
  708             != SECSuccess))
  709         goto cleanup;
  710 
  711     /* 
  712     ** ANSI X9.62, Section 5.3.3, Step 1
  713     **
  714     ** Extract the x co-ordinate of kG into x1
  715     */
  716     CHECK_MPI_OK( mp_read_unsigned_octets(&x1, kGpoint.data + 1, 
  717                                           (mp_size) flen) );
  718 
  719     /* 
  720     ** ANSI X9.62, Section 5.3.3, Step 2
  721     **
  722     ** r = x1 mod n  NOTE: n is the order of the curve
  723     */
  724     CHECK_MPI_OK( mp_mod(&x1, &n, &r) );
  725 
  726     /*
  727     ** ANSI X9.62, Section 5.3.3, Step 3
  728     **
  729     ** verify r != 0 
  730     */
  731     if (mp_cmp_z(&r) == 0) {
  732         PORT_SetError(SEC_ERROR_NEED_RANDOM);
  733         goto cleanup;
  734     }
  735 
  736     /*                                  
  737     ** ANSI X9.62, Section 5.3.3, Step 4
  738     **
  739     ** s = (k**-1 * (HASH(M) + d*r)) mod n 
  740     */
  741     SECITEM_TO_MPINT(*digest, &s);        /* s = HASH(M)     */
  742 
  743     /* In the definition of EC signing, digests are truncated
  744      * to the length of n in bits. 
  745      * (see SEC 1 "Elliptic Curve Digit Signature Algorithm" section 4.1.*/
  746     if (digest->len*8 > ecParams->fieldID.size) {
  747         mpl_rsh(&s,&s,digest->len*8 - ecParams->fieldID.size);
  748     }
  749 
  750 #if EC_DEBUG
  751     mp_todecimal(&n, mpstr);
  752     printf("n : %s (dec)\n", mpstr);
  753     mp_todecimal(&d, mpstr);
  754     printf("d : %s (dec)\n", mpstr);
  755     mp_tohex(&x1, mpstr);
  756     printf("x1: %s\n", mpstr);
  757     mp_todecimal(&s, mpstr);
  758     printf("digest: %s (decimal)\n", mpstr);
  759     mp_todecimal(&r, mpstr);
  760     printf("r : %s (dec)\n", mpstr);
  761     mp_tohex(&r, mpstr);
  762     printf("r : %s\n", mpstr);
  763 #endif
  764 
  765     CHECK_MPI_OK( mp_invmod(&k, &n, &k) );      /* k = k**-1 mod n */
  766     CHECK_MPI_OK( mp_mulmod(&d, &r, &n, &d) );  /* d = d * r mod n */
  767     CHECK_MPI_OK( mp_addmod(&s, &d, &n, &s) );  /* s = s + d mod n */
  768     CHECK_MPI_OK( mp_mulmod(&s, &k, &n, &s) );  /* s = s * k mod n */
  769 
  770 #if EC_DEBUG
  771     mp_todecimal(&s, mpstr);
  772     printf("s : %s (dec)\n", mpstr);
  773     mp_tohex(&s, mpstr);
  774     printf("s : %s\n", mpstr);
  775 #endif
  776 
  777     /*
  778     ** ANSI X9.62, Section 5.3.3, Step 5
  779     **
  780     ** verify s != 0
  781     */
  782     if (mp_cmp_z(&s) == 0) {
  783         PORT_SetError(SEC_ERROR_NEED_RANDOM);
  784         goto cleanup;
  785     }
  786 
  787    /*
  788     **
  789     ** Signature is tuple (r, s)
  790     */
  791     CHECK_MPI_OK( mp_to_fixlen_octets(&r, signature->data, olen) );
  792     CHECK_MPI_OK( mp_to_fixlen_octets(&s, signature->data + olen, olen) );
  793 finish:
  794     signature->len = 2*olen;
  795 
  796     rv = SECSuccess;
  797     err = MP_OKAY;
  798 cleanup:
  799     mp_clear(&x1);
  800     mp_clear(&d);
  801     mp_clear(&k);
  802     mp_clear(&r);
  803     mp_clear(&s);
  804     mp_clear(&n);
  805 
  806     if (kGpoint.data) {
  807         PORT_ZFree(kGpoint.data, 2*flen + 1);
  808     }
  809 
  810     if (err) {
  811         MP_TO_SEC_ERROR(err);
  812         rv = SECFailure;
  813     }
  814 
  815 #if EC_DEBUG
  816     printf("ECDSA signing with seed %s\n",
  817         (rv == SECSuccess) ? "succeeded" : "failed");
  818 #endif
  819 #else
  820     PORT_SetError(SEC_ERROR_UNSUPPORTED_KEYALG);
  821 #endif /* NSS_ENABLE_ECC */
  822 
  823    return rv;
  824 }

  826 /*
  827 ** Computes the ECDSA signature on the digest using the given key 
  828 ** and a random seed.
  829 */
  830 SECStatus 
  831 ECDSA_SignDigest(ECPrivateKey *key, SECItem *signature, const SECItem *digest)
  832 {
  833     SECStatus rv = SECFailure;
  834 #ifdef NSS_ENABLE_ECC
  835     int len;
  836     unsigned char *kBytes= NULL;
  837 
  838     if (!key) {
  839         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  840         return SECFailure;
  841     }
  842 
  843     /* Generate random value k */
  844     len = key->ecParams.order.len;
  845     kBytes = ec_GenerateRandomPrivateKey(key->ecParams.order.data, len);
  846     if (kBytes == NULL) goto cleanup;
  847 
  848     /* Generate ECDSA signature with the specified k value */
  849     rv = ECDSA_SignDigestWithSeed(key, signature, digest, kBytes, len);
  850 
  851 cleanup:    
  852     if (kBytes) {
  853         PORT_ZFree(kBytes, len);
  854     }
  855 
  856 #if EC_DEBUG
  857     printf("ECDSA signing %s\n",
  858         (rv == SECSuccess) ? "succeeded" : "failed");
  859 #endif
  860 #else
  861     PORT_SetError(SEC_ERROR_UNSUPPORTED_KEYALG);
  862 #endif /* NSS_ENABLE_ECC */
  863 
  864     return rv;
  865 }

  867 /*
  868 ** Checks the signature on the given digest using the key provided.
  869 */
  870 SECStatus 
  871 ECDSA_VerifyDigest(ECPublicKey *key, const SECItem *signature, 
  872                  const SECItem *digest)
  873 {
  874     SECStatus rv = SECFailure;
  875 #ifdef NSS_ENABLE_ECC
  876     mp_int r_, s_;           /* tuple (r', s') is received signature) */
  877     mp_int c, u1, u2, v;     /* intermediate values used in verification */
  878     mp_int x1;
  879     mp_int n;
  880     mp_err err = MP_OKAY;
  881     ECParams *ecParams = NULL;
  882     SECItem pointC = { siBuffer, NULL, 0 };
  883     int slen;       /* length in bytes of a half signature (r or s) */
  884     int flen;       /* length in bytes of the field size */
  885     unsigned olen;  /* length in bytes of the base point order */
  886 
  887 #if EC_DEBUG
  888     char mpstr[256];
  889     printf("ECDSA verification called\n");
  890 #endif
  891 
  892     /* Initialize MPI integers. */
  893     /* must happen before the first potential call to cleanup */
  894     MP_DIGITS(&r_) = 0;
  895     MP_DIGITS(&s_) = 0;
  896     MP_DIGITS(&c) = 0;
  897     MP_DIGITS(&u1) = 0;
  898     MP_DIGITS(&u2) = 0;
  899     MP_DIGITS(&x1) = 0;
  900     MP_DIGITS(&v)  = 0;
  901     MP_DIGITS(&n)  = 0;
  902 
  903     /* Check args */
  904     if (!key || !signature || !digest) {
  905         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  906         goto cleanup;
  907     }
  908 
  909     ecParams = &(key->ecParams);
  910     flen = (ecParams->fieldID.size + 7) >> 3;  
  911     olen = ecParams->order.len;  
  912     if (signature->len == 0 || signature->len%2 != 0 ||
  913         signature->len > 2*olen) {
  914         PORT_SetError(SEC_ERROR_INPUT_LEN);
  915         goto cleanup;
  916     }
  917     slen = signature->len/2;
  918 
  919     SECITEM_AllocItem(NULL, &pointC, 2*flen + 1);
  920     if (pointC.data == NULL)
  921         goto cleanup;
  922 
  923     CHECK_MPI_OK( mp_init(&r_) );
  924     CHECK_MPI_OK( mp_init(&s_) );
  925     CHECK_MPI_OK( mp_init(&c)  );
  926     CHECK_MPI_OK( mp_init(&u1) );
  927     CHECK_MPI_OK( mp_init(&u2) );
  928     CHECK_MPI_OK( mp_init(&x1)  );
  929     CHECK_MPI_OK( mp_init(&v)  );
  930     CHECK_MPI_OK( mp_init(&n)  );
  931 
  932     /*
  933     ** Convert received signature (r', s') into MPI integers.
  934     */
  935     CHECK_MPI_OK( mp_read_unsigned_octets(&r_, signature->data, slen) );
  936     CHECK_MPI_OK( mp_read_unsigned_octets(&s_, signature->data + slen, slen) );
  937                                           
  938     /* 
  939     ** ANSI X9.62, Section 5.4.2, Steps 1 and 2
  940     **
  941     ** Verify that 0 < r' < n and 0 < s' < n
  942     */
  943     SECITEM_TO_MPINT(ecParams->order, &n);
  944     if (mp_cmp_z(&r_) <= 0 || mp_cmp_z(&s_) <= 0 ||
  945         mp_cmp(&r_, &n) >= 0 || mp_cmp(&s_, &n) >= 0) {
  946         PORT_SetError(SEC_ERROR_BAD_SIGNATURE);
  947         goto cleanup; /* will return rv == SECFailure */
  948     }
  949 
  950     /*
  951     ** ANSI X9.62, Section 5.4.2, Step 3
  952     **
  953     ** c = (s')**-1 mod n
  954     */
  955     CHECK_MPI_OK( mp_invmod(&s_, &n, &c) );      /* c = (s')**-1 mod n */
  956 
  957     /*
  958     ** ANSI X9.62, Section 5.4.2, Step 4
  959     **
  960     ** u1 = ((HASH(M')) * c) mod n
  961     */
  962     SECITEM_TO_MPINT(*digest, &u1);                  /* u1 = HASH(M)     */
  963 
  964     /* In the definition of EC signing, digests are truncated
  965      * to the length of n in bits. 
  966      * (see SEC 1 "Elliptic Curve Digit Signature Algorithm" section 4.1.*/
  967     if (digest->len*8 > ecParams->fieldID.size) {  /* u1 = HASH(M')     */
  968         mpl_rsh(&u1,&u1,digest->len*8- ecParams->fieldID.size);
  969     }
  970 
  971 #if EC_DEBUG
  972     mp_todecimal(&r_, mpstr);
  973     printf("r_: %s (dec)\n", mpstr);
  974     mp_todecimal(&s_, mpstr);
  975     printf("s_: %s (dec)\n", mpstr);
  976     mp_todecimal(&c, mpstr);
  977     printf("c : %s (dec)\n", mpstr);
  978     mp_todecimal(&u1, mpstr);
  979     printf("digest: %s (dec)\n", mpstr);
  980 #endif
  981 
  982     CHECK_MPI_OK( mp_mulmod(&u1, &c, &n, &u1) );  /* u1 = u1 * c mod n */
  983 
  984     /*
  985     ** ANSI X9.62, Section 5.4.2, Step 4
  986     **
  987     ** u2 = ((r') * c) mod n
  988     */
  989     CHECK_MPI_OK( mp_mulmod(&r_, &c, &n, &u2) );
  990 
  991     /*
  992     ** ANSI X9.62, Section 5.4.3, Step 1
  993     **
  994     ** Compute u1*G + u2*Q
  995     ** Here, A = u1.G     B = u2.Q    and   C = A + B
  996     ** If the result, C, is the point at infinity, reject the signature
  997     */
  998     if (ec_points_mul(ecParams, &u1, &u2, &key->publicValue, &pointC)
  999         != SECSuccess) {
 1000         rv = SECFailure;
 1001         goto cleanup;
 1002     }
 1003     if (ec_point_at_infinity(&pointC)) {
 1004         PORT_SetError(SEC_ERROR_BAD_SIGNATURE);
 1005         rv = SECFailure;
 1006         goto cleanup;
 1007     }
 1008 
 1009     CHECK_MPI_OK( mp_read_unsigned_octets(&x1, pointC.data + 1, flen) );
 1010 
 1011     /*
 1012     ** ANSI X9.62, Section 5.4.4, Step 2
 1013     **
 1014     ** v = x1 mod n
 1015     */
 1016     CHECK_MPI_OK( mp_mod(&x1, &n, &v) );
 1017 
 1018 #if EC_DEBUG
 1019     mp_todecimal(&r_, mpstr);
 1020     printf("r_: %s (dec)\n", mpstr);
 1021     mp_todecimal(&v, mpstr);
 1022     printf("v : %s (dec)\n", mpstr);
 1023 #endif
 1024 
 1025     /*
 1026     ** ANSI X9.62, Section 5.4.4, Step 3
 1027     **
 1028     ** Verification:  v == r'
 1029     */
 1030     if (mp_cmp(&v, &r_)) {
 1031         PORT_SetError(SEC_ERROR_BAD_SIGNATURE);
 1032         rv = SECFailure; /* Signature failed to verify. */
 1033     } else {
 1034         rv = SECSuccess; /* Signature verified. */
 1035     }
 1036 
 1037 #if EC_DEBUG
 1038     mp_todecimal(&u1, mpstr);
 1039     printf("u1: %s (dec)\n", mpstr);
 1040     mp_todecimal(&u2, mpstr);
 1041     printf("u2: %s (dec)\n", mpstr);
 1042     mp_tohex(&x1, mpstr);
 1043     printf("x1: %s\n", mpstr);
 1044     mp_todecimal(&v, mpstr);
 1045     printf("v : %s (dec)\n", mpstr);
 1046 #endif
 1047 
 1048 cleanup:
 1049     mp_clear(&r_);
 1050     mp_clear(&s_);
 1051     mp_clear(&c);
 1052     mp_clear(&u1);
 1053     mp_clear(&u2);
 1054     mp_clear(&x1);
 1055     mp_clear(&v);
 1056     mp_clear(&n);
 1057 
 1058     if (pointC.data) SECITEM_FreeItem(&pointC, PR_FALSE);
 1059     if (err) {
 1060         MP_TO_SEC_ERROR(err);
 1061         rv = SECFailure;
 1062     }
 1063 
 1064 #if EC_DEBUG
 1065     printf("ECDSA verification %s\n",
 1066         (rv == SECSuccess) ? "succeeded" : "failed");
 1067 #endif
 1068 #else
 1069     PORT_SetError(SEC_ERROR_UNSUPPORTED_KEYALG);
 1070 #endif /* NSS_ENABLE_ECC */
 1071 
 1072     return rv;
 1073 }
