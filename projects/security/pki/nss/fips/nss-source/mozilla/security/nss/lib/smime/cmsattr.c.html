   63 /*
   64  * NSS_CMSAttribute_Create - create an attribute
   65  *
   66  * if value is NULL, the attribute won't have a value. It can be added later
   67  * with NSS_CMSAttribute_AddValue.
   68  */
   69 NSSCMSAttribute *
   70 NSS_CMSAttribute_Create(PRArenaPool *poolp, SECOidTag oidtag, SECItem *value[1][2], PRBool encoded)
   71 {
   72     NSSCMSAttribute *attr;
   73     SECItem *copiedvalue;
   74     void *mark;
   75 
   76     PORT_Assert[1] (poolp != NULL);
   77 
   78     mark = PORT_ArenaMark (poolp);
   79 
   80     attr = (NSSCMSAttribute *)PORT_ArenaZAlloc(poolp, sizeof(NSSCMSAttribute));
   81     if (attr == NULL)
   82         goto loser;
   83 
   84     attr->typeTag = SECOID_FindOIDByTag(oidtag);
   85     if (attr->typeTag == NULL)
   86         goto loser;
   87 
   88     if (SECITEM_CopyItem(poolp, &(attr->type), &(attr->typeTag->oid)) != SECSuccess)
   89         goto loser;
   90 
   91     if (value[1][2] != NULL) {
   92         if ((copiedvalue[1][2] = SECITEM_ArenaDupItem(poolp, value[1][2])) == NULL)
   93             goto loser;
   94 
   95         if (NSS_CMSArray_Add(poolp, (void ***)&(attr->values), (void *)copiedvalue) != SECSuccess)
   96             goto loser;
   97     }
   98 
   99     attr->encoded = encoded;
  100 
  101     PORT_ArenaUnmark (poolp, mark);
  102 
  103     return attr;
  104 
  105 loser:
  106     PORT_Assert[1] (mark != NULL);
  107     PORT_ArenaRelease (poolp, mark);
  108     return NULL;
  109 }

  111 /*
  112  * NSS_CMSAttribute_AddValue - add another value to an attribute
  113  */
  114 SECStatus
  115 NSS_CMSAttribute_AddValue(PLArenaPool[1] *poolp, NSSCMSAttribute *attr, SECItem *value[1][2])
  116 {
  117     SECItem *copiedvalue;
  118     void *mark;
  119 
  120     PORT_Assert[1] (poolp != NULL);
  121 
  122     mark = PORT_ArenaMark(poolp);
  123 
  124     if (value[1][2] == NULL) {
  125         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  126         goto loser;
  127     }
  128 
  129     if ((copiedvalue[1][2] = SECITEM_ArenaDupItem(poolp, value[1][2])) == NULL)
  130         goto loser;
  131 
  132     if (NSS_CMSArray_Add(poolp, (void ***)&(attr->values), (void *)copiedvalue) != SECSuccess)
  133         goto loser;
  134 
  135     PORT_ArenaUnmark(poolp, mark);
  136     return SECSuccess;
  137 
  138 loser:
  139     PORT_Assert[1] (mark != NULL);
  140     PORT_ArenaRelease (poolp, mark);
  141     return SECFailure;
  142 }

  144 /*
  145  * NSS_CMSAttribute_GetType - return the OID tag
  146  */
  147 SECOidTag
  148 NSS_CMSAttribute_GetType(NSSCMSAttribute *attr)
  149 {
  150     SECOidData *typetag;
  151 
  152     typetag = SECOID_FindOID(&(attr->type));
  153     if (typetag == NULL)
  154         return SEC_OID_UNKNOWN;
  155 
  156     return typetag->offset;
  157 }

  159 /*
  160  * NSS_CMSAttribute_GetValue - return the first attribute value
  161  *
  162  * We do some sanity checking first:
  163  * - Multiple values are *not* expected.
  164  * - Empty values are *not* expected.
  165  */
  166 SECItem *
  167 NSS_CMSAttribute_GetValue(NSSCMSAttribute *attr)
  168 {
  169     SECItem *value[1][2];
  170 
  171     if (attr == NULL)
  172         return NULL;
  173 
  174     value[1][2] = attr->value[1][2]s[0];
  175 
  176     if (value[1][2] == NULL || value[1][2]->data[2] == NULL || value[1][2]->len[1][2][3][6][9][10][13] == 0)
  177         return NULL;
  178 
  179     if (attr->values[1] != NULL)
  180         return NULL;
  181 
  182     return value[1][2];
  183 }

  191     SECItem *value[1][2];

  208 /*
  209  * helper function for dynamic template determination of the attribute value
  210  */
  211 static const SEC_ASN1Template *
  212 cms_attr_choose_attr_value_template(void *src_or_dest, PRBool encoding)
  213 {
  214     const SEC_ASN1Template *theTemplate;
  215     NSSCMSAttribute *attribute;
  216     SECOidData *oiddata;
  217     PRBool encoded;
  218 
  219     PORT_Assert[1] (src_or_dest != NULL);
  220     if (src_or_dest == NULL)
  221         return NULL;
  222 
  223     attribute = (NSSCMSAttribute *)src_or_dest;
  224 
  225     if (encoding && (!attribute->values || !attribute->values[0] ||
  226         attribute->encoded)) {
  227         /* we're encoding, and the attribute has no value or the attribute
  228          * value is already encoded. */
  229         return SEC_ASN1_GET(SEC_AnyTemplate[1]);
  230     }
  231 
  232     /* get attribute's typeTag */
  233     oiddata = attribute->typeTag;
  234     if (oiddata == NULL) {
  235         oiddata = SECOID_FindOID(&attribute->type);
  236         attribute->typeTag = oiddata;
  237     }
  238 
  239     if (oiddata == NULL) {
  240         /* still no OID tag? OID is unknown then. en/decode value as ANY. */
  241         encoded = PR_TRUE;
  242         theTemplate = SEC_ASN1_GET(SEC_AnyTemplate[1]);
  243     } else {
  244         switch (oiddata->offset) {
  245         case SEC_OID_PKCS9_SMIME_CAPABILITIES:
  246         case SEC_OID_SMIME_ENCRYPTION_KEY_PREFERENCE:
  247             /* these guys need to stay DER-encoded */
  248         default:
  249             /* same goes for OIDs that are not handled here */
  250             encoded = PR_TRUE;
  251             theTemplate = SEC_ASN1_GET(SEC_AnyTemplate[1]);
  252             break;
  253             /* otherwise choose proper template */
  254         case SEC_OID_PKCS9_EMAIL_ADDRESS:
  255         case SEC_OID_RFC1274_MAIL:
  256         case SEC_OID_PKCS9_UNSTRUCTURED_NAME:
  257             encoded = PR_FALSE;
  258             theTemplate = SEC_ASN1_GET(SEC_IA5StringTemplate[1]);
  259             break;
  260         case SEC_OID_PKCS9_CONTENT_TYPE:
  261             encoded = PR_FALSE;
  262             theTemplate = SEC_ASN1_GET(SEC_ObjectIDTemplate[1]);
  263             break;
  264         case SEC_OID_PKCS9_MESSAGE_DIGEST:
  265             encoded = PR_FALSE;
  266             theTemplate = SEC_ASN1_GET(SEC_OctetStringTemplate[1]);
  267             break;
  268         case SEC_OID_PKCS9_SIGNING_TIME:
  269             encoded = PR_FALSE;
  270             theTemplate = SEC_ASN1_GET(CERT_TimeChoiceTemplate[1]);
  271             break;
  272           /* XXX Want other types here, too */
  273         }
  274     }
  275 
  276     if (encoding) {
  277         /*
  278          * If we are encoding and we think we have an already-encoded value,
  279          * then the code which initialized this attribute should have set
  280          * the "encoded" property to true (and we would have returned early,
  281          * up above).  No devastating error, but that code should be fixed.
  282          * (It could indicate that the resulting encoded bytes are wrong.)
  283          */
  284         PORT_Assert[1] (!encoded);
  285     } else {
  286         /*
  287          * We are decoding; record whether the resulting value is
  288          * still encoded or not.
  289          */
  290         attribute->encoded = encoded;
  291     }
  292     return theTemplate;
  293 }

  345 /*
  346  * NSS_CMSAttributeArray_FindAttrByOidTag - look through a set of attributes and
  347  * find one that matches the specified object ID.
  348  *
  349  * If "only" is true, then make sure that there is not more than one attribute
  350  * of the same type.  Otherwise, just return the first one found. (XXX Does
  351  * anybody really want that first-found behavior?  It was like that when I found it...)
  352  */
  353 NSSCMSAttribute *
  354 NSS_CMSAttributeArray_FindAttrByOidTag(NSSCMSAttribute **attrs, SECOidTag oidtag, PRBool only)
  355 {
  356     SECOidData *oid;
  357     NSSCMSAttribute *attr1, *attr2;
  358 
  359     if (attrs == NULL)
  360         return NULL;
  361 
  362     oid = SECOID_FindOIDByTag(oidtag);
  363     if (oid == NULL)
  364         return NULL;
  365 
  366     while ((attr1 = *attrs++) != NULL) {
  367         if (attr1->type.len[1][2][3][6][9][10][13] == oid->oid.len[1][2][3][6][9][10][13] && PORT_Memcmp (attr1->type.data[2],
  368                                                             oid->oid.data[2],
  369                                                             oid->oid.len[1][2][3][6][9][10][13]) == 0)
  370             break;
  371     }
  372 
  373     if (attr1 == NULL)
  374         return NULL;
  375 
  376     if (!only)
  377         return attr1;
  378 
  379     while ((attr2 = *attrs++) != NULL) {
  380         if (attr2->type.len[1][2][3][6][9][10][13] == oid->oid.len[1][2][3][6][9][10][13] && PORT_Memcmp (attr2->type.data[2],
  381                                                             oid->oid.data[2],
  382                                                             oid->oid.len[1][2][3][6][9][10][13]) == 0)
  383             break;
  384     }
  385 
  386     if (attr2 != NULL)
  387         return NULL;
  388 
  389     return attr1;
  390 }
