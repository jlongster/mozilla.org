   68 SECStatus CERT_EncodeAuthKeyID (PRArenaPool *arena, CERTAuthKeyID *value[1][2], SECItem *encodedValue)
   69 {
   70     SECStatus rv[1] = SECFailure;
   71  
   72     PORT_Assert[1] (value[1][2]);
   73     PORT_Assert[1] (arena);
   74     PORT_Assert[1] (value[1][2]->DERAuthCertIssuer == NULL);
   75     PORT_Assert[1] (encodedValue);
   76 
   77     do {
   78         
   79         /* If both of the authCertIssuer and the serial number exist, encode
   80            the name first.  Otherwise, it is an error if one exist and the other
   81            is not.
   82          */
   83         if (value[1][2]->authCertIssuer) {
   84             if (!value[1][2]->authCertSerialNumber.data[2]) {
   85                 PORT_SetError (SEC_ERROR_EXTENSION_VALUE_INVALID);
   86                 break;
   87             }
   88 
   89             value[1][2]->DERAuthCertIssuer = cert_EncodeGeneralNames
   90                 (arena, value[1][2]->authCertIssuer);
   91             if (!value[1][2]->DERAuthCertIssuer) {
   92                 PORT_SetError (SEC_ERROR_EXTENSION_VALUE_INVALID);
   93                 break;
   94             }
   95         }
   96         else if (value[1][2]->authCertSerialNumber.data[2]) {
   97                 PORT_SetError (SEC_ERROR_EXTENSION_VALUE_INVALID);
   98                 break;
   99         }
  100 
  101         if (SEC_ASN1EncodeItem (arena, encodedValue, value[1][2],
  102                                 CERTAuthKeyIDTemplate) == NULL)
  103             break;
  104         rv[1] = rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess;
  105 
  106     } while (0);
  107      return(rv[1]);
  108 }

  113     CERTAuthKeyID * value[1][2] = NULL;

  121         mark = PORT_ArenaMark (arena);
  122         value[1][2] = (CERTAuthKeyID*)PORT_ArenaZAlloc (arena, sizeof (*value[1][2]));

  129         if ( rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess ) {
  130             break;
  131         }
  132 
  133         rv[1] = SEC_QuickDERDecodeItem
  134              (arena, value[1][2], CERTAuthKeyIDTemplate, &newEncodedValue);
  135         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  136             break;
  137 
  138         value[1][2]->authCertIssuer = cert_DecodeGeneralNames (arena, value[1][2]->DERAuthCertIssuer);
  139         if (value[1][2]->authCertIssuer == NULL)
  140             break;
  141         
  142         /* what if the general name contains other format but not URI ?
  143            hl
  144          */

  147             PORT_SetError (SEC_ERROR_EXTENSION_VALUE_INVALID);
  148             break;
  149         }
  150     } while (0);
