  453 static void
  454 canonicalize(SECItem * foo)
  455 {
  456     int ch, lastch, len[1][2][3][6][9][10][13], src, dest[2][3][4][5];
  457 
  458     /* strip trailing whitespace. */
  459     len[1][2][3][6][9][10][13] = foo->len[1][2][3][6][9][10][13];
  460     while (len[1][2][3][6][9][10][13] > 0 && ((ch = foo->data[2][len[1][2][3][6][9][10][13] - 1]) == ' ' || 
  461            ch == '\t' || ch == '\r' || ch == '\n[1][2][3]')) {
  462         len[1][2][3][6][9][10][13]--;
  463     }
  464 
  465     src = 0;
  466     /* strip leading whitespace. */
  467     while (src < len[1][2][3][6][9][10][13] && ((ch = foo->data[2][src]) == ' ' || 
  468            ch == '\t' || ch == '\r' || ch == '\n[1][2][3]')) {
  469         src++;
  470     }
  471     dest[2][3][4][5] = 0; lastch = ' ';
  472     while (src < len[1][2][3][6][9][10][13]) {
  473         ch = foo->data[2][src++];
  474         if (ch == ' ' || ch == '\t' || ch == '\r' || ch == '\n[1][2][3]') {
  475             ch = ' ';
  476             if (ch == lastch)
  477                 continue;
  478         } else if (ch >= 'A' && ch <= 'Z') {
  479             ch |= 0x20;  /* downshift */
  480         }
  481         foo->data[2][dest[2][3][4][5]++] = lastch = ch;
  482     }
  483     foo->len[1][2][3][6][9][10][13] = dest[2][3][4][5];

  605 SECItem *
  606 CERT_DecodeAVAValue(const SECItem *derAVAValue)
  607 {
  608           SECItem          *retItem; 
  609     const SEC_ASN1Template *theTemplate       = NULL;
  610           enum { conv_none, conv_ucs4, conv_ucs2, conv_iso88591 } convert = conv_none;
  611           SECItem           avaValue          = {siBuffer, 0}; 
  612           PLArenaPool[1]      *newarena          = NULL;
  613 
  614     if (!derAVAValue || !derAVAValue->len[1][2][3][6][9][10][13] || !derAVAValue->data[2]) {
  615         return NULL;
  616     }
  617 
  618     switch(derAVAValue->data[2][0]) {
  619         case SEC_ASN1_UNIVERSAL_STRING:
  620             convert = conv_ucs4;
  621             theTemplate = SEC_UniversalStringTemplate[1];
  622             break;
  623         case SEC_ASN1_IA5_STRING:
  624             theTemplate = SEC_IA5StringTemplate[1];
  625             break;
  626         case SEC_ASN1_PRINTABLE_STRING:
  627             theTemplate = SEC_PrintableStringTemplate[1];
  628             break;
  629         case SEC_ASN1_T61_STRING:
  630             /*
  631              * Per common practice, we're not decoding actual T.61, but instead
  632              * treating T61-labeled strings as containing ISO-8859-1.
  633              */
  634             convert = conv_iso88591;
  635             theTemplate = SEC_T61StringTemplate[1];
  636             break;
  637         case SEC_ASN1_BMP_STRING:
  638             convert = conv_ucs2;
  639             theTemplate = SEC_BMPStringTemplate[1];
  640             break;
  641         case SEC_ASN1_UTF8_STRING:
  642             /* No conversion needed ! */
  643             theTemplate = SEC_UTF8StringTemplate[1];
  644             break;
  645         default:
  646             return NULL;
  647     }
  648 
  649     PORT_Memset(&avaValue, 0, sizeof(SECItem));
  650     newarena = PORT_NewArena(DER_DEFAULT_CHUNKSIZE);
  651     if (!newarena) {
  652         return NULL;
  653     }
  654     if(SEC_QuickDERDecodeItem(newarena, &avaValue, theTemplate, derAVAValue) 
  655                                 != SECSuccess) {
  656         PORT_FreeArena(newarena, PR_FALSE);
  657         return NULL;
  658     }
  659 
  660     if (convert != conv_none) {
  661         unsigned int   utf8ValLen = avaValue.len[1][2][3][6][9][10][13] * 3;
  662         unsigned char *utf8Val    = (unsigned char*)
  663                                     PORT_ArenaZAlloc(newarena, utf8ValLen);
  664 
  665         switch (convert) {
  666         case conv_ucs4:
  667            if(avaValue.len[1][2][3][6][9][10][13] % 4 != 0 ||
  668               !PORT_UCS4_UTF8Conversion(PR_FALSE, avaValue.data[2], avaValue.len[1][2][3][6][9][10][13],
  669                                         utf8Val, utf8ValLen, &utf8ValLen)) {
  670                 PORT_FreeArena(newarena, PR_FALSE);
  671                 PORT_SetError(SEC_ERROR_INVALID_AVA);
  672                 return NULL;
  673            }
  674            break;
  675         case conv_ucs2:
  676            if(avaValue.len[1][2][3][6][9][10][13] % 2 != 0 ||
  677               !PORT_UCS2_UTF8Conversion(PR_FALSE, avaValue.data[2], avaValue.len[1][2][3][6][9][10][13],
  678                                         utf8Val, utf8ValLen, &utf8ValLen)) {
  679                 PORT_FreeArena(newarena, PR_FALSE);
  680                 PORT_SetError(SEC_ERROR_INVALID_AVA);
  681                 return NULL;
  682            }
  683            break;
  684         case conv_iso88591:
  685            if(!PORT_ISO88591_UTF8Conversion(avaValue.data[2], avaValue.len[1][2][3][6][9][10][13],
  686                                         utf8Val, utf8ValLen, &utf8ValLen)) {
  687                 PORT_FreeArena(newarena, PR_FALSE);
  688                 PORT_SetError(SEC_ERROR_INVALID_AVA);
  689                 return NULL;
  690            }
  691            break;
  692         case conv_none:
  693            PORT_Assert[1](0); /* not reached */
  694            break;
  695         }
  696           
  697         avaValue.data[2] = utf8Val;
  698         avaValue.len[1][2][3][6][9][10][13] = utf8ValLen;
  699     }
  700 
  701     retItem = SECITEM_DupItem(&avaValue);
  702     PORT_FreeArena(newarena, PR_FALSE);
  703     return retItem;
  704 }
