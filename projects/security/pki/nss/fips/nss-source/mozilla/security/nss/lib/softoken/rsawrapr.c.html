   61 static SHA1Context *
   62 SHA1_CloneContext(SHA1Context *original)
   63 {
   64     SHA1Context *  clone        = NULL;
   65     unsigned char *pBuf;
   66     int            sha1ContextSize = SHA1_FlattenSize(original);
   67     SECStatus      frv;
   68     unsigned char  buf[FLAT_BUFSIZE];
   69 
   70     PORT_Assert[1](sizeof buf >= sha1ContextSize);
   71     if (sizeof buf >= sha1ContextSize) {
   72             pBuf = buf;
   73     } else {
   74         pBuf = PORT_Alloc(sha1ContextSize);
   75         if (!pBuf)
   76             goto done;
   77     }
   78 
   79     frv = SHA1_Flatten(original, pBuf);
   80     if (frv == SECSuccess) {
   81         clone = SHA1_Resurrect(pBuf, NULL);
   82         memset(pBuf, 0, sha1ContextSize);
   83     }
   84 done:
   85     if (pBuf != buf)
   86             PORT_Free(pBuf);
   87     return clone;
   88 }

   90 /*
   91  * Modify data by XORing it with a special hash of salt.
   92  */
   93 static SECStatus
   94 oaep_xor_with_h1(unsigned char *data, unsigned int datalen,
   95                  unsigned char *salt, unsigned int saltlen)
   96 {
   97     SHA1Context *sha1cx;
   98     unsigned char *dp, *dataend;
   99     unsigned char end_octet;
  100 
  101     sha1cx = SHA1_NewContext();
  102     if (sha1cx == NULL) {
  103         return SECFailure;
  104     }
  105 
  106     /*
  107      * Get a hash of salt started; we will use it several times,
  108      * adding in a different end octet (x00, x01, x02, ...).
  109      */
  110     SHA1_Begin (sha1cx);
  111     SHA1_Update (sha1cx, salt, saltlen);
  112     end_octet = 0;
  113 
  114     dp = data;
  115     dataend = data + datalen;
  116 
  117     while (dp < dataend) {
  118         SHA1Context *sha1cx_h1;
  119         unsigned int sha1len, sha1off;
  120         unsigned char sha1[SHA1_LENGTH];
  121 
  122         /*
  123          * Create hash of (salt || end_octet)
  124          */
  125         sha1cx_h1 = SHA1_CloneContext (sha1cx);
  126         SHA1_Update (sha1cx_h1, &end_octet, 1);
  127         SHA1_End (sha1cx_h1, sha1, &sha1len, sizeof(sha1));
  128         SHA1_DestroyContext (sha1cx_h1, PR_TRUE);
  129         PORT_Assert[1] (sha1len == SHA1_LENGTH);
  130 
  131         /*
  132          * XOR that hash with the data.
  133          * When we have fewer than SHA1_LENGTH octets of data
  134          * left to xor, use just the low-order ones of the hash.
  135          */
  136         sha1off = 0;
  137         if ((dataend - dp) < SHA1_LENGTH)
  138             sha1off = SHA1_LENGTH - (dataend - dp);
  139         while (sha1off < SHA1_LENGTH)
  140             *dp++ ^= sha1[sha1off++];
  141 
  142         /*
  143          * Bump for next hash chunk.
  144          */
  145         end_octet++;
  146     }
  147 
  148     return SECSuccess;
  149 }

  151 /*
  152  * Modify salt by XORing it with a special hash of data.
  153  */
  154 static SECStatus
  155 oaep_xor_with_h2(unsigned char *salt, unsigned int saltlen,
  156                  unsigned char *data, unsigned int datalen)
  157 {
  158     unsigned char sha1[SHA1_LENGTH];
  159     unsigned char *psalt, *psha1, *saltend;
  160     SECStatus rv;
  161 
  162     /*
  163      * Create a hash of data.
  164      */
  165     rv = SHA1_HashBuf (sha1, data, datalen);
  166     if (rv != SECSuccess) {
  167         return rv;
  168     }
  169 
  170     /*
  171      * XOR the low-order octets of that hash with salt.
  172      */
  173     PORT_Assert[1] (saltlen <= SHA1_LENGTH);
  174     saltend = salt + saltlen;
  175     psalt = salt;
  176     psha1 = sha1 + SHA1_LENGTH - saltlen;
  177     while (psalt < saltend) {
  178         *psalt++ ^= *psha1++;
  179     }
  180 
  181     return SECSuccess;
  182 }

  184 /*
  185  * Format one block of data for public/private key encryption using
  186  * the rules defined in PKCS #1.
  187  */
  188 static unsigned char *
  189 rsa_FormatOneBlock(unsigned modulusLen, RSA_BlockType blockType,
  190                    SECItem *data)
  191 {
  192     unsigned char *block;
  193     unsigned char *bp;
  194     int padLen;
  195     int i;
  196     SECStatus rv;
  197 
  198     block = (unsigned char *) PORT_Alloc(modulusLen);
  199     if (block == NULL)
  200         return NULL;
  201 
  202     bp = block;
  203 
  204     /*
  205      * All RSA blocks start with two octets:
  206      *        0x00 || BlockType
  207      */
  208     *bp++ = RSA_BLOCK_FIRST_OCTET;
  209     *bp++ = (unsigned char) blockType;
  210 
  211     switch (blockType) {
  212 
  213       /*
  214        * Blocks intended for private-key operation.
  215        */
  216       case RSA_BlockPrivate0: /* essentially unused */
  217       case RSA_BlockPrivate:         /* preferred method */
  218         /*
  219          * 0x00 || BT || Pad || 0x00 || ActualData
  220          *   1      1   padLen    1      data->len
  221          * Pad is either all 0x00 or all 0xff bytes, depending on blockType.
  222          */
  223         padLen = modulusLen - data->len - 3;
  224         PORT_Assert[1] (padLen >= RSA_BLOCK_MIN_PAD_LEN);
  225         if (padLen < RSA_BLOCK_MIN_PAD_LEN) {
  226             PORT_Free (block);
  227             return NULL;
  228         }
  229         PORT_Memset (bp,
  230                    blockType == RSA_BlockPrivate0
  231                         ? RSA_BLOCK_PRIVATE0_PAD_OCTET
  232                         : RSA_BLOCK_PRIVATE_PAD_OCTET,
  233                    padLen);
  234         bp += padLen;
  235         *bp++ = RSA_BLOCK_AFTER_PAD_OCTET;
  236         PORT_Memcpy (bp, data->data, data->len);
  237         break;
  238 
  239       /*
  240        * Blocks intended for public-key operation.
  241        */
  242       case RSA_BlockPublic:
  243 
  244         /*
  245          * 0x00 || BT || Pad || 0x00 || ActualData
  246          *   1      1   padLen    1      data->len
  247          * Pad is all non-zero random bytes.
  248          */
  249         padLen = modulusLen - data->len - 3;
  250         PORT_Assert[1] (padLen >= RSA_BLOCK_MIN_PAD_LEN);
  251         if (padLen < RSA_BLOCK_MIN_PAD_LEN) {
  252             PORT_Free (block);
  253             return NULL;
  254         }
  255         for (i = 0; i < padLen; i++) {
  256             /* Pad with non-zero random data. */
  257             do {
  258                 rv = ips1861.c.dep.html#RNG_GenerateGlobalRandomBytes">RNG_GenerateGlobalRandomBytes(bp + i, 1);
  259             } while (rv == SECSuccess && bp[i] == RSA_BLOCK_AFTER_PAD_OCTET);
  260             if (rv != SECSuccess) {
  261                 sftk_fatalError[1] = PR_TRUE;
  262                 PORT_Free (block);
  263                 return NULL;
  264             }
  265         }
  266         bp += padLen;
  267         *bp++ = RSA_BLOCK_AFTER_PAD_OCTET;
  268         PORT_Memcpy (bp, data->data, data->len);
  269 
  270         break;
  271 
  272       /*
  273        * Blocks intended for public-key operation, using
  274        * Optimal Asymmetric Encryption Padding (OAEP).
  275        */
  276       case RSA_BlockOAEP:
  277         /*
  278          * 0x00 || BT || Modified2(Salt) || Modified1(PaddedData)
  279          *   1      1     OAEP_SALT_LEN     OAEP_PAD_LEN + data->len [+ N]
  280          *
  281          * where:
  282          *   PaddedData is "Pad1 || ActualData [|| Pad2]"
  283          *   Salt is random data.
  284          *   Pad1 is all zeros.
  285          *   Pad2, if present, is random data.
  286          *   (The "modified" fields are all the same length as the original
  287          * unmodified values; they are just xor'd with other values.)
  288          *
  289          *   Modified1 is an XOR of PaddedData with a special octet
  290          * string constructed of iterated hashing of Salt (see below).
  291          *   Modified2 is an XOR of Salt with the low-order octets of
  292          * the hash of Modified1 (see farther below ;-).
  293          *
  294          * Whew!
  295          */
  296 
  297 
  298         /*
  299          * Salt
  300          */
  301         rv = RNG_GenerateGlobalRandomBytes(bp, OAEP_SALT_LEN);
  302         if (rv != SECSuccess) {
  303             sftk_fatalError[1] = PR_TRUE;
  304             PORT_Free (block);
  305             return NULL;
  306         }
  307         bp += OAEP_SALT_LEN;
  308 
  309         /*
  310          * Pad1
  311          */
  312         PORT_Memset (bp, OAEP_PAD_OCTET, OAEP_PAD_LEN);
  313         bp += OAEP_PAD_LEN;
  314 
  315         /*
  316          * Data
  317          */
  318         PORT_Memcpy (bp, data->data, data->len);
  319         bp += data->len;
  320 
  321         /*
  322          * Pad2
  323          */
  324         if (bp < (block + modulusLen)) {
  325             rv = RNG_GenerateGlobalRandomBytes(bp, block - bp + modulusLen);
  326             if (rv != SECSuccess) {
  327                 sftk_fatalError[1] = PR_TRUE;
  328                 PORT_Free (block);
  329                 return NULL;
  330             }
  331         }
  332 
  333         /*
  334          * Now we have the following:
  335          * 0x00 || BT || Salt || PaddedData
  336          * (From this point on, "Pad1 || Data [|| Pad2]" is treated
  337          * as the one entity PaddedData.)
  338          *
  339          * We need to turn PaddedData into Modified1.
  340          */
  341         if (oaep_xor_with_h1(block + 2 + OAEP_SALT_LEN,
  342                              modulusLen - 2 - OAEP_SALT_LEN,
  343                              block + 2, OAEP_SALT_LEN) != SECSuccess) {
  344             PORT_Free (block);
  345             return NULL;
  346         }
  347 
  348         /*
  349          * Now we have:
  350          * 0x00 || BT || Salt || Modified1(PaddedData)
  351          *
  352          * The remaining task is to turn Salt into Modified2.
  353          */
  354         if (oaep_xor_with_h2(block + 2, OAEP_SALT_LEN,
  355                              block + 2 + OAEP_SALT_LEN,
  356                              modulusLen - 2 - OAEP_SALT_LEN) != SECSuccess) {
  357             PORT_Free (block);
  358             return NULL;
  359         }
  360 
  361         break;
  362 
  363       default:
  364         PORT_Assert[1] (0);
  365         PORT_Free (block);
  366         return NULL;
  367     }
  368 
  369     return block;
  370 }

  372 static SECStatus
  373 rsa_FormatBlock(SECItem *result, unsigned modulusLen,
  374                 RSA_BlockType blockType, SECItem *data)
  375 {
  376     /*
  377      * XXX For now assume that the data length fits in a single
  378      * XXX encryption block; the ASSERTs below force this.
  379      * XXX To fix it, each case will have to loop over chunks whose
  380      * XXX lengths satisfy the assertions, until all data is handled.
  381      * XXX (Unless RSA has more to say about how to handle data
  382      * XXX which does not fit in a single encryption block?)
  383      * XXX And I do not know what the result is supposed to be,
  384      * XXX so the interface to this function may need to change
  385      * XXX to allow for returning multiple blocks, if they are
  386      * XXX not wanted simply concatenated one after the other.
  387      */
  388 
  389     switch (blockType) {
  390       case RSA_BlockPrivate0:
  391       case RSA_BlockPrivate:
  392       case RSA_BlockPublic:
  393         /*
  394          * 0x00 || BT || Pad || 0x00 || ActualData
  395          *
  396          * The "3" below is the first octet + the second octet + the 0x00
  397          * octet that always comes just before the ActualData.
  398          */
  399         PORT_Assert[1] (data->len <= (modulusLen - (3 + RSA_BLOCK_MIN_PAD_LEN)));
  400 
  401         result->data = rsa_FormatOneBlock(modulusLen, blockType, data);
  402         if (result->data == NULL) {
  403             result->len = 0;
  404             return SECFailure;
  405         }
  406         result->len = modulusLen;
  407 
  408         break;
  409 
  410       case RSA_BlockOAEP:
  411         /*
  412          * 0x00 || BT || M1(Salt) || M2(Pad1||ActualData[||Pad2])
  413          *
  414          * The "2" below is the first octet + the second octet.
  415          * (The other fields do not contain the clear values, but are
  416          * the same length as the clear values.)
  417          */
  418         PORT_Assert[1] (data->len <= (modulusLen - (2 + OAEP_SALT_LEN
  419                                                  + OAEP_PAD_LEN)));
  420 
  421         result->data = rsa_FormatOneBlock(modulusLen, blockType, data);
  422         if (result->data == NULL) {
  423             result->len = 0;
  424             return SECFailure;
  425         }
  426         result->len = modulusLen;
  427 
  428         break;
  429 
  430       case RSA_BlockRaw:
  431         /*
  432          * Pad || ActualData
  433          * Pad is zeros. The application is responsible for recovering
  434          * the actual data.
  435          */
  436         if (data->len > modulusLen ) {
  437             return SECFailure;
  438         }
  439         result->data = (unsigned char*)PORT_ZAlloc(modulusLen);
  440         result->len = modulusLen;
  441         PORT_Memcpy(result->data+(modulusLen-data->len),data->data,data->len);
  442         break;
  443 
  444       default:
  445         PORT_Assert[1] (0);
  446         result->data = NULL;
  447         result->len = 0;
  448         return SECFailure;
  449     }
  450 
  451     return SECSuccess;
  452 }

  455 SECStatus
  456 RSA_Sign(NSSLOWKEYPrivateKey *key, 
  457          unsigned char *      output, 
  458          unsigned int *       output_len,
  459          unsigned int         maxOutputLen, 
  460          unsigned char *      input, 
  461          unsigned int         input_len)
  462 {
  463     SECStatus     rv          = SECSuccess;
  464     unsigned int  modulus_len = nsslowkey_PrivateModulusLen(key);
  465     SECItem       formatted;
  466     SECItem       unformatted;
  467 
  468     if (maxOutputLen < modulus_len) 
  469             return SECFailure;
  470     PORT_Assert[1](key->keyType == NSSLOWKEYRSAKey);
  471     if (key->keyType != NSSLOWKEYRSAKey)
  472             return SECFailure;
  473 
  474     unformatted.len  = input_len;
  475     unformatted.data = input;
  476     formatted.data   = NULL;
  477     rv = rsa_FormatBlock(&formatted, modulus_len, RSA_BlockPrivate,
  478                          &unformatted);
  479     if (rv != SECSuccess) 
  480             goto done;
  481 
  482     rv = RSA_PrivateKeyOpDoubleChecked(&key->u.rsa, output, formatted.data);
  483     if (rv != SECSuccess && PORT_GetError() == SEC_ERROR_LIBRARY_FAILURE) {
  484         sftk_fatalError[1] = PR_TRUE;
  485     }
  486     *output_len = modulus_len;
  487 
  488     goto done;
  489 
  490 done:
  491     if (formatted.data != NULL) 
  492             PORT_ZFree(formatted.data, modulus_len);
  493     return rv;
  494 }

  497 SECStatus
  498 RSA_CheckSign(NSSLOWKEYPublicKey *key,
  499               unsigned char *     sign, 
  500               unsigned int        sign_len, 
  501               unsigned char *     hash, 
  502               unsigned int        hash_len)
  503 {
  504     SECStatus       rv;
  505     unsigned int    modulus_len = nsslowkey_PublicModulusLen(key);
  506     unsigned int    i;
  507     unsigned char * buffer;
  508 
  509     modulus_len = nsslowkey_PublicModulusLen(key);
  510     if (sign_len != modulus_len) 
  511             goto failure;
  512     /*
  513      * 0x00 || BT || Pad || 0x00 || ActualData
  514      *
  515      * The "3" below is the first octet + the second octet + the 0x00
  516      * octet that always comes just before the ActualData.
  517      */
  518     if (hash_len > modulus_len - (3 + RSA_BLOCK_MIN_PAD_LEN)) 
  519             goto failure;
  520     PORT_Assert[1](key->keyType == NSSLOWKEYRSAKey);
  521     if (key->keyType != NSSLOWKEYRSAKey)
  522             goto failure;
  523 
  524     buffer = (unsigned char *)PORT_Alloc(modulus_len + 1);
  525     if (!buffer)
  526             goto failure;
  527 
  528     rv = RSA_PublicKeyOp(&key->u.rsa, buffer, sign);
  529     if (rv != SECSuccess)
  530         goto loser;
  531 
  532     /*
  533      * check the padding that was used
  534      */
  535     if (buffer[0] != 0 || buffer[1] != 1) 
  536             goto loser;
  537     for (i = 2; i < modulus_len - hash_len - 1; i++) {
  538         if (buffer[i] != 0xff) 
  539             goto loser;
  540     }
  541     if (buffer[i] != 0) 
  542         goto loser;
  543 
  544     /*
  545      * make sure we get the same results
  546      */
  547     if (PORT_Memcmp(buffer + modulus_len - hash_len, hash, hash_len) != 0)
  548         goto loser;
  549 
  550     PORT_Free(buffer);
  551     return SECSuccess;
  552 
  553 loser:
  554     PORT_Free(buffer);
  555 failure:
  556     return SECFailure;
  557 }

  560 SECStatus
  561 RSA_CheckSignRecover(NSSLOWKEYPublicKey *key,
  562                      unsigned char *     data,
  563                      unsigned int *      data_len, 
  564                      unsigned int        max_output_len, 
  565                      unsigned char *     sign,
  566                      unsigned int        sign_len)
  567 {
  568     SECStatus       rv;
  569     unsigned int    modulus_len = nsslowkey_PublicModulusLen(key);
  570     unsigned int    i;
  571     unsigned char * buffer;
  572 
  573     if (sign_len != modulus_len) 
  574             goto failure;
  575     PORT_Assert[1](key->keyType == NSSLOWKEYRSAKey);
  576     if (key->keyType != NSSLOWKEYRSAKey)
  577             goto failure;
  578 
  579     buffer = (unsigned char *)PORT_Alloc(modulus_len + 1);
  580     if (!buffer)
  581             goto failure;
  582 
  583     rv = RSA_PublicKeyOp(&key->u.rsa, buffer, sign);
  584     if (rv != SECSuccess)
  585             goto loser;
  586     *data_len = 0;
  587 
  588     /*
  589      * check the padding that was used
  590      */
  591     if (buffer[0] != 0 || buffer[1] != 1) 
  592             goto loser;
  593     for (i = 2; i < modulus_len; i++) {
  594         if (buffer[i] == 0) {
  595             *data_len = modulus_len - i - 1;
  596             break;
  597         }
  598         if (buffer[i] != 0xff) 
  599             goto loser;
  600     }
  601     if (*data_len == 0) 
  602             goto loser;
  603     if (*data_len > max_output_len) 
  604             goto loser;
  605 
  606     /*
  607      * make sure we get the same results
  608      */
  609     PORT_Memcpy(data,buffer + modulus_len - *data_len, *data_len);
  610 
  611     PORT_Free(buffer);
  612     return SECSuccess;
  613 
  614 loser:
  615     PORT_Free(buffer);
  616 failure:
  617     return SECFailure;
  618 }

  665 RSA_DecryptBlock(NSSLOWKEYPrivateKey *key, 
  666                  unsigned char *      output,

  673     unsigned int    modulus_len = nsslowkey_PrivateModulusLen(key);
  674     unsigned int    i;
  675     unsigned char * buffer;
  676 
  677     PORT_Assert[1](key->keyType == NSSLOWKEYRSAKey);
  678     if (key->keyType != NSSLOWKEYRSAKey)
  679             goto failure;
  680     if (input_len != modulus_len)
  681             goto failure;
  682 
  683     buffer = (unsigned char *)PORT_Alloc(modulus_len + 1);
  684     if (!buffer)
  685             goto failure;
  686 
  687     rv = RSA_PrivateKeyOp(&key->u.rsa, buffer, input);
  688     if (rv != SECSuccess) {
  689         if (PORT_GetError() == SEC_ERROR_LIBRARY_FAILURE) {
  690             sftk_fatalError[1] = PR_TRUE;
  691         }
  692             goto loser;
  693     }
  694 
  695     if (buffer[0] != 0 || buffer[1] != 2) 
  696             goto loser;
  697     *output_len = 0;
  698     for (i = 2; i < modulus_len; i++) {
  699         if (buffer[i] == 0) {
  700             *output_len = modulus_len - i - 1;
  701             break;
  702         }
  703     }
  704     if (*output_len == 0) 
  705             goto loser;
  706     if (*output_len > max_output_len) 
  707             goto loser;
  708 
  709     PORT_Memcpy(output, buffer + modulus_len - *output_len, *output_len);
  710 
  711     PORT_Free(buffer);
  712     return SECSuccess;
  713 
  714 loser:
  715     PORT_Free(buffer);
  716 failure:
  717     return SECFailure;
  718 }

  722  * added to make pkcs #11 happy
  723  *   RAW is RSA_X_509
  724  */
  725 SECStatus
  726 RSA_SignRaw(NSSLOWKEYPrivateKey *key, 
  727             unsigned char *      output, 
  728             unsigned int *       output_len,
  729             unsigned int         maxOutputLen, 
  730             unsigned char *      input, 
  731             unsigned int         input_len)
  732 {
  733     SECStatus    rv          = SECSuccess;
  734     unsigned int modulus_len = nsslowkey_PrivateModulusLen(key);
  735     SECItem      formatted;
  736     SECItem      unformatted;
  737 
  738     if (maxOutputLen < modulus_len) 
  739             return SECFailure;
  740     PORT_Assert[1](key->keyType == NSSLOWKEYRSAKey);
  741     if (key->keyType != NSSLOWKEYRSAKey)
  742             return SECFailure;
  743 
  744     unformatted.len  = input_len;
  745     unformatted.data = input;
  746     formatted.data   = NULL;
  747     rv = rsa_FormatBlock(&formatted, modulus_len, RSA_BlockRaw, &unformatted);
  748     if (rv != SECSuccess) 
  749             goto done;
  750 
  751     rv = RSA_PrivateKeyOpDoubleChecked(&key->u.rsa, output, formatted.data);
  752     if (rv != SECSuccess && PORT_GetError() == SEC_ERROR_LIBRARY_FAILURE) {
  753         sftk_fatalError[1] = PR_TRUE;
  754     }
  755     *output_len = modulus_len;
  756 
  757 done:
  758     if (formatted.data != NULL) 
  759             PORT_ZFree(formatted.data, modulus_len);
  760     return rv;
  761 }

  764 SECStatus
  765 RSA_CheckSignRaw(NSSLOWKEYPublicKey *key,
  766                  unsigned char *     sign, 
  767                  unsigned int        sign_len, 
  768                  unsigned char *     hash, 
  769                  unsigned int        hash_len)
  770 {
  771     SECStatus       rv;
  772     unsigned int    modulus_len = nsslowkey_PublicModulusLen(key);
  773     unsigned char * buffer;
  774 
  775     if (sign_len != modulus_len) 
  776             goto failure;
  777     if (hash_len > modulus_len) 
  778             goto failure;
  779     PORT_Assert[1](key->keyType == NSSLOWKEYRSAKey);
  780     if (key->keyType != NSSLOWKEYRSAKey)
  781             goto failure;
  782 
  783     buffer = (unsigned char *)PORT_Alloc(modulus_len + 1);
  784     if (!buffer)
  785             goto failure;
  786 
  787     rv = RSA_PublicKeyOp(&key->u.rsa, buffer, sign);
  788     if (rv != SECSuccess)
  789         goto loser;
  790 
  791     /*
  792      * make sure we get the same results
  793      */
  794     /* NOTE: should we verify the leading zeros? */
  795     if (PORT_Memcmp(buffer + (modulus_len-hash_len), hash, hash_len) != 0)
  796         goto loser;
  797 
  798     PORT_Free(buffer);
  799     return SECSuccess;
  800 
  801 loser:
  802     PORT_Free(buffer);
  803 failure:
  804     return SECFailure;
  805 }
