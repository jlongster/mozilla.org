   37 #ifndef _ALGHMAC_H_
   38 #define _ALGHMAC_H_
   39 
   40 typedef struct HMACContextStr HMACContext;
   41 
   42 SEC_BEGIN_PROTOS
   43 
   44 /* destroy HMAC context */
   45 extern void
   46 HMAC_Destroy(HMACContext *cx, PRBool freeit);
   47 
   48 /* create HMAC context
   49  *  hash_obj    hash object from SECRawHashObjects[]
   50  *  secret        the secret with which the HMAC is performed.
   51  *  secret_len        the length of the secret.
   52  *  isFIPS        true if conforming to FIPS 198.
   53  *
   54  * NULL is returned if an error occurs.
   55  */
   56 extern HMACContext *
   57 HMAC_Create(const SECHashObject *hash_obj, const unsigned char *secret, 
   58             unsigned int secret_len, PRBool isFIPS);
   59 
   60 /* like HMAC_Create, except caller allocates HMACContext. */
   61 SECStatus
   62 HMAC_Init(HMACContext *cx, const SECHashObject *hash_obj, 
   63           const unsigned char *secret, unsigned int secret_len, PRBool isFIPS);
   64 
   65 /* reset HMAC for a fresh round */
   66 extern void
   67 HMAC_Begin(HMACContext *cx);
   68 
   69 /* update HMAC 
   70  *  cx                HMAC Context
   71  *  data        the data to perform HMAC on
   72  *  data_len        the length of the data to process
   73  */
   74 extern void 
   75 HMAC_Update(HMACContext *cx, const unsigned char *data, unsigned int data_len);
   76 
   77 /* Finish HMAC -- place the results within result
   78  *  cx                HMAC context
   79  *  result        buffer for resulting hmac'd data
   80  *  result_len        where the resultant hmac length is stored
   81  *  max_result_len  maximum possible length that can be stored in result
   82  */
   83 extern SECStatus
   84 HMAC_Finish(HMACContext *cx, unsigned char *result, unsigned int *result_len,
   85             unsigned int max_result_len);
   86 
   87 /* clone a copy of the HMAC state.  this is usefult when you would
   88  * need to keep a running hmac but also need to extract portions 
   89  * partway through the process.
   90  */
   91 extern HMACContext *
   92 HMAC_Clone(HMACContext *cx);
   93 
   94 SEC_END_PROTOS[1]
   95 
   96 #endif
