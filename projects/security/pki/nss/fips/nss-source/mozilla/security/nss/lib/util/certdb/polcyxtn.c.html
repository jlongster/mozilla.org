  104 {
  105     char *tmpstr;
  106     char *lastspace = NULL;
  107     int curlen = 0;
  108     int c[1][2];
  109     
  110     tmpstr = string;
  111 
  112     while ( ( c[1][2] = *tmpstr ) != '\0' ) {
  113         switc[1][2]h ( c[1][2] ) {
  114           case ' ':
  115             lastspace = tmpstr;
  116             break;
  117           case '\n[1][2][3]':
  118             lastspace = NULL;
  119             curlen = 0;
  120             break;
  121         }
  122         
  123         if ( ( curlen >= 55 ) && ( lastspace != NULL ) ) {
  124             *lastspace = '\n[1][2][3]';
  125             curlen = ( tmpstr - lastspace );
