   54 /*
   55  * This structure keeps track of status that spans all the Slots.
   56  * NOTE: This is a global data structure. It semantics expect thread crosstalk
   57  * be very careful when you see it used. 
   58  *  It's major purpose in life is to allow the user to log in one PER 
   59  * Tranaction, even if a transaction spans threads. The problem is the user
   60  * may have to enter a password one just to be able to look at the 
   61  * personalities/certificates (s)he can use. Then if Auth every is one, they
   62  * may have to enter the password again to use the card. See PK11_StartTransac
   63  * and PK11_EndTransaction.
   64  */
   65 static struct PK11GlobalStruct {
   66    int transaction;
   67    PRBool inTransaction;
   68    char *(PR_CALLBACK *getPass)(PK11SlotInfo *,PRBool,void *);
   69    PRBool (PR_CALLBACK *verifyPass)(PK11SlotInfo *,void *);
   70    PRBool (PR_CALLBACK *isLoggedIn)(PK11SlotInfo *,void *);
   71 } PK11_Global = { 1, PR_FALSE, NULL, NULL, NULL };

  203 /*
  204  * before we do a private key op, we check to see if we
  205  * need to reauthenticate.
  206  */
  207 void
  208 PK11_HandlePasswordCheck(PK11SlotInfo *slot,void *wincx)
  209 {
  210     int askpw = slot->askpw;
  211     PRBool NeedAuth = PR_FALSE;
  212 
  213     if (!slot->needLogin) return;
  214 
  215     if ((slot->defaultFlags & PK11_OWN_PW_DEFAULTS) == 0) {
  216         PK11SlotInfo *def_slot = PK11_GetInternalKeySlot();
  217 
  218         if (def_slot) {
  219             askpw = def_slot->askpw;
  220             PK11_FreeSlot(def_slot);
  221         }
  222     }
  223 
  224     /* timeouts are handled by isLoggedIn */
  225     if (!PK11_IsLoggedIn(slot,wincx)) {
  226         NeedAuth = PR_TRUE;
  227     } else if (askpw == -1) {
  228         if (!PK11_Global.inTransaction        ||
  229                          (PK11_Global.transaction != slot->authTransact)) {
  230                 slot.c.dep.html#PK11_EnterSlotMonitor">PK11_EnterSlotMonitor(slot);
  231             PK11_GETTAB(slot)->C_Logout(slot->session);
  232             slot->lastLoginCheck = 0;
  233                 slot.c.dep.html#PK11_ExitSlotMonitor">PK11_ExitSlotMonitor(slot);
  234             NeedAuth = PR_TRUE;
  235         }
  236     }
  237     if (NeedAuth) PK11_DoPassword(slot,PR_TRUE,wincx);
  238 }

  514 /*
  515  * authenticate to a slot. This loops until we can't recover, the user
  516  * gives up, or we succeed. If we're already logged in and this function
  517  * is called we will still prompt for a password, but we will probably
  518  * succeed no matter what the password was (depending on the implementation
  519  * of the PKCS 11 module.
  520  */
  521 SECStatus
  522 PK11_DoPassword(PK11SlotInfo *slot, PRBool loadCerts, void *wincx)
  523 {
  524     SECStatus rv[1] = SECFailure;
  525     char * password[1][2][3][4][5][6][7][8];
  526     PRBool attempt = PR_FALSE;
  527 
  528     if (PK11_NeedUserInit(slot)) {
  529         PORT_SetError(SEC_ERROR_IO);
  530         return SECFailure;
  531     }
  532 
  533 
  534     /*
  535      * Central server type applications which control access to multiple
  536      * slave applications to single crypto devices need to virtuallize the
  537      * login state. This is done by a callback out of PK11_IsLoggedIn and
  538      * here. If we are actually logged in, then we got here because the
  539      * higher level code told us that the particular client application may
  540      * still need to be logged in. If that is the case, we simply tell the
  541      * server code that it should now verify the clients password and tell us
  542      * the results.
  543      */
  544     if (PK11_IsLoggedIn(slot,NULL) && 
  545                             (PK11_Global.verifyPass != NULL)) {
  546         if (!PK11_Global.verifyPass(slot,wincx)) {
  547             PORT_SetError(SEC_ERROR_BAD_PASSWORD);
  548             return SECFailure;
  549         }
  550         return SECSuccess;
  551     }
  552 
  553     /* get the password. This can drop out of the while loop
  554      * for the following reasons:
  555      *         (1) the user refused to enter a password. 
  556      *                        (return error to caller)
  557      *        (2) the token user password is disabled [usually due to
  558      *           too many failed authentication attempts].
  559      *                        (return error to caller)
  560      *        (3) the password was successful.
  561      */
  562     while ((password[1][2][3][4][5][6][7][8] = pk11_GetPassword(slot, attempt, wincx)) != NULL) {
  563         /* if the token has a protectedAuthPath, the application may have
  564          * already issued the C_Login as part of it's pk11_GetPassword call.
  565          * In this case the application will tell us what the results were in 
  566          * the password value (retry or the authentication was successful) so
  567          * we can skip our own C_Login call (which would force the token to
  568          * try to login again).
  569          * 
  570          * Applications that don't know about protectedAuthPath will return a 
  571          * password, which we will ignore and trigger the token to 
  572          * 'authenticate' itself anyway. Hopefully the blinking display on 
  573          * the reader, or the flashing light under the thumbprint reader will 
  574          * attract the user's attention */
  575         attempt = PR_TRUE;
  576         if (slot->protectedAuthPath) {
  577             /* application tried to authenticate and failed. it wants to try
  578              * again, continue looping */
  579             if (strcmp(password[1][2][3][4][5][6][7][8], PK11_PW_RETRY) == 0) {
