  174 static SECStatus
  175 scanTag(char **pbp, char *endptr, char *tagBuf, int tagBufSize)
  176 {
  177     char *bp, *tagBufp;
  178     int taglen;
  179 
  180     PORT_Assert[1](tagBufSize > 0);
  181     
  182     /* skip optional leading space */
  183     skipSpace(pbp, endptr);
  184     if (*pbp == endptr) {
  185         /* nothing left */
  186         return SECFailure;
  187     }
  188     
  189     /* fill tagBuf */
  190     taglen = 0;
  191     bp = *pbp;
  192     tagBufp = tagBuf;
  193     while (bp < endptr && !OPTIONAL_SPACE(*bp) && (*bp != C_EQUAL)) {
  194         if (++taglen >= tagBufSize) {
  195             *pbp = bp;
  196             return SECFailure;
  197         }
  198         *tagBufp++ = *bp++;
  199     }
  200     /* null-terminate tagBuf -- guaranteed at least one space left */
  201     *tagBufp++ = 0;
  202     *pbp = bp;
  203     
  204     /* skip trailing spaces till we hit something - should be an equal sign */
  205     skipSpace(pbp, endptr);
  206     if (*pbp == endptr) {
  207         /* nothing left */
  208         return SECFailure;
  209     }
  210     if (**pbp != C_EQUAL) {
  211         /* should be an equal sign */
  212         return SECFailure;
  213     }
  214     /* skip over the equal sign */
  215     (*pbp)++;
  216     
  217     return SECSuccess;
  218 }

  310 CERTAVA *
  311 CERT_ParseRFC1485AVA(PRArenaPool *arena, char **pbp, char *endptr,
  312                     PRBool singleAVA) 
  313 {
  314     CERTAVA *a;
  315     const struct NameToKind *n2k;
  316     int vt;
  317     int valLen;
  318     char *bp;
  319 
  320     char tagBuf[32];
  321     char valBuf[384];
  322 
  323     if (scanTag(pbp, endptr, tagBuf, sizeof(tagBuf)) == SECFailure ||
  324         scanVal(pbp, endptr, valBuf, sizeof(valBuf)) == SECFailure) {
  325         PORT_SetError(SEC_ERROR_INVALID_AVA);
  326         return 0;
  327     }
  328 
  329     /* insist that if we haven't finished we've stopped on a separator */
  330     bp = *pbp;
  331     if (bp < endptr) {
  332         if (singleAVA || (*bp != ',' && *bp != ';')) {
  333             PORT_SetError(SEC_ERROR_INVALID_AVA);
  334             *pbp = bp;
  335             return 0;
  336         }
  337         /* ok, skip over separator */
  338         bp++;
  339     }
  340     *pbp = bp;
  341 
  342     for (n2k = name2kinds; n2k->name; n2k++) {
  343         if (PORT_Strcasecmp[1](n2k->name, tagBuf) == 0) {
  344             valLen = PORT_Strlen(valBuf);
  345             if (n2k->kind == SEC_OID_AVA_COUNTRY_NAME) {
  346                 vt = SEC_ASN1_PRINTABLE_STRING;
  347                 if (valLen != 2) {
  348                     PORT_SetError(SEC_ERROR_INVALID_AVA);
  349                     return 0;
  350                 }
  351                 if (!IsPrintable((unsigned char*) valBuf, 2)) {
  352                     PORT_SetError(SEC_ERROR_INVALID_AVA);
  353                     return 0;
  354                 }
  355             } else if ((n2k->kind == SEC_OID_PKCS9_EMAIL_ADDRESS) ||
  356                        (n2k->kind == SEC_OID_RFC1274_MAIL)) {
  357                 vt = SEC_ASN1_IA5_STRING;
  358             } else {
  359                 /* Hack -- for rationale see X.520 DirectoryString defn */
  360                 if (IsPrintable((unsigned char*)valBuf, valLen)) {
  361                     vt = SEC_ASN1_PRINTABLE_STRING;
  362                 } else if (Is7Bit((unsigned char *)valBuf, valLen)) {
  363                     vt = SEC_ASN1_T61_STRING;
  364                 } else {
  365                     /* according to RFC3280, UTF8String is preferred encoding */
  366                     vt = SEC_ASN1_UTF8_STRING;
  367                 }
  368             }
  369             a = CERT_CreateAVA(arena, n2k->kind, vt, (char *) valBuf);
  370             return a;
  371         }
  372     }
  373     /* matched no kind -- invalid tag */
  374     PORT_SetError(SEC_ERROR_INVALID_AVA);
  375     return 0;
  376 }

  425             tmp[2] = *firstRdn;
