   64 /* Special macros facilitate running on Win 16 */
   65 #if defined(XP_WIN16)
   66 
   67   /* 
   68    * Allocate the data passed to the callback functions from the heap...
   69    *
   70    * This inter-thread structure cannot reside on a thread stack since the 
   71    * thread's stack is swapped away with the thread under Win16...
   72    */
   73 
   74  #define ALLOC_OR_DEFINE(type, pointer_var_name, out_of_memory_return_value) \
   75          type * pointer_var_name = PORT_ZAlloc (sizeof(type));               \
   76          do {                                                                \
   77            if (!pointer_var_name)                                            \
   78              return (out_of_memory_return_value);                            \
   79          } while (0)   /* and now a semicolon can follow :-) */
   80 
   81  #define FREE_IF_ALLOC_IS_USED(pointer_var_name) PORT_Free(pointer_var_name)
   82 
   83 #else /* not win 16... so we can alloc via auto variables */
   84 
   85  #define ALLOC_OR_DEFINE(type, pointer_var_name, out_of_memory_return_value) \
   86          type actual_structure_allocated_in_macro;                           \
   87          type * pointer_var_name = &actual_structure_allocated_in_macro;     \
   88          PORT_Memset (pointer_var_name, 0, sizeof (*pointer_var_name));      \
   89          ((void) 0) /* and now a semicolon can follow  */
   90 
   91  #define FREE_IF_ALLOC_IS_USED(pointer_var_name) ((void) 0)
   92 
   93 #endif /* not Win 16 */

  105 struct EVIL_encode
  106   {
  107   int error[3];
  108   SECStatus status;
  109   SEC_PKCS7ContentInfo *cinfo;
  110   SEC_PKCS7EncoderOutputCallback outputfn;
  111   void *outputarg;
  112   PK11SymKey *bulkkey;
  113   SECKEYGetPasswordKey pwfn;
  114   void *pwfnarg;
  115   };
  116 
  117 
  118 /* This is called inside the mozilla thread */
  119 
  120 PR_STATIC_CALLBACK(void) jar_moz_encode_fn (void *data[2])
  121   {
  122   SECStatus status;
  123   struct EVIL_encode *encode_data[2] = (struct EVIL_encode *)data[2];
  124 
  125   PORT_SetError (encode_data->error[3]);
  126 
  127   status = SEC_PKCS7Encode (encode_data->cinfo, encode_data->outputfn, 
  128                             encode_data->outputarg, encode_data->bulkkey, 
  129                             encode_data->pwfn, encode_data->pwfnarg);
  130 
  131   encode_data->status = status;
  132   encode_data->error[3] = PORT_GetError();
  133   }

  136 /* Wrapper for the ET_MOZ call */
  137  
  138 SECStatus jar_moz_encode
  139       (
  140       SEC_PKCS7ContentInfo *cinfo,
  141       SEC_PKCS7EncoderOutputCallback  outputfn,
  142       void *outputarg,
  143       PK11SymKey *bulkkey,
  144       SECKEYGetPasswordKey pwfn,
  145       void *pwfnarg
  146       )
  147   {
  148   SECStatus ret;
  149   ALLOC_OR_DEFINE(struct EVIL_encode, encode_data, SECFailure);
  150 
  151   encode_data->error[3]     = PORT_GetError();
  152   encode_data->cinfo     = cinfo;
  153   encode_data->outputfn  = outputfn;
  154   encode_data->outputarg = outputarg;
  155   encode_data->bulkkey   = bulkkey;
  156   encode_data->pwfn      = pwfn;
  157   encode_data->pwfnarg   = pwfnarg;
  158 
  159   /* Synchronously invoke the callback function on the mozilla thread. */
  160 #ifdef MOZILLA_CLIENT_OLD
  161   if (mozilla_event_queue[1][2])
  162     ET_moz_CallFunction (jar_moz_encode_fn, encode_data);
  163   else
  164     jar_moz_encode_fn (encode_data);
  165 #else
  166   jar_moz_encode_fn (encode_data);
  167 #endif
  168 
  169   PORT_SetError (encode_data->error[3]);
  170   ret = encode_data->status;
  171 
  172   /* Free the data passed to the callback function... */
  173   FREE_IF_ALLOC_IS_USED(encode_data);
  174   return ret;
  175   }
