   38 #ifndef _RIJNDAEL_H_
   39 #define _RIJNDAEL_H_ 1
   40 
   41 #define RIJNDAEL_MIN_BLOCKSIZE 16 /* bytes */
   42 #define RIJNDAEL_MAX_BLOCKSIZE 32 /* bytes */
   43 
   44 typedef SECStatus AESFunc(AESContext *cx, unsigned char *output,
   45                           unsigned int *outputLen, unsigned int maxOutputLen,
   46                           const unsigned char *input, unsigned int inputLen, 
   47                           unsigned int blocksize);
   48 
   49 typedef SECStatus AESBlockFunc(AESContext *cx, 
   50                                unsigned char *output,
   51                                const unsigned char *input);
   52 
   53 /* RIJNDAEL_NUM_ROUNDS
   54  *
   55  * Number of rounds per execution
   56  * Nk - number of key bytes
   57  * Nb - blocksize (in bytes)
   58  */
   59 #define RIJNDAEL_NUM_ROUNDS(Nk, Nb) \
   60     (PR_MAX(Nk, Nb) + 6)
   61 
   62 /* RIJNDAEL_MAX_STATE_SIZE 
   63  *
   64  * Maximum number of bytes in the state (spec includes up to 256-bit block
   65  * size)
   66  */
   67 #define RIJNDAEL_MAX_STATE_SIZE 32
   68 
   69 /*
   70  * This magic number is (Nb_max * (Nr_max + 1))
   71  * where Nb_max is the maximum block size in 32-bit words,
   72  *       Nr_max is the maximum number of rounds, which is Nb_max + 6
   73  */
   74 #define RIJNDAEL_MAX_EXP_KEY_SIZE (8 * 15)
   75 
   76 /* AESContextStr
   77  *
   78  * Values which maintain the state for Rijndael encryption/decryption.
   79  *
   80  * iv          - initialization vector for CBC mode
   81  * Nb          - the number of bytes in a block, specified by user
   82  * Nr          - the number of rounds, specified by a table
   83  * expandedKey - the round keys in 4-byte words, the length is Nr * Nb
   84  * worker      - the encryption/decryption function to use with this context
   85  */
   86 struct AESContextStr
   87 {
   88     unsigned int   Nb;
   89     unsigned int   Nr;
   90     AESFunc       *worker;
   91     unsigned char iv[RIJNDAEL_MAX_BLOCKSIZE];
   92     PRUint32      expandedKey[RIJNDAEL_MAX_EXP_KEY_SIZE];
   93 };
   94 
   95 #endif /* _RIJNDAEL_H_ */
