   55 /*
   56 ** Number of times to attempt to generate a prime (p or q) from a random
   57 ** seed (the seed changes for each iteration).
   58 */
   59 #define MAX_PRIME_GEN_ATTEMPTS 10
   60 /*
   61 ** Number of times to attempt to generate a key.  The primes p and q change
   62 ** for each attempt.
   63 */
   64 #define MAX_KEY_GEN_ATTEMPTS 10
   65 
   66 #define MAX_RSA_MODULUS  1024 /* bytes, 8k bits */
   67 #define MAX_RSA_EXPONENT    8 /* bytes, 64 bits */
   68 
   69 /* exponent should not be greater than modulus */
   70 #define BAD_RSA_KEY_SIZE(modLen, expLen) \
   71     ((expLen) > (modLen) || (modLen) > MAX_RSA_MODULUS || \
   72     (expLen) > MAX_RSA_EXPONENT)

   74 /*
   75 ** RSABlindingParamsStr
   76 **
   77 ** For discussion of Paul Kocher's timing attack against an RSA private key
   78 ** operation, see http://www.cryptography.com/timingattack/paper.html.  The 
   79 ** countermeasure to this attack, known as blinding, is also discussed in 
   80 ** the Handbook of Applied Cryptography, 11.118-11.119.
   81 */
   82 struct RSABlindingParamsStr
   83 {
   84     /* Blinding-specific parameters */
   85     PRCList   link;                  /* link to list of structs            */
   86     SECItem   modulus;               /* list element "key"                 */
   87     mp_int    f, g;                  /* Blinding parameters                */
   88     int       counter;               /* number of remaining uses of (f, g) */
   89 };

  117 static SECStatus
  118 rsa_keygen_from_primes(mp_int *p, mp_int *q, mp_int *e, RSAPrivateKey *key,
  119                        unsigned int keySizeInBits)
  120 {
  121     mp_int n, d, phi;
  122     mp_int psub1, qsub1, tmp[1];
  123     mp_err   err = MP_OKAY;
  124     SECStatus rv = SECSuccess;
  125     MP_DIGITS(&n)     = 0;
  126     MP_DIGITS(&d)     = 0;
  127     MP_DIGITS(&phi)   = 0;
  128     MP_DIGITS(&psub1) = 0;
  129     MP_DIGITS(&qsub1) = 0;
  130     MP_DIGITS(&tmp[1])   = 0;
  131     CHECK_MPI_OK( mp_init(&n)     );
  132     CHECK_MPI_OK( mp_init(&d)     );
  133     CHECK_MPI_OK( mp_init(&phi)   );
  134     CHECK_MPI_OK( mp_init(&psub1) );
  135     CHECK_MPI_OK( mp_init(&qsub1) );
  136     CHECK_MPI_OK( mp_init(&tmp[1])   );
  137     /* 1.  Compute n = p*q */
  138     CHECK_MPI_OK( mp_mul(p, q, &n) );
  139     /*     verify that the modulus has the desired number of bits */
  140     if ((unsigned)mpl_significant_bits(&n) != keySizeInBits) {
  141         PORT_SetError(SEC_ERROR_NEED_RANDOM);
  142         rv = SECFailure;
  143         goto cleanup;
  144     }
  145     /* 2.  Compute phi = (p-1)*(q-1) */
  146     CHECK_MPI_OK( mp_sub_d(p, 1, &psub1) );
  147     CHECK_MPI_OK( mp_sub_d(q, 1, &qsub1) );
  148     CHECK_MPI_OK( mp_mul(&psub1, &qsub1, &phi) );
  149     /* 3.  Compute d = e**-1 mod(phi) */
  150     err = mp_invmod(e, &phi, &d);
  151     /*     Verify that phi(n) and e have no common divisors */
  152     if (err != MP_OKAY) {
  153         if (err == MP_UNDEF) {
  154             PORT_SetError(SEC_ERROR_NEED_RANDOM);
  155             err = MP_OKAY; /* to keep PORT_SetError from being called again */
  156             rv = SECFailure;
  157         }
  158         goto cleanup;
  159     }
  160     MPINT_TO_SECITEM(&n, &key->modulus, key->arena);
  161     MPINT_TO_SECITEM(&d, &key->privateExponent, key->arena);
  162     /* 4.  Compute exponent1 = d mod (p-1) */
  163     CHECK_MPI_OK( mp_mod(&d, &psub1, &tmp[1]) );
  164     MPINT_TO_SECITEM(&tmp[1], &key->exponent1, key->arena);
  165     /* 5.  Compute exponent2 = d mod (q-1) */
  166     CHECK_MPI_OK( mp_mod(&d, &qsub1, &tmp[1]) );
  167     MPINT_TO_SECITEM(&tmp[1], &key->exponent2, key->arena);
  168     /* 6.  Compute coefficient = q**-1 mod p */
  169     CHECK_MPI_OK( mp_invmod(q, p, &tmp[1]) );
  170     MPINT_TO_SECITEM(&tmp[1], &key->coefficient, key->arena);
  171 cleanup:
  172     mp_clear(&n);
  173     mp_clear(&d);
  174     mp_clear(&phi);
  175     mp_clear(&psub1);
  176     mp_clear(&qsub1);
  177     mp_clear(&tmp[1]);
  178     if (err) {
  179         MP_TO_SEC_ERROR(err);
  180         rv = SECFailure;
  181     }
  182     return rv;
  183 }
  184 static SECStatus
  185 generate_prime(mp_int *prime, int primeLen)
  186 {
  187     mp_err   err = MP_OKAY;
  188     SECStatus rv = SECSuccess;
  189     unsigned long counter = 0;
  190     int piter;
  191     unsigned char *pb = NULL;
  192     pb = PORT_Alloc(primeLen);
  193     if (!pb) {
  194         PORT_SetError(SEC_ERROR_NO_MEMORY);
  195         goto cleanup;
  196     }
  197     for (piter = 0; piter < MAX_PRIME_GEN_ATTEMPTS; piter++) {
  198         CHECK_SEC_OK[1]( RNG_GenerateGlobalRandomBytes(pb, primeLen) );
  199         pb[0]          |= 0xC0; /* set two high-order bits */
  200         pb[primeLen-1] |= 0x01; /* set low-order bit       */
  201         CHECK_MPI_OK( mp_read_unsigned_octets(prime, pb, primeLen) );
  202         err = mpp_make_prime(prime, primeLen * 8, PR_FALSE, &counter);
  203         if (err != MP_NO)
  204             goto cleanup;
  205         /* keep going while err == MP_NO */
  206     }
  207 cleanup:
  208     if (pb)
  209         PORT_ZFree(pb, primeLen);
  210     if (err) {
  211         MP_TO_SEC_ERROR(err);
  212         rv = SECFailure;
  213     }
  214     return rv;
  215 }

  217 /*
  218 ** Generate and return a new RSA public and private key.
  219 **        Both keys are encoded in a single RSAPrivateKey structure.
  220 **        "cx" is the random number generator context
  221 **        "keySizeInBits" is the size of the key to be generated, in bits.
  222 **           512, 1024, etc.
  223 **        "publicExponent" when not NULL is a pointer to some data that
  224 **           represents the public exponent to use. The data is a byte
  225 **           encoded integer, in "big endian" order.
  226 */
  227 RSAPrivateKey *
  228 RSA_NewKey(int keySizeInBits, SECItem *publicExponent)
  229 {
  230     unsigned int primeLen;
  231     mp_int p, q, e;
  232     int kiter;
  233     mp_err   err = MP_OKAY;
  234     SECStatus rv = SECSuccess;
  235     int prerr = 0;
  236     RSAPrivateKey *key = NULL;
  237     PRArenaPool *arena = NULL;
  238     /* Require key size to be a multiple of 16 bits. */
  239     if (!publicExponent || keySizeInBits % 16 != 0 ||
  240             BAD_RSA_KEY_SIZE(keySizeInBits/8, publicExponent->len)) {
  241         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  242         return NULL;
  243     }
  244     /* 1. Allocate arena & key */
  245     arena = PORT_NewArena(NSS_FREEBL_DEFAULT_CHUNKSIZE);
  246     if (!arena) {
  247         PORT_SetError(SEC_ERROR_NO_MEMORY);
  248         return NULL;
  249     }
  250     key = (RSAPrivateKey *)PORT_ArenaZAlloc(arena, sizeof(RSAPrivateKey));
  251     if (!key) {
  252         PORT_SetError(SEC_ERROR_NO_MEMORY);
  253         PORT_FreeArena(arena, PR_TRUE);
  254         return NULL;
  255     }
  256     key->arena = arena;
  257     /* length of primes p and q (in bytes) */
  258     primeLen = keySizeInBits / (2 * BITS_PER_BYTE);
  259     MP_DIGITS(&p) = 0;
  260     MP_DIGITS(&q) = 0;
  261     MP_DIGITS(&e) = 0;
  262     CHECK_MPI_OK( mp_init(&p) );
  263     CHECK_MPI_OK( mp_init(&q) );
  264     CHECK_MPI_OK( mp_init(&e) );
  265     /* 2.  Set the version number (PKCS1 v1.5 says it should be zero) */
  266     SECITEM_AllocItem(arena, &key->version, 1);
  267     key->version.data[0] = 0;
  268     /* 3.  Set the public exponent */
  269     SECITEM_CopyItem(arena, &key->publicExponent, publicExponent);
  270     SECITEM_TO_MPINT(*publicExponent, &e);
  271     kiter = 0;
  272     do {
  273         prerr = 0;
  274         PORT_SetError(0);
  275         CHECK_SEC_OK[1]( generate_prime(&p, primeLen) );
  276         CHECK_SEC_OK[1]( generate_prime(&q, primeLen) );
  277         /* Assure q < p */
  278         if (mp_cmp(&p, &q) < 0)
  279             mp_exch(&p, &q);
  280         /* Attempt to use these primes to generate a key */
  281         rv = rsa_keygen_from_primes(&p, &q, &e, key, keySizeInBits);
  282         if (rv == SECSuccess)
  283             break; /* generated two good primes */
  284         prerr = PORT_GetError();
  285         kiter++;
  286         /* loop until have primes */
  287     } while (prerr == SEC_ERROR_NEED_RANDOM && kiter < MAX_KEY_GEN_ATTEMPTS);
  288     if (prerr)
  289         goto cleanup;
  290     MPINT_TO_SECITEM(&p, &key->prime1, arena);
  291     MPINT_TO_SECITEM(&q, &key->prime2, arena);
  292 cleanup:
  293     mp_clear(&p);
  294     mp_clear(&q);
  295     mp_clear(&e);
  296     if (err) {
  297         MP_TO_SEC_ERROR(err);
  298         rv = SECFailure;
  299     }
  300     if (rv && arena) {
  301         PORT_FreeArena(arena, PR_TRUE);
  302         key = NULL;
  303     }
  304     return key;
  305 }

  307 static unsigned int
  308 rsa_modulusLen(SECItem *modulus)
  309 {
  310     unsigned char byteZero = modulus->data[0];
  311     unsigned int modLen = modulus->len - !byteZero;
  312     return modLen;
  313 }

  315 /*
  316 ** Perform a raw public-key operation 
  317 **        Length of input and output buffers are equal to key's modulus len.
  318 */
  319 SECStatus 
  320 RSA_PublicKeyOp(RSAPublicKey  *key, 
  321                 unsigned char *output, 
  322                 const unsigned char *input)
  323 {
  324     unsigned int modLen, expLen, offset;
  325     mp_int n, e, m, c;
  326     mp_err err   = MP_OKAY;
  327     SECStatus rv = SECSuccess;
  328     if (!key || !output || !input) {
  329         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  330         return SECFailure;
  331     }
  332     MP_DIGITS(&n) = 0;
  333     MP_DIGITS(&e) = 0;
  334     MP_DIGITS(&m) = 0;
  335     MP_DIGITS(&c) = 0;
  336     CHECK_MPI_OK( mp_init(&n) );
  337     CHECK_MPI_OK( mp_init(&e) );
  338     CHECK_MPI_OK( mp_init(&m) );
  339     CHECK_MPI_OK( mp_init(&c) );
  340     modLen = rsa_modulusLen(&key->modulus);
  341     expLen = rsa_modulusLen(&key->publicExponent);
  342     /* 1.  Obtain public key (n, e) */
  343     if (BAD_RSA_KEY_SIZE(modLen, expLen)) {
  344             PORT_SetError(SEC_ERROR_INVALID_KEY);
  345         rv = SECFailure;
  346         goto cleanup;
  347     }
  348     SECITEM_TO_MPINT(key->modulus, &n);
  349     SECITEM_TO_MPINT(key->publicExponent, &e);
  350     if (e.used > n.used) {
  351         /* exponent should not be greater than modulus */
  352             PORT_SetError(SEC_ERROR_INVALID_KEY);
  353         rv = SECFailure;
  354         goto cleanup;
  355     }
  356     /* 2. check input out of range (needs to be in range [0..n-1]) */
  357     offset = (key->modulus.data[0] == 0) ? 1 : 0; /* may be leading 0 */
  358     if (memcmp(input, key->modulus.data + offset, modLen) >= 0) {
  359         PORT_SetError(SEC_ERROR_INPUT_LEN);
  360         rv = SECFailure;
  361         goto cleanup;
  362     }
  363     /* 2 bis.  Represent message as integer in range [0..n-1] */
  364     CHECK_MPI_OK( mp_read_unsigned_octets(&m, input, modLen) );
  365     /* 3.  Compute c = m**e mod n */
  366 #ifdef USE_MPI_EXPT_D
  367     /* XXX see which is faster */
  368     if (MP_USED(&e) == 1) {
  369         CHECK_MPI_OK( mp_exptmod_d(&m, MP_DIGIT(&e, 0), &n, &c) );
  370     } else
  371 #endif
  372     CHECK_MPI_OK( mp_exptmod(&m, &e, &n, &c) );
  373     /* 4.  result c is ciphertext */
  374     err = mp_to_fixlen_octets(&c, output, modLen);
  375     if (err >= 0) err = MP_OKAY;
  376 cleanup:
  377     mp_clear(&n);
  378     mp_clear(&e);
  379     mp_clear(&m);
  380     mp_clear(&c);
  381     if (err) {
  382         MP_TO_SEC_ERROR(err);
  383         rv = SECFailure;
  384     }
  385     return rv;
  386 }

  412 /*
  413 **  RSA Private key operation using CRT.
  414 */
  415 static SECStatus 
  416 rsa_PrivateKeyOpCRTNoCheck(RSAPrivateKey *key, mp_int *m, mp_int *c)
  417 {
  418     mp_int p, q, d_p, d_q, qInv;
  419     mp_int m1, m2, h, ctmp;
  420     mp_err   err = MP_OKAY;
  421     SECStatus rv = SECSuccess;
  422     MP_DIGITS(&p)    = 0;
  423     MP_DIGITS(&q)    = 0;
  424     MP_DIGITS(&d_p)  = 0;
  425     MP_DIGITS(&d_q)  = 0;
  426     MP_DIGITS(&qInv) = 0;
  427     MP_DIGITS(&m1)   = 0;
  428     MP_DIGITS(&m2)   = 0;
  429     MP_DIGITS(&h)    = 0;
  430     MP_DIGITS(&ctmp) = 0;
  431     CHECK_MPI_OK( mp_init(&p)    );
  432     CHECK_MPI_OK( mp_init(&q)    );
  433     CHECK_MPI_OK( mp_init(&d_p)  );
  434     CHECK_MPI_OK( mp_init(&d_q)  );
  435     CHECK_MPI_OK( mp_init(&qInv) );
  436     CHECK_MPI_OK( mp_init(&m1)   );
  437     CHECK_MPI_OK( mp_init(&m2)   );
  438     CHECK_MPI_OK( mp_init(&h)    );
  439     CHECK_MPI_OK( mp_init(&ctmp) );
  440     /* copy private key parameters into mp integers */
  441     SECITEM_TO_MPINT(key->prime1,      &p);    /* p */
  442     SECITEM_TO_MPINT(key->prime2,      &q);    /* q */
  443     SECITEM_TO_MPINT(key->exponent1,   &d_p);  /* d_p  = d mod (p-1) */
  444     SECITEM_TO_MPINT(key->exponent2,   &d_q);  /* d_q  = d mod (q-1) */
  445     SECITEM_TO_MPINT(key->coefficient, &qInv); /* qInv = q**-1 mod p */
  446     /* 1. m1 = c**d_p mod p */
  447     CHECK_MPI_OK( mp_mod(c, &p, &ctmp) );
  448     CHECK_MPI_OK( mp_exptmod(&ctmp, &d_p, &p, &m1) );
  449     /* 2. m2 = c**d_q mod q */
  450     CHECK_MPI_OK( mp_mod(c, &q, &ctmp) );
  451     CHECK_MPI_OK( mp_exptmod(&ctmp, &d_q, &q, &m2) );
  452     /* 3.  h = (m1 - m2) * qInv mod p */
  453     CHECK_MPI_OK( mp_submod(&m1, &m2, &p, &h) );
  454     CHECK_MPI_OK( mp_mulmod(&h, &qInv, &p, &h)  );
  455     /* 4.  m = m2 + h * q */
  456     CHECK_MPI_OK( mp_mul(&h, &q, m) );
  457     CHECK_MPI_OK( mp_add(m, &m2, m) );
  458 cleanup:
  459     mp_clear(&p);
  460     mp_clear(&q);
  461     mp_clear(&d_p);
  462     mp_clear(&d_q);
  463     mp_clear(&qInv);
  464     mp_clear(&m1);
  465     mp_clear(&m2);
  466     mp_clear(&h);
  467     mp_clear(&ctmp);
  468     if (err) {
  469         MP_TO_SEC_ERROR(err);
  470         rv = SECFailure;
  471     }
  472     return rv;
  473 }

  475 /*
  476 ** An attack against RSA CRT was described by Boneh, DeMillo, and Lipton in:
  477 ** "On the Importance of Eliminating Errors in Cryptographic Computations",
  478 ** http://theory.stanford.edu/~dabo/papers/faults.ps.gz
  479 **
  480 ** As a defense against the attack, carry out the private key operation, 
  481 ** followed up with a public key operation to invert the result.  
  482 ** Verify that result against the input.
  483 */
  484 static SECStatus 
  485 rsa_PrivateKeyOpCRTCheckedPubKey(RSAPrivateKey *key, mp_int *m, mp_int *c)
  486 {
  487     mp_int n, e, v;
  488     mp_err   err = MP_OKAY;
  489     SECStatus rv = SECSuccess;
  490     MP_DIGITS(&n) = 0;
  491     MP_DIGITS(&e) = 0;
  492     MP_DIGITS(&v) = 0;
  493     CHECK_MPI_OK( mp_init(&n) );
  494     CHECK_MPI_OK( mp_init(&e) );
  495     CHECK_MPI_OK( mp_init(&v) );
  496     CHECK_SEC_OK[1]( rsa_PrivateKeyOpCRTNoCheck(key, m, c) );
  497     SECITEM_TO_MPINT(key->modulus,        &n);
  498     SECITEM_TO_MPINT(key->publicExponent, &e);
  499     /* Perform a public key operation v = m ** e mod n */
  500     CHECK_MPI_OK( mp_exptmod(m, &e, &n, &v) );
  501     if (mp_cmp(&v, c) != 0) {
  502         rv = SECFailure;
  503     }
  504 cleanup:
  505     mp_clear(&n);
  506     mp_clear(&e);
  507     mp_clear(&v);
  508     if (err) {
  509         MP_TO_SEC_ERROR(err);
  510         rv = SECFailure;
  511     }
  512     return rv;
  513 }

  528 static SECStatus
  529 generate_blinding_params(struct RSABlindingParamsStr *rsabp, 
  530                          RSAPrivateKey *key, mp_int *n, unsigned int modLen)
  531 {
  532     SECStatus rv = SECSuccess;
  533     mp_int e, k;
  534     mp_err err = MP_OKAY;
  535     unsigned char *kb = NULL;
  536     MP_DIGITS(&e) = 0;
  537     MP_DIGITS(&k) = 0;
  538     CHECK_MPI_OK( mp_init(&e) );
  539     CHECK_MPI_OK( mp_init(&k) );
  540     SECITEM_TO_MPINT(key->publicExponent, &e);
  541     /* generate random k < n */
  542     kb = PORT_Alloc(modLen);
  543     if (!kb) {
  544         PORT_SetError(SEC_ERROR_NO_MEMORY);
  545         goto cleanup;
  546     }
  547     CHECK_SEC_OK[1]( RNG_GenerateGlobalRandomBytes(kb, modLen) );
  548     CHECK_MPI_OK( mp_read_unsigned_octets(&k, kb, modLen) );
  549     /* k < n */
  550     CHECK_MPI_OK( mp_mod(&k, n, &k) );
  551     /* f = k**e mod n */
  552     CHECK_MPI_OK( mp_exptmod(&k, &e, n, &rsabp->f) );
  553     /* g = k**-1 mod n */
  554     CHECK_MPI_OK( mp_invmod(&k, n, &rsabp->g) );
  555     /* Initialize the counter for this (f, g) */
  556     rsabp->counter = RSA_BLINDING_PARAMS_MAX_REUSE;
  557 cleanup:
  558     if (kb)
  559         PORT_ZFree(kb, modLen);
  560     mp_clear(&k);
  561     mp_clear(&e);
  562     if (err) {
  563         MP_TO_SEC_ERROR(err);
  564         rv = SECFailure;
  565     }
  566     return rv;
  567 }

  569 static SECStatus
  570 init_blinding_params(struct RSABlindingParamsStr *rsabp, RSAPrivateKey *key,
  571                      mp_int *n, unsigned int modLen)
  572 {
  573     SECStatus rv = SECSuccess;
  574     mp_err err = MP_OKAY;
  575     MP_DIGITS(&rsabp->f) = 0;
  576     MP_DIGITS(&rsabp->g) = 0;
  577     /* initialize blinding parameters */
  578     CHECK_MPI_OK( mp_init(&rsabp->f) );
  579     CHECK_MPI_OK( mp_init(&rsabp->g) );
  580     /* List elements are keyed using the modulus */
  581     SECITEM_CopyItem(NULL, &rsabp->modulus, &key->modulus);
  582     CHECK_SEC_OK[1]( generate_blinding_params(rsabp, key, n, modLen) );
  583     return SECSuccess;
  584 cleanup:
  585     mp_clear(&rsabp->f);
  586     mp_clear(&rsabp->g);
  587     if (err) {
  588         MP_TO_SEC_ERROR(err);
  589         rv = SECFailure;
  590     }
  591     return rv;
  592 }

  594 static SECStatus
  595 get_blinding_params(RSAPrivateKey *key, mp_int *n, unsigned int modLen,
  596                     mp_int *f, mp_int *g)
  597 {
  598     SECStatus rv = SECSuccess;
  599     mp_err err = MP_OKAY;
  600     int cmp;
  601     PRCList *el;
  602     struct RSABlindingParamsStr *rsabp = NULL;
  603     /* Init the list if neccessary (the init function is only called once!) */
  604     if (blindingParamsList.lock == NULL) {
  605         if (PR_CallOnce(&coBPInit, init_blinding_params_list) != PR_SUCCESS) {
  606             PORT_SetError(SEC_ERROR_LIBRARY_FAILURE);
  607             return SECFailure;
  608         }
  609     }
  610     /* Acquire the list lock */
  611     PZ_Lock(blindingParamsList.lock);
  612     /* Walk the list looking for the private key */
  613     for (el = PR_NEXT_LINK(&blindingParamsList.head);
  614          el != &blindingParamsList.head;
  615          el = PR_NEXT_LINK(el)) {
  616         rsabp = (struct RSABlindingParamsStr *)el;
  617         cmp = SECITEM_CompareItem(&rsabp->modulus, &key->modulus);
  618         if (cmp == 0) {
  619             /* Check the usage counter for the parameters */
  620             if (--rsabp->counter <= 0) {
  621                 /* Regenerate the blinding parameters */
  622                 CHECK_SEC_OK[1]( generate_blinding_params(rsabp, key, n, modLen) );
  623             }
  624             /* Return the parameters */
  625             CHECK_MPI_OK( mp_copy(&rsabp->f, f) );
  626             CHECK_MPI_OK( mp_copy(&rsabp->g, g) );
  627             /* Now that the params are located, release the list lock. */
  628             PZ_Unlock(blindingParamsList.lock); /* XXX when fails? */
  629             return SECSuccess;
  630         } else if (cmp > 0) {
  631             /* The key is not in the list.  Break to param creation. */
  632             break;
  633         }
  634     }
  635     /* At this point, the key is not in the list.  el should point to the
  636     ** list element that this key should be inserted before.  NOTE: the list
  637     ** lock is still held, so there cannot be a race condition here.
  638     */
  639     rsabp = (struct RSABlindingParamsStr *)
  640               PORT_ZAlloc(sizeof(struct RSABlindingParamsStr));
  641     if (!rsabp) {
  642         PORT_SetError(SEC_ERROR_NO_MEMORY);
  643         goto cleanup;
  644     }
  645     /* Initialize the list pointer for the element */
  646     PR_INIT_CLIST(&rsabp->link);
  647     /* Initialize the blinding parameters 
  648     ** This ties up the list lock while doing some heavy, element-specific
  649     ** operations, but we don't want to insert the element until it is valid,
  650     ** which requires computing the blinding params.  If this proves costly,
  651     ** it could be done after the list lock is released, and then if it fails
  652     ** the lock would have to be reobtained and the invalid element removed.
  653     */
  654     rv = init_blinding_params(rsabp, key, n, modLen);
  655     if (rv != SECSuccess) {
  656         PORT_ZFree(rsabp, sizeof(struct RSABlindingParamsStr));
  657         goto cleanup;
  658     }
  659     /* Insert the new element into the list
  660     ** If inserting in the middle of the list, el points to the link
  661     ** to insert before.  Otherwise, the link needs to be appended to
  662     ** the end of the list, which is the same as inserting before the
  663     ** head (since el would have looped back to the head).
  664     */
  665     PR_INSERT_BEFORE(&rsabp->link, el);
  666     /* Return the parameters */
  667     CHECK_MPI_OK( mp_copy(&rsabp->f, f) );
  668     CHECK_MPI_OK( mp_copy(&rsabp->g, g) );
  669     /* Release the list lock */
  670     PZ_Unlock(blindingParamsList.lock); /* XXX when fails? */
  671     return SECSuccess;
  672 cleanup:
  673     /* It is possible to reach this after the lock is already released.
  674     ** Ignore the error in that case.
  675     */
  676     PZ_Unlock(blindingParamsList.lock);
  677     if (err) {
  678         MP_TO_SEC_ERROR(err);
  679         rv = SECFailure;
  680     }
  681     return SECFailure;
  682 }

  684 /*
  685 ** Perform a raw private-key operation 
  686 **        Length of input and output buffers are equal to key's modulus len.
  687 */
  688 static SECStatus 
  689 rsa_PrivateKeyOp(RSAPrivateKey *key, 
  690                  unsigned char *output, 
  691                  const unsigned char *input,
  692                  PRBool check)
  693 {
  694     unsigned int modLen;
  695     unsigned int offset;
  696     SECStatus rv = SECSuccess;
  697     mp_err err;
  698     mp_int n, c, m;
  699     mp_int f, g;
  700     if (!key || !output || !input) {
  701         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  702         return SECFailure;
  703     }
  704     /* check input out of range (needs to be in range [0..n-1]) */
  705     modLen = rsa_modulusLen(&key->modulus);
  706     offset = (key->modulus.data[0] == 0) ? 1 : 0; /* may be leading 0 */
  707     if (memcmp(input, key->modulus.data + offset, modLen) >= 0) {
  708         PORT_SetError(SEC_ERROR_INVALID_ARGS);
  709         return SECFailure;
  710     }
  711     MP_DIGITS(&n) = 0;
  712     MP_DIGITS(&c) = 0;
  713     MP_DIGITS(&m) = 0;
  714     MP_DIGITS(&f) = 0;
  715     MP_DIGITS(&g) = 0;
  716     CHECK_MPI_OK( mp_init(&n) );
  717     CHECK_MPI_OK( mp_init(&c) );
  718     CHECK_MPI_OK( mp_init(&m) );
  719     CHECK_MPI_OK( mp_init(&f) );
  720     CHECK_MPI_OK( mp_init(&g) );
  721     SECITEM_TO_MPINT(key->modulus, &n);
  722     OCTETS_TO_MPINT(input, &c, modLen);
  723     /* If blinding, compute pre-image of ciphertext by multiplying by
  724     ** blinding factor
  725     */
  726     if (nssRSAUseBlinding) {
  727         CHECK_SEC_OK[1]( get_blinding_params(key, &n, modLen, &f, &g) );
  728         /* c' = c*f mod n */
  729         CHECK_MPI_OK( mp_mulmod(&c, &f, &n, &c) );
  730     }
  731     /* Do the private key operation m = c**d mod n */
  732     if ( key->prime1.len      == 0 ||
  733          key->prime2.len      == 0 ||
  734          key->exponent1.len   == 0 ||
  735          key->exponent2.len   == 0 ||
  736          key->coefficient.len == 0) {
  737         CHECK_SEC_OK[1]( rsa_PrivateKeyOpNoCRT(key, &m, &c, &n, modLen) );
  738     } else if (check) {
  739         CHECK_SEC_OK[1]( rsa_PrivateKeyOpCRTCheckedPubKey(key, &m, &c) );
  740     } else {
  741         CHECK_SEC_OK[1]( rsa_PrivateKeyOpCRTNoCheck(key, &m, &c) );
  742     }
  743     /* If blinding, compute post-image of plaintext by multiplying by
  744     ** blinding factor
  745     */
  746     if (nssRSAUseBlinding) {
  747         /* m = m'*g mod n */
  748         CHECK_MPI_OK( mp_mulmod(&m, &g, &n, &m) );
  749     }
  750     err = mp_to_fixlen_octets(&m, output, modLen);
  751     if (err >= 0) err = MP_OKAY;
  752 cleanup:
  753     mp_clear(&n);
  754     mp_clear(&c);
  755     mp_clear(&m);
  756     mp_clear(&f);
  757     mp_clear(&g);
  758     if (err) {
  759         MP_TO_SEC_ERROR(err);
  760         rv = SECFailure;
  761     }
  762     return rv;
  763 }

  765 SECStatus 
  766 RSA_PrivateKeyOp(RSAPrivateKey *key, 
  767                  unsigned char *output, 
  768                  const unsigned char *input)
  769 {
  770     return rsa_PrivateKeyOp(key, output, input, PR_FALSE);
  771 }

  781 static SECStatus
  782 swap_in_key_value(PRArenaPool *arena, mp_int *mpval, SECItem *buffer)
  783 {
  784     int len;
  785     mp_err err = MP_OKAY;
  786     memset(buffer->data, 0, buffer->len);
  787     len = mp_unsigned_octet_size(mpval);
  788     if (len <= 0) return SECFailure;
  789     if ((unsigned int)len <= buffer->len) {
  790         /* The new value is no longer than the old buffer, so use it */
  791         err = mp_to_unsigned_octets(mpval, buffer->data, len);
  792         if (err >= 0) err = MP_OKAY;
  793         buffer->len = len;
  794     } else if (arena) {
  795         /* The new value is longer, but working within an arena */
  796         (void)SECITEM_AllocItem(arena, buffer, len);
  797         err = mp_to_unsigned_octets(mpval, buffer->data, len);
  798         if (err >= 0) err = MP_OKAY;
  799     } else {
  800         /* The new value is longer, no arena, can't handle this key */
  801         return SECFailure;
  802     }
  803     return (err == MP_OKAY) ? SECSuccess : SECFailure;
  804 }

  806 SECStatus
  807 RSA_PrivateKeyCheck(RSAPrivateKey *key)
  808 {
  809     mp_int p, q, n, psub1, qsub1, e, d, d_p, d_q, qInv, res;
  810     mp_err   err = MP_OKAY;
  811     SECStatus rv = SECSuccess;
  812     MP_DIGITS(&n)    = 0;
  813     MP_DIGITS(&psub1)= 0;
  814     MP_DIGITS(&qsub1)= 0;
  815     MP_DIGITS(&e)    = 0;
  816     MP_DIGITS(&d)    = 0;
  817     MP_DIGITS(&d_p)  = 0;
  818     MP_DIGITS(&d_q)  = 0;
  819     MP_DIGITS(&qInv) = 0;
  820     MP_DIGITS(&res)  = 0;
  821     CHECK_MPI_OK( mp_init(&n)    );
  822     CHECK_MPI_OK( mp_init(&p)    );
  823     CHECK_MPI_OK( mp_init(&q)    );
  824     CHECK_MPI_OK( mp_init(&psub1));
  825     CHECK_MPI_OK( mp_init(&qsub1));
  826     CHECK_MPI_OK( mp_init(&e)    );
  827     CHECK_MPI_OK( mp_init(&d)    );
  828     CHECK_MPI_OK( mp_init(&d_p)  );
  829     CHECK_MPI_OK( mp_init(&d_q)  );
  830     CHECK_MPI_OK( mp_init(&qInv) );
  831     CHECK_MPI_OK( mp_init(&res)  );
  832     SECITEM_TO_MPINT(key->modulus,         &n);
  833     SECITEM_TO_MPINT(key->prime1,          &p);
  834     SECITEM_TO_MPINT(key->prime2,          &q);
  835     SECITEM_TO_MPINT(key->publicExponent,  &e);
  836     SECITEM_TO_MPINT(key->privateExponent, &d);
  837     SECITEM_TO_MPINT(key->exponent1,       &d_p);
  838     SECITEM_TO_MPINT(key->exponent2,       &d_q);
  839     SECITEM_TO_MPINT(key->coefficient,     &qInv);
  840     /* p > q  */
  841     if (mp_cmp(&p, &q) <= 0) {
  842         /* mind the p's and q's (and d_p's and d_q's) */
  843         SECItem tmp[1];
  844         mp_exch(&p, &q);
  845         mp_exch(&d_p,&d_q);
  846         tmp[1] = key->prime1;
  847         key->prime1 = key->prime2;
  848         key->prime2 = tmp[1];
  849         tmp[1] = key->exponent1;
  850         key->exponent1 = key->exponent2;
  851         key->exponent2 = tmp[1];
  852     }
  853 #define VERIFY_MPI_EQUAL(m1, m2) \
  854     if (mp_cmp(m1, m2) != 0) {   \
  855         rv = SECFailure;         \
  856         goto cleanup;            \
  857     }
  858 #define VERIFY_MPI_EQUAL_1(m)    \
  859     if (mp_cmp_d(m, 1) != 0) {   \
  860         rv = SECFailure;         \
  861         goto cleanup;            \
  862     }
  863     /*
  864      * The following errors cannot be recovered from.
  865      */
  866     /* n == p * q */
  867     CHECK_MPI_OK( mp_mul(&p, &q, &res) );
  868     VERIFY_MPI_EQUAL(&res, &n);
  869     /* gcd(e, p-1) == 1 */
  870     CHECK_MPI_OK( mp_sub_d(&p, 1, &psub1) );
  871     CHECK_MPI_OK( mp_gcd(&e, &psub1, &res) );
  872     VERIFY_MPI_EQUAL_1(&res);
  873     /* gcd(e, q-1) == 1 */
  874     CHECK_MPI_OK( mp_sub_d(&q, 1, &qsub1) );
  875     CHECK_MPI_OK( mp_gcd(&e, &qsub1, &res) );
  876     VERIFY_MPI_EQUAL_1(&res);
  877     /* d*e == 1 mod p-1 */
  878     CHECK_MPI_OK( mp_mulmod(&d, &e, &psub1, &res) );
  879     VERIFY_MPI_EQUAL_1(&res);
  880     /* d*e == 1 mod q-1 */
  881     CHECK_MPI_OK( mp_mulmod(&d, &e, &qsub1, &res) );
  882     VERIFY_MPI_EQUAL_1(&res);
  883     /*
  884      * The following errors can be recovered from.
  885      */
  886     /* d_p == d mod p-1 */
  887     CHECK_MPI_OK( mp_mod(&d, &psub1, &res) );
  888     if (mp_cmp(&d_p, &res) != 0) {
  889         /* swap in the correct value */
  890         CHECK_SEC_OK[1]( swap_in_key_value(key->arena, &res, &key->exponent1) );
  891     }
  892     /* d_q == d mod q-1 */
  893     CHECK_MPI_OK( mp_mod(&d, &qsub1, &res) );
  894     if (mp_cmp(&d_q, &res) != 0) {
  895         /* swap in the correct value */
  896         CHECK_SEC_OK[1]( swap_in_key_value(key->arena, &res, &key->exponent2) );
  897     }
  898     /* q * q**-1 == 1 mod p */
  899     CHECK_MPI_OK( mp_mulmod(&q, &qInv, &p, &res) );
  900     if (mp_cmp_d(&res, 1) != 0) {
  901         /* compute the correct value */
  902         CHECK_MPI_OK( mp_invmod(&q, &p, &qInv) );
  903         CHECK_SEC_OK[1]( swap_in_key_value(key->arena, &qInv, &key->coefficient) );
  904     }
  905 cleanup:
  906     mp_clear(&n);
  907     mp_clear(&p);
  908     mp_clear(&q);
  909     mp_clear(&psub1);
  910     mp_clear(&qsub1);
  911     mp_clear(&e);
  912     mp_clear(&d);
  913     mp_clear(&d_p);
  914     mp_clear(&d_q);
  915     mp_clear(&qInv);
  916     mp_clear(&res);
  917     if (err) {
  918         MP_TO_SEC_ERROR(err);
  919         rv = SECFailure;
  920     }
  921     return rv;
  922 }

  924 /* cleanup at shutdown */
  925 void RSA_Cleanup(void)
  926 {
  927     if (!coBPInit.initialized)
  928         return;
  929 
  930     while (!PR_CLIST_IS_EMPTY(&blindingParamsList.head))
  931     {
  932         struct RSABlindingParamsStr * rsabp = (struct RSABlindingParamsStr *)
  933             PR_LIST_HEAD(&blindingParamsList.head);
  934         PR_REMOVE_LINK(&rsabp->link);
  935         mp_clear(&rsabp->f);
  936         mp_clear(&rsabp->g);
  937         SECITEM_FreeItem(&rsabp->modulus,PR_FALSE);
  938         PORT_Free(rsabp);
  939     }
  940 
  941     if (blindingParamsList.lock)
  942     {
  943         PZ_DestroyLock(blindingParamsList.lock);
  944         blindingParamsList.lock = NULL;
  945     }
  946 
  947     coBPInit.initialized = 0;
  948     coBPInit.inProgress = 0;
  949     coBPInit.status = 0;
  950 }
