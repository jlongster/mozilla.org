   84  * FUNCTION: CERT_DisableOCSPChecking
   85  *   Turns off OCSP checking for the given certificate database.
   86  *   This routine disables OCSP checking.  Though it will return
   87  *   SECFailure if OCSP checking is not enabled, it is "safe" to
   88  *   call it that way and just ignore the return value, if it is

   91  *   CERTCertDBHandle *handle
   92  *     Certificate database for which OCSP checking will be disabled.
   93  * RETURN:
   94  *   Returns SECFailure if an error occurred (usually means that OCSP
   95  *   checking was not enabled or status contexts were not initialized --
   96  *   error set will be SEC_ERROR_OCSP_NOT_ENABLED); SECSuccess otherwise.

   98 extern SECStatus
   99 CERT_DisableOCSPChecking(CERTCertDBHandle *handle);

  101 /*
  102  * FUNCTION: CERT_SetOCSPDefaultResponder
  103  *   Specify the location and cert of the default responder.
  104  *   If OCSP checking is already enabled *and* use of a default responder
  105  *   is also already enabled, all OCSP checking from now on will go directly
  106  *   to the specified responder.  If OCSP checking is not enabled, or if
  107  *   it is but use of a default responder is not enabled, the information
  108  *   will be recorded and take effect whenever both are enabled.
  109  * INPUTS:
  110  *   CERTCertDBHandle *handle
  111  *     Cert database on which OCSP checking should use the default responder.
  112  *   char *url
  113  *     The location of the default responder (e.g. "http://foo.com:80/ocsp")
  114  *     Note that the location will not be tested until the first attempt
  115  *     to send a request there.
  116  *   char *name
  117  *     The nickname of the cert to trust (expected) to sign the OCSP responses.
  118  *     If the corresponding cert cannot be found, SECFailure is returned.
  119  * RETURN:
  120  *   Returns SECFailure if an error occurred; SECSuccess otherwise.
  121  *   The most likely error is that the cert for "name" could not be found
  122  *   (probably SEC_ERROR_UNKNOWN_CERT).  Other errors are low-level (no memory,
  123  *   bad database, etc.).
  124  */
  125 extern SECStatus
  126 CERT_SetOCSPDefaultResponder(CERTCertDBHandle *handle,
  127                              const char *url, const char *name);

  130  * FUNCTION: CERT_EnableOCSPDefaultResponder

  134  *   and the nickname of the responder cert must already be specified.)
  135  *   If OCSP checking is not enabled, this will be recorded and take effect
  136  *   whenever it is enabled.

  138  *   CERTCertDBHandle *handle
  139  *     Cert database on which OCSP checking should use the default responder.
  140  * RETURN:
  141  *   Returns SECFailure if an error occurred; SECSuccess otherwise.
  142  *   No errors are especially likely unless the caller did not previously
  143  *   perform a successful call to SetOCSPDefaultResponder (in which case
  144  *   the error set will be SEC_ERROR_OCSP_NO_DEFAULT_RESPONDER).

  146 extern SECStatus
  147 CERT_EnableOCSPDefaultResponder(CERTCertDBHandle *handle);

  173 /*
  174  * FUNCTION: CERT_CreateOCSPRequest
  175  *   Creates a CERTOCSPRequest, requesting the status of the certs in 
  176  *   the given list.
  177  * INPUTS:
  178  *   CERTCertList *certList
  179  *     A list of certs for which status will be requested.
  180  *     Note that all of these certificates should have the same issuer,
  181  *     or it's expected the response will be signed by a trusted responder.
  182  *     If the certs need to be broken up into multiple requests, that
  183  *     must be handled by the caller (and thus by having multiple calls
  184  *     to this routine), who knows about where the request(s) are being
  185  *     sent and whether there are any trusted responders in place.
  186  *   int64 time
  187  *     Indicates the time for which the certificate status is to be 
  188  *     determined -- this may be used in the search for the cert's issuer
  189  *     but has no effect on the request itself.
  190  *   PRBool addServiceLocator
  191  *     If true, the Service Locator extension should be added to the
  192  *     single request(s) for each cert.
  193  *   CERTCertificate *signerCert
  194  *     If non-NULL, means sign the request using this cert.  Otherwise,
  195  *     do not sign.
  196  *     XXX note that request signing is not yet supported; see comment in code
  197  * RETURN:
  198  *   A pointer to a CERTOCSPRequest structure containing an OCSP request
  199  *   for the cert list.  On error, null is returned, with an error set
  200  *   indicating the reason.  This is likely SEC_ERROR_UNKNOWN_ISSUER.
  201  *   (The issuer is needed to create a request for the certificate.)
  202  *   Other errors are low-level problems (no memory, bad database, etc.).
  203  */
  204 extern CERTOCSPRequest *
  205 CERT_CreateOCSPRequest(CERTCertList *certList, int64 time, 
  206                        PRBool addServiceLocator,
  207                        CERTCertificate *signerCert);

  209 /*
  210  * FUNCTION: CERT_AddOCSPAcceptableResponses
  211  *   Add the AcceptableResponses extension to an OCSP Request.
  212  * INPUTS:
  213  *   CERTOCSPRequest *request
  214  *     The request to which the extension should be added.
  215  *   SECOidTag responseType0, ...
  216  *     A list (of one or more) of SECOidTag -- each of the response types
  217  *     to be added.  The last OID *must* be SEC_OID_PKIX_OCSP_BASIC_RESPONSE.
  218  *     (This marks the end of the list, and it must be specified because a
  219  *     client conforming to the OCSP standard is required to handle the basic
  220  *     response type.)  The OIDs are not checked in any way.
  221  * RETURN:
  222  *   SECSuccess if the extension is added; SECFailure if anything goes wrong.
  223  *   All errors are internal or low-level problems (e.g. no memory).
  224  */
  225 extern SECStatus
  226 CERT_AddOCSPAcceptableResponses(CERTOCSPRequest *request,
  227                                 SECOidTag responseType0, ...);

  305 /*
  306  * FUNCTION: CERT_GetEncodedOCSPResponse
  307  *   Creates and sends a request to an OCSP responder, then reads and
  308  *   returns the (encoded) response.
  309  * INPUTS:
  310  *   PRArenaPool *arena
  311  *     Pointer to arena from which return value will be allocated.
  312  *     If NULL, result will be allocated from the heap (and thus should
  313  *     be freed via SECITEM_FreeItem).
  314  *   CERTCertList *certList
  315  *     A list of certs for which status will be requested.
  316  *     Note that all of these certificates should have the same issuer,
  317  *     or it's expected the response will be signed by a trusted responder.
  318  *     If the certs need to be broken up into multiple requests, that
  319  *     must be handled by the caller (and thus by having multiple calls
  320  *     to this routine), who knows about where the request(s) are being
  321  *     sent and whether there are any trusted responders in place.
  322  *   char *location
  323  *     The location of the OCSP responder (a URL).
  324  *   int64 time
  325  *     Indicates the time for which the certificate status is to be 
  326  *     determined -- this may be used in the search for the cert's issuer
  327  *     but has no other bearing on the operation.
  328  *   PRBool addServiceLocator
  329  *     If true, the Service Locator extension should be added to the
  330  *     single request(s) for each cert.
  331  *   CERTCertificate *signerCert
  332  *     If non-NULL, means sign the request using this cert.  Otherwise,
  333  *     do not sign.
  334  *   void *pwArg
  335  *     Pointer to argument for password prompting, if needed.  (Definitely
  336  *     not needed if not signing.)
  337  * OUTPUTS:
  338  *   CERTOCSPRequest **pRequest
  339  *     Pointer in which to store the OCSP request created for the given
  340  *     list of certificates.  It is only filled in if the entire operation
  341  *     is successful and the pointer is not null -- and in that case the
  342  *     caller is then reponsible for destroying it.
  343  * RETURN:
  344  *   Returns a pointer to the SECItem holding the response.
  345  *   On error, returns null with error set describing the reason:
  346  *        SEC_ERROR_UNKNOWN_ISSUER
  347  *        SEC_ERROR_CERT_BAD_ACCESS_LOCATION
  348  *        SEC_ERROR_OCSP_BAD_HTTP_RESPONSE
  349  *   Other errors are low-level problems (no memory, bad database, etc.).
  350  */
  351 extern SECItem *
  352 CERT_GetEncodedOCSPResponse(PRArenaPool *arena, CERTCertList *certList,
  353                             char *location, int64 time,
  354                             PRBool addServiceLocator,
  355                             CERTCertificate *signerCert, void *pwArg,
  356                             CERTOCSPRequest **pRequest);

  358 /*
  359  * FUNCTION: CERT_VerifyOCSPResponseSignature
  360  *   Check the signature on an OCSP Response.  Will also perform a
  361  *   verification of the signer's certificate.  Note, however, that a
  362  *   successful verification does not make any statement about the
  363  *   signer's *authority* to provide status for the certificate(s),
  364  *   that must be checked individually for each certificate.
  365  * INPUTS:
  366  *   CERTOCSPResponse *response
  367  *     Pointer to response structure with signature to be checked.
  368  *   CERTCertDBHandle *handle
  369  *     Pointer to CERTCertDBHandle for certificate DB to use for verification.
  370  *   void *pwArg
  371  *     Pointer to argument for password prompting, if needed.
  372  *   CERTCertificate *issuerCert
  373  *     Issuer of the certificate that generated the OCSP request.
  374  * OUTPUTS:
  375  *   CERTCertificate **pSignerCert
  376  *     Pointer in which to store signer's certificate; only filled-in if
  377  *     non-null.
  378  * RETURN:
  379  *   Returns SECSuccess when signature is valid, anything else means invalid.
  380  *   Possible errors set:
  381  *        SEC_ERROR_OCSP_MALFORMED_RESPONSE - unknown type of ResponderID
  382  *        SEC_ERROR_INVALID_TIME - bad format of "ProducedAt" time
  383  *        SEC_ERROR_UNKNOWN_SIGNER - signer's cert could not be found
  384  *        SEC_ERROR_BAD_SIGNATURE - the signature did not verify
  385  *   Other errors are any of the many possible failures in cert verification
  386  *   (e.g. SEC_ERROR_REVOKED_CERTIFICATE, SEC_ERROR_UNTRUSTED_ISSUER) when
  387  *   verifying the signer's cert, or low-level problems (no memory, etc.)
  388  */
  389 extern SECStatus
  390 CERT_VerifyOCSPResponseSignature(CERTOCSPResponse *response,        
  391                                  CERTCertDBHandle *handle, void *pwArg,
  392                                  CERTCertificate **pSignerCert,
  393                                  CERTCertificate *issuerCert);

  417  *   Checks the status of a certificate via OCSP.  Will only check status for

  447  *        SEC_ERROR_OCSP_UNAUTHORIZED_REQUEST
  448  *        SEC_ERROR_OCSP_UNAUTHORIZED_RESPONSE
  449  *        SEC_ERROR_OCSP_UNKNOWN_CERT
  450  *        SEC_ERROR_OCSP_UNKNOWN_RESPONSE_STATUS
  451  *        SEC_ERROR_OCSP_UNKNOWN_RESPONSE_TYPE

  453  *        SEC_ERROR_BAD_SIGNATURE
  454  *        SEC_ERROR_CERT_BAD_ACCESS_LOCATION
  455  *        SEC_ERROR_INVALID_TIME
  456  *        SEC_ERROR_REVOKED_CERTIFICATE
  457  *        SEC_ERROR_UNKNOWN_ISSUER
  458  *        SEC_ERROR_UNKNOWN_SIGNER
  459  *
  460  *   Other errors are any of the many possible failures in cert verification
  461  *   (e.g. SEC_ERROR_REVOKED_CERTIFICATE, SEC_ERROR_UNTRUSTED_ISSUER) when
  462  *   verifying the signer's cert, or low-level problems (error allocating
  463  *   memory, error performing ASN.1 decoding, etc.).
