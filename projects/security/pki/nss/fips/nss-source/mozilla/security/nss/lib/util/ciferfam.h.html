   37 /*
   38  * ciferfam.h - cipher familie IDs used for configuring ciphers for export
   39  *              control
   40  *
   41  * $Id$
   42  */
   43 
   44 #ifndef _CIFERFAM_H_
   45 #define _CIFERFAM_H_
   46 
   47 /* Cipher Suite "Families" */
   48 #define CIPHER_FAMILY_PKCS12                        "PKCS12"
   49 #define CIPHER_FAMILY_SMIME                        "SMIME"
   50 #define CIPHER_FAMILY_SSL2                        "SSLv2"
   51 #define CIPHER_FAMILY_SSL3                        "SSLv3"
   52 #define CIPHER_FAMILY_SSL                        "SSL"
   53 #define CIPHER_FAMILY_ALL                        ""
   54 #define CIPHER_FAMILY_UNKNOWN                        "UNKNOWN"
   55 
   56 #define CIPHER_FAMILYID_MASK                        0xFFFF0000L
   57 #define CIPHER_FAMILYID_SSL                        0x00000000L
   58 #define CIPHER_FAMILYID_SMIME                        0x00010000L
   59 #define CIPHER_FAMILYID_PKCS12                        0x00020000L
   60 
   61 /* SMIME "Cipher Suites" */
   62 /*
   63  * Note that it is assumed that the cipher number itself can be used
   64  * as a bit position in a mask, and that mask is currently 32 bits wide.
   65  * So, if you want to add a cipher that is greater than 0033, secmime.c
   66  * needs to be made smarter at the same time.
   67  */
   68 #define        SMIME_RC2_CBC_40                (CIPHER_FAMILYID_SMIME | 0001)
   69 #define        SMIME_RC2_CBC_64                (CIPHER_FAMILYID_SMIME | 0002)
   70 #define        SMIME_RC2_CBC_128                (CIPHER_FAMILYID_SMIME | 0003)
   71 #define        SMIME_DES_CBC_56                (CIPHER_FAMILYID_SMIME | 0011)
   72 #define        SMIME_DES_EDE3_168                (CIPHER_FAMILYID_SMIME | 0012)
   73 #define        SMIME_RC5PAD_64_16_40                (CIPHER_FAMILYID_SMIME | 0021)
   74 #define        SMIME_RC5PAD_64_16_64                (CIPHER_FAMILYID_SMIME | 0022)
   75 #define        SMIME_RC5PAD_64_16_128                (CIPHER_FAMILYID_SMIME | 0023)
   76 #define        SMIME_FORTEZZA                        (CIPHER_FAMILYID_SMIME | 0031)
   77 
   78 /* PKCS12 "Cipher Suites" */
   79 
   80 #define        PKCS12_RC2_CBC_40                (CIPHER_FAMILYID_PKCS12 | 0001)
   81 #define        PKCS12_RC2_CBC_128                (CIPHER_FAMILYID_PKCS12 | 0002)
   82 #define        PKCS12_RC4_40                        (CIPHER_FAMILYID_PKCS12 | 0011)
   83 #define        PKCS12_RC4_128                        (CIPHER_FAMILYID_PKCS12 | 0012)
   84 #define        PKCS12_DES_56                        (CIPHER_FAMILYID_PKCS12 | 0021)
   85 #define        PKCS12_DES_EDE3_168                (CIPHER_FAMILYID_PKCS12 | 0022)
   86 
   87 /* SMIME version numbers are negative, to avoid colliding with SSL versions */
   88 #define SMIME_LIBRARY_VERSION_1_0                        -0x0100
   89 
   90 #endif /* _CIFERFAM_H_ */
