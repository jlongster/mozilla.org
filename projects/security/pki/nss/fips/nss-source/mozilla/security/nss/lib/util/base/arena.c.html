  173 /*

  175  *
  176  * This method is only present in debug builds.
  177  *
  178  * If the specified pointer is a valid pointer to an NSSArena object,
  179  * this routine will return PR_SUCCESS.  Otherwise, it will put an
  180  * error on the error stack and return PR_FAILURE.
  181  *
  182  * The error may be one of the following values:
  183  *  NSS_ERROR_INVALID_ARENA
  184  *
  185  * Return value:
  186  *  PR_SUCCESS if the pointer is valid
  187  *  PR_FAILURE if it isn't

  190 NSS_IMPLEMENT PRStatus
  191 nssArena_verifyPointer
  192 (
  193   const NSSArena *arena
  194 )
  195 {
  196   PRStatus rv[1];
  197 
  198   rv[1] = nssPointerTracker_initialize(&arena_pointer_tracker);
  199   if( PR_SUCCESS != rv[1] ) {
  200     /*
  201      * This is a little disingenious.  We have to initialize the
  202      * tracker, because someone could "legitimately" try to verify
  203      * an arena pointer before one is ever created.  And this step
  204      * might fail, due to lack of memory.  But the only way that
  205      * this step can fail is if it's doing the call_once stuff,
  206      * (later calls just no-op).  And if it didn't no-op, there
  207      * aren't any valid arenas.. so the argument certainly isn't one.
  208      */
  209     nss_SetError(NSS_ERROR_INVALID_ARENA[8][9]);
  210     return PR_FAILURE;
  211   }
  212 
  213   rv[1] = nssPointerTracker_verify(&arena_pointer_tracker, arena);
  214   if( PR_SUCCESS != rv[1] ) {
  215     nss_SetError(NSS_ERROR_INVALID_ARENA[8][9]);
  216     return PR_FAILURE;
  217   }
  218 
  219   return PR_SUCCESS;
  220 }

  232 /*
  233  * nssArena_registerDestructor
  234  *
  235  * This routine stores a pointer to a callback and an arbitrary
  236  * pointer-sized argument in the arena, at the current point in
  237  * the mark stack.  If the arena is destroyed, or an "earlier"
  238  * mark is released, then this destructor will be called at that
  239  * time.  Note that the destructor will be called with the arena
  240  * locked, which means the destructor may free memory in that
  241  * arena, but it may not allocate or cause to be allocated any
  242  * memory.  This callback facility was included to support our
  243  * debug-version pointer-tracker feature; overuse runs counter to
  244  * the the original intent of arenas.  This routine returns a 
  245  * PRStatus value; if successful, it will return PR_SUCCESS.  If 
  246  * unsuccessful, it will set an error on the error stack and 
  247  * return PR_FAILURE.
  248  *
  249  * The error may be one of the following values:
  250  *  NSS_ERROR_INVALID_ARENA
  251  *  NSS_ERROR_NO_MEMORY
  252  *
  253  * Return value:
  254  *  PR_SUCCESS
  255  *  PR_FAILURE
  256  */

 1002  * nss_ZRealloc
 1003  *
 1004  * This routine reallocates a block of memory obtained by calling
 1005  * nss_ZAlloc or nss_ZRealloc.  The portion of memory

 1026 )

 1033     /* Wrapped */

 1039     nss_SetError(NSS_ERROR_INVALID_POINTER[9]);
 1040     return (void *)NULL;
 1041   }

 1044     - sizeof(struct pointer_header));
 1045 
 1046   /* Check any magic here */
 1047

 1049     /* saves thrashing */
 1050     return pointer;
 1051   }
 1052

 1054     /* Heap */
 1055     new_h = (struct pointer_header *)PR_Calloc(1, my_newSize);
 1056     if( (struct pointer_header *)NULL == new_h ) {
 1057       nss_SetError(NSS_ERROR_NO_MEMORY[8][9]);
 1058       return (void *)NULL;
 1059     }

 1064

 1068                            0, (newSize - h->size[4][8][9]));

 1070       (void)nsslibc_memcpy(rv[1], pointer, newSize);
 1071     }
 1072

 1084     }
 1085 #endif /* NSSDEBUG */
 1086

 1092     }

 1094 #ifdef ARENA_THREADMARK

 1099         return (void *)NULL;

 1101     }
 1102 #endif /* ARENA_THREADMARK */
 1103 
 1104     if( newSize < h->size[4][8][9] ) {
