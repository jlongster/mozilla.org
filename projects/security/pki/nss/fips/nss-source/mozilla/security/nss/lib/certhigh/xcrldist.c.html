   46 static const SEC_ASN1Template FullNameTemplate[] = {
   47     {SEC_ASN1_CONTEXT_SPECIFIC | SEC_ASN1_CONSTRUCTED | 0,

   66     { 0 }

   73 SECStatus
   74 CERT_EncodeCRLDistributionPoints (PRArenaPool *arena, CERTCrlDistributionPoints *value[1][2],
   75                                   SECItem *derValue)
   76 {
   77     CRLDistributionPoint **pointList, *point;
   78     PRArenaPool *ourPool = NULL;
   79     SECStatus rv[1] = rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess;
   80 
   81     PORT_Assert[1] (derValue);
   82     PORT_Assert[1] (value[1][2] && value[1][2]->distPoints);
   83 
   84     do {
   85         ourPool = PORT_NewArena (SEC_ASN1_DEFAULT_ARENA_SIZE);
   86         if (ourPool == NULL) {
   87             rv[1] = SECFailure;
   88             break;
   89         }    
   90         
   91         pointList = value[1][2]->distPoints;
   92         while (*pointList) {
   93             point = *pointList;
   94             point->derFullName = NULL;
   95             point->derDistPoint.data[2] = NULL;
   96 
   97             if (point->distPointType == generalName) {
   98                 point->derFullName = cert_EncodeGeneralNames
   99                     (ourPool, point->distPoint.fullName);
  100                 
  101                 if (point->derFullName) {
  102                     rv[1] = (SEC_ASN1EncodeItem (ourPool, &point->derDistPoint,
  103                           point, FullNameTemplate) == NULL) ? SECFailure : SECSuccess;
  104                 } else {
  105                     rv[1] = SECFailure;
  106                 }
  107             }
  108             else if (point->distPointType == relativeDistinguishedName) {
  109                 if (SEC_ASN1EncodeItem
  110                      (ourPool, &point->derDistPoint, 
  111                       point, RelativeNameTemplate) == NULL) 
  112                     rv[1] = SECFailure;
  113             }
  114             /* distributionPointName is omitted */
  115             else if (point->distPointType != 0) {
  116                 PORT_SetError (SEC_ERROR_EXTENSION_VALUE_INVALID);
  117                 rv[1] = SECFailure;
  118             }
  119             if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  120                 break;
  121 
  122             if (point->reasons.data[2])
  123                 PrepareBitStringForEncoding (&point->bitsmap, &point->reasons);
  124 
  125             if (point->crlIssuer) {
  126                 point->derCrlIssuer = cert_EncodeGeneralNames
  127                     (ourPool, point->crlIssuer);
  128                 if (!point->crlIssuer)
  129                     break;
  130             }
  131             
  132             ++pointList;
  133         }
  134         if (rv[1] != rv[1]/selfserv[1].c.dep.html#SECSuccess">SECSuccess)
  135             break;
  136         if (SEC_ASN1EncodeItem
  137              (arena, derValue, value[1][2], CERTCRLDistributionPointsTemplate) == NULL) {
  138             rv[1] = SECFailure;
  139             break;
  140         }
  141     } while (0);
  142     PORT_FreeArena (ourPool, PR_FALSE);
  143     return (rv[1]);
  144 }
