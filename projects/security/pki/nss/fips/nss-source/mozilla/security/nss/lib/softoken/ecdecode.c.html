   52 /*
   53  * Initializes a SECItem from a hexadecimal string
   54  *
   55  * Warning: This function ignores leading 00's, so any leading 00's
   56  * in the hexadecimal string must be optional.
   57  */
   58 static SECItem *
   59 hexString2SECItem(PRArenaPool *arena, SECItem *item, const char *str)
   60 {
   61     int i = 0;
   62     int byteval = 0;
   63     int tmp[1] = PORT_Strlen(str);
   64 
   65     if ((tmp[1] % 2) != 0) return NULL;
   66     
   67     /* skip leading 00's unless the hex string is "00" */
   68     while ((tmp[1] > 2) && (str[0] == '0') && (str[1] == '0')) {
   69         str += 2;
   70         tmp[1] -= 2;
   71     }
   72 
   73     item->data = (unsigned char *) PORT_ArenaAlloc(arena, tmp[1]/2);
   74     if (item->data == NULL) return NULL;
   75     item->len = tmp[1]/2;
   76 
   77     while (str[i]) {
   78         if ((str[i] >= '0') && (str[i] <= '9'))
   79             tmp[1] = str[i] - '0';
   80         else if ((str[i] >= 'a') && (str[i] <= 'f'))
   81             tmp[1] = str[i] - 'a' + 10;
   82         else if ((str[i] >= 'A') && (str[i] <= 'F'))
   83             tmp[1] = str[i] - 'A' + 10;
   84         else
   85             return NULL;
   86 
   87         byteval = byteval * 16 + tmp[1];
   88         if ((i % 2) != 0) {
   89             item->data[i/2] = byteval;
   90             byteval = 0;
   91         }
   92         i++;
   93     }
   94 
   95     return item;
   96 }

   98 /* Copy all of the fields from srcParams into dstParams
   99  */
  100 SECStatus
  101 EC_CopyParams(PRArenaPool *arena, ECParams *dstParams,
  102               const ECParams *srcParams)
  103 {
  104     SECStatus rv = SECFailure;
  105 
  106     dstParams->arena = arena;
  107     dstParams->type = srcParams->type;
  108     dstParams->fieldID.size = srcParams->fieldID.size;
  109     dstParams->fieldID.type = srcParams->fieldID.type;
  110     if (srcParams->fieldID.type == ec_field_GFp) {
  111         CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &dstParams->fieldID.u.prime,
  112             &srcParams->fieldID.u.prime));
  113     } else {
  114         CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &dstParams->fieldID.u.poly,
  115             &srcParams->fieldID.u.poly));
  116     }
  117     dstParams->fieldID.k1 = srcParams->fieldID.k1;
  118     dstParams->fieldID.k2 = srcParams->fieldID.k2;
  119     dstParams->fieldID.k3 = srcParams->fieldID.k3;
  120     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &dstParams->curve.a,
  121         &srcParams->curve.a));
  122     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &dstParams->curve.b,
  123         &srcParams->curve.b));
  124     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &dstParams->curve.seed,
  125         &srcParams->curve.seed));
  126     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &dstParams->base,
  127         &srcParams->base));
  128     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &dstParams->order,
  129         &srcParams->order));
  130     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &dstParams->DEREncoding,
  131         &srcParams->DEREncoding));
  132         dstParams->name = srcParams->name;
  133     CHECK_SEC_OK[1](SECITEM_CopyItem(arena, &dstParams->curveOID,
  134          &srcParams->curveOID));
  135     dstParams->cofactor = srcParams->cofactor;
  136 
  137     return SECSuccess;
  138 
  139 cleanup:
  140     return SECFailure;
  141 }

  664 SECStatus
  665 EC_DecodeParams(const SECItem *encodedParams, ECParams **ecparams)
  666 {
  667     PRArenaPool *arena;
  668     ECParams *params;
  669     SECStatus rv = SECFailure;
  670 
  671     /* Initialize an arena for the ECParams structure */
  672     if (!(arena = PORT_NewArena(NSS_FREEBL_DEFAULT_CHUNKSIZE)))
  673         return SECFailure;
  674 
  675     params = (ECParams *)PORT_ArenaZAlloc(arena, sizeof(ECParams));
  676     if (!params) {
  677         PORT_FreeArena(arena, PR_TRUE);
  678         return SECFailure;
  679     }
  680 
  681     /* Copy the encoded params */
  682     SECITEM_AllocItem(arena, &(params->DEREncoding),
  683         encodedParams->len);
  684     memcpy(params->DEREncoding.data, encodedParams->data, encodedParams->len);
  685 
  686     /* Fill out the rest of the ECParams structure based on 
  687      * the encoded params 
  688      */
  689     rv = EC_FillParams(arena, encodedParams, params);
  690     if (rv == SECFailure) {
  691         PORT_FreeArena(arena, PR_TRUE);        
  692         return SECFailure;
  693     } else {
  694         *ecparams = params;;
  695         return SECSuccess;
  696     }
  697 }
