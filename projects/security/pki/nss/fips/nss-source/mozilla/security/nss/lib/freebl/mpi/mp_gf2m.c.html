   51 /* Multiply two binary polynomials mp_digits a, b.
   52  * Result is a polynomial with degree < 2 * MP_DIGIT_BITS - 1.
   53  * Output in two mp_digits rh, rl.
   54  */
   55 #if MP_DIGIT_BITS == 32
   56 void 
   57 s_bmul_1x1(mp_digit *rh, mp_digit *rl, const mp_digit a, const mp_digit b)
   58 {
   59     register mp_digit h, l, s;
   60     mp_digit tab[8], top2b = a >> 30; 
   61     register mp_digit a1, a2, a4;
   62 
   63     a1 = a & (0x3FFFFFFF); a2 = a1 << 1; a4 = a2 << 1;
   64 
   65     tab[0] =  0; tab[1] = a1;    tab[2] = a2;    tab[3] = a1^a2;
   66     tab[4] = a4; tab[5] = a1^a4; tab[6] = a2^a4; tab[7] = a1^a2^a4;
   67 
   68     s = tab[b       & 0x7]; l  = s;
   69     s = tab[b >>  3 & 0x7]; l ^= s <<  3; h  = s >> 29;
   70     s = tab[b >>  6 & 0x7]; l ^= s <<  6; h ^= s >> 26;
   71     s = tab[b >>  9 & 0x7]; l ^= s <<  9; h ^= s >> 23;
   72     s = tab[b >> 12 & 0x7]; l ^= s << 12; h ^= s >> 20;
   73     s = tab[b >> 15 & 0x7]; l ^= s << 15; h ^= s >> 17;
   74     s = tab[b >> 18 & 0x7]; l ^= s << 18; h ^= s >> 14;
   75     s = tab[b >> 21 & 0x7]; l ^= s << 21; h ^= s >> 11;
   76     s = tab[b >> 24 & 0x7]; l ^= s << 24; h ^= s >>  8;
   77     s = tab[b >> 27 & 0x7]; l ^= s << 27; h ^= s >>  5;
   78     s = tab[b >> 30      ]; l ^= s << 30; h ^= s >>  2;
   79 
   80     /* compensate for the top two bits of a */
   81 
   82     if (top2b & 01) { l ^= b << 30; h ^= b >> 2; } 
   83     if (top2b & 02) { l ^= b << 31; h ^= b >> 1; } 
   84 
   85     *rh = h; *rl = l;
   86 } 
   87 #else
   88 void 
   89 s_bmul_1x1(mp_digit *rh, mp_digit *rl, const mp_digit a, const mp_digit b)
   90 {
   91     register mp_digit h, l, s;
   92     mp_digit tab[16], top3b = a >> 61;
   93     register mp_digit a1, a2, a4, a8;
   94 
   95     a1 = a & (0x1FFFFFFFFFFFFFFFULL); a2 = a1 << 1; 
   96     a4 = a2 << 1; a8 = a4 << 1;
   97     tab[ 0] = 0;     tab[ 1] = a1;       tab[ 2] = a2;       tab[ 3] = a1^a2;
   98     tab[ 4] = a4;    tab[ 5] = a1^a4;    tab[ 6] = a2^a4;    tab[ 7] = a1^a2^a4;
   99     tab[ 8] = a8;    tab[ 9] = a1^a8;    tab[10] = a2^a8;    tab[11] = a1^a2^a8;
  100     tab[12] = a4^a8; tab[13] = a1^a4^a8; tab[14] = a2^a4^a8; tab[15] = a1^a2^a4^a8;
  101 
  102     s = tab[b       & 0xF]; l  = s;
  103     s = tab[b >>  4 & 0xF]; l ^= s <<  4; h  = s >> 60;
  104     s = tab[b >>  8 & 0xF]; l ^= s <<  8; h ^= s >> 56;
  105     s = tab[b >> 12 & 0xF]; l ^= s << 12; h ^= s >> 52;
  106     s = tab[b >> 16 & 0xF]; l ^= s << 16; h ^= s >> 48;
  107     s = tab[b >> 20 & 0xF]; l ^= s << 20; h ^= s >> 44;
  108     s = tab[b >> 24 & 0xF]; l ^= s << 24; h ^= s >> 40;
  109     s = tab[b >> 28 & 0xF]; l ^= s << 28; h ^= s >> 36;
  110     s = tab[b >> 32 & 0xF]; l ^= s << 32; h ^= s >> 32;
  111     s = tab[b >> 36 & 0xF]; l ^= s << 36; h ^= s >> 28;
  112     s = tab[b >> 40 & 0xF]; l ^= s << 40; h ^= s >> 24;
  113     s = tab[b >> 44 & 0xF]; l ^= s << 44; h ^= s >> 20;
  114     s = tab[b >> 48 & 0xF]; l ^= s << 48; h ^= s >> 16;
  115     s = tab[b >> 52 & 0xF]; l ^= s << 52; h ^= s >> 12;
  116     s = tab[b >> 56 & 0xF]; l ^= s << 56; h ^= s >>  8;
  117     s = tab[b >> 60      ]; l ^= s << 60; h ^= s >>  4;
  118 
  119     /* compensate for the top three bits of a */
  120 
  121     if (top3b & 01) { l ^= b << 61; h ^= b >> 3; } 
  122     if (top3b & 02) { l ^= b << 62; h ^= b >> 2; } 
  123     if (top3b & 04) { l ^= b << 63; h ^= b >> 1; } 
  124 
  125     *rh = h; *rl = l;
  126 }

  129 /* Compute xor-multiply of two binary polynomials  (a1, a0) x (b1, b0)  
  130  * result is a binary polynomial in 4 mp_digits r[4].
  131  * The caller MUST ensure that r has the right amount of space allocated.
  132  */
  133 void 
  134 s_bmul_2x2(mp_digit *r, const mp_digit a1, const mp_digit a0, const mp_digit b1,
  135            const mp_digit b0)
  136 {
  137     mp_digit m1, m0;
  138     /* r[3] = h1, r[2] = h0; r[1] = l1; r[0] = l0 */
  139     s_bmul_1x1(r+3, r+2, a1, b1);
  140     s_bmul_1x1(r+1, r, a0, b0);
  141     s_bmul_1x1(&m1, &m0, a0 ^ a1, b0 ^ b1);
  142     /* Correction on m1 ^= l1 ^ h1; m0 ^= l0 ^ h0; */
  143     r[2] ^= m1 ^ r[1] ^ r[3];  /* h0 ^= m1 ^ l1 ^ h1; */
  144     r[1]  = r[3] ^ r[2] ^ r[0] ^ m1 ^ m0;  /* l1 ^= l0 ^ h0 ^ m0; */
  145 }

  147 /* Compute xor-multiply of two binary polynomials  (a2, a1, a0) x (b2, b1, b0)  
  148  * result is a binary polynomial in 6 mp_digits r[6].
  149  * The caller MUST ensure that r has the right amount of space allocated.
  150  */
  151 void 
  152 s_bmul_3x3(mp_digit *r, const mp_digit a2, const mp_digit a1, const mp_digit a0, 
  153         const mp_digit b2, const mp_digit b1, const mp_digit b0)
  154 {
  155         mp_digit zm[4];
  156 
  157         s_bmul_1x1(r+5, r+4, a2, b2);         /* fill top 2 words */
  158         s_bmul_2x2(zm, a1, a2^a0, b1, b2^b0); /* fill middle 4 words */
  159         s_bmul_2x2(r, a1, a0, b1, b0);        /* fill bottom 4 words */
  160 
  161         zm[3] ^= r[3];
  162         zm[2] ^= r[2]; 
  163         zm[1] ^= r[1] ^ r[5];
  164         zm[0] ^= r[0] ^ r[4];
  165 
  166         r[5]  ^= zm[3];
  167         r[4]  ^= zm[2];
  168         r[3]  ^= zm[1];
  169         r[2]  ^= zm[0];
  170 }

  172 /* Compute xor-multiply of two binary polynomials  (a3, a2, a1, a0) x (b3, b2, b1, b0)  
  173  * result is a binary polynomial in 8 mp_digits r[8].
  174  * The caller MUST ensure that r has the right amount of space allocated.
  175  */
  176 void s_bmul_4x4(mp_digit *r, const mp_digit a3, const mp_digit a2, const mp_digit a1, 
  177         const mp_digit a0, const mp_digit b3, const mp_digit b2, const mp_digit b1, 
  178         const mp_digit b0)
  179 {
  180         mp_digit zm[4];
  181 
  182         s_bmul_2x2(r+4, a3, a2, b3, b2);            /* fill top 4 words */
  183         s_bmul_2x2(zm, a3^a1, a2^a0, b3^b1, b2^b0); /* fill middle 4 words */
  184         s_bmul_2x2(r, a1, a0, b1, b0);              /* fill bottom 4 words */
  185 
  186         zm[3] ^= r[3] ^ r[7]; 
  187         zm[2] ^= r[2] ^ r[6]; 
  188         zm[1] ^= r[1] ^ r[5]; 
  189         zm[0] ^= r[0] ^ r[4]; 
  190 
  191         r[5]  ^= zm[3];    
  192         r[4]  ^= zm[2];
  193         r[3]  ^= zm[1];    
  194         r[2]  ^= zm[0];
  195 }

  197 /* Compute addition of two binary polynomials a and b,
  198  * store result in c; c could be a or b, a and b could be equal; 
  199  * c is the bitwise XOR of a and b.
  200  */
  201 mp_err
  202 mp_badd(const mp_int *a, const mp_int *b, mp_int *c)
  203 {
  204     mp_digit *pa, *pb, *pc;
  205     mp_size ix;
  206     mp_size used_pa, used_pb;
  207     mp_err res = MP_OKAY;
  208 
  209     /* Add all digits up to the precision of b.  If b had more
  210      * precision than a initially, swap a, b first
  211      */
  212     if (MP_USED(a) >= MP_USED(b)) {
  213         pa = MP_DIGITS(a);
  214         pb = MP_DIGITS(b);
  215         used_pa = MP_USED(a);
  216         used_pb = MP_USED(b);
  217     } else {
  218         pa = MP_DIGITS(b);
  219         pb = MP_DIGITS(a);
  220         used_pa = MP_USED(b);
  221         used_pb = MP_USED(a);
  222     }
  223 
  224     /* Make sure c has enough precision for the output value */
  225     MP_CHECKOK( s_mp_pad(c, used_pa) );
  226 
  227     /* Do word-by-word xor */
  228     pc = MP_DIGITS(c);
  229     for (ix = 0; ix < used_pb; ix++) {
  230         (*pc++) = (*pa++) ^ (*pb++);
  231     }
  232 
  233     /* Finish the rest of digits until we're actually done */
  234     for (; ix < used_pa; ++ix) {
  235         *pc++ = *pa++;
  236     }
  237 
  238     MP_USED(c) = used_pa;
  239     MP_SIGN(c) = ZPOS;
  240     s_mp_clamp[1](c);
  241 
  242 CLEANUP:
  243     return res;
  244 }

  248 /* Compute binary polynomial multiply d = a * b */
  249 static void 
  250 s_bmul_d(const mp_digit *a, mp_size a_len, mp_digit b, mp_digit *d)
  251 {
  252     mp_digit a_i, a0b0, a1b1, carry = 0;
  253     while (a_len--) {
  254         a_i = *a++;
  255         s_bmul_1x1(&a1b1, &a0b0, a_i, b);
  256         *d++ = a0b0 ^ carry;
  257         carry = a1b1;
  258     }
  259     *d = carry;
  260 }

  262 /* Compute binary polynomial xor multiply accumulate d ^= a * b */
  263 static void 
  264 s_bmul_d_add(const mp_digit *a, mp_size a_len, mp_digit b, mp_digit *d)
  265 {
  266     mp_digit a_i, a0b0, a1b1, carry = 0;
  267     while (a_len--) {
  268         a_i = *a++;
  269         s_bmul_1x1(&a1b1, &a0b0, a_i, b);
  270         *d++ ^= a0b0 ^ carry;
  271         carry = a1b1;
  272     }
  273     *d ^= carry;
  274 }

  276 /* Compute binary polynomial xor multiply c = a * b.  
  277  * All parameters may be identical.
  278  */
  279 mp_err 
  280 mp_bmul(const mp_int *a, const mp_int *b, mp_int *c)
  281 {
  282     mp_digit *pb, b_i;
  283     mp_int tmp[1];
  284     mp_size ib, a_used, b_used;
  285     mp_err res = MP_OKAY;
  286 
  287     MP_DIGITS(&tmp[1]) = 0;
  288 
  289     ARGCHK(a != NULL && b != NULL && c != NULL, MP_BADARG);
  290 
  291     if (a == c) {
  292         MP_CHECKOK( mp_init_copy(&tmp[1], a) );
  293         if (a == b)
  294             b = &tmp[1];
  295         a = &tmp[1];
  296     } else if (b == c) {
  297         MP_CHECKOK( mp_init_copy(&tmp[1], b) );
  298         b = &tmp[1];
  299     }
  300 
  301     if (MP_USED(a) < MP_USED(b)) {
  302         const mp_int *xch = b;      /* switch a and b if b longer */
  303         b = a;
  304         a = xch;
  305     }
  306 
  307     MP_USED(c) = 1; MP_DIGIT(c, 0) = 0;
  308     MP_CHECKOK( s_mp_pad(c, USED(a) + USED(b)) );
  309 
  310     pb = MP_DIGITS(b);
  311     s_bmul_d(MP_DIGITS(a), MP_USED(a), *pb++, MP_DIGITS(c));
  312 
  313     /* Outer loop:  Digits of b */
  314     a_used = MP_USED(a);
  315     b_used = MP_USED(b);
  316         MP_USED(c) = a_used + b_used;
  317     for (ib = 1; ib < b_used; ib++) {
  318         b_i = *pb++;
  319 
  320         /* Inner product:  Digits of a */
  321         if (b_i)
  322             s_bmul_d_add(MP_DIGITS(a), a_used, b_i, MP_DIGITS(c) + ib);
  323         else
  324             MP_DIGIT(c, ib + a_used) = b_i;
  325     }
  326 
  327     s_mp_clamp[1](c);
  328 
  329     SIGN(c) = ZPOS;
  330 
  331 CLEANUP:
  332     mp_clear(&tmp[1]);
  333     return res;
  334 }

  337 /* Compute modular reduction of a and store result in r.  
  338  * r could be a. 
  339  * For modular arithmetic, the irreducible polynomial f(t) is represented 
  340  * as an array of int[], where f(t) is of the form: 
  341  *     f(t) = t^p[0] + t^p[1] + ... + t^p[k]
  342  * where m = p[0] > p[1] > ... > p[k] = 0.
  343  */
  344 mp_err
  345 mp_bmod(const mp_int *a, const unsigned int p[], mp_int *r)
  346 {
  347     int j, k;
  348     int n, dN, d0, d1;
  349     mp_digit zz, *z, tmp[1];
  350     mp_size used;
  351     mp_err res = MP_OKAY;
  352 
  353     /* The algorithm does the reduction in place in r, 
  354      * if a != r, copy a into r first so reduction can be done in r
  355      */
  356     if (a != r) {
  357         MP_CHECKOK( mp_copy(a, r) );
  358     }
  359     z = MP_DIGITS(r);
  360 
  361     /* start reduction */
  362     dN = p[0] / MP_DIGIT_BITS;
  363     used = MP_USED(r);
  364 
  365     for (j = used - 1; j > dN;) {
  366 
  367         zz = z[j];
  368         if (zz == 0) {
  369             j--; continue;
  370         }
  371         z[j] = 0;
  372 
  373         for (k = 1; p[k] > 0; k++) {
  374             /* reducing component t^p[k] */
  375             n = p[0] - p[k];
  376             d0 = n % MP_DIGIT_BITS;  
  377             d1 = MP_DIGIT_BITS - d0;
  378             n /= MP_DIGIT_BITS;
  379             z[j-n] ^= (zz>>d0);
  380             if (d0) 
  381                 z[j-n-1] ^= (zz<<d1);
  382         }
  383 
  384         /* reducing component t^0 */
  385         n = dN;  
  386         d0 = p[0] % MP_DIGIT_BITS;
  387         d1 = MP_DIGIT_BITS - d0;
  388         z[j-n] ^= (zz >> d0);
  389         if (d0) 
  390             z[j-n-1] ^= (zz << d1);
  391 
  392     }
  393 
  394     /* final round of reduction */
  395     while (j == dN) {
  396 
  397         d0 = p[0] % MP_DIGIT_BITS;
  398         zz = z[dN] >> d0;  
  399         if (zz == 0) break;
  400         d1 = MP_DIGIT_BITS - d0;
  401 
  402         /* clear up the top d1 bits */
  403         if (d0) z[dN] = (z[dN] << d1) >> d1; 
  404         *z ^= zz; /* reduction t^0 component */
  405 
  406         for (k = 1; p[k] > 0; k++) {
  407             /* reducing component t^p[k]*/
  408             n = p[k] / MP_DIGIT_BITS;
  409             d0 = p[k] % MP_DIGIT_BITS;
  410             d1 = MP_DIGIT_BITS - d0;
  411             z[n] ^= (zz << d0);
  412             tmp[1] = zz >> d1;
  413             if (d0 && tmp[1])
  414                 z[n+1] ^= tmp[1];
  415         }
  416     }
  417 
  418     s_mp_clamp[1](r);
  419 CLEANUP:
  420     return res;
  421 }

  423 /* Compute the product of two polynomials a and b, reduce modulo p, 
  424  * Store the result in r.  r could be a or b; a could be b.
  425  */
  426 mp_err 
  427 mp_bmulmod(const mp_int *a, const mp_int *b, const unsigned int p[], mp_int *r)
  428 {
  429     mp_err res;
  430     
  431     if (a == b) return mp_bsqrmod(a, p, r);
  432     if ((res = mp_bmul(a, b, r) ) != MP_OKAY)
  433         return res;
  434     return mp_bmod(r, p, r);
  435 }

  437 /* Compute binary polynomial squaring c = a*a mod p .  
  438  * Parameter r and a can be identical.
  439  */
  440 
  441 mp_err 
  442 mp_bsqrmod(const mp_int *a, const unsigned int p[], mp_int *r)
  443 {
  444     mp_digit *pa, *pr, a_i;
  445     mp_int tmp[1];
  446     mp_size ia, a_used;
  447     mp_err res;
  448 
  449     ARGCHK(a != NULL && r != NULL, MP_BADARG);
  450     MP_DIGITS(&tmp[1]) = 0;
  451 
  452     if (a == r) {
  453         MP_CHECKOK( mp_init_copy(&tmp[1], a) );
  454         a = &tmp[1];
  455     }
  456 
  457     MP_USED(r) = 1; MP_DIGIT(r, 0) = 0;
  458     MP_CHECKOK( s_mp_pad(r, 2*USED(a)) );
  459 
  460     pa = MP_DIGITS(a);
  461     pr = MP_DIGITS(r);
  462     a_used = MP_USED(a);
  463         MP_USED(r) = 2 * a_used;
  464 
  465     for (ia = 0; ia < a_used; ia++) {
  466         a_i = *pa++;
  467         *pr++ = gf2m_SQR0(a_i);
  468         *pr++ = gf2m_SQR1(a_i);
  469     }
  470 
  471     MP_CHECKOK( mp_bmod(r, p, r) );
  472     s_mp_clamp[1](r);
  473     SIGN(r) = ZPOS;
  474 
  475 CLEANUP:
  476     mp_clear(&tmp[1]);
  477     return res;
  478 }

  480 /* Compute binary polynomial y/x mod p, y divided by x, reduce modulo p.
  481  * Store the result in r. r could be x or y, and x could equal y.
  482  * Uses algorithm Modular_Division_GF(2^m) from 
  483  *     Chang-Shantz, S.  "From Euclid's GCD to Montgomery Multiplication to 
  484  *     the Great Divide".
  485  */
  486 int 
  487 mp_bdivmod(const mp_int *y, const mp_int *x, const mp_int *pp, 
  488     const unsigned int p[], mp_int *r)
  489 {
  490     mp_int aa, bb, uu;
  491     mp_int *a, *b, *u, *v;
  492     mp_err res = MP_OKAY;
  493 
  494     MP_DIGITS(&aa) = 0;
  495     MP_DIGITS(&bb) = 0;
  496     MP_DIGITS(&uu) = 0;
  497 
  498     MP_CHECKOK( mp_init_copy(&aa, x) );
  499     MP_CHECKOK( mp_init_copy(&uu, y) );
  500     MP_CHECKOK( mp_init_copy(&bb, pp) );
  501     MP_CHECKOK( s_mp_pad(r, USED(pp)) );
  502     MP_USED(r) = 1; MP_DIGIT(r, 0) = 0;
  503 
  504     a = &aa; b= &bb; u=&uu; v=r;
  505     /* reduce x and y mod p */
  506     MP_CHECKOK( mp_bmod(a, p, a) );
  507     MP_CHECKOK( mp_bmod(u, p, u) );
  508 
  509     while (!mp_isodd(a)) {
  510         s_mp_div2(a);
  511         if (mp_isodd(u)) {
  512             MP_CHECKOK( mp_badd(u, pp, u) );
  513         }
